001package org.xbib.io.compress.bzip2;
002
003import java.io.ByteArrayInputStream;
004import java.io.ByteArrayOutputStream;
005import java.io.IOException;
006
007import org.junit.Test;
008
009/**
010 * Tests BZip2HuffmanStageDecoder
011 */
012public class BZip2HuffmanStageDecoderTests {
013
014        /**
015         * Tests decoding an invalid symbol
016         * @throws Exception 
017         */
018        @Test(expected=IOException.class)
019        public void testExceptionOnInvalidSymbol() throws Exception {
020
021                byte[][] tableCodeLengths = { { 23, 23, 23, 22, 22, 21, 21, 19, 18, 17, 16, 15, 14, 13, 12, 11, 10, 9, 8, 7, 6, 5, 4, 3, 3, 3, 3, 3, 3, 3 } };
022                byte[] selectors = new byte[1024];
023
024                ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
025                BZip2BitOutputStream bitOutputStream = new BZip2BitOutputStream (outputStream);
026                bitOutputStream.writeBits (23, 8388607); // This value would be the 4th 23-length code
027                bitOutputStream.flush();
028
029                BZip2BitInputStream bitInputStream = new BZip2BitInputStream (new ByteArrayInputStream (outputStream.toByteArray()));
030                BZip2HuffmanStageDecoder decoder = new BZip2HuffmanStageDecoder (bitInputStream, tableCodeLengths[0].length, tableCodeLengths, selectors);
031
032                decoder.nextSymbol();
033
034        }
035
036
037        /**
038         * Tests that an exception is thrown when there is data beyond the available selectors
039         * @throws Exception
040         */
041        @Test(expected=IOException.class)
042        public void testExceptionOnTooMuchData() throws Exception {
043
044                byte[][] tableCodeLengths = { { 1, 1 } };
045                byte[] selectors = new byte[1];
046
047                ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
048                BZip2BitOutputStream bitOutputStream = new BZip2BitOutputStream (outputStream);
049                for (int i = 0; i < 51; i++) {
050                        bitOutputStream.writeBits (1, 0);
051                }
052                bitOutputStream.flush();
053
054                BZip2BitInputStream bitInputStream = new BZip2BitInputStream (new ByteArrayInputStream (outputStream.toByteArray()));
055                BZip2HuffmanStageDecoder decoder = new BZip2HuffmanStageDecoder (bitInputStream, tableCodeLengths[0].length, tableCodeLengths, selectors);
056
057                for (int i = 0; i < 51; i++) {
058                        decoder.nextSymbol();
059                }
060
061        }
062
063}
