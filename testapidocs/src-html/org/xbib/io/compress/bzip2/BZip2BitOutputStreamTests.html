014
015        // Boolean
016
017        /**
018         * Test writing 8 zeroes
019         * @throws java.io.IOException
020         */
021        @Test
022        public void testBooleanFalse8() throws IOException {
023
024                byte[] expected = { 0 };
025                ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
026                BZip2BitOutputStream outputStream = new BZip2BitOutputStream (byteArrayOutputStream);
027
028                for (int i = 0; i < 8; i++) {
029                        outputStream.writeBoolean (false);
030                }
031
032                outputStream.flush();
033                assertArrayEquals (expected, byteArrayOutputStream.toByteArray());
034
035        }
036

059        /**
060         * Test writing a single 1 in any position as a boolean
061         * @throws java.io.IOException
062         */
063        @Test
064        public void testBooleanSingleOne() throws IOException {
065
066                for (int i = 0; i < 8; i++) {
067
068                        byte[] expected = { (byte)(1 << (7 - i)) };
069                        ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
070                        BZip2BitOutputStream outputStream = new BZip2BitOutputStream (byteArrayOutputStream);
071        
072                        for (int j = 0; j < 8; j++) {
073                                outputStream.writeBoolean (j == i);
074                        }
075
076                        outputStream.flush();
077                        assertArrayEquals (expected, byteArrayOutputStream.toByteArray());
078
079                }
080
081        }
