018import java.io.File;

020import java.nio.file.Path;
021import java.nio.file.Paths;
022
023import static org.junit.Assert.assertEquals;
024import static org.junit.Assert.assertTrue;
025
026public class KnapsackSplitTests extends AbstractNodeTestHelper {
027
028    private final static Logger logger = LogManager.getLogger(KnapsackSplitTests.class);
029
030    @Test
031    public void testSplit() {
032        try {
033            File exportFile = File.createTempFile("split-import-", ".bulk");
034            Path exportPath = Paths.get(URI.create("file:" + exportFile.getAbsolutePath()));
035            for (int i = 0; i < 100; i++) {
036                client("1").index(new IndexRequest().index("index1").type("test1").id("doc" + i)
037                        .source("content", "Hello World " + i).refresh(true)).actionGet();
038            }
039            // 275 bytes = ~10 docs per archive
040            KnapsackExportRequestBuilder requestBuilder = new KnapsackExportRequestBuilder(client("1"))
041                    .setPath(exportPath)
042                    .setBytesToTransfer(ByteSizeValue.parseBytesSizeValue("275b", ""));
043            KnapsackExportResponse knapsackExportResponse = requestBuilder.execute().actionGet();
044            assertTrue(knapsackExportResponse.isRunning());
045            KnapsackStateRequestBuilder knapsackStateRequestBuilder =
046                    new KnapsackStateRequestBuilder(client("2"));
047            KnapsackStateResponse knapsackStateResponse = knapsackStateRequestBuilder.execute().actionGet();
048            // assertTrue(knapsackStateResponse.isExportActive(exportPath)); // why does this fail???
049            Thread.sleep(2000L);
050            // delete index
051            client("1").admin().indices().delete(new DeleteIndexRequest("index1")).actionGet();
052            KnapsackImportRequestBuilder knapsackImportRequestBuilder = new KnapsackImportRequestBuilder(client("1"))
053                    .setPath(exportPath)
054                    .setMaxActionsPerBulkRequest(100);
055            KnapsackImportResponse knapsackImportResponse = knapsackImportRequestBuilder.execute().actionGet();
056            Thread.sleep(2000L);
057            // count
058            long count = client("1").prepareCount("index1").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getCount();
059            assertEquals(10L, count);
060        } catch (Throwable t) {
061            logger.error(t.getMessage(), t);
062        }
063    }
064}
