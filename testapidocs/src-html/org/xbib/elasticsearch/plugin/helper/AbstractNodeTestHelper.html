029
030import static org.elasticsearch.common.settings.Settings.settingsBuilder;
031import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
032
033public abstract class AbstractNodeTestHelper {
034
035    private final static ESLogger logger = ESLoggerFactory.getLogger("test");
036
037    private Map<String, Node> nodes = new HashMap<>();
038
039    private Map<String, AbstractClient> clients = new HashMap<>();
040
041    private AtomicInteger counter = new AtomicInteger();
042
043    private String cluster;
044
045    private String host;
046
047    private int port;
048
049    protected void setClusterName() {
050        this.cluster = "test-aggregations-cluster-"
051                + getLocalAddress().getHostName()
052                + "-" + System.getProperty("user.name")
053                + "-" + counter.incrementAndGet();
054    }
055
056    protected String getClusterName() {
057        return cluster;
058    }
059
060    protected String getHome() {
061        return System.getProperty("path.home");
062    }
063
064    protected Settings getNodeSettings() {
065        return settingsBuilder()
066                .put("cluster.name", cluster)
067                .put("path.home", getHome())
068                .put("http.enabled", false)
069                .put("index.number_of_replicas", 0)
070                .put("plugin.types", KnapsackPlugin.class.getName())
071                .build();
072    }
073
074    @Before
075    public void startNodes() {
076        try {
077            setClusterName();
078            startNode("1");
079            startNode("2");
080            findNodeAddress();
081            ClientHelper.waitForCluster(client("1"), ClusterHealthStatus.GREEN, TimeValue.timeValueSeconds(30));
082        } catch (Throwable t) {
083            logger.error("startNodes failed", t);
084        }
085    }
086
087    @After
088    public void stopNodes() {
089        logger.info("stopping nodes");
090        try {
091            //ClientHelper.waitForCluster(client("1"), ClusterHealthStatus.YELLOW, TimeValue.timeValueSeconds(30));
092            //logger.info("waiting for recovery");
093            //ClientHelper.waitForRecovery(client("1"));
094            //logger.info("recovered");
095            //client("1").admin().indices().prepareDelete("_all").execute().actionGet();
096        } catch (Throwable e) {
097            logger.error("stop failed", e);
098        } finally {
099            logger.info("closing nodes");
100            try {
101                closeAllNodes();
102            } catch (Throwable t) {
103
104            }
105        }
106    }
107
108    protected Node startNode(String id) {
109        return buildNode(id).start();
110    }
111
112    public AbstractClient client(String id) {
113        return clients.get(id);
114    }
115
116    private Node buildNode(String id) {
117        Settings finalSettings = settingsBuilder()
118                .put(getNodeSettings())
119                .put("name", id)
120                .build();
121        Node node = nodeBuilder().settings(finalSettings).local(true).build();
122        AbstractClient client = (AbstractClient)node.client();
123        nodes.put(id, node);
124        clients.put(id, client);
125        return node;
126    }
127
128    protected void findNodeAddress() {
129        NodesInfoRequest nodesInfoRequest = new NodesInfoRequest().transport(true);
130        NodesInfoResponse response = client("1").admin().cluster().nodesInfo(nodesInfoRequest).actionGet();
131        Object obj = response.iterator().next().getTransport().getAddress()
132                .publishAddress();
133        if (obj instanceof InetSocketTransportAddress) {
134            InetSocketTransportAddress address = (InetSocketTransportAddress) obj;
135            host = address.address().getHostName();
136            port = address.address().getPort();
137        }
138    }
139
140    public void closeAllNodes() throws IOException {
141        for (AbstractClient client : clients.values()) {
142            client.close();
143        }
144        clients.clear();
145        for (Node node : nodes.values()) {
146            if (node != null) {
147                node.close();
148            }
149        }
150        nodes.clear();
151        deleteFiles();
152    }
153
154    private void deleteFiles() throws IOException {
155        Path directory = Paths.get(getHome() + "/data");
156        Files.walkFileTree(directory, new SimpleFileVisitor<Path>() {
157            @Override
158            public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
159                Files.delete(file);
160                return FileVisitResult.CONTINUE;
161            }
162
163            @Override
164            public FileVisitResult postVisitDirectory(Path dir, IOException exc) throws IOException {
165                Files.delete(dir);
166                return FileVisitResult.CONTINUE;
167            }
168
169        });
170
171    }
172
173    private final static InetAddress localAddress;
174
175    static {
176        InetAddress address;
177        try {
178            address = InetAddress.getLocalHost();
179        } catch (Throwable e) {
180            address = InetAddress.getLoopbackAddress();
181        }
182        localAddress = address;
183    }
184
185    public static InetAddress getLocalAddress() {
186        return localAddress;
187    }
188
189}
