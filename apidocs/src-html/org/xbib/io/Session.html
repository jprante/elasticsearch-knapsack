002    package org.xbib.io;
003    
004    import java.io.IOException;
005    
006    /**
007     * The Session interface is used for being opened, receive
008     * operations, and being closed. Sessions must be opened before the first
009     * operation and closed after the last operation.
010     */
011    public interface Session<P extends Packet> {
012    
013        enum Mode {
014    
015            READ, WRITE, READ_WRITE, APPEND, DEFERRED_WRITE, DRY, CONTROL, DELETE;
016        }
017    
018        /**
019         * Open valve with a given input/output mode
020         *
021         * @throws java.io.IOException if valve can not be opened
022         */
023        void open(Mode mode) throws IOException;
024    
025        P newPacket();
026    
027        P read() throws IOException;
028    
029        void write(P packet) throws IOException;
030    
031        /**
032         * Close valve
033         *
034         * @throws java.io.IOException if valve can not be closed
035         */
036        void close() throws IOException;
037    
038        /**
039         * Checks if this session has been successfully opened.
040         *
041         * @return true if the session is open
042         */
043        boolean isOpen();
044    }
