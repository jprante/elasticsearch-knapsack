002package org.xbib.io;
003
004import java.io.IOException;
005import java.net.URI;
006
007/**
008 * A Connection is an access to a resource via a scheme or a protocol.
009 * Each connection can serve multiple sessions in parallel.
010 */
011public interface Connection<S extends Session> {
012
013    /**
014     * Set URI of this connection
015     *
016     * @param uri
017     */
018    Connection setURI(URI uri);
019
020    /**
021     * Get URI of this connection
022     */
023    URI getURI();
024
025    /**
026     * Close connection and close all sessions
027     */
028    void close() throws IOException;
029
030    /**
031     * Create a new session on this connection
032     *
033     * @return the session
034     * @throws IOException if the session can not be created
035     */
036    S createSession() throws IOException;
037
038}
