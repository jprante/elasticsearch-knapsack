002package org.xbib.io;
003
004import java.util.Map;
005import java.util.ServiceLoader;
006import java.util.Set;
007import java.util.WeakHashMap;
008
009public class StreamCodecService {
010
011    private final static Map<String, StreamCodec> codecs = new WeakHashMap();
012    private final static StreamCodecService instance = new StreamCodecService();
013
014    private StreamCodecService() {
015        ServiceLoader<StreamCodec> loader = ServiceLoader.load(StreamCodec.class);
016        for (StreamCodec codec : loader) {
017            if (!codecs.containsKey(codec.getName())) {
018                codecs.put(codec.getName(), codec);
019            }
020        }
021    }
022
023    public static StreamCodecService getInstance() {
024        return instance;
025    }
026
027    public StreamCodec getCodec(String suffix) {
028        if (codecs.containsKey(suffix)) {
029            return codecs.get(suffix);
030        }
031        throw new IllegalArgumentException("Stream codec for " + suffix + " not found in " + codecs);
032    }
033    
034    public static Set<String> getCodecs() {
035        return codecs.keySet();
036    }
037}
