032    package org.xbib.io;
033    
034    import java.io.IOException;
035    import java.io.InputStream;
036    import java.io.OutputStream;
037    import java.io.Reader;
038    import java.io.Writer;
039    
040    public final class StreamUtil {
041    
042        public final static int BUFSIZE = 8192;
043    
044        public static void copy(InputStream in, OutputStream out) throws IOException {
045            byte[] buffer = new byte[BUFSIZE];
046            int len;
047            while ((len = in.read(buffer)) != -1) {
048                out.write(buffer, 0, len);
049            }
050        }
051    
052        public static void copy(Reader reader, Writer writer) throws IOException {
053            char[] buffer = new char[BUFSIZE];
054            int len;
055            while ((len = reader.read(buffer)) != -1) {
056                writer.write(buffer, 0, len);
057            }
058        }
059    
060    }
