002package org.xbib.io.commons;
003
004import org.xbib.io.Connection;
005
006import java.io.IOException;
007import java.net.URI;
008import java.util.ArrayList;
009import java.util.List;
010
011
012public class TarConnection implements Connection<TarSession> {
013
014    private List<TarSession> sessions = new ArrayList<TarSession>();
015    private URI uri;
016
017    @Override
018    public TarConnection setURI(URI uri) {
019        this.uri = uri;
020        return this;
021    }
022
023    @Override
024    public URI getURI() {
025        return uri;
026    }
027
028    @Override
029    public TarSession createSession() throws IOException {
030        TarSession session = new TarSession();
031        session.setName(uri.getSchemeSpecificPart());
032        session.setScheme(uri.getScheme());
033        sessions.add(session);
034        return session;
035    }
036
037    @Override
038    public void close() throws IOException {
039        for (TarSession session : sessions) {
040            session.close();
041        }
042    }
043}
