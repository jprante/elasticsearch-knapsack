002package org.xbib.io;
003
004import java.io.IOException;
005import java.util.concurrent.atomic.AtomicInteger;
006import java.util.concurrent.atomic.AtomicLong;
007import org.elasticsearch.action.ActionListener;
008import org.elasticsearch.action.bulk.BulkRequestBuilder;
009import org.elasticsearch.action.bulk.BulkResponse;
010import org.elasticsearch.action.index.IndexRequest;
011import org.elasticsearch.client.Client;
012import org.elasticsearch.client.Requests;
013import org.elasticsearch.common.logging.ESLogger;
014
015
016public class BulkOperation {
017
018    private String index;
019
020    private String type;
021
022    private String id;
023
024    private Client client;
025
026    private ESLogger logger;
027
028    private int bulkSize = 100;
029
030    private int maxActiveRequests = 30;
031
032    private long millisBeforeContinue = 60000L;
033
034    private int totalTimeouts;
035
036    private static final int MAX_TOTAL_TIMEOUTS = 10;
037
038    private static final AtomicInteger activeBulks = new AtomicInteger(0);
039
040    private static final AtomicLong counter = new AtomicLong(0);
041
042    private ThreadLocal<BulkRequestBuilder> currentBulk = new ThreadLocal();
043
044    public BulkOperation(Client client, ESLogger logger) {
045        this.client = client;
046        this.logger = logger;
047        this.id = null;
048        this.totalTimeouts = 0;
049    }
050    
051    public BulkOperation setIndex(String index) {
052        this.index = index;
053        return this;
054    }
055    
056    public String getIndex() {
057        return index;
058    }
059    
060    public BulkOperation setType(String type) {
061        this.type = type;
062        return this;
063    }
064    
065    public String getType() {
066        return type;
067    }
068
069    public BulkOperation setId(String id) {
070        this.id = id;
071        return this;
072    }
073    
074    public String getId() {
075        return id;
076    }
077
078    public long getCount() {
079        return counter.longValue();
080    }
081    
082    public BulkOperation setBulkSize(int bulkSize) {
083        this.bulkSize = bulkSize;
084        return this;
085    }
086
087    public BulkOperation setMaxActiveRequests(int maxActiveRequests) {
088        this.maxActiveRequests = maxActiveRequests;
089        return this;
090    }
091
092    public BulkOperation setMillisBeforeContinue(long millis) {
093        this.millisBeforeContinue = millis;
094        return this;
095    }
096
097    public void create(String index, String type, String id, String source) {
098        if (!isNullOrEmpty(index)) {
099            setIndex(index);
100        }
101        if (!isNullOrEmpty(type)) {
102            setType(type);
103        }
104        if (!isNullOrEmpty(id)) {
105           setId(id);
106        }
107        if (id == null) {
108            return; // skip if ID is null
109        }
110        if (currentBulk.get() == null) {
111            currentBulk.set(client.prepareBulk());
112        }
113        IndexRequest request = Requests.indexRequest(getIndex()).type(getType()).id(getId()).create(true).source(source);
114        currentBulk.get().add(request);
115        if (currentBulk.get().numberOfActions() >= bulkSize) {
116            processBulk();
117        }
118    }    
119    
120    public void index(String index, String type, String id, String source) {
121        if (!isNullOrEmpty(index)) {
122            setIndex(index);
123        }
124        if (!isNullOrEmpty(type)) {
125            setType(type);
126        }
127        if (!isNullOrEmpty(id)) {
128           setId(id);
129        }
130        if (id == null) {
131            return; // skip if ID is null
132        }
133        if (currentBulk.get() == null) {
134            currentBulk.set(client.prepareBulk());
135        }
136        IndexRequest request = Requests.indexRequest(getIndex()).type(getType()).id(getId()).source(source);
137        currentBulk.get().add(request);
138        if (currentBulk.get().numberOfActions() >= bulkSize) {
139            processBulk();
140        }
141    }
142    
143    public void delete(String index, String type, String id) {
144        if (!isNullOrEmpty(index)) {
145            setIndex(index);
146        }
147        if (!isNullOrEmpty(type)) {
148            setType(type);
149        }
150        if (!isNullOrEmpty(id)) {
151           setId(id);
152        }
153        if (id == null) {
154            return; // skip
155        }
156        if (currentBulk.get() == null) {
157            currentBulk.set(client.prepareBulk());
158        }
159        currentBulk.get().add(Requests.deleteRequest(getIndex()).type(getType()).id(getId()));
160        if (currentBulk.get().numberOfActions() >= bulkSize) {
161            processBulk();
162        }        
163    }
164    
165    public void flush() throws IOException {
166        if (totalTimeouts > MAX_TOTAL_TIMEOUTS) {
167            // waiting some minutes is much too long, do not wait any longer            
168            throw new IOException("total timeouts exceeded limit of + " + MAX_TOTAL_TIMEOUTS + ", aborting");
169        }
170        if (currentBulk.get() != null && currentBulk.get().numberOfActions() > 0) {
171            processBulk();
172        }
173        // wait for all outstanding bulk requests
174        while (activeBulks.intValue() > 0) {
175            logger.info("waiting for {} active bulk requests", activeBulks);
176            synchronized (activeBulks) {
177                try {
178                    activeBulks.wait(millisBeforeContinue);
179                } catch (InterruptedException e) {
180                    logger.warn("timeout while waiting, continuing after {} ms", millisBeforeContinue);
181                    totalTimeouts++;
182                }
183            }
184        }
185    }
186
187    private void processBulk() {
188        while (activeBulks.intValue() >= maxActiveRequests) {
189            logger.info("waiting for {} active bulk requests", activeBulks);
190            synchronized (activeBulks) {
191                try {
192                    activeBulks.wait(millisBeforeContinue);
193                } catch (InterruptedException e) {
194                    logger.warn("timeout while waiting, continuing after {} ms", millisBeforeContinue);
195                    totalTimeouts++;
196                }
197            }
198        }
199        final int currentOnGoingBulks = activeBulks.incrementAndGet();
200        final int numberOfActions = currentBulk.get().numberOfActions();
201        logger.info("submitting new bulk request ({} docs, {} requests currently active)", numberOfActions, currentOnGoingBulks);
202        try {
203            currentBulk.get().execute(new ActionListener<BulkResponse>() {
204
205                @Override
206                public void onResponse(BulkResponse bulkResponse) {
207                    if (bulkResponse.hasFailures()) {
208                        logger.error("bulk request has failures: {}", bulkResponse.buildFailureMessage());
209                    } else {
210                        final long totalActions = counter.addAndGet(numberOfActions);
211                        logger.info("bulk request success ({} millis, {} docs, total of {} docs)",
212                                bulkResponse.getTookInMillis(), numberOfActions, totalActions);
213                    }
214                    activeBulks.decrementAndGet();
215                    synchronized (activeBulks) {
216                        activeBulks.notifyAll();
217                    }
218                }
219
220                @Override
221                public void onFailure(Throwable e) {
222                    logger.error("bulk request failed", e);
223                }
224            });
225        } catch (Exception e) {
226            logger.error("unhandled exception, failed to execute bulk request", e);
227        } finally {
228            currentBulk.set(client.prepareBulk());
229        }
230    }
231
232    private boolean isNullOrEmpty(String s) {
233        return s == null || s.length() == 0;
234    }
