044    /**

048     */
049    public TarArchiveInputStream(InputStream is) {
050        this.encoding = ArchiveEntryEncodingHelper.getEncoding(null);
051        this.readBuf = null;
052        this.hasHitEOF = false;
053        this.inStream = is;
054        this.blockSize = DEFAULT_BLOCK_SIZE;
055        this.recordSize = DEFAULT_RECORD_SIZE;
056        this.recsPerBlock = this.blockSize / this.recordSize;
057        this.blockBuffer = new byte[this.blockSize];
058        this.currRecIdx = this.recsPerBlock;
059    }

084    /**
085     * Get the available data that can be read from the current
086     * entry in the archive. This does not indicate how much data
087     * is left in the entire archive, only in the current entry.
088     * This value is determined from the entry's size header field
089     * and the amount of data already read from the current entry.
090     * Integer.MAX_VALUE is returen in case more than Integer.MAX_VALUE
091     * bytes are left in the current entry in the archive.
092     *
093     * @return The number of available bytes for the current entry.
094     * @throws IOException
095     */
096    @Override
097    public int available() throws IOException {
098        if (entrySize - entryOffset > Integer.MAX_VALUE) {
099            return Integer.MAX_VALUE;
100        }
101        return (int) (entrySize - entryOffset);
102    }

104    /**
105     * Skip bytes in the input buffer. This skips bytes in the
106     * current entry's data, not the entire archive, and will
107     * stop at the end of the current entry's data if the number
108     * to skip extends beyond that point.
109     *
110     * @param numToSkip The number of bytes to skip.
111     * @return the number actually skipped
112     * @throws IOException on error
113     */
114    @Override
115    public long skip(long numToSkip) throws IOException {
116        // REVIEW
117        // This is horribly inefficient, but it ensures that we
118        // properly skip over bytes
119        //
120        byte[] skipBuf = new byte[1024];
121        long skip = numToSkip;
122        while (skip > 0) {
123            int realSkip = (int) (skip > skipBuf.length ? skipBuf.length : skip);
124            int numRead = read(skipBuf, 0, realSkip);
125            if (numRead == -1) {
126                break;
127            }
128            skip -= numRead;
129        }
130        return (numToSkip - skip);
131    }

140    /**
141     * Get the next entry in this tar archive. This will skip
142     * over any remaining data in the current entry, if there
143     * is one, and place the input stream at the header of the
144     * next entry, and read the header and instantiate a new
145     * TarEntry from the header bytes and return that entry.
146     * If there are no more entries in the archive, null will
147     * be returned to indicate that the end of the archive has
148     * been reached.
149     *
150     * @return The next TarEntry in the archive, or null.
151     * @throws IOException on error
152     */
153    public synchronized TarArchiveInputEntry getNextTarEntry() throws IOException {
154        if (hasHitEOF) {
155            return null;
156        }
157        if (entry != null) {
158            long numToSkip = entrySize - entryOffset;
159            while (numToSkip > 0) {
160                long skipped = skip(numToSkip);
161                if (skipped <= 0) {
162                    throw new RuntimeException("failed to skip current tar entry");
163                }
164                numToSkip -= skipped;
165            }
166            readBuf = null;
167        }
168        byte[] headerBuf = getRecord();
169        if (hasHitEOF) {
170            entry = null;
171            return null;
172        }
173        try {
174            this.entry = new TarArchiveInputEntry(headerBuf, encoding);
175        } catch (IllegalArgumentException e) {
176            throw new IOException("error detected parsing the header", e);
177        }
178        this.entryOffset = 0;
179        if (entry.isGNULongNameEntry()) {
180            StringBuilder longName = new StringBuilder();
181            byte[] buf = new byte[256];
182            int length;
183            while ((length = read(buf)) >= 0) {
184                longName.append(new String(buf, 0, length));
185            }
186            getNextEntry();
187            if (entry == null) {
188                return null;
189            }
190            if (longName.length() > 0 && longName.charAt(longName.length() - 1) == 0) {
191                longName.deleteCharAt(longName.length() - 1);
192            }
193            entry.setName(longName.toString());
194        }
195        if (entry.isPaxHeader()) {
196            paxHeaders();
197        }
198        this.entrySize = entry.getEntrySize();
199        return entry;
200    }
201
202    /**
203     * Get the next record in this tar archive. This will skip
204     * over any remaining data in the current entry, if there
205     * is one, and place the input stream at the header of the
206     * next entry.
207     * If there are no more entries in the archive, null will
208     * be returned to indicate that the end of the archive has
209     * been reached.
210     *
211     * @return The next header in the archive, or null.
212     * @throws IOException on error
213     */
214    private byte[] getRecord() throws IOException {
215        if (hasHitEOF) {
216            return null;
217        }
218        byte[] headerBuf = readRecord();
219        if (headerBuf == null) {
220            hasHitEOF = true;
221        } else if (isEOFRecord(headerBuf)) {
222            hasHitEOF = true;
223        }
224        return hasHitEOF ? null : headerBuf;
225    }
226
227    /**
228     * Read a record from the input stream and return the data.
229     *
230     * @return The record data.
231     * @throws IOException on error
232     */
233    private byte[] readRecord() throws IOException {
234        if (currRecIdx >= recsPerBlock && !readBlock()) {
235            return null;
236        }
237        byte[] result = new byte[recordSize];
238        System.arraycopy(blockBuffer, (currRecIdx * recordSize), result, 0, recordSize);
239        currRecIdx++;
240        return result;
241    }
242
243    private boolean readBlock() throws IOException {
244        currRecIdx = 0;
245        int offset = 0;
246        int bytesNeeded = blockSize;
247        while (bytesNeeded > 0) {
248            long numBytes = inStream.read(blockBuffer, offset, bytesNeeded);
249            if (numBytes == -1) {
250                if (offset == 0) {
251                    return false;
252                }
253                Arrays.fill(blockBuffer, offset, offset + bytesNeeded, (byte) 0);
254                break;
255            }
256            offset += numBytes;
257            bytesNeeded -= numBytes;
258        }
259        return true;
260    }
261
262    /**
263     * Determine if an archive record indicate End of Archive. End of
264     * archive is indicated by a record that consists entirely of null bytes.
265     *
266     * @param record The record data to check.
267     * @return true if the record data is an End of Archive
268     */
269    private boolean isEOFRecord(byte[] record) {
270        for (int i = 0, sz = getRecordSize(); i < sz; ++i) {
271            if (record[i] != 0) {
272                return false;
273            }
274        }
275        return true;
276    }

279    private void paxHeaders() throws IOException {
280        Map<String, String> headers = parsePaxHeaders(this);
281        getNextEntry(); // Get the actual file entry
282        applyPaxHeadersToCurrentEntry(headers);
283    }
284
285    private Map<String, String> parsePaxHeaders(InputStream i) throws IOException {
286        Map<String, String> headers = new HashMap<String, String>();
287        // Format is "length keyword=value\n";
288        while (true) { // get length
289            int ch;
290            int len = 0;
291            int read = 0;
292            while ((ch = i.read()) != -1) {
293                read++;
294                if (ch == ' ') { // End of length string
295                    // Get keyword
296                    ByteArrayOutputStream coll = new ByteArrayOutputStream();
297                    while ((ch = i.read()) != -1) {
298                        read++;
299                        if (ch == '=') { // end of keyword
300                            String keyword = coll.toString("UTF-8");
301                            // Get rest of entry
302                            byte[] rest = new byte[len - read];
303                            int got = i.read(rest);
304                            if (got != len - read) {
305                                throw new IOException("Failed to read "
306                                        + "Paxheader. Expected "
307                                        + (len - read)
308                                        + " bytes, read "
309                                        + got);
310                            }
311                            // Drop trailing NL
312                            String value = new String(rest, 0,
313                                    len - read - 1, Charset.forName("UTF-8"));
314                            headers.put(keyword, value);
315                            break;
316                        }
317                        coll.write((byte) ch);
318                    }
319                    break; // Processed single header
320                }
321                len *= 10;
322                len += ch - '0';
323            }
324            if (ch == -1) { // EOF
325                break;
326            }
327        }
328        return headers;
329    }
330
331    private void applyPaxHeadersToCurrentEntry(Map<String, String> headers) {
332        /*
333         * The following headers are defined for Pax.
334         * atime, ctime, charset: cannot use these without changing TarArchiveEntry fields
335         * mtime
336         * comment
337         * gid, gname
338         * linkpath
339         * size
340         * uid,uname
341         * SCHILY.devminor, SCHILY.devmajor: don't have setters/getters for those
342         */
343        for (Entry<String, String> ent : headers.entrySet()) {
344            String key = ent.getKey();
345            String val = ent.getValue();
346            if ("path".equals(key)) {
347                entry.setName(val);
348            } else if ("linkpath".equals(key)) {
349                entry.setLinkName(val);
350            } else if ("gid".equals(key)) {
351                entry.setGroupId(Integer.parseInt(val));
352            } else if ("gname".equals(key)) {
353                entry.setGroupName(val);
354            } else if ("uid".equals(key)) {
355                entry.setUserId(Integer.parseInt(val));
356            } else if ("uname".equals(key)) {
357                entry.setUserName(val);
358            } else if ("size".equals(key)) {
359                entry.setEntrySize(Long.parseLong(val));
360            } else if ("mtime".equals(key)) {
361                long mtime = (long) (Double.parseDouble(val) * 1000);
362                entry.setLastModified(new Date(mtime));
363            } else if ("SCHILY.devminor".equals(key)) {
364                entry.setDevMinor(Integer.parseInt(val));
365            } else if ("SCHILY.devmajor".equals(key)) {
366                entry.setDevMajor(Integer.parseInt(val));
367            }
368        }
369    }

376    /**
377     * Reads bytes from the current tar archive entry.
378     * This method is aware of the boundaries of the current
379     * entry in the archive and will deal with them as if they
380     * were this stream's start and EOF.
381     *
382     * @param buf       The buffer into which to place bytes read.
383     * @param offset    The offset at which to place bytes read.
384     * @param numToRead The number of bytes to read.
385     * @return The number of bytes read, or -1 at EOF
386     * @throws IOException on error
387     */
388    @Override
389    public int read(byte[] buf, int offset, int numToRead) throws IOException {
390        int totalRead = 0;
391        if (entryOffset >= entrySize) {
392            return -1;
393        }
394        if ((numToRead + entryOffset) > entrySize) {
395            numToRead = (int) (entrySize - entryOffset);
396        }
397        if (readBuf != null) {
398            int sz = (numToRead > readBuf.length) ? readBuf.length : numToRead;
399            System.arraycopy(readBuf, 0, buf, offset, sz);
400            if (sz >= readBuf.length) {
401                readBuf = null;
402            } else {
403                int newLen = readBuf.length - sz;
404                byte[] newBuf = new byte[newLen];
405                System.arraycopy(readBuf, sz, newBuf, 0, newLen);
406                readBuf = newBuf;
407            }
408            totalRead += sz;
409            numToRead -= sz;
410            offset += sz;
411        }
412        while (numToRead > 0) {
413            byte[] rec = readRecord();
414            if (rec == null) {
415                throw new IOException("unexpected EOF with " + numToRead + " bytes unread");
416            }
417            int sz = numToRead;
418            int recLen = rec.length;
419            if (recLen > sz) {
420                System.arraycopy(rec, 0, buf, offset, sz);
421                readBuf = new byte[recLen - sz];
422                System.arraycopy(rec, sz, readBuf, 0, recLen - sz);
423            } else {
424                sz = recLen;
425                System.arraycopy(rec, 0, buf, offset, recLen);
426            }
427            totalRead += sz;
428            numToRead -= sz;
429            offset += sz;
430        }
431        entryOffset += totalRead;
432        return totalRead;
433    }
