132    /**
133     * Grow a byte buffer, so it has a minimal capacity or at least
134     * the double capacity of the original buffer
135     *
136     * @param b           The original buffer.
137     * @param newCapacity The minimal requested new capacity.
138     * @return A byte buffer <code>r</code> with
139     * <code>r.capacity() = max(b.capacity()*2,newCapacity)</code> and
140     * all the data contained in <code>b</code> copied to the beginning
141     * of <code>r</code>.
142     */
143    static ByteBuffer growBuffer(ByteBuffer b, int newCapacity) {
144        b.limit(b.position());
145        b.rewind();
146
147        int c2 = b.capacity() * 2;
148        ByteBuffer on = ByteBuffer.allocate(c2 < newCapacity ? newCapacity : c2);
149
150        on.put(b);
151        return on;
152    }

155    /**
156     * The hexadecimal digits <code>0,...,9,A,...,F</code> encoded as
157     * ASCII bytes.
158     */
159    private static final byte[] HEX_DIGITS =
160            new byte[]{
161                    0x30, 0x31, 0x32, 0x33, 0x34, 0x35, 0x36, 0x37, 0x38, 0x39, 0x41,
162                    0x42, 0x43, 0x44, 0x45, 0x46
163            };
164
165    /**
166     * Append <code>%Uxxxx</code> to the given byte buffer.
167     * The caller must assure, that <code>bb.remaining()&gt;=6</code>.
168     *
169     * @param bb The byte buffer to write to.
170     * @param c  The character to write.
171     */
172    public static void appendSurrogate(ByteBuffer bb, char c) {
173
174        bb.put((byte) '%');
175        bb.put((byte) 'U');
176
177        bb.put(HEX_DIGITS[(c >> 12) & 0x0f]);
178        bb.put(HEX_DIGITS[(c >> 8) & 0x0f]);
179        bb.put(HEX_DIGITS[(c >> 4) & 0x0f]);
180        bb.put(HEX_DIGITS[c & 0x0f]);
181    }

183    /**
184     * name of the encoding UTF-8
185     */
186    public static final String UTF8 = "UTF8";
187
188    /**
189     * variant name of the encoding UTF-8 used for comparisions.
190     */
191    private static final String UTF_DASH_8 = "UTF_8";
192
193    /**
194     * name of the encoding UTF-8
195     */
196    public static final ArchiveEntryEncoding UTF8_ENCODING = new FallbackArchiveEntryEncoding(UTF8);
197
198    /**
199     * Instantiates an encoding.
200     *
201     * @param name The name of the encoding. Specify {@code null} for
202     *             the platform's default encoding.
203     * @return An encoding for the given encoding name.
204     */
205    public static ArchiveEntryEncoding getEncoding(String name) {
206        // fallback encoding is good enough for utf-8.
207        if (isUTF8(name)) {
208            return UTF8_ENCODING;
209        }
210        if (name == null) {
211            return new FallbackArchiveEntryEncoding();
212        }
213        SimpleEncodingHolder h = simpleEncodings.get(name);
214        if (h != null) {
215            return h.getEncoding();
216        }
217        try {
218            Charset cs = Charset.forName(name);
219            return new NioArchiveEntryEncoding(cs);
220        } catch (UnsupportedCharsetException e) {
221            return new FallbackArchiveEntryEncoding(name);
222        }
223    }
224
225    /**
226     * Whether a given encoding - or the platform's default encoding
227     * if the parameter is null - is UTF-8.
228     */
229    public static boolean isUTF8(String encoding) {
230        if (encoding == null) {
231            // check platform's default encoding
232            encoding = System.getProperty("file.encoding");
233        }
234        return UTF8.equalsIgnoreCase(encoding)
235                || UTF_DASH_8.equalsIgnoreCase(encoding);
236    }
