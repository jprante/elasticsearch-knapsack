022import java.io.IOException;
023import java.io.OutputStream;

026/**
027 * CPIOArchiveOutputStream is a stream for writing CPIO streams. All formats of
028 * CPIO are supported (old ASCII, old binary, new portable format and the new
029 * portable format with CRC).
030 * <p>
031 * <p>
032 * An entry can be written by creating an instance of CpioArchiveEntry and fill
033 * it with the necessary values and put it into the CPIO stream. Afterwards
034 * write the contents of the file into the CPIO stream. Either close the stream
035 * by calling finish() or put a next entry into the cpio stream.
036 * <p>
037 * <code>
038 * CpioArchiveOutputStream out = new CpioArchiveOutputStream(
039 *         new FileOutputStream(new File("test.cpio")));
040 * CpioArchiveEntry entry = new CpioArchiveEntry();
041 * entry.setName("testfile");
042 * String contents = &quot;12345&quot;;
043 * entry.setFileSize(contents.length());
044 * entry.setMode(CpioConstants.C_ISREG); // regular file
045 * ... set other attributes, e.g. time, number of links
046 * out.putArchiveEntry(entry);
047 * out.write(testContents.getBytes());
048 * out.close();
049 * </code>
050 * <p>
051 * Note: This implementation should be compatible to cpio 2.5
052 */
053public class CpioArchiveOutputStream extends ArchiveOutputStream<CpioArchiveEntry> implements CpioConstants {
054
055    private CpioArchiveEntry entry;
056
057    private boolean closed = false;
058
059    private boolean finished;
060
061    private final short entryFormat;
062
063    private final HashMap<String, CpioArchiveEntry> names = new HashMap<String, CpioArchiveEntry>();
064
065    private long crc = 0;
066
067    private long written;
068
069    private final int blockSize;
070
071    private long nextArtificalDeviceAndInode = 1;
072
073    private final CountingOutputStream out;
074
075    /**
076     * Construct the cpio output stream. The format for this CPIO stream is the
077     * "new" format
078     *
079     * @param out The cpio stream
080     */
081    public CpioArchiveOutputStream(OutputStream out) {
082        this(out, FORMAT_NEW);
083    }

085    /**
086     * Construct the cpio output stream with a specified format and a
087     * blocksize of {@link CpioConstants#BLOCK_SIZE BLOCK_SIZE}.
088     *
089     * @param out    The cpio stream
090     * @param format The format of the stream
091     */
092    public CpioArchiveOutputStream(OutputStream out, short format) {
093        this(out, format, BLOCK_SIZE);
094    }

097    /**
098     * Construct the cpio output stream with a specified format
099     *
100     * @param out       The cpio stream
101     * @param format    The format of the stream
102     * @param blockSize The block size of the archive.
103     */
104    public CpioArchiveOutputStream(OutputStream out,
105                                   short format, int blockSize) {
106        this.out = new CountingOutputStream(out);
107        switch (format) {
108            case FORMAT_NEW:
109            case FORMAT_NEW_CRC:
110            case FORMAT_OLD_ASCII:
111            case FORMAT_OLD_BINARY:
112                break;
113            default:
114                throw new IllegalArgumentException("Unknown format: " + format);
115
116        }
117        this.entryFormat = format;
118        this.blockSize = blockSize;
119    }
120
121    /**
122     * Check to make sure that this stream has not been closed
123     *
124     * @throws java.io.IOException if the stream is already closed
125     */
126    private void ensureOpen() throws IOException {
127        if (this.closed) {
128            throw new IOException("Stream closed");
129        }
130    }
131
132    @Override
133    public CpioArchiveEntry newArchiveEntry() {
134        return new CpioArchiveEntry(FORMAT_NEW);
135    }
136
137    /**
138     * Begins writing a new CPIO file entry and positions the stream to the
139     * start of the entry data. Closes the current entry if still active. The
140     * current time will be used if the entry has no set modification time and
141     * the default header format will be used if no other format is specified in
142     * the entry.
143     *
144     * @param entry the CPIO cpioEntry to be written
145     * @throws java.io.IOException if an I/O error has occurred or if a CPIO file error has
146     *                             occurred
147     * @throws ClassCastException  if entry is not an instance of CpioArchiveEntry
148     */
149    @Override
150    public void putArchiveEntry(CpioArchiveEntry entry) throws IOException {
151        if (finished) {
152            throw new IOException("stream has already been finished");
153        }
154        ensureOpen();
155        if (this.entry != null) {
156            closeArchiveEntry(); // close previous entry
157        }
158        if (entry.getTime() == -1) {
159            entry.setTime(System.currentTimeMillis() / 1000);
160        }
161        if (entry.getFormat() != this.entryFormat) {
162            throw new IOException("header format " + entry.getFormat() + " does not match existing format: " + this.entryFormat);
163        }
164        if (this.names.put(entry.getName(), entry) != null) {
165            throw new IOException("duplicate entry: " + entry.getName());
166        }
167        writeHeader(entry);
168        this.entry = entry;
169        this.written = 0;
170    }
171
172    private void writeHeader(final CpioArchiveEntry e) throws IOException {
173        switch (e.getFormat()) {
174            case FORMAT_NEW:
175                out.write(ArchiveUtils.toAsciiBytes(MAGIC_NEW));
176                writeNewEntry(e);
177                break;
178            case FORMAT_NEW_CRC:
179                out.write(ArchiveUtils.toAsciiBytes(MAGIC_NEW_CRC));
180                writeNewEntry(e);
181                break;
182            case FORMAT_OLD_ASCII:
183                out.write(ArchiveUtils.toAsciiBytes(MAGIC_OLD_ASCII));
184                writeOldAsciiEntry(e);
185                break;
186            case FORMAT_OLD_BINARY:
187                boolean swapHalfWord = true;
188                writeBinaryLong(MAGIC_OLD_BINARY, 2, swapHalfWord);
189                writeOldBinaryEntry(e, swapHalfWord);
190                break;
191        }
192    }
193
194    private void writeNewEntry(final CpioArchiveEntry entry) throws IOException {
195        long inode = entry.getInode();
196        long devMin = entry.getDeviceMin();
197        if (CPIO_TRAILER.equals(entry.getName())) {
198            inode = devMin = 0;
199        } else {
200            if (inode == 0 && devMin == 0) {
201                inode = nextArtificalDeviceAndInode & 0xFFFFFFFF;
202                devMin = (nextArtificalDeviceAndInode++ >> 32) & 0xFFFFFFFF;
203            } else {
204                nextArtificalDeviceAndInode =
205                        Math.max(nextArtificalDeviceAndInode,
206                                inode + 0x100000000L * devMin) + 1;
207            }
208        }
209        writeAsciiLong(inode, 8, 16);
210        writeAsciiLong(entry.getMode(), 8, 16);
211        writeAsciiLong(entry.getUID(), 8, 16);
212        writeAsciiLong(entry.getGID(), 8, 16);
213        writeAsciiLong(entry.getNumberOfLinks(), 8, 16);
214        writeAsciiLong(entry.getTime(), 8, 16);
215        writeAsciiLong(entry.getEntrySize(), 8, 16);
216        writeAsciiLong(entry.getDeviceMaj(), 8, 16);
217        writeAsciiLong(devMin, 8, 16);
218        writeAsciiLong(entry.getRemoteDeviceMaj(), 8, 16);
219        writeAsciiLong(entry.getRemoteDeviceMin(), 8, 16);
220        writeAsciiLong(entry.getName().length() + 1, 8, 16);
221        writeAsciiLong(entry.getChksum(), 8, 16);
222        writeCString(entry.getName());
223        pad(entry.getHeaderPadCount());
224    }
225
226    private void writeOldAsciiEntry(final CpioArchiveEntry entry)
227            throws IOException {
228        long inode = entry.getInode();
229        long device = entry.getDevice();
230        if (CPIO_TRAILER.equals(entry.getName())) {
231            inode = device = 0;
232        } else {
233            if (inode == 0 && device == 0) {
234                inode = nextArtificalDeviceAndInode & 0777777;
235                device = (nextArtificalDeviceAndInode++ >> 18) & 0777777;
236            } else {
237                nextArtificalDeviceAndInode =
238                        Math.max(nextArtificalDeviceAndInode,
239                                inode + 01000000 * device) + 1;
240            }
241        }
242        writeAsciiLong(device, 6, 8);
243        writeAsciiLong(inode, 6, 8);
244        writeAsciiLong(entry.getMode(), 6, 8);
245        writeAsciiLong(entry.getUID(), 6, 8);
246        writeAsciiLong(entry.getGID(), 6, 8);
247        writeAsciiLong(entry.getNumberOfLinks(), 6, 8);
248        writeAsciiLong(entry.getRemoteDevice(), 6, 8);
249        writeAsciiLong(entry.getTime(), 11, 8);
250        writeAsciiLong(entry.getName().length() + 1, 6, 8);
251        writeAsciiLong(entry.getEntrySize(), 11, 8);
252        writeCString(entry.getName());
253    }
254
255    private void writeOldBinaryEntry(final CpioArchiveEntry entry,
256                                     final boolean swapHalfWord) throws IOException {
257        long inode = entry.getInode();
258        long device = entry.getDevice();
259        if (CPIO_TRAILER.equals(entry.getName())) {
260            inode = device = 0;
261        } else {
262            if (inode == 0 && device == 0) {
263                inode = nextArtificalDeviceAndInode & 0xFFFF;
264                device = (nextArtificalDeviceAndInode++ >> 16) & 0xFFFF;
265            } else {
266                nextArtificalDeviceAndInode =
267                        Math.max(nextArtificalDeviceAndInode,
268                                inode + 0x10000 * device) + 1;
269            }
270        }
271
272        writeBinaryLong(device, 2, swapHalfWord);
273        writeBinaryLong(inode, 2, swapHalfWord);
274        writeBinaryLong(entry.getMode(), 2, swapHalfWord);
275        writeBinaryLong(entry.getUID(), 2, swapHalfWord);
276        writeBinaryLong(entry.getGID(), 2, swapHalfWord);
277        writeBinaryLong(entry.getNumberOfLinks(), 2, swapHalfWord);
278        writeBinaryLong(entry.getRemoteDevice(), 2, swapHalfWord);
279        writeBinaryLong(entry.getTime(), 4, swapHalfWord);
280        writeBinaryLong(entry.getName().length() + 1, 2, swapHalfWord);
281        writeBinaryLong(entry.getEntrySize(), 4, swapHalfWord);
282        writeCString(entry.getName());
283        pad(entry.getHeaderPadCount());
284    }

286    @Override
287    public void closeArchiveEntry() throws IOException {
288        if (finished) {
289            throw new IOException("stream has already been finished");
290        }
291        ensureOpen();
292        if (entry == null) {
293            throw new IOException("trying to close non-existent entry");
294        }
295        if (this.entry.getEntrySize() != this.written) {
296            throw new IOException("invalid entry size (expected "
297                    + this.entry.getEntrySize() + " but got " + this.written + " bytes)");
298        }
299        pad(this.entry.getDataPadCount());
300        if (this.entry.getFormat() == FORMAT_NEW_CRC && this.crc != this.entry.getChksum()) {
301            throw new IOException("CRC Error");
302        }
303        this.entry = null;
304        this.crc = 0;
305        this.written = 0;
306    }

308    /**
309     * Writes an array of bytes to the current CPIO entry data. This method will
310     * block until all the bytes are written.
311     *
312     * @param b   the data to be written
313     * @param off the start offset in the data
314     * @param len the number of bytes that are written
315     * @throws java.io.IOException if an I/O error has occurred or if a CPIO file error has occurred
316     */
317    @Override
318    public void write(final byte[] b, final int off, final int len) throws IOException {
319        ensureOpen();
320        getWatcher().updateBytesTransferred(len);
321        if (off < 0 || len < 0 || off > b.length - len) {
322            throw new IndexOutOfBoundsException();
323        } else if (len == 0) {
324            return;
325        }
326        if (this.entry == null) {
327            throw new IOException("no current CPIO entry");
328        }
329        if (this.written + len > this.entry.getEntrySize()) {
330            throw new IOException("attempt to write past end of STORED entry");
331        }
332        out.write(b, off, len);
333        this.written += len;
334        if (this.entry.getFormat() == FORMAT_NEW_CRC) {
335            for (int pos = 0; pos < len; pos++) {
336                this.crc += b[pos] & 0xFF;
337            }
338        }
339    }

341    /**
342     * Finishes writing the contents of the CPIO output stream without closing
343     * the underlying stream. Use this method when applying multiple filters in
344     * succession to the same output stream.
345     *
346     * @throws java.io.IOException if an I/O exception has occurred or if a CPIO file error has
347     *                             occurred
348     */
349    @Override
350    public void finish() throws IOException {
351        ensureOpen();
352        if (finished) {
353            throw new IOException("This archive has already been finished");
354        }
355        if (this.entry != null) {
356            throw new IOException("This archive contains unclosed entries.");
357        }
358        this.entry = new CpioArchiveEntry(this.entryFormat);
359        this.entry.setName(CPIO_TRAILER);
360        this.entry.setNumberOfLinks(1);
361        writeHeader(this.entry);
362        closeArchiveEntry();
363
364        int lengthOfLastBlock = (int) (out.getBytesWritten() % blockSize);
365        if (lengthOfLastBlock != 0) {
366            pad(blockSize - lengthOfLastBlock);
367        }
368        finished = true;
369    }
370
371    /**
372     * Closes the CPIO output stream as well as the stream being filtered.
373     *
374     * @throws java.io.IOException if an I/O error has occurred or if a CPIO file error has
375     *                             occurred
376     */
377    @Override
378    public void close() throws IOException {
379        if (!finished) {
380            finish();
381        }
382        if (!this.closed) {
383            out.close();
384            this.closed = true;
385        }
386    }
387
388    private void pad(int count) throws IOException {
389        if (count > 0) {
390            byte buff[] = new byte[count];
391            out.write(buff);
392        }
393    }
394
395    private void writeBinaryLong(final long number, final int length,
396                                 final boolean swapHalfWord) throws IOException {
397        byte tmp[] = CpioUtil.long2byteArray(number, length, swapHalfWord);
398        out.write(tmp);
399    }
400
401    private void writeAsciiLong(final long number, final int length,
402                                final int radix) throws IOException {
403        StringBuilder tmp = new StringBuilder();
404        String tmpStr;
405        if (radix == 16) {
406            tmp.append(Long.toHexString(number));
407        } else if (radix == 8) {
408            tmp.append(Long.toOctalString(number));
409        } else {
410            tmp.append(Long.toString(number));
411        }
412
413        if (tmp.length() <= length) {
414            long insertLength = length - tmp.length();
415            for (int pos = 0; pos < insertLength; pos++) {
416                tmp.insert(0, "0");
417            }
418            tmpStr = tmp.toString();
419        } else {
420            tmpStr = tmp.substring(tmp.length() - length);
421        }
422        byte[] b = ArchiveUtils.toAsciiBytes(tmpStr);
423        out.write(b);
424    }
425
426    /**
427     * Writes an ASCII string to the stream followed by \0
428     *
429     * @param str the String to write
430     * @throws java.io.IOException if the string couldn't be written
431     */
432    private void writeCString(final String str) throws IOException {
433        byte[] b = ArchiveUtils.toAsciiBytes(str);
434        out.write(b);
435        out.write('\0');
436    }
