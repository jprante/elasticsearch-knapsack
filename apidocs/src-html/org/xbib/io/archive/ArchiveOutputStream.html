016package org.xbib.io.archive;
017
018import org.xbib.io.BytesProgressWatcher;
019
020import java.io.IOException;
021import java.io.OutputStream;
022
023/**
024 * Archive output stream implementations are expected to override the
025 * {@link #write(byte[], int, int)} method to improve performance.
026 * They should also override {@link #close()} to ensure that any necessary
027 * trailers are added.
028 * The normal sequence of calls for working with ArchiveOutputStreams is:
029 * + create ArchiveOutputStream object
030 * + write header (optional, Zip only)
031 * + repeat as needed
032 * - putArchiveEntry() (writes entry header)
033 * - write() (writes entry data)
034 * - closeArchiveEntry() (closes entry)
035 * + finish() (ends the addition of entries)
036 * + write additional data if format supports it (optional)
037 * + close()
038 */
039public abstract class ArchiveOutputStream<E extends ArchiveEntry> extends OutputStream {
040
041    private BytesProgressWatcher watcher;
042
043    public ArchiveOutputStream setWatcher(BytesProgressWatcher watcher) {
044        this.watcher = watcher;
045        return this;
046    }
047
048    public BytesProgressWatcher getWatcher() {
049        return watcher;
050    }
051
052    /**
053     * Temporary buffer used for the {@link #write(int)} method
054     */
055    private final byte[] oneByte = new byte[1];
056
057    /**
058     * Creates a new archive entry for this archive output stream
059     *
060     * @return the new archive entry
061     * @throws IOException
062     */
063    public abstract E newArchiveEntry() throws IOException;
064
065    /**
066     * Writes the headers for an archive entry to the output stream.
067     * The caller must then write the content to the stream and call
068     * {@link #closeArchiveEntry()} to complete the process.
069     *
070     * @param entry describes the entry
071     * @throws java.io.IOException
072     */
073    public abstract void putArchiveEntry(E entry) throws IOException;
074
075    /**
076     * Closes the archive entry, writing any trailer information that may
077     * be required.
078     *
079     * @throws java.io.IOException
080     */
081    public abstract void closeArchiveEntry() throws IOException;
082
083    /**
084     * Finishes the addition of entries to this stream, without closing it.
085     * Additional data can be written, if the format supports it.
086     * <p>
087     * The finish() method throws an Exception if the user forgets to close the entry
088     * .
089     *
090     * @throws java.io.IOException
091     */
092    public abstract void finish() throws IOException;
093
094    /**
095     * Writes a byte to the current archive entry.
096     * <p>
097     * This method simply calls write( byte[], 0, 1 ).
098     *
099     * @param b The byte to be written.
100     * @throws java.io.IOException on error
101     */
102    @Override
103    public void write(int b) throws IOException {
104        oneByte[0] = (byte) (b & 0xFF);
105        write(oneByte, 0, 1);
106    }
107
108}
