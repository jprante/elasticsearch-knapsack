016package org.xbib.io.archive.esbulk;
017
018import org.xbib.io.archive.ArchiveInputStream;
019import org.xbib.io.archive.ArchiveUtils;
020
021import java.io.BufferedReader;
022import java.io.ByteArrayInputStream;
023import java.io.IOException;
024import java.io.InputStream;
025import java.io.InputStreamReader;
026import java.io.UnsupportedEncodingException;
027import java.nio.charset.Charset;
028import java.util.regex.Matcher;
029import java.util.regex.Pattern;
030
031public class EsBulkArchiveInputStream extends ArchiveInputStream<EsBulkArchiveEntry> {
032
033    private final static Pattern indexPattern = Pattern.compile("\"_index\"\\s*:\\s*\"(.*?)\"");
034    private final static Pattern typePattern = Pattern.compile("\"_type\"\\s*:\\s*\"(.*?)\"");
035    private final static Pattern idPattern = Pattern.compile("\"_id\"\\s*:\\s*\"(.*?)\"");
036
037    private final BufferedReader reader;
038
039    private ByteArrayInputStream in;
040
041    public EsBulkArchiveInputStream(InputStream in) throws UnsupportedEncodingException {
042        this.reader = new BufferedReader(new InputStreamReader(in, "UTF-8"));
043    }
044
045    @Override
046    public EsBulkArchiveEntry getNextEntry() throws IOException {
047        String meta = reader.readLine();
048        if (meta == null) {
049            return null;
050        }
051        String data = reader.readLine();
052        if (data == null) {
053            return null;
054        }
055        EsBulkArchiveEntry entry = new EsBulkArchiveEntry();
056        StringBuilder sb = new StringBuilder();
057        Matcher m = indexPattern.matcher(meta);
058        if (m.find()) {
059            sb.append(ArchiveUtils.encode(m.group(1), Charset.forName("UTF-8")));
060        } else {
061            throw new IOException("no _index found");
062        }
063        m = typePattern.matcher(meta);
064        if (m.find()) {
065            sb.append('/').append(ArchiveUtils.encode(m.group(1), Charset.forName("UTF-8")));
066        } else {
067            throw new IOException("no _type found");
068        }
069        m = idPattern.matcher(meta);
070        if (m.find()) {
071            sb.append('/').append(ArchiveUtils.encode(m.group(1), Charset.forName("UTF-8")));
072        } else {
073            throw new IOException("no _id found");
074        }
075        entry.setName(sb.toString());
076        byte[] b = data.getBytes("UTF-8");
077        entry.setEntrySize(b.length);
078        this.in = new ByteArrayInputStream(b);
079        return entry;
080    }
081
082    @Override
083    public int read(byte[] buffer, int start, int length) throws IOException {
084        return in.read(buffer, start, length);
085    }
086
087    @Override
088    public void close() throws IOException {
089        reader.close();
090    }
091
092    @Override
093    public long skip(long value) throws IOException {
094        throw new UnsupportedOperationException();
095    }
096}
