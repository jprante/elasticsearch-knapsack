003    import java.io.IOException;
004    import java.io.InputStream;

006    /**
007     * Safe {@link ChunkDecoder} implementation that can be used on any platform.
008     */
009    public class VanillaChunkDecoder extends ChunkDecoder {
010    
011        public VanillaChunkDecoder() {
012        }
013    
014        @Override
015        public final int decodeChunk(final InputStream is, final byte[] inputBuffer, final byte[] outputBuffer)
016                throws IOException {
017            int bytesInOutput;
018            /* note: we do NOT read more than 5 bytes because otherwise might need to shuffle bytes
019             * for output buffer (could perhaps optimize in future?)
020             */
021            int bytesRead = readHeader(is, inputBuffer);
022            if ((bytesRead < HEADER_BYTES)
023                    || inputBuffer[0] != LZFChunk.BYTE_Z || inputBuffer[1] != LZFChunk.BYTE_V) {
024                if (bytesRead == 0) { // probably fine, clean EOF
025                    return -1;
026                }
027                throw new IOException("Corrupt input data, block did not start with 2 byte signature ('ZV') followed by type byte, 2-byte length)");
028            }
029            int type = inputBuffer[2];
030            int compLen = uint16(inputBuffer, 3);
031            if (type == LZFChunk.BLOCK_TYPE_NON_COMPRESSED) { // uncompressed
032                readFully(is, false, outputBuffer, 0, compLen);
033                bytesInOutput = compLen;
034            } else { // compressed
035                readFully(is, true, inputBuffer, 0, 2 + compLen); // first 2 bytes are uncompressed length
036                int uncompLen = uint16(inputBuffer, 0);
037                decodeChunk(inputBuffer, 2, outputBuffer, 0, uncompLen);
038                bytesInOutput = uncompLen;
039            }
040            return bytesInOutput;
041        }

043        @Override
044        public final void decodeChunk(byte[] in, int inPos, byte[] out, int outPos, int outEnd)
045                throws IOException {
046            do {
047                int ctrl = in[inPos++] & 255;
048                if (ctrl < LZFChunk.MAX_LITERAL) { // literal run
049                    switch (ctrl) {
050                        case 31:
051                            out[outPos++] = in[inPos++];
052                        case 30:
053                            out[outPos++] = in[inPos++];
054                        case 29:
055                            out[outPos++] = in[inPos++];
056                        case 28:
057                            out[outPos++] = in[inPos++];
058                        case 27:
059                            out[outPos++] = in[inPos++];
060                        case 26:
061                            out[outPos++] = in[inPos++];
062                        case 25:
063                            out[outPos++] = in[inPos++];
064                        case 24:
065                            out[outPos++] = in[inPos++];
066                        case 23:
067                            out[outPos++] = in[inPos++];
068                        case 22:
069                            out[outPos++] = in[inPos++];
070                        case 21:
071                            out[outPos++] = in[inPos++];
072                        case 20:
073                            out[outPos++] = in[inPos++];
074                        case 19:
075                            out[outPos++] = in[inPos++];
076                        case 18:
077                            out[outPos++] = in[inPos++];
078                        case 17:
079                            out[outPos++] = in[inPos++];
080                        case 16:
081                            out[outPos++] = in[inPos++];
082                        case 15:
083                            out[outPos++] = in[inPos++];
084                        case 14:
085                            out[outPos++] = in[inPos++];
086                        case 13:
087                            out[outPos++] = in[inPos++];
088                        case 12:
089                            out[outPos++] = in[inPos++];
090                        case 11:
091                            out[outPos++] = in[inPos++];
092                        case 10:
093                            out[outPos++] = in[inPos++];
094                        case 9:
095                            out[outPos++] = in[inPos++];
096                        case 8:
097                            out[outPos++] = in[inPos++];
098                        case 7:
099                            out[outPos++] = in[inPos++];
100                        case 6:
101                            out[outPos++] = in[inPos++];
102                        case 5:
103                            out[outPos++] = in[inPos++];
104                        case 4:
105                            out[outPos++] = in[inPos++];
106                        case 3:
107                            out[outPos++] = in[inPos++];
108                        case 2:
109                            out[outPos++] = in[inPos++];
110                        case 1:
111                            out[outPos++] = in[inPos++];
112                        case 0:
113                            out[outPos++] = in[inPos++];
114                    }
115                    continue;
116                }
117                // back reference
118                int len = ctrl >> 5;
119                ctrl = -((ctrl & 0x1f) << 8) - 1;
120                if (len < 7) { // 2 bytes; length of 3 - 8 bytes
121                    ctrl -= in[inPos++] & 255;
122                    out[outPos] = out[outPos++ + ctrl];
123                    out[outPos] = out[outPos++ + ctrl];
124                    switch (len) {
125                        case 6:
126                            out[outPos] = out[outPos++ + ctrl];
127                        case 5:
128                            out[outPos] = out[outPos++ + ctrl];
129                        case 4:
130                            out[outPos] = out[outPos++ + ctrl];
131                        case 3:
132                            out[outPos] = out[outPos++ + ctrl];
133                        case 2:
134                            out[outPos] = out[outPos++ + ctrl];
135                        case 1:
136                            out[outPos] = out[outPos++ + ctrl];
137                    }
138                    continue;
139                }
140    
141                // long version (3 bytes, length of up to 264 bytes)
142                len = in[inPos++] & 255;
143                ctrl -= in[inPos++] & 255;
144    
145                // First: if there is no overlap, can just use arraycopy:
146                if ((ctrl + len) < -9) {
147                    len += 9;
148                    if (len <= 32) {
149                        copyUpTo32WithSwitch(out, outPos + ctrl, out, outPos, len - 1);
150                    } else {
151                        System.arraycopy(out, outPos + ctrl, out, outPos, len);
152                    }
153                    outPos += len;
154                    continue;
155                }
156    
157                // otherwise manual copy: so first just copy 9 bytes we know are needed
158                out[outPos] = out[outPos++ + ctrl];
159                out[outPos] = out[outPos++ + ctrl];
160                out[outPos] = out[outPos++ + ctrl];
161                out[outPos] = out[outPos++ + ctrl];
162                out[outPos] = out[outPos++ + ctrl];
163                out[outPos] = out[outPos++ + ctrl];
164                out[outPos] = out[outPos++ + ctrl];
165                out[outPos] = out[outPos++ + ctrl];
166                out[outPos] = out[outPos++ + ctrl];
167    
168                // then loop
169                // Odd: after extensive profiling, looks like magic number
170                // for unrolling is 4: with 8 performance is worse (even
171                // bit less than with no unrolling).
172                len += outPos;
173                final int end = len - 3;
174                while (outPos < end) {
175                    out[outPos] = out[outPos++ + ctrl];
176                    out[outPos] = out[outPos++ + ctrl];
177                    out[outPos] = out[outPos++ + ctrl];
178                    out[outPos] = out[outPos++ + ctrl];
179                }
180                switch (len - outPos) {
181                    case 3:
182                        out[outPos] = out[outPos++ + ctrl];
183                    case 2:
184                        out[outPos] = out[outPos++ + ctrl];
185                    case 1:
186                        out[outPos] = out[outPos++ + ctrl];
187                }
188            } while (outPos < outEnd);
189    
190            // sanity check to guard against corrupt data:
191            if (outPos != outEnd) {
192                throw new IOException("Corrupt data: overrun in decompress, input offset " + inPos + ", output offset " + outPos);
193            }
194        }
