016package org.xbib.io.compress.lzf;
017
018import java.io.IOException;
019
020/**
021 * Encoder that handles splitting of input into chunks to encode, calls
022 * {@link ChunkEncoder} to compress individual chunks and combines resulting
023 * chunks into contiguous output byte array.
024 */
025public class LZFEncoder {
026    // Static methods only, no point in instantiating
027
028    private LZFEncoder() {
029    }
030
031    public static byte[] encode(byte[] data) throws IOException {
032        return encode(data, data.length);
033    }
034
035    /**
036     * Method for compressing given input data using LZF encoding and block
037     * structure (compatible with lzf command line utility). Result consists of
038     * a sequence of chunks.
039     */
040    public static byte[] encode(byte[] data, int length) throws IOException {
041        return encode(data, 0, length);
042    }
043
044    /**
045     * Method for compressing given input data using LZF encoding and block
046     * structure (compatible with lzf command line utility). Result consists of
047     * a sequence of chunks.
048     */
049    public static byte[] encode(byte[] data, int offset, int length) throws IOException {
050        ChunkEncoder enc = new ChunkEncoder(length);
051        byte[] result = encode(enc, data, offset, length);
052        // important: may be able to reuse buffers
053        enc.close();
054        return result;
055    }
056
057    public static byte[] encode(ChunkEncoder enc, byte[] data, int length)
058            throws IOException {
059        return encode(enc, data, 0, length);
060    }
061
062    public static byte[] encode(ChunkEncoder enc, byte[] data, int offset, int length)
063            throws IOException {
064        int left = length;
065        int chunkLen = Math.min(LZFChunk.MAX_CHUNK_LEN, left);
066        LZFChunk first = enc.encodeChunk(data, offset, chunkLen);
067        left -= chunkLen;
068        // shortcut: if it all fit in, no need to coalesce:
069        if (left < 1) {
070            return first.getData();
071        }
072        // otherwise need to get other chunks:
073        int resultBytes = first.length();
074        offset += chunkLen;
075        LZFChunk last = first;
076
077        do {
078            chunkLen = Math.min(left, LZFChunk.MAX_CHUNK_LEN);
079            LZFChunk chunk = enc.encodeChunk(data, offset, chunkLen);
080            offset += chunkLen;
081            left -= chunkLen;
082            resultBytes += chunk.length();
083            last.setNext(chunk);
084            last = chunk;
085        } while (left > 0);
086        // and then coalesce returns into single contiguous byte array
087        byte[] result = new byte[resultBytes];
088        int ptr = 0;
089        for (; first != null; first = first.next()) {
090            ptr = first.copyTo(result, ptr);
091        }
092        return result;
093    }
094}
