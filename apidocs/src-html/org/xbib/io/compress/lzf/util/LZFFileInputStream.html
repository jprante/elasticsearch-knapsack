001package org.xbib.io.compress.lzf.util;
002
003import java.io.File;
004import java.io.FileDescriptor;
005import java.io.FileInputStream;
006import java.io.FileNotFoundException;
007import java.io.IOException;
008import java.io.InputStream;
009import java.io.OutputStream;
010import org.xbib.io.compress.BufferRecycler;
011import org.xbib.io.compress.lzf.ChunkDecoder;
012import org.xbib.io.compress.lzf.LZFChunk;
013
014/**
015 * Helper class that allows use of LZF compression even if a library requires
016 * use of {@link FileInputStream}. <p> Note that use of this class is not
017 * recommended unless you absolutely must use a {@link FileInputStream}
018 * instance; otherwise basic {@link LZFInputStream} (which uses aggregation for
019 * underlying streams) is more appropriate <p> <p> Implementation note: much of
020 * the code is just copied from {@link LZFInputStream}, so care must be taken to
021 * keep implementations in sync if there are fixes.
022 *
023 * @since 0.8
024 */
025public class LZFFileInputStream
026        extends FileInputStream {
027
028    /**
029     * Underlying decoder in use.
030     */
031    protected final ChunkDecoder _decompressor;
032    /**
033     * Object that handles details of buffer recycling
034     */
035    protected final BufferRecycler _recycler;
036    /**
037     * Flag that indicates if we have already called 'inputStream.close()' (to
038     * avoid calling it multiple times)
039     */
040    protected boolean _inputStreamClosed;
041    /**
042     * Flag that indicates whether we force full reads (reading of as many bytes
043     * as requested), or 'optimal' reads (up to as many as available, but at
044     * least one). Default is false, meaning that 'optimal' read is used.
045     */
046    protected boolean _cfgFullReads = false;
047    /**
048     * the current buffer of compressed bytes (from which to decode)
049     *
050     */
051    protected byte[] _inputBuffer;
052    /**
053     * the buffer of uncompressed bytes from which content is read
054     *
055     */
056    protected byte[] _decodedBytes;
057    /**
058     * The current position (next char to output) in the uncompressed bytes
059     * buffer.
060     *
061     */
062    protected int _bufferPosition = 0;
063    /**
064     * Length of the current uncompressed bytes buffer
065     *
066     */
067    protected int _bufferLength = 0;
068    /**
069     * Wrapper object we use to allow decoder to read directly from the stream,
070     * without ending in infinite loop...
071     */
072    protected final Wrapper _wrapper;
073
074    /*
075     // Construction, configuration
076     */
077    public LZFFileInputStream(File file) throws FileNotFoundException {
078        this(file, ChunkDecoderFactory.optimalInstance());
079    }
080
081    public LZFFileInputStream(FileDescriptor fdObj) {
082        this(fdObj, ChunkDecoderFactory.optimalInstance());
083    }
084
085    public LZFFileInputStream(String name) throws FileNotFoundException {
086        this(name, ChunkDecoderFactory.optimalInstance());
087    }
088
089    public LZFFileInputStream(File file, ChunkDecoder decompressor) throws FileNotFoundException {
090        super(file);
091        _decompressor = decompressor;
092        _recycler = BufferRecycler.instance();
093        _inputStreamClosed = false;
094        _inputBuffer = _recycler.allocInputBuffer(LZFChunk.MAX_CHUNK_LEN);
095        _decodedBytes = _recycler.allocDecodeBuffer(LZFChunk.MAX_CHUNK_LEN);
096        _wrapper = new Wrapper();
097    }
098
099    public LZFFileInputStream(FileDescriptor fdObj, ChunkDecoder decompressor) {
100        super(fdObj);
101        _decompressor = decompressor;
102        _recycler = BufferRecycler.instance();
103        _inputStreamClosed = false;
104        _inputBuffer = _recycler.allocInputBuffer(LZFChunk.MAX_CHUNK_LEN);
105        _decodedBytes = _recycler.allocDecodeBuffer(LZFChunk.MAX_CHUNK_LEN);
106        _wrapper = new Wrapper();
107    }
108
109    public LZFFileInputStream(String name, ChunkDecoder decompressor) throws FileNotFoundException {
110        super(name);
111        _decompressor = decompressor;
112        _recycler = BufferRecycler.instance();
113        _inputStreamClosed = false;
114        _inputBuffer = _recycler.allocInputBuffer(LZFChunk.MAX_CHUNK_LEN);
115        _decodedBytes = _recycler.allocDecodeBuffer(LZFChunk.MAX_CHUNK_LEN);
116        _wrapper = new Wrapper();
117    }
118
119    /**
120     * Method that can be used define whether reads should be "full" or
121     * "optimal": former means that full compressed blocks are read right away
122     * as needed, optimal that only smaller chunks are read at a time, more
123     * being read as needed.
124     */
125    public void setUseFullReads(boolean b) {
126        _cfgFullReads = b;
127    }
128
129    /*
130     // FileInputStream overrides
131     */
132    @Override
133    public int available() {
134        // if closed, return -1;
135        if (_inputStreamClosed) {
136            return -1;
137        }
138        int left = (_bufferLength - _bufferPosition);
139        return (left <= 0) ? 0 : left;
140    }
141
142    @Override
143    public void close() throws IOException {
144        _bufferPosition = _bufferLength = 0;
145        byte[] buf = _inputBuffer;
146        if (buf != null) {
147            _inputBuffer = null;
148            _recycler.releaseInputBuffer(buf);
149        }
150        buf = _decodedBytes;
151        if (buf != null) {
152            _decodedBytes = null;
153            _recycler.releaseDecodeBuffer(buf);
154        }
155        if (!_inputStreamClosed) {
156            _inputStreamClosed = true;
157            super.close();
158        }
159    }
160
161    // fine as is: don't override
162    // public FileChannel getChannel();
163    // final, can't override:
164    //public FileDescriptor getFD();
165    @Override
166    public int read() throws IOException {
167        if (!readyBuffer()) {
168            return -1;
169        }
170        return _decodedBytes[_bufferPosition++] & 255;
171    }
172
173    @Override
174    public int read(byte[] b) throws IOException {
175        return read(b, 0, b.length);
176    }
177
178    @Override
179    public int read(byte[] buffer, int offset, int length) throws IOException {
180        if (!readyBuffer()) {
181            return -1;
182        }
183        if (length < 1) {
184            return 0;
185        }
186        // First let's read however much data we happen to have...
187        int chunkLength = Math.min(_bufferLength - _bufferPosition, length);
188        System.arraycopy(_decodedBytes, _bufferPosition, buffer, offset, chunkLength);
189        _bufferPosition += chunkLength;
190
191        if (chunkLength == length || !_cfgFullReads) {
192            return chunkLength;
193        }
194        // Need more data, then
195        int totalRead = chunkLength;
196        do {
197            offset += chunkLength;
198            if (!readyBuffer()) {
199                break;
200            }
201            chunkLength = Math.min(_bufferLength - _bufferPosition, (length - totalRead));
202            System.arraycopy(_decodedBytes, _bufferPosition, buffer, offset, chunkLength);
203            _bufferPosition += chunkLength;
204            totalRead += chunkLength;
205        } while (totalRead < length);
206
207        return totalRead;
208    }
209
210    /**
211     * Overridden to just skip at most a single chunk at a time
212     */
213    @Override
214    public long skip(long n) throws IOException {
215        if (!readyBuffer()) {
216            return -1L;
217        }
218        int left = (_bufferLength - _bufferPosition);
219        // either way, just skip whatever we have decoded
220        if (left > n) {
221            left = (int) n;
222        }
223        _bufferPosition += left;
224        return left;
225    }
226
227    /*
228     // Extended public API
229     */
230    /**
231     * Convenience method that will read and uncompress all data available, and
232     * write it using given {@link OutputStream}. This avoids having to make an
233     * intermediate copy of uncompressed data which would be needed when doing
234     * the same manually.
235     *
236     * @param out OutputStream to use for writing content
237     *
238     * @return Number of bytes written (uncompressed)
239     *
240     */
241    public int readAndWrite(OutputStream out) throws IOException {
242        int total = 0;
243
244        while (readyBuffer()) {
245            int avail = _bufferLength - _bufferPosition;
246            out.write(_decodedBytes, _bufferPosition, avail);
247            _bufferPosition += avail; // to ensure it looks like we consumed it all
248            total += avail;
249        }
250        return total;
251    }
252
253    /*
254     // Internal methods
255     */
256    /**
257     * Fill the uncompressed bytes buffer by reading the underlying inputStream.
258     *
259     * @throws IOException
260     */
261    protected boolean readyBuffer() throws IOException {
262        if (_inputStreamClosed) {
263            throw new IOException("Input stream closed");
264        }
265        if (_bufferPosition < _bufferLength) {
266            return true;
267        }
268        _bufferLength = _decompressor.decodeChunk(_wrapper, _inputBuffer, _decodedBytes);
269        if (_bufferLength < 0) {
270            return false;
271        }
272        _bufferPosition = 0;
273        return (_bufferPosition < _bufferLength);
274    }
275
276    protected final int readRaw(byte[] buffer, int offset, int length) throws IOException {
277        return super.read(buffer, offset, length);
278    }
279
280    /*
281     // Helper class(es)
282     */
283    /**
284     * This simple wrapper is needed to re-route read calls so that they will
285     * use "raw" reads
286     */
287    private final class Wrapper extends InputStream {
288
289        @Override
290        public int read() throws IOException {
291            throw new UnsupportedOperationException();
292        }
293
294        @Override
295        public int read(byte[] buffer, int offset, int length) throws IOException {
296            return readRaw(buffer, offset, length);
297        }
298    }
