023package org.xbib.io.compress.bzip2;
024
025import java.io.IOException;
026import java.io.OutputStream;
027
028
029/**
030 * <p>An OutputStream wrapper that compresses BZip2 data</p>
031 *
032 * <p>Instances of this class are not threadsafe.</p>
033 */
034public class BZip2OutputStream extends OutputStream {
035
036    /**
037     * The stream to which compressed BZip2 data is written
038     */
039    private OutputStream outputStream;
040
041    /**
042     * An OutputStream wrapper that provides bit-level writes
043     */
044    private BZip2BitOutputStream bitOutputStream;
045
046    /**
047     * (@code true} if the compressed stream has been finished, otherwise {@code false}
048     */
049    private boolean streamFinished = false;
050
051    /**
052     * The declared maximum block size of the stream (before final run-length decoding)
053     */
054    private final int streamBlockSize;
055
056    /**
057     * The merged CRC of all blocks compressed so far
058     */
059    private int streamCRC = 0;
060
061    /**
062     * The compressor for the current block
063     */
064    private BZip2BlockCompressor blockCompressor;
065
066    /**
067     * Constructs a BZip2 stream compressor with the maximum (900,000 byte) block size
068     *
069     * @param outputStream The output stream to write to
070     * @throws java.io.IOException on any I/O error writing to the output stream
071     */
072    public BZip2OutputStream(final OutputStream outputStream) throws IOException {
073        this(outputStream, 9);
074    }
075
076    /**
077     * @param outputStream        The output stream to write to
078     * @param blockSizeMultiplier The BZip2 block size as a multiple of 100,000 bytes (minimum 1,
079     *                            maximum 9). Larger block sizes require more memory for both compression and decompression,
080     *                            but give better compression ratios. <code>9</code> will usually be the best value to use
081     * @throws java.io.IOException on any I/O error writing to the output stream
082     */
083    public BZip2OutputStream(final OutputStream outputStream, final int blockSizeMultiplier) throws IOException {
084        if (outputStream == null) {
085            throw new IllegalArgumentException("Null output stream");
086        }
087        if ((blockSizeMultiplier < 1) || (blockSizeMultiplier > 9)) {
088            throw new IllegalArgumentException("Invalid BZip2 block size" + blockSizeMultiplier);
089        }
090        this.streamBlockSize = blockSizeMultiplier * 100000;
091        this.outputStream = outputStream;
092        this.bitOutputStream = new BZip2BitOutputStream(this.outputStream);
093        this.bitOutputStream.writeBits(16, BZip2Constants.STREAM_START_MARKER_1);
094        this.bitOutputStream.writeBits(8, BZip2Constants.STREAM_START_MARKER_2);
095        this.bitOutputStream.writeBits(8, '0' + blockSizeMultiplier);
096        initialiseNextBlock();
097    }
098
099    @Override
100    public void write(final int value) throws IOException {
101        if (this.outputStream == null) {
102            throw new BZip2Exception("Stream closed");
103        }
104        if (this.streamFinished) {
105            throw new BZip2Exception("Write beyond end of stream");
106        }
107        if (!this.blockCompressor.write(value & 0xff)) {
108            closeBlock();
109            initialiseNextBlock();
110            this.blockCompressor.write(value & 0xff);
111        }
112    }
113
114    @Override
115    public void write(final byte[] data, int offset, int length) throws IOException {
116        if (this.outputStream == null) {
117            throw new BZip2Exception("Stream closed");
118        }
119        if (this.streamFinished) {
120            throw new BZip2Exception("Write beyond end of stream");
121        }
122        int bytesWritten;
123        while (length > 0) {
124            if ((bytesWritten = this.blockCompressor.write(data, offset, length)) < length) {
125                closeBlock();
126                initialiseNextBlock();
127            }
128            offset += bytesWritten;
129            length -= bytesWritten;
130        }
131    }
132
133    @Override
134    public void close() throws IOException {
135        if (this.outputStream != null) {
136            finish();
137            this.outputStream.close();
138            this.outputStream = null;
139        }
140    }
141
142    /**
143     * Initialises a new block for compression
144     */
145    private void initialiseNextBlock() {
146        this.blockCompressor = new BZip2BlockCompressor(this.bitOutputStream, this.streamBlockSize);
147    }
148
149    /**
150     * Compress and write out the block currently in progress. If no bytes have been written to the
151     * block, it is discarded
152     *
153     * @throws java.io.IOException on any I/O error writing to the output stream
154     */
155    private void closeBlock() throws IOException {
156
157        if (this.blockCompressor.isEmpty()) {
158            return;
159        }
160
161        this.blockCompressor.close();
162        int blockCRC = this.blockCompressor.getCRC();
163        this.streamCRC = ((this.streamCRC << 1) | (this.streamCRC >>> 31)) ^ blockCRC;
164
165    }
166
167
168    /**
169     * Compresses and writes out any as yet unwritten data, then writes the end of the BZip2 stream.
170     * The underlying OutputStream is not closed
171     *
172     * @throws java.io.IOException on any I/O error writing to the output stream
173     */
174    public void finish() throws IOException {
175
176        if (!this.streamFinished) {
177            this.streamFinished = true;
178            try {
179                closeBlock();
180                this.bitOutputStream.writeBits(24, BZip2Constants.STREAM_END_MARKER_1);
181                this.bitOutputStream.writeBits(24, BZip2Constants.STREAM_END_MARKER_2);
182                this.bitOutputStream.writeInteger(this.streamCRC);
183                this.bitOutputStream.flush();
184                this.outputStream.flush();
185            } finally {
186                this.blockCompressor = null;
187            }
188        }
189
190    }
191
192}
