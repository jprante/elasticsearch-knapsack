023package org.xbib.io.compress.bzip2;
024
025import java.io.IOException;
026
027
028/*
029 * Block encoding consists of the following stages:
030 * 1. Run-Length Encoding[1] - write()
031 * 2. Burrows Wheeler Transform - close() (through BZip2DivSufSort)
032 * 3. Write block header - close()
033 * 4. Move To Front Transform - close() (through BZip2HuffmanStageEncoder)
034 * 5. Run-Length Encoding[2] - close()  (through BZip2HuffmanStageEncoder)
035 * 6. Create and write Huffman tables - close() (through BZip2HuffmanStageEncoder)
036 * 7. Huffman encode and write data - close() (through BZip2HuffmanStageEncoder)
037 */
038
039/**
040 * Compresses and writes a single BZip2 block
041 */
042public class BZip2BlockCompressor {
043
044    /**
045     * The stream to which compressed BZip2 data is written
046     */
047    private final BZip2BitOutputStream bitOutputStream;
048
049    /**
050     * CRC builder for the block
051     */
052    private final CRC32 crc = new CRC32();
053
054    /**
055     * The RLE'd block data
056     */
057    private final byte[] block;
058
059    /**
060     * Current length of the data within the {@link block} array
061     */
062    private int blockLength = 0;
063
064    /**
065     * A limit beyond which new data will not be accepted into the block
066     */
067    private final int blockLengthLimit;
068
069    /**
070     * The values that are present within the RLE'd block data. For each index, {@code true} if that
071     * value is present within the data, otherwise {@code false}
072     */
073    private final boolean[] blockValuesPresent = new boolean[256];
074
075    /**
076     * The Burrows Wheeler Transformed block data
077     */
078    private final int[] bwtBlock;
079
080    /**
081     * The current RLE value being accumulated (undefined when {@link #rleLength} is 0)
082     */
083    private int rleCurrentValue = -1;
084
085    /**
086     * The repeat count of the current RLE value
087     */
088    private int rleLength = 0;
089
090
091    /**
092     * Write the Huffman symbol to output byte map
093     *
094     * @throws java.io.IOException on any I/O error writing the data
095     */
096    private void writeSymbolMap() throws IOException {
097
098        BZip2BitOutputStream bitOutputStream = this.bitOutputStream;
099
100        final boolean[] blockValuesPresent = this.blockValuesPresent;
101        final boolean[] condensedInUse = new boolean[16];
102
103        for (int i = 0; i < 16; i++) {
104            for (int j = 0, k = i << 4; j < 16; j++, k++) {
105                if (blockValuesPresent[k]) {
106                    condensedInUse[i] = true;
107                }
108            }
109        }
110
111        for (int i = 0; i < 16; i++) {
112            bitOutputStream.writeBoolean(condensedInUse[i]);
113        }
114
115        for (int i = 0; i < 16; i++) {
116            if (condensedInUse[i]) {
117                for (int j = 0, k = i * 16; j < 16; j++, k++) {
118                    bitOutputStream.writeBoolean(blockValuesPresent[k]);
119                }
120            }
121        }
122
123    }
124
125
126    /**
127     * Writes an RLE run to the block array, updating the block CRC and present values array as required
128     *
129     * @param value     The value to write
130     * @param runLength The run length of the value to write
131     */
132    private void writeRun(final int value, int runLength) {
133
134        final int blockLength = this.blockLength;
135        final byte[] block = this.block;
136
137        this.blockValuesPresent[value] = true;
138        this.crc.updateCRC(value, runLength);
139
140        final byte byteValue = (byte) value;
141        switch (runLength) {
142            case 1:
143                block[blockLength] = byteValue;
144                this.blockLength = blockLength + 1;
145                break;
146
147            case 2:
148                block[blockLength] = byteValue;
149                block[blockLength + 1] = byteValue;
150                this.blockLength = blockLength + 2;
151                break;
152
153            case 3:
154                block[blockLength] = byteValue;
155                block[blockLength + 1] = byteValue;
156                block[blockLength + 2] = byteValue;
157                this.blockLength = blockLength + 3;
158                break;
159
160            default:
161                runLength -= 4;
162                this.blockValuesPresent[runLength] = true;
163                block[blockLength] = byteValue;
164                block[blockLength + 1] = byteValue;
165                block[blockLength + 2] = byteValue;
166                block[blockLength + 3] = byteValue;
167                block[blockLength + 4] = (byte) runLength;
168                this.blockLength = blockLength + 5;
169                break;
170        }
171
172    }
173
174
175    /**
176     * Writes a byte to the block, accumulating to an RLE run where possible
177     *
178     * @param value The byte to write
179     * @return {@code true} if the byte was written, or {@code false} if the block is already full
180     */
181    public boolean write(final int value) {
182
183        if (this.blockLength > this.blockLengthLimit) {
184            return false;
185        }
186
187        final int rleCurrentValue = this.rleCurrentValue;
188        final int rleLength = this.rleLength;
189
190        if (rleLength == 0) {
191            this.rleCurrentValue = value;
192            this.rleLength = 1;
193        } else if (rleCurrentValue != value) {
194            // This path commits us to write 6 bytes - one RLE run (5 bytes) plus one extra
195            writeRun(rleCurrentValue & 0xff, rleLength);
196            this.rleCurrentValue = value;
197            this.rleLength = 1;
198        } else {
199            if (rleLength == 254) {
200                writeRun(rleCurrentValue & 0xff, 255);
201                this.rleLength = 0;
202            } else {
203                this.rleLength = rleLength + 1;
204            }
205        }
206
207        return true;
208
209    }
210
211
212    /**
213     * Writes an array to the block
214     *
215     * @param data   The array to write
216     * @param offset The offset within the input data to write from
217     * @param length The number of bytes of input data to write
218     * @return The actual number of input bytes written. May be less than the number requested, or
219     * zero if the block is already full
220     */
221    public int write(final byte[] data, int offset, int length) {
222
223        int written = 0;
224
225        while (length-- > 0) {
226            if (!write(data[offset++])) {
227                break;
228            }
229            written++;
230        }
231
232        return written;
233
234    }
235
236
237    /**
238     * Compresses and writes out the block
239     *
240     * @throws java.io.IOException on any I/O error writing the data
241     */
242    public void close() throws IOException {
243
244        // If an RLE run is in progress, write it out
245        if (this.rleLength > 0) {
246            writeRun(this.rleCurrentValue & 0xff, this.rleLength);
247        }
248
249        // Apply a one byte block wrap required by the BWT implementation
250        this.block[this.blockLength] = this.block[0];
251
252        // Perform the Burrows Wheeler Transform
253        BZip2DivSufSort divSufSort = new BZip2DivSufSort(this.block, this.bwtBlock, this.blockLength);
254        int bwtStartPointer = divSufSort.bwt();
255
256        // Write out the block header
257        this.bitOutputStream.writeBits(24, BZip2Constants.BLOCK_HEADER_MARKER_1);
258        this.bitOutputStream.writeBits(24, BZip2Constants.BLOCK_HEADER_MARKER_2);
259        this.bitOutputStream.writeInteger(this.crc.getCRC());
260        this.bitOutputStream.writeBoolean(false); // Randomised block flag. We never create randomised blocks
261        this.bitOutputStream.writeBits(24, bwtStartPointer);
262
263        // Write out the symbol map
264        writeSymbolMap();
265
266        // Perform the Move To Front Transform and Run-Length Encoding[2] stages
267        BZip2MTFAndRLE2StageEncoder mtfEncoder = new BZip2MTFAndRLE2StageEncoder(this.bwtBlock, this.blockLength, this.blockValuesPresent);
268        mtfEncoder.encode();
269
270        // Perform the Huffman Encoding stage and write out the encoded data
271        BZip2HuffmanStageEncoder huffmanEncoder = new BZip2HuffmanStageEncoder(this.bitOutputStream, mtfEncoder.getMtfBlock(), mtfEncoder.getMtfLength(), mtfEncoder.getMtfAlphabetSize(), mtfEncoder.getMtfSymbolFrequencies());
272        huffmanEncoder.encode();
273
274    }
275
276
277    /**
278     * Determines if any bytes have been written to the block
279     *
280     * @return {@code true} if one or more bytes has been written to the block, otherwise
281     * {@code false}
282     */
283    public boolean isEmpty() {
284
285        return ((this.blockLength == 0) && (this.rleLength == 0));
286
287    }
288
289
290    /**
291     * Gets the CRC of the completed block. Only valid after calling {@link #close()}
292     *
293     * @return The block's CRC
294     */
295    public int getCRC() {
296
297        return this.crc.getCRC();
298
299    }
300
301
302    /**
303     * @param bitOutputStream The BZip2BitOutputStream to which compressed BZip2 data is written
304     * @param blockSize       The declared block size in bytes. Up to this many bytes will be accepted
305     *                        into the block after Run-Length Encoding is applied
306     */
307    public BZip2BlockCompressor(final BZip2BitOutputStream bitOutputStream, final int blockSize) {
308
309        this.bitOutputStream = bitOutputStream;
310
311        // One extra byte is added to allow for the block wrap applied in close()
312        this.block = new byte[blockSize + 1];
313        this.bwtBlock = new int[blockSize + 1];
314        this.blockLengthLimit = blockSize - 6; // 5 bytes for one RLE run plus one byte - see {@link #write(int)}
315
316    }
