017        /*
018        index of the last char in the block, so
019        the block size == last + 1.

024        index in zptr[] of original string after sorting.

028        /*
029        always: in the range 0 .. 9.

032        private int blockSize100k;
033        private boolean blockRandomised;
034        private int bsBuff;
035        private int bsLive;
036        private CRC mCrc = new CRC();
037        private boolean[] inUse = new boolean[256];

085        private void makeMaps() {
086            nInUse = 0;
087            for (int j = 0; j < 256; j++) {
088                if (inUse[j]) {
089                    seqToUnseq[nInUse] = (char) j;
090                    nInUse++;
091                }
092            }
093        }
094    
095        private void initStream() throws IOException {
096            initialize();
097            initBlock();
098            setupBlock();
099        }
100    
101        private int read() throws IOException {
102            if (streamEnd) {
103                return -1;
104            } else {
105                int retChar = currentChar;
106                switch (currentState) {
107                    case START_BLOCK_STATE:
108                        break;
109                    case RAND_PART_A_STATE:
110                        break;
111                    case RAND_PART_B_STATE:
112                        setupRandPartB();
113                        break;
114                    case RAND_PART_C_STATE:
115                        setupRandPartC();
116                        break;
117                    case NO_RAND_PART_A_STATE:
118                        break;
119                    case NO_RAND_PART_B_STATE:
120                        setupNoRandPartB();
121                        break;
122                    case NO_RAND_PART_C_STATE:
123                        setupNoRandPartC();
124                        break;
125                    default:
126                        break;
127                }
128                return retChar;
129            }
130        }

132        private void initialize() throws IOException {
133            char magic1, magic2;
134            char magic3, magic4;
135            magic1 = bsGetUChar();
136            magic2 = bsGetUChar();
137            magic3 = bsGetUChar();
138            magic4 = bsGetUChar();
139            if (magic1 != 'B' || magic2 != 'Z' || magic3 != 'h' || magic4 < '1' || magic4 > '9') {
140                bsFinishedWithStream();
141                streamEnd = true;
142                return;
143            }
144    
145            setDecompressStructureSizes(magic4 - '0');
146            computedCombinedCRC = 0;
147        }
148    
149        private void initBlock() throws IOException {
150            char magic1, magic2, magic3, magic4;
151            char magic5, magic6;
152            magic1 = bsGetUChar();
153            magic2 = bsGetUChar();
154            magic3 = bsGetUChar();
155            magic4 = bsGetUChar();
156            magic5 = bsGetUChar();
157            magic6 = bsGetUChar();
158            if (magic1 == 0x17 && magic2 == 0x72 && magic3 == 0x45 && magic4 == 0x38 && magic5 == 0x50 && magic6 == 0x90) {
159                complete();
160                return;
161            }
162    
163            if (magic1 != 0x31 || magic2 != 0x41 || magic3 != 0x59 || magic4 != 0x26 || magic5 != 0x53 || magic6 != 0x59) {
164                throw new IOException("bad block header");
165            }
166    
167            storedBlockCRC = bsGetInt32();
168    
169            if (bsR(1) == 1) {
170                blockRandomised = true;
171            } else {
172                blockRandomised = false;
173            }
174    
175            //        currBlockNo++;
176            getAndMoveToFrontDecode();
177    
178            mCrc.initialiseCRC();
179            currentState = START_BLOCK_STATE;
180        }
181    
182        private void endBlock() throws IOException {
183            computedBlockCRC = mCrc.getFinalCRC();
184            /* A bad CRC is considered a fatal error. */
185            if (storedBlockCRC != computedBlockCRC) {
186                throw new IOException("CRC error");
187            }
188    
189            computedCombinedCRC = (computedCombinedCRC << 1) | (computedCombinedCRC >>> 31);
190            computedCombinedCRC ^= computedBlockCRC;
191        }
192    
193        private void complete() throws IOException {
194            storedCombinedCRC = bsGetInt32();
195            if (storedCombinedCRC != computedCombinedCRC) {
196                throw new IOException("CRC error");
197            }
198    
199            bsFinishedWithStream();
200            streamEnd = true;
201        }
202    
203        private void bsFinishedWithStream() {
204            try {
205                if (this.bsStream != null && this.bsStream != System.in) {
206                    this.bsStream.close();
207                    this.bsStream = null;
208    
209                }
210            } catch (IOException ioe) {
211                //ignore
212            }
213        }

215        private void bsSetStream() {
216            bsStream = null;
217            bsLive = 0;
218            bsBuff = 0;
219        }

221        private int bsR(int n) throws IOException {
222            int v;
223            while (bsLive < n) {
224                int zzi;
225                char thech = 0;
226    
227                thech = (char) bsStream.read();
228    
229                if (thech == -1) {
230                    throw new IOException("unexpected end of file");
231                }
232                zzi = thech;
233                bsBuff = (bsBuff << 8) | (zzi & 0xff);
234                bsLive += 8;
235            }
236    
237            v = (bsBuff >> (bsLive - n)) & ((1 << n) - 1);
238            bsLive -= n;
239            return v;
240        }

242        private char bsGetUChar() throws IOException {
243            return (char) bsR(8);
244        }

246        private int bsGetint() throws IOException {
247            int u = 0;
248            u = (u << 8) | bsR(8);
249            u = (u << 8) | bsR(8);
250            u = (u << 8) | bsR(8);
251            u = (u << 8) | bsR(8);
252            return u;
253        }

263        private void hbCreateDecodeTables(int[] limit, int[] base,
264                                          int[] perm, char[] length,
265                                          int minLen, int maxLen, int alphaSize) {
266            int pp, i, j, vec;
267    
268            pp = 0;
269            for (i = minLen; i <= maxLen; i++) {
270                for (j = 0; j < alphaSize; j++) {
271                    if (length[j] == i) {
272                        perm[pp] = j;
273                        pp++;
274                    }
275                }
276            }
277    
278            for (i = 0; i < MAX_CODE_LEN; i++) {
279                base[i] = 0;
280            }
281            for (i = 0; i < alphaSize; i++) {
282                base[length[i] + 1]++;
283            }
284    
285            for (i = 1; i < MAX_CODE_LEN; i++) {
286                base[i] += base[i - 1];
287            }
288    
289            for (i = 0; i < MAX_CODE_LEN; i++) {
290                limit[i] = 0;
291            }
292            vec = 0;
293    
294            for (i = minLen; i <= maxLen; i++) {
295                vec += (base[i + 1] - base[i]);
296                limit[i] = vec - 1;
297                vec <<= 1;
298            }
299            for (i = minLen + 1; i <= maxLen; i++) {
300                base[i] = ((limit[i - 1] + 1) << 1) - base[i];
301            }
302        }

304        private void recvDecodingTables() throws IOException {
305            char len[][] = new char[N_GROUPS][MAX_ALPHA_SIZE];
306            int i, j, t, nGroups, nSelectors, alphaSize;
307            int minLen, maxLen;
308            boolean[] inUse16 = new boolean[16];
309    
310            /* Receive the mapping table */
311            for (i = 0; i < 16; i++) {
312                if (bsR(1) == 1) {
313                    inUse16[i] = true;
314                } else {
315                    inUse16[i] = false;
316                }
317            }
318    
319            for (i = 0; i < 256; i++) {
320                inUse[i] = false;
321            }
322    
323            for (i = 0; i < 16; i++) {
324                if (inUse16[i]) {
325                    for (j = 0; j < 16; j++) {
326                        if (bsR(1) == 1) {
327                            inUse[i * 16 + j] = true;
328                        }
329                    }
330                }
331            }
332    
333            makeMaps();
334            alphaSize = nInUse + 2;
335    
336            /* Now the selectors */
337            nGroups = bsR(3);
338            nSelectors = bsR(15);
339            for (i = 0; i < nSelectors; i++) {
340                j = 0;
341                while (bsR(1) == 1) {
342                    j++;
343                }
344                selectorMtf[i] = (char) j;
345            }
346    
347            /* Undo the MTF values for the selectors. */
348            {
349                char[] pos = new char[N_GROUPS];
350                char tmp, v;
351                for (v = 0; v < nGroups; v++) {
352                    pos[v] = v;
353                }
354    
355                for (i = 0; i < nSelectors; i++) {
356                    v = selectorMtf[i];
357                    tmp = pos[v];
358                    while (v > 0) {
359                        pos[v] = pos[v - 1];
360                        v--;
361                    }
362                    pos[0] = tmp;
363                    selector[i] = tmp;
364                }
365            }
366    
367            /* Now the coding tables */
368            for (t = 0; t < nGroups; t++) {
369                int curr = bsR(5);
370                for (i = 0; i < alphaSize; i++) {
371                    while (bsR(1) == 1) {
372                        if (bsR(1) == 0) {
373                            curr++;
374                        } else {
375                            curr--;
376                        }
377                    }
378                    len[t][i] = (char) curr;
379                }
380            }
381    
382            /* Create the Huffman decoding tables */
383            for (t = 0; t < nGroups; t++) {
384                minLen = 32;
385                maxLen = 0;
386                for (i = 0; i < alphaSize; i++) {
387                    if (len[t][i] > maxLen) {
388                        maxLen = len[t][i];
389                    }
390                    if (len[t][i] < minLen) {
391                        minLen = len[t][i];
392                    }
393                }
394                hbCreateDecodeTables(limit[t], base[t], perm[t], len[t], minLen,
395                        maxLen, alphaSize);
396                minLens[t] = minLen;
397            }
398        }

400        private void getAndMoveToFrontDecode() throws IOException {
401            char[] yy = new char[256];
402            int i, j, nextSym, limitLast;
403            int EOB, groupNo, groupPos;
404    
405            limitLast = BASE_BLOCK_SIZE * blockSize100k;
406            origPtr = bsGetIntVS(24);
407    
408            recvDecodingTables();
409            EOB = nInUse + 1;
410            groupNo = -1;
411            groupPos = 0;
412    
413            /*
414            Setting up the unzftab entries here is not strictly
415            necessary, but it does save having to do it later
416            in a separate pass, and so saves a block's worth of
417            cache misses.
418             */
419            for (i = 0; i <= 255; i++) {
420                unzftab[i] = 0;
421            }
422    
423            for (i = 0; i <= 255; i++) {
424                yy[i] = (char) i;
425            }
426    
427            last = -1;
428    
429            {
430                int zt, zn, zvec, zj;
431                if (groupPos == 0) {
432                    groupNo++;
433                    groupPos = G_SIZE;
434                }
435                groupPos--;
436                zt = selector[groupNo];
437                zn = minLens[zt];
438                zvec = bsR(zn);
439                while (zvec > limit[zt][zn]) {
440                    zn++;
441                    {
442                        {
443                            while (bsLive < 1) {
444                                int zzi;
445                                char thech = 0;
446    
447                                thech = (char) bsStream.read();
448    
449                                if (thech == -1) {
450                                    throw new IOException("unexpected end of file");
451                                }
452                                zzi = thech;
453                                bsBuff = (bsBuff << 8) | (zzi & 0xff);
454                                bsLive += 8;
455                            }
456                        }
457                        zj = (bsBuff >> (bsLive - 1)) & 1;
458                        bsLive--;
459                    }
460                    zvec = (zvec << 1) | zj;
461                }
462                nextSym = perm[zt][zvec - base[zt][zn]];
463            }
464    
465            while (true) {
466    
467                if (nextSym == EOB) {
468                    break;
469                }
470    
471                if (nextSym == RUNA || nextSym == RUNB) {
472                    char ch;
473                    int s = -1;
474                    int N = 1;
475                    do {
476                        if (nextSym == RUNA) {
477                            s = s + (0 + 1) * N;
478                        } else if (nextSym == RUNB) {
479                            s = s + (1 + 1) * N;
480                        }
481                        N = N * 2;
482                        {
483                            int zt, zn, zvec, zj;
484                            if (groupPos == 0) {
485                                groupNo++;
486                                groupPos = G_SIZE;
487                            }
488                            groupPos--;
489                            zt = selector[groupNo];
490                            zn = minLens[zt];
491                            zvec = bsR(zn);
492                            while (zvec > limit[zt][zn]) {
493                                zn++;
494                                {
495                                    {
496                                        while (bsLive < 1) {
497                                            int zzi;
498                                            char thech = 0;
499    
500                                            thech = (char) bsStream.read();
501    
502                                            if (thech == -1) {
503                                                throw new IOException("unexpected end of file");
504                                            }
505                                            zzi = thech;
506                                            bsBuff = (bsBuff << 8) | (zzi & 0xff);
507                                            bsLive += 8;
508                                        }
509                                    }
510                                    zj = (bsBuff >> (bsLive - 1)) & 1;
511                                    bsLive--;
512                                }
513                                zvec = (zvec << 1) | zj;
514                            }
515                            nextSym = perm[zt][zvec - base[zt][zn]];
516                        }
517                    } while (nextSym == RUNA || nextSym == RUNB);
518    
519                    s++;
520                    ch = seqToUnseq[yy[0]];
521                    unzftab[ch] += s;
522    
523                    while (s > 0) {
524                        last++;
525                        ll8[last] = ch;
526                        s--;
527                    }
528    
529                    if (last >= limitLast) {
530                        throw new IOException("block overrun");
531                    }
532                    continue;
533                } else {
534                    char tmp;
535                    last++;
536                    if (last >= limitLast) {
537                        throw new IOException("block overrun");
538                    }
539    
540                    tmp = yy[nextSym - 1];
541                    unzftab[seqToUnseq[tmp]]++;
542                    ll8[last] = seqToUnseq[tmp];
543    
544                    /*
545                    This loop is hammered during decompression,
546                    hence the unrolling.
547    
548                    for (j = nextSym-1; j > 0; j--) yy[j] = yy[j-1];
549                     */
550    
551                    j = nextSym - 1;
552                    for (; j > 3; j -= 4) {
553                        yy[j] = yy[j - 1];
554                        yy[j - 1] = yy[j - 2];
555                        yy[j - 2] = yy[j - 3];
556                        yy[j - 3] = yy[j - 4];
557                    }
558                    for (; j > 0; j--) {
559                        yy[j] = yy[j - 1];
560                    }
561    
562                    yy[0] = tmp;
563                    {
564                        int zt, zn, zvec, zj;
565                        if (groupPos == 0) {
566                            groupNo++;
567                            groupPos = G_SIZE;
568                        }
569                        groupPos--;
570                        zt = selector[groupNo];
571                        zn = minLens[zt];
572                        zvec = bsR(zn);
573                        while (zvec > limit[zt][zn]) {
574                            zn++;
575                            {
576                                {
577                                    while (bsLive < 1) {
578                                        int zzi;
579                                        char thech = 0;
580    
581                                        thech = (char) bsStream.read();
582    
583                                        zzi = thech;
584                                        bsBuff = (bsBuff << 8) | (zzi & 0xff);
585                                        bsLive += 8;
586                                    }
587                                }
588                                zj = (bsBuff >> (bsLive - 1)) & 1;
589                                bsLive--;
590                            }
591                            zvec = (zvec << 1) | zj;
592                        }
593                        nextSym = perm[zt][zvec - base[zt][zn]];
594                    }
595                    continue;
596                }
597            }
598        }

600        private void setupBlock() throws IOException {
601            int i;
602            int[] cftab = new int[257];
603            char ch;
604    
605            cftab[0] = 0;
606            for (i = 1; i <= 256; i++) {
607                cftab[i] = unzftab[i - 1];
608            }
609            for (i = 1; i <= 256; i++) {
610                cftab[i] += cftab[i - 1];
611            }
612    
613            for (i = 0; i <= last; i++) {
614                ch = ll8[i];
615                tt[cftab[ch]] = i;
616                cftab[ch]++;
617            }
618            cftab = null;
619    
620            tPos = tt[origPtr];
621    
622            count = 0;
623            i2 = 0;
624            ch2 = 256;   /* not a char and not EOF */
625    
626            if (blockRandomised) {
627                rNToGo = 0;
628                rTPos = 0;
629                setupRandPartA();
630            } else {
631                setupNoRandPartA();
632            }
633        }
634    
635        private void setupRandPartA() throws IOException {
636            if (i2 <= last) {
637                chPrev = ch2;
638                ch2 = ll8[tPos];
639                tPos = tt[tPos];
640                if (rNToGo == 0) {
641                    rNToGo = NUMS[rTPos];
642                    rTPos++;
643                    if (rTPos == 512) {
644                        rTPos = 0;
645                    }
646                }
647                rNToGo--;
648                ch2 ^= ((rNToGo == 1) ? 1 : 0);
649                i2++;
650    
651                currentChar = ch2;
652                currentState = RAND_PART_B_STATE;
653                mCrc.updateCRC(ch2);
654            } else {
655                endBlock();
656                initBlock();
657                setupBlock();
658            }
659        }
660    
661        private void setupNoRandPartA() throws IOException {
662            if (i2 <= last) {
663                chPrev = ch2;
664                ch2 = ll8[tPos];
665                tPos = tt[tPos];
666                i2++;
667    
668                currentChar = ch2;
669                currentState = NO_RAND_PART_B_STATE;
670                mCrc.updateCRC(ch2);
671            } else {
672                endBlock();
673                initBlock();
674                setupBlock();
675            }
676        }
677    
678        private void setupRandPartB() throws IOException {
679            if (ch2 != chPrev) {
680                currentState = RAND_PART_A_STATE;
681                count = 1;
682                setupRandPartA();
683            } else {
684                count++;
685                if (count >= 4) {
686                    z = ll8[tPos];
687                    tPos = tt[tPos];
688                    if (rNToGo == 0) {
689                        rNToGo = NUMS[rTPos];
690                        rTPos++;
691                        if (rTPos == 512) {
692                            rTPos = 0;
693                        }
694                    }
695                    rNToGo--;
696                    z ^= ((rNToGo == 1) ? 1 : 0);
697                    j2 = 0;
698                    currentState = RAND_PART_C_STATE;
699                    setupRandPartC();
700                } else {
701                    currentState = RAND_PART_A_STATE;
702                    setupRandPartA();
703                }
704            }
705        }
706    
707        private void setupRandPartC() throws IOException {
708            if (j2 < (int) z) {
709                currentChar = ch2;
710                mCrc.updateCRC(ch2);
711                j2++;
712            } else {
713                currentState = RAND_PART_A_STATE;
714                i2++;
715                count = 0;
716                setupRandPartA();
717            }
718        }
719    
720        private void setupNoRandPartB() throws IOException {
721            if (ch2 != chPrev) {
722                currentState = NO_RAND_PART_A_STATE;
723                count = 1;
724                setupNoRandPartA();
725            } else {
726                count++;
727                if (count >= 4) {
728                    z = ll8[tPos];
729                    tPos = tt[tPos];
730                    currentState = NO_RAND_PART_C_STATE;
731                    j2 = 0;
732                    setupNoRandPartC();
733                } else {
734                    currentState = NO_RAND_PART_A_STATE;
735                    setupNoRandPartA();
736                }
737            }
738        }
739    
740        private void setupNoRandPartC() throws IOException {
741            if (j2 < (int) z) {
742                currentChar = ch2;
743                mCrc.updateCRC(ch2);
744                j2++;
745            } else {
746                currentState = NO_RAND_PART_A_STATE;
747                i2++;
748                count = 0;
749                setupNoRandPartA();
750            }
751        }

753        private void setDecompressStructureSizes(int newSize100k) throws IOException {
754            if (!(0 <= newSize100k && newSize100k <= 9 && 0 <= blockSize100k && blockSize100k <= 9)) {
755                throw new IOException("invalid block size");
756            }
757    
758            blockSize100k = newSize100k;
759    
760            if (newSize100k == 0) {
761                return;
762            }
763    
764            int n = BASE_BLOCK_SIZE * newSize100k;
765            ll8 = new char[n];
766            tt = new int[n];
767        }

769        public void setInput(byte[] b) throws IOException {
770            this.setInput(b, 0, b.length);
771        }

773        public void setInput(byte[] b, int off, int len) throws IOException {
774            this.inputBuffer.write(b, off, len);
775    
776            if (!this.readHeader) {
777                if (this.inputBuffer.size() > 10) {
778                    this.bsStream = new ByteArrayInputStream(this.inputBuffer.toByteArray());
779                    this.inputBuffer.reset();
780    
781                    this.initStream();
782    
783                    if (this.streamEnd == true) {
784                        throw new IOException("invalid BZip2 stream");
785                    }
786    
787                    this.readHeader = true;
788                }
789            } else if (this.inputBuffer.size() > 0 && (this.bsStream == null || !(this.bsStream.available() > 0))) {
790                this.flush();
791            } else if (this.inputBuffer.size() > 8192) {
792                this.flush();
793            }
794        }

803            int rc = 0;
804    
805            while (this.bsStream != null && this.bsStream.available() > 0 && rc < b.length) {
806                b[rc++] = (byte) this.read();
807            }

814        public void flush() {
815            if (this.inputBuffer.size() > 0) {
816                if (this.bsStream == null || !(this.bsStream.available() > 0)) {
817                    this.bsStream = new ByteArrayInputStream(this.inputBuffer.toByteArray());
818                } else {
819                    byte[] buff = new byte[this.bsStream.available() + this.inputBuffer.size()];
820    
821                    int i = 0;
822    
823                    try {
824                        i = this.bsStream.read(buff);
825                    } catch (IOException e) {
826                    }
827    
828                    if (i == 0) {
829                        this.bsStream = new ByteArrayInputStream(this.inputBuffer.toByteArray());
830                    } else {
831                        byte[] o = this.inputBuffer.toByteArray();
832    
833                        System.arraycopy(o, 0, buff, i, o.length);
834    
835                        this.bsStream = new ByteArrayInputStream(buff);
836                    }
837                }
838    
839                this.inputBuffer.reset();
840            }
