035public final class Inflate {
036
037    // preset dictionary flag in zlib header
038    private static final int PRESET_DICT = 0x20;
039    private static final int Z_FINISH = 4;
040    private static final int Z_DEFLATED = 8;
041    private static final int Z_OK = 0;
042    private static final int Z_STREAM_END = 1;
043    private static final int Z_NEED_DICT = 2;
044    private static final int Z_STREAM_ERROR = -2;
045    private static final int Z_DATA_ERROR = -3;
046    private static final int Z_BUF_ERROR = -5;
047    private static final int METHOD = 0;   // waiting for method byte
048    private static final int FLAG = 1;     // waiting for flag byte
049    private static final int DICT4 = 2;    // four dictionary check bytes to go
050    private static final int DICT3 = 3;    // three dictionary check bytes to go
051    private static final int DICT2 = 4;    // two dictionary check bytes to go
052    private static final int DICT1 = 5;    // one dictionary check byte to go
053    private static final int DICT0 = 6;    // waiting for inflateSetDictionary
054    private static final int BLOCKS = 7;   // decompressing blocks
055    private static final int CHECK4 = 8;   // four check bytes to go
056    private static final int CHECK3 = 9;   // three check bytes to go
057    private static final int CHECK2 = 10;  // two check bytes to go
058    private static final int CHECK1 = 11;  // one check byte to go
059    private static final int DONE = 12;    // finished check, done
060    private static final int BAD = 13;     // got an error--stay here
061    private int mode;                            // current inflate mode
062    // mode dependent information
063    private int method;        // if FLAGS, method byte
064    // if CHECK, check values to compare
065    private long[] was = new long[1]; // computed check value
066    private long need;               // stream check value
067    // if BAD, inflateSync's marker bytes count
068    private int marker;
069    // mode independent information
070    private int nowrap;          // flag for no wrapper
071    private int wbits;            // log2(window size)  (8..15, defaults to 15)
072    private InfBlocks blocks;     // current inflate_blocks state
073
074    protected int inflateReset(ZStream z) {
075        if (z == null || z.istate == null) {
076            return Z_STREAM_ERROR;
077        }
078
079        z.totalin = z.totalout = 0;
080        z.msg = null;
081        z.istate.mode = z.istate.nowrap != 0 ? BLOCKS : METHOD;
082        z.istate.blocks.reset(z, null);
083        return Z_OK;
084    }
085
086    protected int inflateEnd(ZStream z) {
087        if (blocks != null) {
088            blocks.free(z);
089        }
090        blocks = null;
091        //    ZFREE(z, z->state);
092        return Z_OK;
093    }
094
095    protected int inflateInit(ZStream z, int w) {
096        z.msg = null;
097        blocks = null;
098
099        // handle undocumented nowrap option (no zlib header or check)
100        nowrap = 0;
101        if (w < 0) {
102            w = -w;
103            nowrap = 1;
104        }
105
106        // set window size
107        if (w < 8 || w > 15) {
108            inflateEnd(z);
109            return Z_STREAM_ERROR;
110        }
111        wbits = w;
112
113        z.istate.blocks = new InfBlocks(z,
114                z.istate.nowrap != 0 ? null : this,
115                1 << w);
116
117        // reset state
118        inflateReset(z);
119        return Z_OK;
120    }
121
122    protected int inflate(ZStream z, int f) {
123        int r;
124        int b;
125
126        if (z == null || z.istate == null || z.nextin == null) {
127            return Z_STREAM_ERROR;
128        }
129        f = f == Z_FINISH ? Z_BUF_ERROR : Z_OK;
130        r = Z_BUF_ERROR;
131        while (true) {
132            switch (z.istate.mode) {
133                case METHOD:
134                    if (z.availin == 0) {
135                        return r;
136                    }
137                    r = f;
138                    z.availin--;
139                    z.totalin++;
140                    if (((z.istate.method = z.nextin[z.nextinindex++]) & 0xf) != Z_DEFLATED) {
141                        z.istate.mode = BAD;
142                        z.msg = "unknown compression method";
143                        z.istate.marker = 5;       // can't try inflateSync
144                        break;
145                    }
146                    if ((z.istate.method >> 4) + 8 > z.istate.wbits) {
147                        z.istate.mode = BAD;
148                        z.msg = "invalid window size";
149                        z.istate.marker = 5;       // can't try inflateSync
150                        break;
151                    }
152                    z.istate.mode = FLAG;
153                    break;
154                case FLAG:
155                    if (z.availin == 0) {
156                        return r;
157                    }
158                    r = f;
159                    z.availin--;
160                    z.totalin++;
161                    b = (z.nextin[z.nextinindex++]) & 0xff;
162                    if ((((z.istate.method << 8) + b) % 31) != 0) {
163                        z.istate.mode = BAD;
164                        z.msg = "incorrect header check";
165                        z.istate.marker = 5;       // can't try inflateSync
166                        break;
167                    }
168
169                    if ((b & PRESET_DICT) == 0) {
170                        z.istate.mode = BLOCKS;
171                        break;
172                    }
173                    z.istate.mode = DICT4;
174                    break;
175                case DICT4:
176                    if (z.availin == 0) {
177                        return r;
178                    }
179                    r = f;
180                    z.availin--;
181                    z.totalin++;
182                    z.istate.need = ((z.nextin[z.nextinindex++] & 0xff) << 24) & 0xff000000L;
183                    z.istate.mode = DICT3;
184                    break;
185                case DICT3:
186                    if (z.availin == 0) {
187                        return r;
188                    }
189                    r = f;
190                    z.availin--;
191                    z.totalin++;
192                    z.istate.need += ((z.nextin[z.nextinindex++] & 0xff) << 16) & 0xff0000L;
193                    z.istate.mode = DICT2;
194                    break;
195                case DICT2:
196                    if (z.availin == 0) {
197                        return r;
198                    }
199                    r = f;
200                    z.availin--;
201                    z.totalin++;
202                    z.istate.need += ((z.nextin[z.nextinindex++] & 0xff) << 8) & 0xff00L;
203                    z.istate.mode = DICT1;
204                    break;
205                case DICT1:
206                    if (z.availin == 0) {
207                        return r;
208                    }
209                    r = f;
210                    z.availin--;
211                    z.totalin++;
212                    z.istate.need += (z.nextin[z.nextinindex++] & 0xffL);
213                    z.adler = z.istate.need;
214                    z.istate.mode = DICT0;
215                    return Z_NEED_DICT;
216                case DICT0:
217                    z.istate.mode = BAD;
218                    z.msg = "need dictionary";
219                    z.istate.marker = 0;       // can try inflateSync
220                    return Z_STREAM_ERROR;
221                case BLOCKS:
222                    r = z.istate.blocks.proc(z, r);
223                    if (r == Z_DATA_ERROR) {
224                        z.istate.mode = BAD;
225                        z.istate.marker = 0;     // can try inflateSync
226                        break;
227                    }
228                    if (r == Z_OK) {
229                        r = f;
230                    }
231                    if (r != Z_STREAM_END) {
232                        return r;
233                    }
234                    r = f;
235                    z.istate.blocks.reset(z, z.istate.was);
236                    if (z.istate.nowrap != 0) {
237                        z.istate.mode = DONE;
238                        break;
239                    }
240                    z.istate.mode = CHECK4;
241                    break;
242                case CHECK4:
243                    if (z.availin == 0) {
244                        return r;
245                    }
246                    r = f;
247                    z.availin--;
248                    z.totalin++;
249                    z.istate.need = ((z.nextin[z.nextinindex++] & 0xff) << 24) & 0xff000000L;
250                    z.istate.mode = CHECK3;
251                    break;
252                case CHECK3:
253                    if (z.availin == 0) {
254                        return r;
255                    }
256                    r = f;
257                    z.availin--;
258                    z.totalin++;
259                    z.istate.need += ((z.nextin[z.nextinindex++] & 0xff) << 16) & 0xff0000L;
260                    z.istate.mode = CHECK2;
261                    break;
262                case CHECK2:
263                    if (z.availin == 0) {
264                        return r;
265                    }
266                    r = f;
267                    z.availin--;
268                    z.totalin++;
269                    z.istate.need += ((z.nextin[z.nextinindex++] & 0xff) << 8) & 0xff00L;
270                    z.istate.mode = CHECK1;
271                    break;
272                case CHECK1:
273                    if (z.availin == 0) {
274                        return r;
275                    }
276                    r = f;
277                    z.availin--;
278                    z.totalin++;
279                    z.istate.need += (z.nextin[z.nextinindex++] & 0xffL);
280                    if (((int) (z.istate.was[0])) != ((int) (z.istate.need))) {
281                        z.istate.mode = BAD;
282                        z.msg = "incorrect data check";
283                        z.istate.marker = 5;       // can't try inflateSync
284                        break;
285                    }
286                    z.istate.mode = DONE;
287                    break;
288                case DONE:
289                    return Z_STREAM_END;
290                case BAD:
291                    return Z_DATA_ERROR;
292                default:
293                    return Z_STREAM_ERROR;
294            }
295        }
296    }

298    protected int inflateSetDictionary(ZStream z, byte[] dictionary, int dictLength) {
299        int index = 0;
300        int length = dictLength;
301        if (z == null || z.istate == null || z.istate.mode != DICT0) {
302            return Z_STREAM_ERROR;
303        }
304
305        if (Adler32.adler32(1L, dictionary, 0, dictLength) != z.adler) {
306            return Z_DATA_ERROR;
307        }
308
309        z.adler = Adler32.adler32(0, null, 0, 0);
310
311        if (length >= (1 << z.istate.wbits)) {
312            length = (1 << z.istate.wbits) - 1;
313            index = dictLength - length;
314        }
315        z.istate.blocks.setDictionary(dictionary, index, length);
316        z.istate.mode = BLOCKS;
317        return Z_OK;
318    }
319    private static byte[] mark = {(byte) 0, (byte) 0, (byte) 0xff, (byte) 0xff};
320
321    protected int inflateSync(ZStream z) {
322        int n;       // number of bytes to look at
323        int p;       // pointer to bytes
324        int m;       // number of marker bytes found in a row
325        long r, w;   // temporaries to save total_in and total_out
326
327        // set up
328        if (z == null || z.istate == null) {
329            return Z_STREAM_ERROR;
330        }
331        if (z.istate.mode != BAD) {
332            z.istate.mode = BAD;
333            z.istate.marker = 0;
334        }
335        if ((n = z.availin) == 0) {
336            return Z_BUF_ERROR;
337        }
338        p = z.nextinindex;
339        m = z.istate.marker;
340
341        // search
342        while (n != 0 && m < 4) {
343            if (z.nextin[p] == mark[m]) {
344                m++;
345            } else if (z.nextin[p] != 0) {
346                m = 0;
347            } else {
348                m = 4 - m;
349            }
350            p++;
351            n--;
352        }
353
354        // restore
355        z.totalin += p - z.nextinindex;
356        z.nextinindex = p;
357        z.availin = n;
358        z.istate.marker = m;
359
360        // return no joy or set up to restart on a new block
361        if (m != 4) {
362            return Z_DATA_ERROR;
363        }
364        r = z.totalin;
365        w = z.totalout;
366        inflateReset(z);
367        z.totalin = r;
368        z.totalout = w;
369        z.istate.mode = BLOCKS;
370        return Z_OK;
371    }
