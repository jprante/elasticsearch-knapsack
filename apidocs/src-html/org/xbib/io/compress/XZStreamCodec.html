002package org.xbib.io.compress;
003
004import org.xbib.io.StreamCodec;
005import org.xbib.io.compress.xz.LZMA2Options;
006import org.xbib.io.compress.xz.XZInputStream;
007import org.xbib.io.compress.xz.XZOutputStream;
008
009import java.io.IOException;
010import java.io.InputStream;
011import java.io.OutputStream;
012
013public class XZStreamCodec implements StreamCodec<XZInputStream, XZOutputStream> {
014
015    @Override
016    public String getName() {
017        return "xz";
018    }
019
020    @Override
021    public XZInputStream decode(InputStream in) throws IOException {
022        return new XZInputStream(in);
023    }
024
025    @Override
026    public XZInputStream decode(InputStream in, int bufsize) throws IOException {
027        return new XZInputStream(in, bufsize / 1024); // KB limit
028    }
029
030    @Override
031    public XZOutputStream encode(OutputStream out) throws IOException {
032        return new XZOutputStream(out, new LZMA2Options());
033    }
034
035    @Override
036    public XZOutputStream encode(OutputStream out, int bufsize) throws IOException {
037        return new XZOutputStream(out, new LZMA2Options()); // ignore bufsize
038    }
039}
