002    package org.xbib.io.compress.xz.rangecoder;
003    
004    import org.xbib.io.compress.xz.CorruptedInputException;
005    
006    import java.io.DataInputStream;
007    import java.io.IOException;
008    
009    public final class RangeDecoder extends RangeCoder {
010        private static final int INIT_SIZE = 5;
011    
012        private final byte[] buf;
013        private int pos = 0;
014        private int end = 0;
015    
016        private int range = 0;
017        private int code = 0;
018    
019        public RangeDecoder(int inputSizeMax) {
020            buf = new byte[inputSizeMax - INIT_SIZE];
021        }
022    
023        public void prepareInputBuffer(DataInputStream in, int len)
024                throws IOException {
025            if (len < INIT_SIZE) {
026                throw new CorruptedInputException();
027            }
028    
029            if (in.readUnsignedByte() != 0x00) {
030                throw new CorruptedInputException();
031            }
032    
033            code = in.readInt();
034            range = 0xFFFFFFFF;
035    
036            pos = 0;
037            end = len - INIT_SIZE;
038            in.readFully(buf, 0, end);
039        }
040    
041        public boolean isInBufferOK() {
042            return pos <= end;
043        }
044    
045        public boolean isFinished() {
046            return pos == end && code == 0;
047        }
048    
049        public void normalize() throws IOException {
050            if ((range & TOP_MASK) == 0) {
051                try {
052                    // If the input is corrupt, this might throw
053                    // ArrayIndexOutOfBoundsException.
054                    code = (code << SHIFT_BITS) | (buf[pos++] & 0xFF);
055                    range <<= SHIFT_BITS;
056                } catch (ArrayIndexOutOfBoundsException e) {
057                    throw new CorruptedInputException();
058                }
059            }
060        }
061    
062        public int decodeBit(short[] probs, int index) throws IOException {
063            normalize();
064    
065            int prob = probs[index];
066            int bound = (range >>> BIT_MODEL_TOTAL_BITS) * prob;
067            int bit;
068    
069            // Compare code and bound as if they were unsigned 32-bit integers.
070            if ((code ^ 0x80000000) < (bound ^ 0x80000000)) {
071                range = bound;
072                probs[index] = (short) (
073                        prob + ((BIT_MODEL_TOTAL - prob) >>> MOVE_BITS));
074                bit = 0;
075            } else {
076                range -= bound;
077                code -= bound;
078                probs[index] = (short) (prob - (prob >>> MOVE_BITS));
079                bit = 1;
080            }
081    
082            return bit;
083        }
084    
085        public int decodeBitTree(short[] probs) throws IOException {
086            int symbol = 1;
087    
088            do {
089                symbol = (symbol << 1) | decodeBit(probs, symbol);
090            } while (symbol < probs.length);
091    
092            return symbol - probs.length;
093        }
094    
095        public int decodeReverseBitTree(short[] probs) throws IOException {
096            int symbol = 1;
097            int i = 0;
098            int result = 0;
099    
100            do {
101                int bit = decodeBit(probs, symbol);
102                symbol = (symbol << 1) | bit;
103                result |= bit << i++;
104            } while (symbol < probs.length);
105    
106            return result;
107        }
108    
109        public int decodeDirectBits(int count) throws IOException {
110            int result = 0;
111    
112            do {
113                normalize();
114    
115                range >>>= 1;
116                int t = (code - range) >>> 31;
117                code -= range & (t - 1);
118                result = (result << 1) | (1 - t);
119            } while (--count != 0);
120    
121            return result;
122        }
123    }
