002    package org.xbib.io.compress.xz.simple;
003    
004    public final class IA64 implements SimpleFilter {
005        private static final int[] BRANCH_TABLE = {
006                0, 0, 0, 0, 0, 0, 0, 0,
007                0, 0, 0, 0, 0, 0, 0, 0,
008                4, 4, 6, 6, 0, 0, 7, 7,
009                4, 4, 0, 0, 4, 4, 0, 0};
010    
011        private final boolean isEncoder;
012        private int pos;
013    
014        public IA64(boolean isEncoder, int startPos) {
015            this.isEncoder = isEncoder;
016            pos = startPos;
017        }
018    
019        public int code(byte[] buf, int off, int len) {
020            int end = off + len - 16;
021            int i;
022    
023            for (i = off; i <= end; i += 16) {
024                int instrTemplate = buf[i] & 0x1F;
025                int mask = BRANCH_TABLE[instrTemplate];
026    
027                for (int slot = 0, bitPos = 5; slot < 3; ++slot, bitPos += 41) {
028                    if (((mask >>> slot) & 1) == 0) {
029                        continue;
030                    }
031    
032                    int bytePos = bitPos >>> 3;
033                    int bitRes = bitPos & 7;
034    
035                    long instr = 0;
036                    for (int j = 0; j < 6; ++j) {
037                        instr |= (buf[i + bytePos + j] & 0xFFL) << (8 * j);
038                    }
039    
040                    long instrNorm = instr >>> bitRes;
041    
042                    if (((instrNorm >>> 37) & 0x0F) != 0x05
043                            || ((instrNorm >>> 9) & 0x07) != 0x00) {
044                        continue;
045                    }
046    
047                    int src = (int) ((instrNorm >>> 13) & 0x0FFFFF);
048                    src |= ((int) (instrNorm >>> 36) & 1) << 20;
049                    src <<= 4;
050    
051                    int dest;
052                    if (isEncoder) {
053                        dest = src + (pos + i - off);
054                    } else {
055                        dest = src - (pos + i - off);
056                    }
057    
058                    dest >>>= 4;
059    
060                    instrNorm &= ~(0x8FFFFFL << 13);
061                    instrNorm |= (dest & 0x0FFFFFL) << 13;
062                    instrNorm |= (dest & 0x100000L) << (36 - 20);
063    
064                    instr &= (1 << bitRes) - 1;
065                    instr |= instrNorm << bitRes;
066    
067                    for (int j = 0; j < 6; ++j) {
068                        buf[i + bytePos + j] = (byte) (instr >>> (8 * j));
069                    }
070                }
071            }
072    
073            i -= off;
074            pos += i;
075            return i;
076        }
077    }
