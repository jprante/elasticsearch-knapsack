011package org.xbib.io.compress.xz.simple;
012
013public final class ARM implements SimpleFilter {
014    private final boolean isEncoder;
015    private int pos;
016
017    public ARM(boolean isEncoder, int startPos) {
018        this.isEncoder = isEncoder;
019        pos = startPos + 8;
020    }
021
022    public int code(byte[] buf, int off, int len) {
023        int end = off + len - 4;
024        int i;
025
026        for (i = off; i <= end; i += 4) {
027            if ((buf[i + 3] & 0xFF) == 0xEB) {
028                int src = ((buf[i + 2] & 0xFF) << 16)
029                          | ((buf[i + 1] & 0xFF) << 8)
030                          | (buf[i] & 0xFF);
031                src <<= 2;
032
033                int dest;
034                if (isEncoder)
035                    dest = src + (pos + i - off);
036                else
037                    dest = src - (pos + i - off);
038
039                dest >>>= 2;
040                buf[i + 2] = (byte)(dest >>> 16);
041                buf[i + 1] = (byte)(dest >>> 8);
042                buf[i] = (byte)dest;
043            }
044        }
045
046        i -= off;
047        pos += i;
048        return i;
049    }
050}
