002package org.xbib.io.compress.xz;
003
004/**
005 * Thrown when the compressed input data is corrupt.
006 * However, it is possible that some or all of the data
007 * already read from the input stream was corrupt too.
008 */
009public class CorruptedInputException extends XZIOException {
010    private static final long serialVersionUID = 3L;
011
012    /**
013     * Creates a new CorruptedInputException with
014     * the default error detail message.
015     */
016    public CorruptedInputException() {
017        super("Compressed data is corrupt");
018    }
019
020    /**
021     * Creates a new CorruptedInputException with
022     * the specified error detail message.
023     *
024     * @param s error detail message
025     */
026    public CorruptedInputException(String s) {
027        super(s);
028    }
029}
