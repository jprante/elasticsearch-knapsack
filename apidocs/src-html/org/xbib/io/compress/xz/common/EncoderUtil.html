002    package org.xbib.io.compress.xz.common;
003    
004    import java.io.IOException;
005    import java.io.OutputStream;
006    import java.util.zip.CRC32;
007    
008    public class EncoderUtil extends Util {
009        public static void writeCRC32(OutputStream out, byte[] buf)
010                throws IOException {
011            CRC32 crc32 = new CRC32();
012            crc32.update(buf);
013            long value = crc32.getValue();
014    
015            for (int i = 0; i < 4; ++i) {
016                out.write((byte) (value >>> (i * 8)));
017            }
018        }
019    
020        public static void encodeVLI(OutputStream out, long num)
021                throws IOException {
022            while (num >= 0x80) {
023                out.write((byte) (num | 0x80));
024                num >>>= 7;
025            }
026    
027            out.write((byte) num);
028        }
029    }
