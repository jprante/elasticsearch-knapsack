003    
004    public class Util {
005        public static final int STREAM_HEADER_SIZE = 12;
006        public static final long BACKWARD_SIZE_MAX = 1L << 34;
007        public static final int BLOCK_HEADER_SIZE_MAX = 1024;
008        public static final long VLI_MAX = Long.MAX_VALUE;
009        public static final int VLI_SIZE_MAX = 9;
010    
011        public static int getVLISize(long num) {
012            int size = 0;
013            do {
014                ++size;
015                num >>= 7;
016            } while (num != 0);
017    
018            return size;
019        }
