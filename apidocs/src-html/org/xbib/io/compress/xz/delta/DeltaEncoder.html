001package org.xbib.io.compress.xz.delta;
002
003public class DeltaEncoder extends DeltaCoder {
004    public DeltaEncoder(int distance) {
005        super(distance);
006    }
007
008    public void encode(byte[] in, int in_off, int len, byte[] out) {
009        for (int i = 0; i < len; ++i) {
010            byte tmp = history[(distance + pos) & DISTANCE_MASK];
011            history[pos-- & DISTANCE_MASK] = in[in_off + i];
012            out[i] = (byte) (in[in_off + i] - tmp);
013        }
014    }
015}
