002package org.xbib.io.compress.xz.lz;
003
004import org.xbib.io.compress.xz.CorruptedInputException;
005
006import java.io.DataInputStream;
007import java.io.IOException;
008
009public final class LZDecoder {
010    private final byte[] buf;
011    private int start = 0;
012    private int pos = 0;
013    private int full = 0;
014    private int limit = 0;
015    private int pendingLen = 0;
016    private int pendingDist = 0;
017
018    public LZDecoder(int dictSize, byte[] presetDict) {
019        buf = new byte[dictSize];
020
021        if (presetDict != null) {
022            pos = Math.min(presetDict.length, dictSize);
023            full = pos;
024            start = pos;
025            System.arraycopy(presetDict, presetDict.length - pos, buf, 0, pos);
026        }
027    }
028
029    public void reset() {
030        start = 0;
031        pos = 0;
032        full = 0;
033        limit = 0;
034        buf[buf.length - 1] = 0x00;
035    }
036
037    public void setLimit(int outMax) {
038        if (buf.length - pos <= outMax) {
039            limit = buf.length;
040        } else {
041            limit = pos + outMax;
042        }
043    }
044
045    public boolean hasSpace() {
046        return pos < limit;
047    }
048
049    public boolean hasPending() {
050        return pendingLen > 0;
051    }
052
053    public int getPos() {
054        return pos;
055    }
056
057    public int getByte(int dist) {
058        int offset = pos - dist - 1;
059        if (dist >= pos) {
060            offset += buf.length;
061        }
062
063        return buf[offset] & 0xFF;
064    }
065
066    public void putByte(byte b) {
067        buf[pos++] = b;
068
069        if (full < pos) {
070            full = pos;
071        }
072    }
073
074    public void repeat(int dist, int len) throws IOException {
075        if (dist < 0 || dist >= full) {
076            throw new CorruptedInputException();
077        }
078
079        int left = Math.min(limit - pos, len);
080        pendingLen = len - left;
081        pendingDist = dist;
082
083        int back = pos - dist - 1;
084        if (dist >= pos) {
085            back += buf.length;
086        }
087
088        do {
089            buf[pos++] = buf[back++];
090            if (back == buf.length) {
091                back = 0;
092            }
093        } while (--left > 0);
094
095        if (full < pos) {
096            full = pos;
097        }
098    }
099
100    public void repeatPending() throws IOException {
101        if (pendingLen > 0) {
102            repeat(pendingDist, pendingLen);
103        }
104    }
105
106    public void copyUncompressed(DataInputStream inData, int len)
107            throws IOException {
108        int copySize = Math.min(buf.length - pos, len);
109        inData.readFully(buf, pos, copySize);
110        pos += copySize;
111
112        if (full < pos) {
113            full = pos;
114        }
115    }
116
117    public int flush(byte[] out, int outOff) {
118        int copySize = pos - start;
119        if (pos == buf.length) {
120            pos = 0;
121        }
122
123        System.arraycopy(buf, start, out, outOff, copySize);
124        start = pos;
125
126        return copySize;
127    }
128}
