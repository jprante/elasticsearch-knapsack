002    package org.xbib.io.compress.xz.index;
003    
004    import org.xbib.io.compress.xz.CorruptedInputException;
005    import org.xbib.io.compress.xz.MemoryLimitException;
006    import org.xbib.io.compress.xz.SeekableInputStream;
007    import org.xbib.io.compress.xz.UnsupportedOptionsException;
008    import org.xbib.io.compress.xz.common.DecoderUtil;
009    import org.xbib.io.compress.xz.common.StreamFlags;
010    
011    import java.io.EOFException;
012    import java.io.IOException;
013    import java.util.zip.CheckedInputStream;
014    
015    public class IndexDecoder extends IndexBase {
016        private final BlockInfo info = new BlockInfo();
017        private final long streamPadding;
018        private final int memoryUsage;
019        private final long[] unpadded;
020        private final long[] uncompressed;
021        private long largestBlockSize = 0;
022    
023        /**
024         * Current position in the arrays. This is initialized to <code>-1</code>
025         * because then it is possible to use <code>hasNext()</code> and
026         * <code>getNext()</code> to get BlockInfo of the first Block.
027         */
028        private int pos = -1;
029    
030        public IndexDecoder(SeekableInputStream in, StreamFlags streamFooterFlags,
031                            long streamPadding, int memoryLimit)
032                throws IOException {
033            super(new CorruptedInputException("XZ Index is corrupt"));
034            info.streamFlags = streamFooterFlags;
035            this.streamPadding = streamPadding;
036    
037            // If endPos is exceeded before the CRC32 field has been decoded,
038            // the Index is corrupt.
039            long endPos = in.position() + streamFooterFlags.backwardSize - 4;
040    
041            java.util.zip.CRC32 crc32 = new java.util.zip.CRC32();
042            CheckedInputStream inChecked = new CheckedInputStream(in, crc32);
043    
044            // Index Indicator
045            if (inChecked.read() != 0x00) {
046                throw new CorruptedInputException("XZ Index is corrupt");
047            }
048    
049            try {
050                // Number of Records
051                long count = DecoderUtil.decodeVLI(inChecked);
052    
053                // Catch Record counts that obviously too high to be valid.
054                // This test isn't exact because it ignores Index Indicator,
055                // Number of Records, and CRC32 fields, but this is good enough
056                // to catch the most obvious problems.
057                if (count >= streamFooterFlags.backwardSize / 2) {
058                    throw new CorruptedInputException("XZ Index is corrupt");
059                }
060    
061                // If the Record count doesn't fit into an int, we cannot
062                // allocate the arrays to hold the Records.
063                if (count > Integer.MAX_VALUE) {
064                    throw new UnsupportedOptionsException("XZ Index has over "
065                            + Integer.MAX_VALUE + " Records");
066                }
067    
068                // Calculate approximate memory requirements and check the
069                // memory usage limit.
070                memoryUsage = 1 + (int) ((16L * count + 1023) / 1024);
071                if (memoryLimit >= 0 && memoryUsage > memoryLimit) {
072                    throw new MemoryLimitException(memoryUsage, memoryLimit);
073                }
074    
075                // Allocate the arrays for the Records.
076                unpadded = new long[(int) count];
077                uncompressed = new long[(int) count];
078                int record = 0;
079    
080                // Decode the Records.
081                for (int i = (int) count; i > 0; --i) {
082                    // Get the next Record.
083                    long unpaddedSize = DecoderUtil.decodeVLI(inChecked);
084                    long uncompressedSize = DecoderUtil.decodeVLI(inChecked);
085    
086                    // Check that the input position stays sane. Since this is
087                    // checked only once per loop iteration instead of for
088                    // every input byte read, it's still possible that
089                    // EOFException gets thrown with corrupt input.
090                    if (in.position() > endPos) {
091                        throw new CorruptedInputException("XZ Index is corrupt");
092                    }
093    
094                    // Add the new Record.
095                    unpadded[record] = blocksSum + unpaddedSize;
096                    uncompressed[record] = uncompressedSum + uncompressedSize;
097                    ++record;
098                    super.add(unpaddedSize, uncompressedSize);
099                    assert record == recordCount;
100    
101                    // Remember the uncompressed size of the largest Block.
102                    if (largestBlockSize < uncompressedSize) {
103                        largestBlockSize = uncompressedSize;
104                    }
105                }
106            } catch (EOFException e) {
107                // EOFException is caught just in case a corrupt input causes
108                // DecoderUtil.decodeVLI to read too much at once.
109                throw new CorruptedInputException("XZ Index is corrupt");
110            }
111    
112            // Validate that the size of the Index field matches
113            // Backward Size.
114            int indexPaddingSize = getIndexPaddingSize();
115            if (in.position() + indexPaddingSize != endPos) {
116                throw new CorruptedInputException("XZ Index is corrupt");
117            }
118    
119            // Index Padding
120            while (indexPaddingSize-- > 0) {
121                if (inChecked.read() != 0x00) {
122                    throw new CorruptedInputException("XZ Index is corrupt");
123                }
124            }
125    
126            // CRC32
127            long value = crc32.getValue();
128            for (int i = 0; i < 4; ++i) {
129                if (((value >>> (i * 8)) & 0xFF) != in.read()) {
130                    throw new CorruptedInputException("XZ Index is corrupt");
131                }
132            }
133        }
134    
135        public BlockInfo locate(long target) {
136            assert target < uncompressedSum;
137    
138            int left = 0;
139            int right = unpadded.length - 1;
140    
141            while (left < right) {
142                int i = left + (right - left) / 2;
143    
144                if (uncompressed[i] <= target) {
145                    left = i + 1;
146                } else {
147                    right = i;
148                }
149            }
150    
151            pos = left;
152            return getInfo();
153        }
154    
155        public int getMemoryUsage() {
156            return memoryUsage;
157        }
158    
159        public long getStreamAndPaddingSize() {
160            return getStreamSize() + streamPadding;
161        }
162    
163        public long getUncompressedSize() {
164            return uncompressedSum;
165        }
166    
167        public long getLargestBlockSize() {
168            return largestBlockSize;
169        }
170    
171        public boolean hasNext() {
172            return pos + 1 < recordCount;
173        }
174    
175        public BlockInfo getNext() {
176            ++pos;
177            return getInfo();
178        }
179    
180        private BlockInfo getInfo() {
181            if (pos == 0) {
182                info.compressedOffset = 0;
183                info.uncompressedOffset = 0;
184            } else {
185                info.compressedOffset = (unpadded[pos - 1] + 3) & ~3;
186                info.uncompressedOffset = uncompressed[pos - 1];
187            }
188    
189            info.unpaddedSize = unpadded[pos] - info.compressedOffset;
190            info.uncompressedSize = uncompressed[pos] - info.uncompressedOffset;
191    
192            info.compressedOffset += DecoderUtil.STREAM_HEADER_SIZE;
193            return info;
194        }
195    }
