002package org.xbib.io.compress.xz.index;
003
004import org.xbib.io.compress.xz.XZIOException;
005import org.xbib.io.compress.xz.common.EncoderUtil;
006
007import java.io.IOException;
008import java.io.OutputStream;
009import java.util.ArrayList;
010import java.util.Iterator;
011import java.util.zip.CheckedOutputStream;
012
013public class IndexEncoder extends IndexBase {
014
015    private final ArrayList<IndexRecord> records = new ArrayList<IndexRecord>();
016
017    public IndexEncoder() {
018        super(new XZIOException("XZ Stream or its Index has grown too big"));
019    }
020
021    public void add(long unpaddedSize, long uncompressedSize)
022            throws XZIOException {
023        super.add(unpaddedSize, uncompressedSize);
024        records.add(new IndexRecord(unpaddedSize, uncompressedSize));
025    }
026
027    public void encode(OutputStream out) throws IOException {
028        java.util.zip.CRC32 crc32 = new java.util.zip.CRC32();
029        CheckedOutputStream outChecked = new CheckedOutputStream(out, crc32);
030
031        // Index Indicator
032        outChecked.write(0x00);
033
034        // Number of Records
035        EncoderUtil.encodeVLI(outChecked, recordCount);
036
037        // List of Records
038        for (Iterator i = records.iterator(); i.hasNext(); ) {
039            IndexRecord record = (IndexRecord) i.next();
040            EncoderUtil.encodeVLI(outChecked, record.unpadded);
041            EncoderUtil.encodeVLI(outChecked, record.uncompressed);
042        }
043
044        // Index Padding
045        for (int i = getIndexPaddingSize(); i > 0; --i) {
046            outChecked.write(0x00);
047        }
048
049        // CRC32
050        long value = crc32.getValue();
051        for (int i = 0; i < 4; ++i) {
052            out.write((byte) (value >>> (i * 8)));
053        }
054    }
055}
