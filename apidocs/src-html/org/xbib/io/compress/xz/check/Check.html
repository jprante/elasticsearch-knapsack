002    package org.xbib.io.compress.xz.check;
003    
004    import org.xbib.io.compress.xz.UnsupportedOptionsException;
005    import org.xbib.io.compress.xz.XZ;
006    
007    public abstract class Check {
008        int size;
009        String name;
010    
011        public abstract void update(byte[] buf, int off, int len);
012    
013        public abstract byte[] finish();
014    
015        public void update(byte[] buf) {
016            update(buf, 0, buf.length);
017        }
018    
019        public int getSize() {
020            return size;
021        }
022    
023        public String getName() {
024            return name;
025        }
026    
027        public static Check getInstance(int checkType)
028                throws UnsupportedOptionsException {
029            switch (checkType) {
030                case XZ.CHECK_NONE:
031                    return new None();
032    
033                case XZ.CHECK_CRC32:
034                    return new CRC32();
035    
036                case XZ.CHECK_CRC64:
037                    return new CRC64();
038    
039                case XZ.CHECK_SHA256:
040                    try {
041                        return new SHA256();
042                    } catch (java.security.NoSuchAlgorithmException e) {
043                    }
044    
045                    break;
046            }
047    
048            throw new UnsupportedOptionsException(
049                    "Unsupported Check ID " + checkType);
050        }
051    }
