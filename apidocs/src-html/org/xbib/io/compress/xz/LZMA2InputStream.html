001package org.xbib.io.compress.xz;
002
003import org.xbib.io.compress.xz.lz.LZDecoder;
004import org.xbib.io.compress.xz.lzma.LZMADecoder;
005import org.xbib.io.compress.xz.rangecoder.RangeDecoder;
006
007import java.io.DataInputStream;
008import java.io.IOException;
009import java.io.InputStream;
010
011/**
012 * Decompresses a raw LZMA2 stream (no XZ headers).
013 */
014public class LZMA2InputStream extends InputStream {
015    /**
016     * Smallest valid LZMA2 dictionary size.
017     * <p>
018     * Very tiny dictionaries would be a performance problem, so
019     * the minimum is 4 KiB.
020     */
021    public static final int DICT_SIZE_MIN = 4096;
022
023    /**
024     * Largest dictionary size supported by this implementation.
025     * <p>
026     * The LZMA2 algorithm allows dictionaries up to one byte less than 4 GiB.
027     * This implementation supports only 16 bytes less than 2 GiB for raw
028     * LZMA2 streams, and for .xz files the maximum is 1.5 GiB. This
029     * limitation is due to Java using signed 32-bit integers for array
030     * indexing. The limitation shouldn't matter much in practice since so
031     * huge dictionaries are not normally used.
032     */
033    public static final int DICT_SIZE_MAX = Integer.MAX_VALUE & ~15;
034
035    private static final int COMPRESSED_SIZE_MAX = 1 << 16;
036
037    private DataInputStream in;
038
039    private final LZDecoder lz;
040    private final RangeDecoder rc = new RangeDecoder(COMPRESSED_SIZE_MAX);
041    private LZMADecoder lzma;
042
043    private int uncompressedSize = 0;
044    private boolean isLZMAChunk;
045
046    private boolean needDictReset = true;
047    private boolean needProps = true;
048    private boolean endReached = false;
049
050    private IOException exception = null;
051
052    /**
053     * Gets approximate decompressor memory requirements as kibibytes for
054     * the given dictionary size.
055     *
056     * @param dictSize LZMA2 dictionary size as bytes, must be
057     *                 in the range [<code>DICT_SIZE_MIN</code>,
058     *                 <code>DICT_SIZE_MAX</code>]
059     * @return approximate memory requirements as kibibytes (KiB)
060     */
061    public static int getMemoryUsage(int dictSize) {
062        // The base state is aroudn 30-40 KiB (probabilities etc.),
063        // range decoder needs COMPRESSED_SIZE_MAX bytes for buffering,
064        // and LZ decoder needs a dictionary buffer.
065        return 40 + COMPRESSED_SIZE_MAX / 1024 + getDictSize(dictSize) / 1024;
066    }
067
068    private static int getDictSize(int dictSize) {
069        if (dictSize < DICT_SIZE_MIN || dictSize > DICT_SIZE_MAX) {
070            throw new IllegalArgumentException(
071                    "Unsupported dictionary size " + dictSize);
072        }
073
074        // Round dictionary size upward to a multiple of 16. This way LZMA
075        // can use LZDecoder.getPos() for calculating LZMA's posMask.
076        // Note that this check is needed only for raw LZMA2 streams; it is
077        // redundant with .xz.
078        return (dictSize + 15) & ~15;
079    }
080
081    /**
082     * Creates a new input stream that decompresses raw LZMA2 data
083     * from <code>in</code>.
084     * <p>
085     * The caller needs to know the dictionary size used when compressing;
086     * the dictionary size isn't stored as part of a raw LZMA2 stream.
087     * <p>
088     * Specifying a too small dictionary size will prevent decompressing
089     * the stream. Specifying a too big dictionary is waste of memory but
090     * decompression will work.
091     * <p>
092     * There is no need to specify a dictionary bigger than
093     * the uncompressed size of the data even if a bigger dictionary
094     * was used when compressing. If you know the uncompressed size
095     * of the data, this might allow saving some memory.
096     *
097     * @param in       input stream from which LZMA2-compressed
098     *                 data is read
099     * @param dictSize LZMA2 dictionary size as bytes, must be
100     *                 in the range [<code>DICT_SIZE_MIN</code>,
101     *                 <code>DICT_SIZE_MAX</code>]
102     */
103    public LZMA2InputStream(InputStream in, int dictSize) {
104        this(in, dictSize, null);
105    }
106
107    /**
108     * Creates a new LZMA2 decompressor using a preset dictionary.
109     * <p>
110     * This is like <code>LZMAInputStream(InputStream, int)</code> except
111     * that the dictionary may be initialized using a preset dictionary.
112     * If a preset dictionary was used when compressing the data, the
113     * same preset dictionary must be provided when decompressing.
114     *
115     * @param in         input stream from which LZMA2-compressed
116     *                   data is read
117     * @param dictSize   LZMA2 dictionary size as bytes, must be
118     *                   in the range [<code>DICT_SIZE_MIN</code>,
119     *                   <code>DICT_SIZE_MAX</code>]
120     * @param presetDict preset dictionary or <code>null</code>
121     *                   to use no preset dictionary
122     */
123    public LZMA2InputStream(InputStream in, int dictSize, byte[] presetDict) {
124        // Check for null because otherwise null isn't detect
125        // in this constructor.
126        if (in == null) {
127            throw new NullPointerException();
128        }
129
130        this.in = new DataInputStream(in);
131        this.lz = new LZDecoder(getDictSize(dictSize), presetDict);
132
133        if (presetDict != null && presetDict.length > 0) {
134            needDictReset = false;
135        }
136    }
137
138    /**
139     * Decompresses the next byte from this input stream.
140     * <p>
141     * Reading lots of data with <code>read()</code> from this input stream
142     * may be inefficient. Wrap it in <code>java.io.BufferedInputStream</code>
143     * if you need to read lots of data one byte at a time.
144     *
145     * @return the next decompressed byte, or <code>-1</code>
146     * to indicate the end of the compressed stream
147     * @throws CorruptedInputException
148     * @throws XZIOException           if the stream has been closed
149     * @throws java.io.IOException     may be thrown by <code>in</code>
150     */
151    public int read() throws IOException {
152        byte[] buf = new byte[1];
153        return read(buf, 0, 1) == -1 ? -1 : (buf[0] & 0xFF);
154    }
155
156    /**
157     * Decompresses into an array of bytes.
158     * <p>
159     * If <code>len</code> is zero, no bytes are read and <code>0</code>
160     * is returned. Otherwise this will block until <code>len</code>
161     * bytes have been decompressed, the end of LZMA2 stream is reached,
162     * or an exception is thrown.
163     *
164     * @param buf target buffer for uncompressed data
165     * @param off start offset in <code>buf</code>
166     * @param len maximum number of uncompressed bytes to read
167     * @return number of bytes read, or <code>-1</code> to indicate
168     * the end of the compressed stream
169     * @throws CorruptedInputException
170     * @throws XZIOException           if the stream has been closed
171     * @throws java.io.IOException     may be thrown by <code>in</code>
172     */
173    public int read(byte[] buf, int off, int len) throws IOException {
174        if (off < 0 || len < 0 || off + len < 0 || off + len > buf.length) {
175            throw new IndexOutOfBoundsException();
176        }
177
178        if (len == 0) {
179            return 0;
180        }
181
182        if (in == null) {
183            throw new XZIOException("Stream closed");
184        }
185
186        if (exception != null) {
187            throw exception;
188        }
189
190        if (endReached) {
191            return -1;
192        }
193
194        try {
195            int size = 0;
196
197            while (len > 0) {
198                if (uncompressedSize == 0) {
199                    decodeChunkHeader();
200                    if (endReached) {
201                        return size == 0 ? -1 : size;
202                    }
203                }
204
205                int copySizeMax = Math.min(uncompressedSize, len);
206
207                if (!isLZMAChunk) {
208                    lz.copyUncompressed(in, copySizeMax);
209                } else {
210                    lz.setLimit(copySizeMax);
211                    lzma.decode();
212                }
213
214                int copiedSize = lz.flush(buf, off);
215                off += copiedSize;
216                len -= copiedSize;
217                size += copiedSize;
218                uncompressedSize -= copiedSize;
219
220                if (uncompressedSize == 0) {
221                    if (!rc.isFinished() || lz.hasPending()) {
222                        throw new CorruptedInputException();
223                    }
224                }
225            }
226
227            return size;
228
229        } catch (IOException e) {
230            exception = e;
231            throw e;
232        }
233    }
234
235    private void decodeChunkHeader() throws IOException {
236        int control = in.readUnsignedByte();
237
238        if (control == 0x00) {
239            endReached = true;
240            return;
241        }
242
243        if (control >= 0xE0 || control == 0x01) {
244            needProps = true;
245            needDictReset = false;
246            lz.reset();
247        } else if (needDictReset) {
248            throw new CorruptedInputException();
249        }
250
251        if (control >= 0x80) {
252            isLZMAChunk = true;
253
254            uncompressedSize = (control & 0x1F) << 16;
255            uncompressedSize += in.readUnsignedShort() + 1;
256
257            int compressedSize = in.readUnsignedShort() + 1;
258
259            if (control >= 0xC0) {
260                needProps = false;
261                decodeProps();
262
263            } else if (needProps) {
264                throw new CorruptedInputException();
265
266            } else if (control >= 0xA0) {
267                lzma.reset();
268            }
269
270            rc.prepareInputBuffer(in, compressedSize);
271
272        } else if (control > 0x02) {
273            throw new CorruptedInputException();
274
275        } else {
276            isLZMAChunk = false;
277            uncompressedSize = in.readUnsignedShort() + 1;
278        }
279    }
280
281    private void decodeProps() throws IOException {
282        int props = in.readUnsignedByte();
283
284        if (props > (4 * 5 + 4) * 9 + 8) {
285            throw new CorruptedInputException();
286        }
287
288        int pb = props / (9 * 5);
289        props -= pb * 9 * 5;
290        int lp = props / 9;
291        int lc = props - lp * 9;
292
293        if (lc + lp > 4) {
294            throw new CorruptedInputException();
295        }
296
297        lzma = new LZMADecoder(lz, rc, lc, lp, pb);
298    }
299
300    /**
301     * Returns the number of uncompressed bytes that can be read
302     * without blocking. The value is returned with an assumption
303     * that the compressed input data will be valid. If the compressed
304     * data is corrupt, <code>CorruptedInputException</code> may get
305     * thrown before the number of bytes claimed to be available have
306     * been read from this input stream.
307     * <p>
308     * In LZMAInputStream, the return value will be non-zero when the
309     * decompressor is in the middle of an LZMA2 chunk. The return value
310     * will then be the number of uncompressed bytes remaining from that
311     * chunk.
312     *
313     * @return the number of uncompressed bytes that can be read
314     * without blocking
315     */
316    public int available() throws IOException {
317        if (in == null) {
318            throw new XZIOException("Stream closed");
319        }
320
321        if (exception != null) {
322            throw exception;
323        }
324
325        return uncompressedSize;
326    }
327
328    /**
329     * Closes the stream and calls <code>in.close()</code>.
330     * If the stream was already closed, this does nothing.
331     *
332     * @throws java.io.IOException if thrown by <code>in.close()</code>
333     */
334    public void close() throws IOException {
335        if (in != null) {
336            try {
337                in.close();
338            } finally {
339                in = null;
340            }
341        }
342    }
