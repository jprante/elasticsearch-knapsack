011package org.xbib.io.compress.xz;
012
013import java.io.InputStream;
014import java.io.IOException;
015
016/**
017 * Base class for filter-specific options classes.
018 */
019public abstract class FilterOptions implements Cloneable {
020    /**
021     * Gets how much memory the encoder will need with
022     * the given filter chain. This function simply calls
023     * <code>getEncoderMemoryUsage()</code> for every filter
024     * in the array and returns the sum of the returned values.
025     */
026    public static int getEncoderMemoryUsage(FilterOptions[] options) {
027        int m = 0;
028
029        for (int i = 0; i < options.length; ++i)
030            m += options[i].getEncoderMemoryUsage();
031
032        return m;
033    }
034
035    /**
036     * Gets how much memory the decoder will need with
037     * the given filter chain. This function simply calls
038     * <code>getDecoderMemoryUsage()</code> for every filter
039     * in the array and returns the sum of the returned values.
040     */
041    public static int getDecoderMemoryUsage(FilterOptions[] options) {
042        int m = 0;
043
044        for (int i = 0; i < options.length; ++i)
045            m += options[i].getDecoderMemoryUsage();
046
047        return m;
048    }
049
050    /**
051     * Gets how much memory the encoder will need with these options.
052     */
053    public abstract int getEncoderMemoryUsage();
054
055    /**
056     * Gets a raw (no XZ headers) encoder output stream using these options.
057     * Raw streams are an advanced feature. In most cases you want to store
058     * the compressed data in the .xz container format instead of using
059     * a raw stream. To use this filter in a .xz file, pass this object
060     * to XZOutputStream.
061     */
062    public abstract FinishableOutputStream getOutputStream(
063            FinishableOutputStream out);
064
065    /**
066     * Gets how much memory the decoder will need to decompress the data
067     * that was encoded with these options.
068     */
069    public abstract int getDecoderMemoryUsage();
070
071    /**
072     * Gets a raw (no XZ headers) decoder input stream using these options.
073     */
074    public abstract InputStream getInputStream(InputStream in)
075            throws IOException;
076
077    abstract FilterEncoder getFilterEncoder();
078
079    FilterOptions() {}
080}
