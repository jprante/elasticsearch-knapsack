001package org.xbib.io.compress.xz;
002
003import java.io.IOException;
004import java.io.InputStream;
005
006/**
007 * Base class for filter-specific options classes.
008 */
009public abstract class FilterOptions implements Cloneable {
010    /**
011     * Gets how much memory the encoder will need with
012     * the given filter chain. This function simply calls
013     * <code>getEncoderMemoryUsage()</code> for every filter
014     * in the array and returns the sum of the returned values.
015     */
016    public static int getEncoderMemoryUsage(FilterOptions[] options) {
017        int m = 0;
018
019        for (int i = 0; i < options.length; ++i) {
020            m += options[i].getEncoderMemoryUsage();
021        }
022
023        return m;
024    }
025
026    /**
027     * Gets how much memory the decoder will need with
028     * the given filter chain. This function simply calls
029     * <code>getDecoderMemoryUsage()</code> for every filter
030     * in the array and returns the sum of the returned values.
031     */
032    public static int getDecoderMemoryUsage(FilterOptions[] options) {
033        int m = 0;
034
035        for (int i = 0; i < options.length; ++i) {
036            m += options[i].getDecoderMemoryUsage();
037        }
038
039        return m;
040    }
041
042    /**
043     * Gets how much memory the encoder will need with these options.
044     */
045    public abstract int getEncoderMemoryUsage();
046
047    /**
048     * Gets a raw (no XZ headers) encoder output stream using these options.
049     * Raw streams are an advanced feature. In most cases you want to store
050     * the compressed data in the .xz container format instead of using
051     * a raw stream. To use this filter in a .xz file, pass this object
052     * to XZOutputStream.
053     */
054    public abstract FinishableOutputStream getOutputStream(
055            FinishableOutputStream out);
056
057    /**
058     * Gets how much memory the decoder will need to decompress the data
059     * that was encoded with these options.
060     */
061    public abstract int getDecoderMemoryUsage();
062
063    /**
064     * Gets a raw (no XZ headers) decoder input stream using these options.
065     */
066    public abstract InputStream getInputStream(InputStream in)
067            throws IOException;
068
069    abstract FilterEncoder getFilterEncoder();
070
071    FilterOptions() {
072    }
073}
