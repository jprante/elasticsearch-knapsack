001package org.xbib.io.compress.gzip;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.util.zip.CRC32;
006import java.util.zip.DataFormatException;
007import java.util.zip.Deflater;
008import java.util.zip.Inflater;
009import java.util.zip.ZipException;
010import org.xbib.io.compress.BufferRecycler;
011
012/**
013 * Optimized variant of {@link java.util.zip.GZIPInputStream} that reuses
014 * underlying {@link java.util.zip.Deflater} instance}.
015 */
016public class OptimizedGZIPInputStream extends InputStream {
017
018    /**
019     * What kinds of chunks do we feed underlying {@link Inflater}?
020     */
021    private final static int INPUT_BUFFER_SIZE = 4096;
022
023    /**
024     * Enumeration used for keeping track of decoding state within stream
025     */
026    enum State {
027
028        GZIP_HEADER, GZIP_CONTENT, GZIP_TRAILER, GZIP_COMPLETE;
029    };
030
031    /*
032     * Helper objects
033     */
034    protected Inflater _inflater;
035    protected final CRC32 _crc;
036    /**
037     * Object that handles details of buffer recycling
038     */
039    protected final BufferRecycler _bufferRecycler;
040    protected final GZIPRecycler _gzipRecycler;
041
042    /*
043     ///////////////////////////////////////////////////////////////////////
044     // State
045     ///////////////////////////////////////////////////////////////////////
046     */
047    protected byte[] _buffer;
048    protected int _bufferPtr;
049    protected int _bufferEnd;
050    /**
051     * Temporary buffer used for single-byte reads, skipping.
052     */
053    protected byte[] _tmpBuffer;
054    /**
055     * Underlying input stream from which compressed data is to be read from.
056     */
057    protected InputStream _rawInput;
058    /**
059     * Flag set to true during handling of header processing
060     */
061    protected OptimizedGZIPInputStream.State _state;
062
063    /*
064     * Construction
065     */
066    public OptimizedGZIPInputStream(InputStream in) throws IOException {
067        super();
068        _bufferRecycler = BufferRecycler.instance();
069        _gzipRecycler = GZIPRecycler.instance();
070        _rawInput = in;
071        _buffer = _bufferRecycler.allocInputBuffer(INPUT_BUFFER_SIZE);
072
073        _bufferPtr = _bufferEnd = 0;
074        _inflater = _gzipRecycler.allocInflater();
075        _crc = new CRC32();
076
077        // And then need to process header...
078        _readHeader();
079        _state = State.GZIP_CONTENT;
080        _crc.reset();
081        // and if all is good, kick start inflater etc
082        if (_bufferPtr >= _bufferEnd) { // need more data
083            _loadMore();
084        }
085        _inflater.setInput(_buffer, _bufferPtr, _bufferEnd - _bufferPtr);
086    }
087
088    /*
089     ///////////////////////////////////////////////////////////////////////
090     // InputStream implementation
091     ///////////////////////////////////////////////////////////////////////
092     */
093    @Override
094    public int available() {
095        if (_state == State.GZIP_COMPLETE) {
096            return 0;
097        }
098        // not sure if this makes sense but...
099        return _inflater.finished() ? 0 : 1;
100    }
101
102    @Override
103    public void close() throws IOException {
104        _state = State.GZIP_COMPLETE;
105        if (_rawInput != null) {
106            _rawInput.close();
107            _rawInput = null;
108        }
109        byte[] b = _buffer;
110        if (b != null) {
111            _buffer = null;
112            _bufferRecycler.releaseInputBuffer(b);
113        }
114        b = _tmpBuffer;
115        if (b != null) {
116            _tmpBuffer = null;
117            _bufferRecycler.releaseDecodeBuffer(b);
118        }
119        Inflater i = _inflater;
120        if (i != null) {
121            _inflater = null;
122            _gzipRecycler.releaseInflater(i);
123        }
124    }
125
126    @Override
127    public void mark(int limit) {
128        // not supported... but not lethal to call
129    }
130
131    @Override
132    public boolean markSupported() {
133        return false;
134    }
135
136    @Override
137    public final int read() throws IOException {
138        byte[] tmp = _getTmpBuffer();
139        int count = read(tmp, 0, 1);
140        if (count < 0) {
141            return -1;
142        }
143        return tmp[0] & 0xFF;
144    }
145
146    @Override
147    public final int read(byte[] buf) throws IOException {
148        return read(buf, 0, buf.length);
149    }
150
151    @Override
152    public final int read(byte[] buf, int offset, int len) throws IOException {
153        if (buf == null) {
154            throw new NullPointerException();
155        }
156        if (offset < 0 || len < 0 || len > buf.length - offset) {
157            throw new IndexOutOfBoundsException();
158        }
159        if (_state == State.GZIP_COMPLETE) { // closed or EOF
160            return -1;
161        }
162        if (len == 0) {
163            return 0;
164        }
165        try {
166            int count;
167            while ((count = _inflater.inflate(buf, offset, len)) == 0) {
168                if (_inflater.finished() || _inflater.needsDictionary()) {
169                    _readTrailer();
170                    _state = State.GZIP_COMPLETE;
171                    return -1;
172                }
173                if (_inflater.needsInput()) {
174                    _loadMore();
175                    _inflater.setInput(_buffer, _bufferPtr, _bufferEnd - _bufferPtr);
176                    _bufferPtr = _bufferEnd;
177                }
178            }
179            _crc.update(buf, offset, count);
180            return count;
181        } catch (DataFormatException e) {
182            String s = e.getMessage();
183            throw new ZipException(s != null ? s : "Invalid ZLIB data format");
184        }
185    }
186
187    @Override
188    public void reset() throws IOException {
189        throw new IOException("mark/reset not supported");
190    }
191
192    @Override
193    public long skip(long n) throws IOException {
194        if (n < 0) {
195            throw new IllegalArgumentException("negative skip length");
196        }
197        byte[] tmp = _getTmpBuffer();
198        long total = 0;
199
200        while (true) {
201            int max = (int) (n - total);
202            if (max == 0) {
203                break;
204            }
205            int count = read(tmp, 0, Math.min(max, tmp.length));
206            total += count;
207        }
208        return total;
209    }
210
211    /*
212     ///////////////////////////////////////////////////////////////////////
213     // Internal methods
214     ///////////////////////////////////////////////////////////////////////
215     */
216    protected byte[] _getTmpBuffer() {
217        if (_tmpBuffer == null) {
218            _tmpBuffer = _bufferRecycler.allocDecodeBuffer(INPUT_BUFFER_SIZE);
219        }
220        return _tmpBuffer;
221    }
222
223    protected final void _readHeader() throws IOException {
224        _state = State.GZIP_HEADER;
225
226        // Check header magic
227        int sig = _readShort();
228        if (sig != GZIPUncompressor.GZIP_MAGIC) {
229            throw new ZipException("Not in GZIP format (got 0x" + Integer.toHexString(sig)
230                    + ", should be 0x" + Integer.toHexString(GZIPUncompressor.GZIP_MAGIC) + ")");
231        }
232        // Check compression method
233        if (_readByte() != Deflater.DEFLATED) {
234            throw new ZipException("Unsupported compression method (only support Deflate, " + Deflater.DEFLATED + ")");
235        }
236        // Read flags
237        int flg = _readByte();
238        // Skip MTIME, XFL, and OS fields
239        _skipBytes(6);
240        // Skip optional extra field
241        if ((flg & GZIPUncompressor.FEXTRA) != 0) {
242            _skipBytes(_readShort());
243        }
244        // Skip optional file name
245        if ((flg & GZIPUncompressor.FNAME) != 0) {
246            while (_readByte() != 0) ;
247        }
248        // Skip optional file comment
249        if ((flg & GZIPUncompressor.FCOMMENT) != 0) {
250            while (_readByte() != 0) ;
251        }
252        // Check optional header CRC
253        if ((flg & GZIPUncompressor.FHCRC) != 0) {
254            int act = (int) _crc.getValue() & 0xffff;
255            int exp = _readShort();
256            if (act != exp) {
257                throw new IOException("Corrupt GZIP header (header CRC 0x"
258                        + Integer.toHexString(act) + ", expected 0x "
259                        + Integer.toHexString(exp));
260            }
261        }
262    }
263
264    protected final void _readTrailer() throws IOException {
265        int actCrc = (int) _crc.getValue();
266        // does Inflater have leftovers?
267        int remains = _inflater.getRemaining();
268        if (remains > 0) {
269            // ok, let's update ptr to indicate where we are at...
270            _bufferPtr = _bufferEnd - remains;
271        } else { // if not, just load more
272            _loadMore(8);
273        }
274        int expCrc = _readInt();
275        int expCount = _readInt();
276        int actCount32 = (int) _inflater.getBytesWritten();
277
278        if (actCount32 != expCount) {
279            throw new ZipException("Corrupt trailer: expected byte count " + expCount + ", read " + actCount32);
280        }
281        if (expCrc != actCrc) {
282            throw new ZipException("Corrupt trailer: expected CRC " + Integer.toHexString(expCrc) + ", computed " + Integer.toHexString(actCrc));
283        }
284    }
285
286    private final void _skipBytes(int count) throws IOException {
287        while (--count >= 0) {
288            _readByte();
289        }
290    }
291
292    private final int _readByte() throws IOException {
293        if (_bufferPtr >= _bufferEnd) {
294            _loadMore();
295        }
296        byte b = _buffer[_bufferPtr++];
297        if (_state == State.GZIP_HEADER) {
298            _crc.update(b);
299        }
300        return b & 0xFF;
301    }
302
303    private final int _readShort() throws IOException {
304        // LSB... blech
305        return _readByte() | (_readByte() << 8);
306    }
307
308    private final int _readInt() throws IOException {
309        // LSB... yuck
310        return _readByte() | (_readByte() << 8)
311                | (_readByte() << 16) | (_readByte() << 24);
312    }
313
314    private final void _loadMore() throws IOException {
315        // let's read at most 8k; deflater has to buffer some of data
316        _loadMore(Math.min(_buffer.length, INPUT_BUFFER_SIZE));
317    }
318
319    private final void _loadMore(int max) throws IOException {
320        int count = _rawInput.read(_buffer, 0, max);
321        if (count < 1) {
322            String prob = (count < 0)
323                    ? "Unexpected end of input" : "Strange underlying stream (returned 0 bytes for read)";
324            throw new IOException(prob + " when reading " + _state);
325        }
326        _bufferPtr = 0;
327        _bufferEnd = count;
328    }
329}
