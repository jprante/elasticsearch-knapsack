016package org.xbib.io;
017
018import java.util.Map;
019import java.util.TreeMap;
020
021/**
022 * Utility class that tracks the number of bytes transferred from a source, and
023 * uses this information to calculate transfer rates and estimate end times. The
024 * watcher stores the number of bytes that will be transferred, the number of
025 * bytes that have been transferred in the current session and the time this has
026 * taken, and the number of bytes and time taken overal (eg for transfers that
027 * have been restarted).
028 */
029public class BytesProgressWatcher {
030
031    /**
032     * The number of seconds worth of historical byte transfer information that
033     * will be stored and used to calculate the recent transfer rate.
034     */
035    public static final int SECONDS_OF_HISTORY = 5;
036
037    private boolean isStarted = false;
038
039    private long bytesToTransfer = 0;
040
041    private long startTimeAllTransfersMS = -1;
042
043    private long totalBytesInAllTransfers = 0;
044
045    private long startTimeCurrentTransferMS = -1;
046
047    private long totalBytesInCurrentTransfer = 0;
048
049    private long endTimeCurrentTransferMS = -1;
050
051    private Map<Long, Long> historyOfBytesBySecond = new TreeMap<Long, Long>();
052
053    private long earliestHistorySecond = Long.MAX_VALUE;
054
055    /**
056     * Construct a watcher for a transfer that will involve a given number of
057     * bytes.
058     *
059     * @param bytesToTransfer the number of bytes that will be transferred, eg
060     *                        the size of a file being uploaded.
061     */
062    public BytesProgressWatcher(long bytesToTransfer) {
063        this.bytesToTransfer = bytesToTransfer;
064    }
065
066    /**
067     * @return the count of bytes that will be transferred by the object watched
068     * by this class.
069     */
070    public synchronized long getBytesToTransfer() {
071        return bytesToTransfer;
072    }
073
074    /**
075     * Resets the byte count and timer variables for a watcher. This method is
076     * called automatically when a transfer is started (ie the first bytes are
077     * registered in the method {@link #updateBytesTransferred(long)}), or when
078     * a transfer is restarted (eg due to transmission errors).
079     */
080    public synchronized void resetWatcher() {
081        startTimeCurrentTransferMS = System.currentTimeMillis();
082        if (startTimeAllTransfersMS == -1) {
083            startTimeAllTransfersMS = startTimeCurrentTransferMS;
084        }
085        endTimeCurrentTransferMS = -1;
086        totalBytesInCurrentTransfer = 0;
087        isStarted = true;
088    }
089
090    /**
091     * Notifies this watcher that bytes have been transferred.
092     *
093     * @param byteCount the number of bytes that have been transferred.
094     */
095    public synchronized void updateBytesTransferred(long byteCount) {
096        // Start the monitor when we are notified of the first bytes transferred.
097        if (!isStarted) {
098            resetWatcher();
099        }
100
101        // Store the total byte count for the current transfer, and for all transfers.
102        totalBytesInCurrentTransfer += byteCount;
103        totalBytesInAllTransfers += byteCount;
104
105        // Recognise when all the expected bytes have been transferred and mark the end time.
106        if (totalBytesInCurrentTransfer >= bytesToTransfer) {
107            endTimeCurrentTransferMS = System.currentTimeMillis();
108        }
109
110        // Keep historical records of the byte counts transferred in a given second.
111        Long currentSecond = System.currentTimeMillis() / 1000;
112        Long bytesInSecond = historyOfBytesBySecond.get(currentSecond);
113        if (bytesInSecond != null) {
114            historyOfBytesBySecond.put(currentSecond, byteCount + bytesInSecond);
115        } else {
116            historyOfBytesBySecond.put(currentSecond, byteCount);
117        }
118
119        // Remember the earliest second value for which we have historical info.
120        if (currentSecond < earliestHistorySecond) {
121            earliestHistorySecond = currentSecond;
122        }
123
124        // Remove any history records we are no longer interested in.
125        long removeHistoryBeforeSecond = currentSecond - SECONDS_OF_HISTORY;
126        for (long sec = earliestHistorySecond; sec < removeHistoryBeforeSecond; sec++) {
127            historyOfBytesBySecond.remove(sec);
128        }
129        earliestHistorySecond = removeHistoryBeforeSecond;
130    }
131
132    /**
133     * @return the number of bytes that have so far been transferred in the most
134     * recent transfer session.
135     */
136    public synchronized long getBytesTransferred() {
137        return totalBytesInCurrentTransfer;
138    }
139
140    /**
141     * @return the number of bytes that are remaining to be transferred.
142     */
143    public synchronized long getBytesRemaining() {
144        return bytesToTransfer - totalBytesInCurrentTransfer;
145    }
146
147    /**
148     * @return an estimate of the time (in seconds) it will take for the
149     * transfer to completed, based on the number of bytes remaining to transfer
150     * and the overall bytes/second rate.
151     */
152    public synchronized long getRemainingTime() {
153        BytesProgressWatcher[] progressWatchers = new BytesProgressWatcher[1];
154        progressWatchers[0] = this;
155
156        long bytesRemaining = bytesToTransfer - totalBytesInCurrentTransfer;
157        double remainingSecs =
158                (double) bytesRemaining / calculateOverallBytesPerSecond(progressWatchers);
159        return Math.round(remainingSecs);
160    }
161
162    /**
163     * @return the byte rate (per second) based on the historical information
164     * for the last
165     * {@link #SECONDS_OF_HISTORY} seconds before the current time.
166     */
167    public synchronized double getRecentByteRatePerSecond() {
168        if (!isStarted) {
169            return 0;
170        }
171
172        long currentSecond = System.currentTimeMillis() / 1000;
173        long startSecond = 1 + (currentSecond - SECONDS_OF_HISTORY);
174        long endSecond = (endTimeCurrentTransferMS != -1
175                ? endTimeCurrentTransferMS / 1000
176                : currentSecond);
177
178        if (currentSecond - SECONDS_OF_HISTORY > endSecond) {
179            // This item finished too long ago, ignore it now.
180            historyOfBytesBySecond.clear();
181            return 0;
182        }
183
184        // Count the number of bytes transferred from SECONDS_OF_HISTORY ago to the second before now.
185        long sumOfBytes = 0;
186        long numberOfSecondsInHistory = 0;
187        for (long sec = startSecond; sec <= endSecond; sec++) {
188            numberOfSecondsInHistory++;
189            Long bytesInSecond = historyOfBytesBySecond.get(sec);
190            if (bytesInSecond != null) {
191                sumOfBytes += bytesInSecond;
192            }
193        }
194        return (numberOfSecondsInHistory == 0 ? 0 : (double) sumOfBytes / numberOfSecondsInHistory);
195    }
196
197    /**
198     * @return the number of milliseconds time elapsed for a transfer. The value
199     * returned is the time elapsed so far if the transfer is ongoing, the total
200     * time taken for the transfer if it is complete, or 0 if the transfer has
201     * not yet started.
202     */
203    public synchronized long getElapsedTimeMS() {
204        if (!isStarted) {
205            return 0;
206        }
207        if (endTimeCurrentTransferMS != -1) {
208            // Transfer is complete, report the time it took.
209            return endTimeCurrentTransferMS - startTimeCurrentTransferMS;
210        } else {
211            return System.currentTimeMillis() - startTimeCurrentTransferMS;
212        }
213    }
214
215    /**
216     * @return the number of bytes that have been transferred over all sessions,
217     * including any sessions that have been restarted.
218     */
219    public synchronized long getTotalBytesInAllTransfers() {
220        return totalBytesInAllTransfers;
221    }
222
223    protected synchronized boolean isStarted() {
224        return isStarted;
225    }
226
227    /**
228     * @return the time (in milliseconds) when the first bytes were transferred,
229     * regardless of how many times the transfer was reset.
230     */
231    public synchronized long getHistoricStartTimeMS() {
232        return startTimeAllTransfersMS;
233    }
234
235    /**
236     * @param progressWatchers all the watchers involved in the same byte
237     *                         transfer operation.
238     * @return the total number of bytes to transfer.
239     */
240    public static long sumBytesToTransfer(BytesProgressWatcher[] progressWatchers) {
241        long sumOfBytes = 0;
242        for (BytesProgressWatcher progressWatcher : progressWatchers) {
243            sumOfBytes += progressWatcher.getBytesToTransfer();
244        }
245        return sumOfBytes;
246    }
247
248    /**
249     * @param progressWatchers all the watchers involved in the same byte
250     *                         transfer operation.
251     * @return the total number of bytes already transferred.
252     */
253    public static long sumBytesTransferred(BytesProgressWatcher[] progressWatchers) {
254        long sumOfBytes = 0;
255        for (BytesProgressWatcher progressWatcher : progressWatchers) {
256            sumOfBytes += progressWatcher.getBytesTransferred();
257        }
258        return sumOfBytes;
259    }
260
261    /**
262     * @param progressWatchers all the watchers involved in the same byte
263     *                         transfer operation.
264     * @return an estimate of the time (in seconds) it will take for the
265     * transfer to completed, based on the number of bytes remaining to transfer
266     * and the overall bytes/second rate.
267     */
268    public static long calculateRemainingTime(BytesProgressWatcher[] progressWatchers) {
269        long bytesRemaining = sumBytesToTransfer(progressWatchers)
270                - sumBytesTransferred(progressWatchers);
271        double bytesPerSecond = calculateOverallBytesPerSecond(progressWatchers);
272        if (Math.abs(bytesPerSecond) < 0.001d) {
273            // No transfer has occurred yet.
274            return 0;
275        }
276        double remainingSecs =
277                (double) bytesRemaining / bytesPerSecond;
278        return Math.round(remainingSecs);
279    }
280
281    /**
282     * @param progressWatchers all the watchers involved in the same byte
283     *                         transfer operation.
284     * @return the overall rate of bytes/second over all transfers for all
285     * watchers.
286     */
287    public static double calculateOverallBytesPerSecond(BytesProgressWatcher[] progressWatchers) {
288        long initialStartTime = Long.MAX_VALUE; // The oldest start time of any monitor.
289
290        long bytesTotal = 0;
291        for (BytesProgressWatcher progressWatcher : progressWatchers) {
292            // Ignore any watchers that have not yet started.
293            if (!progressWatcher.isStarted()) {
294                continue;
295            }
296
297            // Add up all the bytes transferred by all started watchers.
298            bytesTotal += progressWatcher.getTotalBytesInAllTransfers();
299
300            // Find the earliest starting time of any monitor.
301            if (progressWatcher.getHistoricStartTimeMS() < initialStartTime) {
302                initialStartTime = progressWatcher.getHistoricStartTimeMS();
303            }
304        }
305
306        // Determine how much time has elapsed since the earliest watcher start time.
307        long elapsedTimeSecs = (System.currentTimeMillis() - initialStartTime) / 1000;
308
309        // Calculate the overall rate of bytes/second over all transfers for all watchers.
310
311        return elapsedTimeSecs == 0 ? bytesTotal : (double) bytesTotal / elapsedTimeSecs;
312    }
313
314    /**
315     * @param progressWatchers all the watchers involved in the same byte
316     *                         transfer operation.
317     * @return the rate of bytes/second that has been achieved recently (ie
318     * within the last
319     * {@link #SECONDS_OF_HISTORY} seconds).
320     */
321    public static long calculateRecentByteRatePerSecond(BytesProgressWatcher[] progressWatchers) {
322        double sumOfRates = 0;
323        for (BytesProgressWatcher progressWatcher : progressWatchers) {
324            if (progressWatcher.isStarted()) {
325                sumOfRates += progressWatcher.getRecentByteRatePerSecond();
326            }
327        }
328        return Math.round(sumOfRates);
329    }
