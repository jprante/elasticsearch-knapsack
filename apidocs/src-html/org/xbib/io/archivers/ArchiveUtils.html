002package org.xbib.io.archivers;
003
004import java.io.UnsupportedEncodingException;
005
006/**
007 * Generic Archive utilities
008 */
009public class ArchiveUtils {
010
011    /**
012     * Private constructor to prevent instantiation of this utility class.
013     */
014    private ArchiveUtils() {
015    }
016
017    /**
018     * Generates a string containing the name, isDirectory setting and size of an entry.
019     * <p/>
020     * For example:<br/>
021     * <tt>-    2000 main.c</tt><br/>
022     * <tt>d     100 testfiles</tt><br/>
023     *
024     * @return the representation of the entry
025     */
026    public static String toString(ArchiveEntry entry) {
027        StringBuilder sb = new StringBuilder();
028        sb.append(entry.isDirectory() ? 'd' : '-');// c.f. "ls -l" output
029        String size = Long.toString((entry.getEntrySize()));
030        sb.append(' ');
031        // Pad output to 7 places, leading spaces
032        for (int i = 7; i > size.length(); i--) {
033            sb.append(' ');
034        }
035        sb.append(size);
036        sb.append(' ').append(entry.getName());
037        return sb.toString();
038    }
039
040    /**
041     * Check if buffer contents matches Ascii String.
042     *
043     * @param expected
044     * @param buffer
045     * @param offset
046     * @param length
047     * @return {@code true} if buffer is the same as the expected string
048     */
049    public static boolean matchAsciiBuffer(
050            String expected, byte[] buffer, int offset, int length) {
051        byte[] buffer1;
052        try {
053            buffer1 = expected.getBytes("ASCII");
054        } catch (UnsupportedEncodingException e) {
055            throw new RuntimeException(e); // Should not happen
056        }
057        return isEqual(buffer1, 0, buffer1.length, buffer, offset, length, false);
058    }
059
060    /**
061     * Convert a string to Ascii bytes.
062     * Used for comparing "magic" strings which need to be independent of the default Locale.
063     *
064     * @param inputString
065     * @return the bytes
066     */
067    public static byte[] toAsciiBytes(String inputString) {
068        try {
069            return inputString.getBytes("ASCII");
070        } catch (UnsupportedEncodingException e) {
071            throw new RuntimeException(e); // Should never happen
072        }
073    }
074
075    /**
076     * Convert an input byte array to a String using the ASCII character set.
077     *
078     * @param inputBytes
079     * @return the bytes, interpreted as an Ascii string
080     */
081    public static String toAsciiString(final byte[] inputBytes) {
082        try {
083            return new String(inputBytes, "ASCII");
084        } catch (UnsupportedEncodingException e) {
085            throw new RuntimeException(e); // Should never happen
086        }
087    }
088
089    /**
090     * Convert an input byte array to a String using the ASCII character set.
091     *
092     * @param inputBytes input byte array
093     * @param offset     offset within array
094     * @param length     length of array
095     * @return the bytes, interpreted as an Ascii string
096     */
097    public static String toAsciiString(final byte[] inputBytes, int offset, int length) {
098        try {
099            return new String(inputBytes, offset, length, "ASCII");
100        } catch (UnsupportedEncodingException e) {
101            throw new RuntimeException(e); // Should never happen
102        }
103    }
104
105    /**
106     * Compare byte buffers, optionally ignoring trailing nulls
107     *
108     * @param buffer1
109     * @param offset1
110     * @param length1
111     * @param buffer2
112     * @param offset2
113     * @param length2
114     * @param ignoreTrailingNulls
115     * @return {@code true} if buffer1 and buffer2 have same contents, having regard to trailing nulls
116     */
117    public static boolean isEqual(
118            final byte[] buffer1, final int offset1, final int length1,
119            final byte[] buffer2, final int offset2, final int length2,
120            boolean ignoreTrailingNulls) {
121        int minLen = length1 < length2 ? length1 : length2;
122        for (int i = 0; i < minLen; i++) {
123            if (buffer1[offset1 + i] != buffer2[offset2 + i]) {
124                return false;
125            }
126        }
127        if (length1 == length2) {
128            return true;
129        }
130        if (ignoreTrailingNulls) {
131            if (length1 > length2) {
132                for (int i = length2; i < length1; i++) {
133                    if (buffer1[offset1 + i] != 0) {
134                        return false;
135                    }
136                }
137            } else {
138                for (int i = length1; i < length2; i++) {
139                    if (buffer2[offset2 + i] != 0) {
140                        return false;
141                    }
142                }
143            }
144            return true;
145        }
146        return false;
147    }
148
149}
