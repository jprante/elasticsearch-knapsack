002    package org.xbib.io.archivers.zip;
003    
004    /**
005     * Wrapper for extra field data that doesn't conform to the recommended format of header-tag + size + data.
006     * <p/>
007     * <p>The header-id is artificial (and not listed as a known ID in
008     * {@link <a href="http://www.pkware.com/documents/casestudies/APPNOTE.TXT">
009     * APPNOTE.TXT</a>}).  Since it isn't used anywhere except to satisfy the
010     * ZipExtraField contract it shouldn't matter anyway.</p>
011     */
012    public final class UnparseableExtraFieldData implements ZipExtraField {
013        private static final ZipShort HEADER_ID = new ZipShort(0xACC1);
014    
015        private byte[] localFileData;
016        private byte[] centralDirectoryData;
017    
018        /**
019         * The Header-ID.
020         *
021         * @return a completely arbitrary value that should be ignored.
022         */
023        public ZipShort getHeaderId() {
024            return HEADER_ID;
025        }
026    
027        /**
028         * Length of the complete extra field in the local file data.
029         *
030         * @return The LocalFileDataLength value
031         */
032        public ZipShort getLocalFileDataLength() {
033            return new ZipShort(localFileData == null ? 0 : localFileData.length);
034        }
035    
036        /**
037         * Length of the complete extra field in the central directory.
038         *
039         * @return The CentralDirectoryLength value
040         */
041        public ZipShort getCentralDirectoryLength() {
042            return centralDirectoryData == null
043                    ? getLocalFileDataLength()
044                    : new ZipShort(centralDirectoryData.length);
045        }
046    
047        /**
048         * The actual data to put into local file data.
049         *
050         * @return The LocalFileDataData value
051         */
052        public byte[] getLocalFileDataData() {
053            return ZipUtil.copy(localFileData);
054        }
055    
056        /**
057         * The actual data to put into central directory.
058         *
059         * @return The CentralDirectoryData value
060         */
061        public byte[] getCentralDirectoryData() {
062            return centralDirectoryData == null
063                    ? getLocalFileDataData() : ZipUtil.copy(centralDirectoryData);
064        }
065    
066        /**
067         * Populate data from this array as if it was in local file data.
068         *
069         * @param buffer the buffer to read data from
070         * @param offset offset into buffer to read data
071         * @param length the length of data
072         */
073        public void parseFromLocalFileData(byte[] buffer, int offset, int length) {
074            localFileData = new byte[length];
075            System.arraycopy(buffer, offset, localFileData, 0, length);
076        }
077    
078        /**
079         * Populate data from this array as if it was in central directory data.
080         *
081         * @param buffer the buffer to read data from
082         * @param offset offset into buffer to read data
083         * @param length the length of data
084         */
085        public void parseFromCentralDirectoryData(byte[] buffer, int offset,
086                                                  int length) {
087            centralDirectoryData = new byte[length];
088            System.arraycopy(buffer, offset, centralDirectoryData, 0, length);
089            if (localFileData == null) {
090                parseFromLocalFileData(buffer, offset, length);
091            }
092        }
093    
094    }
