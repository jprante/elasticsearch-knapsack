002    package org.xbib.io.archivers.zip;
003    
004    /**
005     * The different modes {@link ZipArchiveOutputStream} can operate in.
006     *
007     * @see ZipArchiveOutputStream#setUseZip64
008     */
009    public enum Zip64Mode {
010        /**
011         * Use Zip64 extensions for all entries, even if it is clear it is
012         * not required.
013         */
014        Always,
015        /**
016         * Don't use Zip64 extensions for any entries.
017         * <p/>
018         * <p>This will cause a {@link Zip64RequiredException} to be
019         * thrown if {@link ZipArchiveOutputStream} detects it needs Zip64
020         * support.</p>
021         */
022        Never,
023        /**
024         * Use Zip64 extensions for all entries where they are required,
025         * don't use them for entries that clearly don't require them.
026         */
027        AsNeeded
028    }
