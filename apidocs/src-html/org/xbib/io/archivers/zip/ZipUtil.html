002    package org.xbib.io.archivers.zip;
003    
004    import java.io.IOException;
005    import java.util.Calendar;
006    import java.util.zip.CRC32;
007    import java.util.zip.ZipEntry;
008    
009    /**
010     * Utility class for handling DOS and Java time conversions.
011     */
012    public abstract class ZipUtil {
013        /**
014         * Smallest date/time ZIP can handle.
015         */
016        private static final byte[] DOS_TIME_MIN = ZipLong.getBytes(0x00002100L);
017    
018        /**
019         * Convert a Date object to a DOS date/time field.
020         * <p/>
021         * <p>Stolen from InfoZip's <code>fileio.c</code></p>
022         *
023         * @param t number of milliseconds since the epoch
024         * @return the date as a byte array
025         */
026        public static byte[] toDosTime(long t) {
027            Calendar c = Calendar.getInstance();
028            c.setTimeInMillis(t);
029    
030            int year = c.get(Calendar.YEAR);
031            if (year < 1980) {
032                return copy(DOS_TIME_MIN); // stop callers from changing the array
033            }
034            int month = c.get(Calendar.MONTH) + 1;
035            long value = ((year - 1980) << 25)
036                    | (month << 21)
037                    | (c.get(Calendar.DAY_OF_MONTH) << 16)
038                    | (c.get(Calendar.HOUR_OF_DAY) << 11)
039                    | (c.get(Calendar.MINUTE) << 5)
040                    | (c.get(Calendar.SECOND) >> 1);
041            return ZipLong.getBytes(value);
042        }
043    
044    
045        /**
046         * Converts DOS time to Java time (number of milliseconds since
047         * epoch).
048         */
049        public static long dosToJavaTime(long dosTime) {
050            Calendar cal = Calendar.getInstance();
051            // CheckStyle:MagicNumberCheck OFF - no point
052            cal.set(Calendar.YEAR, (int) ((dosTime >> 25) & 0x7f) + 1980);
053            cal.set(Calendar.MONTH, (int) ((dosTime >> 21) & 0x0f) - 1);
054            cal.set(Calendar.DATE, (int) (dosTime >> 16) & 0x1f);
055            cal.set(Calendar.HOUR_OF_DAY, (int) (dosTime >> 11) & 0x1f);
056            cal.set(Calendar.MINUTE, (int) (dosTime >> 5) & 0x3f);
057            cal.set(Calendar.SECOND, (int) (dosTime << 1) & 0x3e);
058            // CheckStyle:MagicNumberCheck ON
059            return cal.getTime().getTime();
060        }
061    
062        /**
063         * If the entry has Unicode*ExtraFields and the CRCs of the
064         * names/comments match those of the extra fields, transfer the
065         * known Unicode values from the extra field.
066         */
067        static void setNameAndCommentFromExtraFields(ZipArchiveEntry ze,
068                                                     byte[] originalNameBytes,
069                                                     byte[] commentBytes) {
070            UnicodePathExtraField name = (UnicodePathExtraField)
071                    ze.getExtraField(UnicodePathExtraField.UPATH_ID);
072            String originalName = ze.getName();
073            String newName = getUnicodeStringIfOriginalMatches(name,
074                    originalNameBytes);
075            if (newName != null && !originalName.equals(newName)) {
076                ze.setName(newName);
077            }
078    
079            if (commentBytes != null && commentBytes.length > 0) {
080                UnicodeCommentExtraField cmt = (UnicodeCommentExtraField)
081                        ze.getExtraField(UnicodeCommentExtraField.UCOM_ID);
082                String newComment =
083                        getUnicodeStringIfOriginalMatches(cmt, commentBytes);
084                if (newComment != null) {
085                    ze.setComment(newComment);
086                }
087            }
088        }
089    
090        /**
091         * If the stored CRC matches the one of the given name, return the
092         * Unicode name of the given field.
093         * <p/>
094         * <p>If the field is null or the CRCs don't match, return null
095         * instead.</p>
096         */
097        private static String getUnicodeStringIfOriginalMatches(AbstractUnicodeExtraField f,
098                                                                byte[] orig) {
099            if (f != null) {
100                CRC32 crc32 = new CRC32();
101                crc32.update(orig);
102                long origCRC32 = crc32.getValue();
103    
104                if (origCRC32 == f.getNameCRC32()) {
105                    try {
106                        return ZipEncodingHelper
107                                .UTF8_ZIP_ENCODING.decode(f.getUnicodeName());
108                    } catch (IOException ex) {
109                        // UTF-8 unsupported?  should be impossible the
110                        // Unicode*ExtraField must contain some bad bytes
111    
112                        // TODO log this anywhere?
113                        return null;
114                    }
115                }
116            }
117            return null;
118        }
119    
120        /**
121         * Create a copy of the given array - or return null if the
122         * argument is null.
123         */
124        static byte[] copy(byte[] from) {
125            if (from != null) {
126                byte[] to = new byte[from.length];
127                System.arraycopy(from, 0, to, 0, to.length);
128                return to;
129            }
130            return null;
131        }
132    
133        /**
134         * Whether this library is able to read or write the given entry.
135         */
136        static boolean canHandleEntryData(ZipArchiveEntry entry) {
137            return supportsEncryptionOf(entry) && supportsMethodOf(entry);
138        }
139    
140        /**
141         * Whether this library supports the encryption used by the given
142         * entry.
143         *
144         * @return true if the entry isn't encrypted at all
145         */
146        private static boolean supportsEncryptionOf(ZipArchiveEntry entry) {
147            return !entry.getGeneralPurposeBit().usesEncryption();
148        }
149    
150        /**
151         * Whether this library supports the compression method used by
152         * the given entry.
153         *
154         * @return true if the compression method is STORED or DEFLATED
155         */
156        private static boolean supportsMethodOf(ZipArchiveEntry entry) {
157            return entry.getMethod() == ZipEntry.STORED
158                    || entry.getMethod() == ZipEntry.DEFLATED;
159        }
160    
161        /**
162         * Checks whether the entry requires features not (yet) supported
163         * by the library and throws an exception if it does.
164         */
165        static void checkRequestedFeatures(ZipArchiveEntry ze)
166                throws UnsupportedZipFeatureException {
167            if (!supportsEncryptionOf(ze)) {
168                throw
169                        new UnsupportedZipFeatureException(UnsupportedZipFeatureException
170                                .Feature.ENCRYPTION, ze);
171            }
172            if (!supportsMethodOf(ze)) {
173                throw
174                        new UnsupportedZipFeatureException(UnsupportedZipFeatureException
175                                .Feature.METHOD, ze);
176            }
177        }
178    }
