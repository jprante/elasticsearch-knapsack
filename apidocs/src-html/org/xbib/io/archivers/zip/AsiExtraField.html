002    package org.xbib.io.archivers.zip;
003    
004    import java.util.zip.CRC32;
005    import java.util.zip.ZipException;
006    
007    /**
008     * Adds Unix file permission and UID/GID fields as well as symbolic
009     * link handling.
010     * <p/>
011     * <p>This class uses the ASi extra field in the format:
012     * <pre>
013     *         Value         Size            Description
014     *         -----         ----            -----------
015     * (Unix3) 0x756e        Short           tag for this extra block type
016     *         TSize         Short           total data size for this block
017     *         CRC           Long            CRC-32 of the remaining data
018     *         Mode          Short           file permissions
019     *         SizDev        Long            symlink'd size OR major/minor dev num
020     *         UID           Short           user ID
021     *         GID           Short           group ID
022     *         (var.)        variable        symbolic link filename
023     * </pre>
024     * taken from appnote.iz (Info-ZIP note, 981119) found at <a
025     * href="ftp://ftp.uu.net/pub/archiving/zip/doc/">ftp://ftp.uu.net/pub/archiving/zip/doc/</a></p>
026     * <p/>
027     * <p/>
028     * <p>Short is two bytes and Long is four bytes in big endian byte and
029     * word order, device numbers are currently not supported.</p>
030     * <p/>
031     * <p>Since the documentation this class is based upon doesn't mention
032     * the character encoding of the file name at all, it is assumed that
033     * it uses the current platform's default encoding.</p>
034     */
035    public class AsiExtraField implements ZipExtraField, UnixStat, Cloneable {
036    
037        private static final ZipShort HEADER_ID = new ZipShort(0x756E);
038        private static final int WORD = 4;
039        /**
040         * Standard Unix stat(2) file mode.
041         */
042        private int mode = 0;
043        /**
044         * User ID.
045         */
046        private int uid = 0;
047        /**
048         * Group ID.
049         */
050        private int gid = 0;
051        /**
052         * File this entry points to, if it is a symbolic link.
053         * <p/>
054         * <p>empty string - if entry is not a symbolic link.</p>
055         */
056        private String link = "";
057        /**
058         * Is this an entry for a directory?
059         */
060        private boolean dirFlag = false;
061    
062        /**
063         * Instance used to calculate checksums.
064         */
065        private CRC32 crc = new CRC32();
066    
067        /**
068         * Constructor for AsiExtraField.
069         */
070        public AsiExtraField() {
071        }
072    
073        /**
074         * The Header-ID.
075         *
076         * @return the value for the header id for this extrafield
077         */
078        public ZipShort getHeaderId() {
079            return HEADER_ID;
080        }
081    
082        /**
083         * Length of the extra field in the local file data - without
084         * Header-ID or length specifier.
085         *
086         * @return a <code>ZipShort</code> for the length of the data of this extra field
087         */
088        public ZipShort getLocalFileDataLength() {
089            return new ZipShort(WORD         // CRC
090                    + 2         // Mode
091                    + WORD         // SizDev
092                    + 2         // UID
093                    + 2         // GID
094                    + getLinkedFile().getBytes().length);
095            // Uses default charset - see class Javadoc
096        }
097    
098        /**
099         * Delegate to local file data.
100         *
101         * @return the centralDirectory length
102         */
103        public ZipShort getCentralDirectoryLength() {
104            return getLocalFileDataLength();
105        }
106    
107        /**
108         * The actual data to put into local file data - without Header-ID
109         * or length specifier.
110         *
111         * @return get the data
112         */
113        public byte[] getLocalFileDataData() {
114            // CRC will be added later
115            byte[] data = new byte[getLocalFileDataLength().getValue() - WORD];
116            System.arraycopy(ZipShort.getBytes(getMode()), 0, data, 0, 2);
117    
118            byte[] linkArray = getLinkedFile().getBytes(); // Uses default charset - see class Javadoc
119            // CheckStyle:MagicNumber OFF
120            System.arraycopy(ZipLong.getBytes(linkArray.length),
121                    0, data, 2, WORD);
122    
123            System.arraycopy(ZipShort.getBytes(getUserId()),
124                    0, data, 6, 2);
125            System.arraycopy(ZipShort.getBytes(getGroupId()),
126                    0, data, 8, 2);
127    
128            System.arraycopy(linkArray, 0, data, 10, linkArray.length);
129    
130            crc.reset();
131            crc.update(data);
132            long checksum = crc.getValue();
133    
134            byte[] result = new byte[data.length + WORD];
135            System.arraycopy(ZipLong.getBytes(checksum), 0, result, 0, WORD);
136            System.arraycopy(data, 0, result, WORD, data.length);
137            return result;
138        }
139    
140        /**
141         * Delegate to local file data.
142         *
143         * @return the local file data
144         */
145        public byte[] getCentralDirectoryData() {
146            return getLocalFileDataData();
147        }
148    
149        /**
150         * Set the user id.
151         *
152         * @param uid the user id
153         */
154        public void setUserId(int uid) {
155            this.uid = uid;
156        }
157    
158        /**
159         * Get the user id.
160         *
161         * @return the user id
162         */
163        public int getUserId() {
164            return uid;
165        }
166    
167        /**
168         * Set the group id.
169         *
170         * @param gid the group id
171         */
172        public void setGroupId(int gid) {
173            this.gid = gid;
174        }
175    
176        /**
177         * Get the group id.
178         *
179         * @return the group id
180         */
181        public int getGroupId() {
182            return gid;
183        }
184    
185        /**
186         * Indicate that this entry is a symbolic link to the given filename.
187         *
188         * @param name Name of the file this entry links to, empty String
189         *             if it is not a symbolic link.
190         */
191        public void setLinkedFile(String name) {
192            link = name;
193            mode = getMode(mode);
194        }
195    
196        /**
197         * Name of linked file
198         *
199         * @return name of the file this entry links to if it is a
200         * symbolic link, the empty string otherwise.
201         */
202        public String getLinkedFile() {
203            return link;
204        }
205    
206        /**
207         * Is this entry a symbolic link?
208         *
209         * @return true if this is a symbolic link
210         */
211        public boolean isLink() {
212            return getLinkedFile().length() != 0;
213        }
214    
215        /**
216         * File mode of this file.
217         *
218         * @param mode the file mode
219         */
220        public void setMode(int mode) {
221            this.mode = getMode(mode);
222        }
223    
224        /**
225         * File mode of this file.
226         *
227         * @return the file mode
228         */
229        public int getMode() {
230            return mode;
231        }
232    
233        /**
234         * Indicate whether this entry is a directory.
235         *
236         * @param dirFlag if true, this entry is a directory
237         */
238        public void setDirectory(boolean dirFlag) {
239            this.dirFlag = dirFlag;
240            mode = getMode(mode);
241        }
242    
243        /**
244         * Is this entry a directory?
245         *
246         * @return true if this entry is a directory
247         */
248        public boolean isDirectory() {
249            return dirFlag && !isLink();
250        }
251    
252        /**
253         * Populate data from this array as if it was in local file data.
254         *
255         * @param data   an array of bytes
256         * @param offset the start offset
257         * @param length the number of bytes in the array from offset
258         * @throws java.util.zip.ZipException on error
259         */
260        public void parseFromLocalFileData(byte[] data, int offset, int length)
261                throws ZipException {
262    
263            long givenChecksum = ZipLong.getValue(data, offset);
264            byte[] tmp = new byte[length - WORD];
265            System.arraycopy(data, offset + WORD, tmp, 0, length - WORD);
266            crc.reset();
267            crc.update(tmp);
268            long realChecksum = crc.getValue();
269            if (givenChecksum != realChecksum) {
270                throw new ZipException("bad CRC checksum "
271                        + Long.toHexString(givenChecksum)
272                        + " instead of "
273                        + Long.toHexString(realChecksum));
274            }
275    
276            int newMode = ZipShort.getValue(tmp, 0);
277            byte[] linkArray = new byte[(int) ZipLong.getValue(tmp, 2)];
278            uid = ZipShort.getValue(tmp, 6);
279            gid = ZipShort.getValue(tmp, 8);
280    
281            if (linkArray.length == 0) {
282                link = "";
283            } else {
284                System.arraycopy(tmp, 10, linkArray, 0, linkArray.length);
285                link = new String(linkArray); // Uses default charset - see class Javadoc
286            }
287            setDirectory((newMode & DIR_FLAG) != 0);
288            setMode(newMode);
289        }
290    
291        /**
292         * Doesn't do anything special since this class always uses the
293         * same data in central directory and local file data.
294         */
295        public void parseFromCentralDirectoryData(byte[] buffer, int offset,
296                                                  int length)
297                throws ZipException {
298            parseFromLocalFileData(buffer, offset, length);
299        }
300    
301        /**
302         * Get the file mode for given permissions with the correct file type.
303         *
304         * @param mode the mode
305         * @return the type with the mode
306         */
307        protected int getMode(int mode) {
308            int type = FILE_FLAG;
309            if (isLink()) {
310                type = LINK_FLAG;
311            } else if (isDirectory()) {
312                type = DIR_FLAG;
313            }
314            return type | (mode & PERM_MASK);
315        }
316    
317        @Override
318        public Object clone() throws CloneNotSupportedException {
319            AsiExtraField cloned = (AsiExtraField) super.clone();
320            cloned.crc = new CRC32();
321            return cloned;
322        }
323    }
