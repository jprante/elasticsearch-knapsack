117        /**
118         * Grow a byte buffer, so it has a minimal capacity or at least
119         * the double capacity of the original buffer
120         *
121         * @param b           The original buffer.
122         * @param newCapacity The minimal requested new capacity.
123         * @return A byte buffer <code>r</code> with
124         * <code>r.capacity() = max(b.capacity()*2,newCapacity)</code> and
125         * all the data contained in <code>b</code> copied to the beginning
126         * of <code>r</code>.
127         */
128        static ByteBuffer growBuffer(ByteBuffer b, int newCapacity) {
129            b.limit(b.position());
130            b.rewind();
131    
132            int c2 = b.capacity() * 2;
133            ByteBuffer on = ByteBuffer.allocate(c2 < newCapacity ? newCapacity : c2);
134    
135            on.put(b);
136            return on;
137        }

140        /**
141         * The hexadecimal digits <code>0,...,9,A,...,F</code> encoded as
142         * ASCII bytes.
143         */
144        private static final byte[] HEX_DIGITS =
145                new byte[]{
146                        0x30, 0x31, 0x32, 0x33, 0x34, 0x35, 0x36, 0x37, 0x38, 0x39, 0x41,
147                        0x42, 0x43, 0x44, 0x45, 0x46
148                };
149    
150        /**
151         * Append <code>%Uxxxx</code> to the given byte buffer.
152         * The caller must assure, that <code>bb.remaining()&gt;=6</code>.
153         *
154         * @param bb The byte buffer to write to.
155         * @param c  The character to write.
156         */
157        static void appendSurrogate(ByteBuffer bb, char c) {
158    
159            bb.put((byte) '%');
160            bb.put((byte) 'U');
161    
162            bb.put(HEX_DIGITS[(c >> 12) & 0x0f]);
163            bb.put(HEX_DIGITS[(c >> 8) & 0x0f]);
164            bb.put(HEX_DIGITS[(c >> 4) & 0x0f]);
165            bb.put(HEX_DIGITS[c & 0x0f]);
166        }

169        /**
170         * name of the encoding UTF-8
171         */
172        static final String UTF8 = "UTF8";
173    
174        /**
175         * variant name of the encoding UTF-8 used for comparisions.
176         */
177        private static final String UTF_DASH_8 = "UTF_8";
178    
179        /**
180         * name of the encoding UTF-8
181         */
182        static final ZipEncoding UTF8_ZIP_ENCODING = new FallbackZipEncoding(UTF8);
183    
184        /**
185         * Instantiates a zip encoding.
186         *
187         * @param name The name of the zip encoding. Specify {@code null} for
188         *             the platform's default encoding.
189         * @return A zip encoding for the given encoding name.
190         */
191        public static ZipEncoding getZipEncoding(String name) {
192    
193            // fallback encoding is good enough for utf-8.
194            if (isUTF8(name)) {
195                return UTF8_ZIP_ENCODING;
196            }
197    
198            if (name == null) {
199                return new FallbackZipEncoding();
200            }
201    
202            SimpleEncodingHolder h = simpleEncodings.get(name);
203    
204            if (h != null) {
205                return h.getEncoding();
206            }
207    
208            try {
209    
210                Charset cs = Charset.forName(name);
211                return new NioZipEncoding(cs);
212    
213            } catch (UnsupportedCharsetException e) {
214                return new FallbackZipEncoding(name);
215            }
216        }
217    
218        /**
219         * Whether a given encoding - or the platform's default encoding
220         * if the parameter is null - is UTF-8.
221         */
222        static boolean isUTF8(String encoding) {
223            if (encoding == null) {
224                // check platform's default encoding
225                encoding = System.getProperty("file.encoding");
226            }
227            return UTF8.equalsIgnoreCase(encoding)
228                    || UTF_DASH_8.equalsIgnoreCase(encoding);
229        }
