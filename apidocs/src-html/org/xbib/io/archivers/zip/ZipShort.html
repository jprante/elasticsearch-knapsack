002package org.xbib.io.archivers.zip;
003
004import static org.xbib.io.archivers.zip.ZipConstants.BYTE_MASK;
005
006/**
007 * Utility class that represents a two byte integer with conversion
008 * rules for the big endian byte order of ZIP files.
009 */
010public final class ZipShort implements Cloneable {
011    private static final int BYTE_1_MASK = 0xFF00;
012    private static final int BYTE_1_SHIFT = 8;
013
014    private final int value;
015
016    /**
017     * Create instance from a number.
018     *
019     * @param value the int to store as a ZipShort
020     */
021    public ZipShort(int value) {
022        this.value = value;
023    }
024
025    /**
026     * Create instance from bytes.
027     *
028     * @param bytes the bytes to store as a ZipShort
029     */
030    public ZipShort(byte[] bytes) {
031        this(bytes, 0);
032    }
033
034    /**
035     * Create instance from the two bytes starting at offset.
036     *
037     * @param bytes  the bytes to store as a ZipShort
038     * @param offset the offset to start
039     */
040    public ZipShort(byte[] bytes, int offset) {
041        value = ZipShort.getValue(bytes, offset);
042    }
043
044    /**
045     * Get value as two bytes in big endian byte order.
046     *
047     * @return the value as a a two byte array in big endian byte order
048     */
049    public byte[] getBytes() {
050        byte[] result = new byte[2];
051        result[0] = (byte) (value & BYTE_MASK);
052        result[1] = (byte) ((value & BYTE_1_MASK) >> BYTE_1_SHIFT);
053        return result;
054    }
055
056    /**
057     * Get value as Java int.
058     *
059     * @return value as a Java int
060     */
061    public int getValue() {
062        return value;
063    }
064
065    /**
066     * Get value as two bytes in big endian byte order.
067     *
068     * @param value the Java int to convert to bytes
069     * @return the converted int as a byte array in big endian byte order
070     */
071    public static byte[] getBytes(int value) {
072        byte[] result = new byte[2];
073        result[0] = (byte) (value & BYTE_MASK);
074        result[1] = (byte) ((value & BYTE_1_MASK) >> BYTE_1_SHIFT);
075        return result;
076    }
077
078    /**
079     * Helper method to get the value as a java int from two bytes starting at given array offset
080     *
081     * @param bytes  the array of bytes
082     * @param offset the offset to start
083     * @return the corresponding java int value
084     */
085    public static int getValue(byte[] bytes, int offset) {
086        int value = (bytes[offset + 1] << BYTE_1_SHIFT) & BYTE_1_MASK;
087        value += (bytes[offset] & BYTE_MASK);
088        return value;
089    }
090
091    /**
092     * Helper method to get the value as a java int from a two-byte array
093     *
094     * @param bytes the array of bytes
095     * @return the corresponding java int value
096     */
097    public static int getValue(byte[] bytes) {
098        return getValue(bytes, 0);
099    }
100
101    /**
102     * Override to make two instances with same value equal.
103     *
104     * @param o an object to compare
105     * @return true if the objects are equal
106     */
107    @Override
108    public boolean equals(Object o) {
109        if (o == null || !(o instanceof ZipShort)) {
110            return false;
111        }
112        return value == ((ZipShort) o).getValue();
113    }
114
115    /**
116     * Override to make two instances with same value equal.
117     *
118     * @return the value stored in the ZipShort
119     */
120    @Override
121    public int hashCode() {
122        return value;
123    }
124
125    @Override
126    public Object clone() {
127        try {
128            return super.clone();
129        } catch (CloneNotSupportedException cnfe) {
130            // impossible
131            throw new RuntimeException(cnfe);
132        }
133    }
134
135    @Override
136    public String toString() {
137        return "ZipShort value: " + value;
138    }
139}
