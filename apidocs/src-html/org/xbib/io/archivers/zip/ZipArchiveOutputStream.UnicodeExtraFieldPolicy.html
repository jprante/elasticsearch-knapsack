030    /**
031     * Reimplementation of {@link java.util.zip.ZipOutputStream
032     * java.util.zip.ZipOutputStream} that does handle the extended
033     * functionality of this package, especially internal/external file
034     * attributes and extra fields with different layouts for local file
035     * data and central directory entries.
036     * <p/>
037     * <p>This class will try to use {@link java.io.RandomAccessFile
038     * RandomAccessFile} when you know that the output is going to go to a
039     * file.</p>
040     * <p/>
041     * <p>If RandomAccessFile cannot be used, this implementation will use
042     * a Data Descriptor to store size and CRC information for {@link
043     * #DEFLATED DEFLATED} entries, this means, you don't need to
044     * calculate them yourself.  Unfortunately this is not possible for
045     * the {@link #STORED STORED} method, here setting the CRC and
046     * uncompressed size information is required before {@link
047     * #putArchiveEntry(ArchiveEntry)} can be called.</p>
048     * <p/>
049     * <p>It transparently supports Zip64
050     * extensions and thus individual entries and archives larger than 4
051     * GB or with more than 65536 entries in most cases but explicit
052     * control is provided via {@link #setUseZip64}.  If the stream can not
053     * user RandomAccessFile and you try to write a ZipArchiveEntry of
054     * unknown size then Zip64 extensions will be disabled by default.</p>
055     */

230        /**
231         * Creates a new ZIP OutputStream writing to a File.  Will use
232         * random access if possible.
233         *
234         * @param file the file to zip to
235         * @throws java.io.IOException on error
236         */
237        public ZipArchiveOutputStream(File file) throws IOException {
238            OutputStream o = null;
239            RandomAccessFile _raf = null;
240            try {
241                _raf = new RandomAccessFile(file, "rw");
242                _raf.setLength(0);
243            } catch (IOException e) {
244                if (_raf != null) {
245                    try {
246                        _raf.close();
247                    } catch (IOException inner) { // NOPMD
248                        // ignore
249                    }
250                    _raf = null;
251                }
252                o = new FileOutputStream(file);
253            }
254            out = o;
255            raf = _raf;
256        }

328        /**
329         * Whether Zip64 extensions will be used.
330         * <p/>
331         * <p>When setting the mode to {@link Zip64Mode#Never Never},
332         * {@link #putArchiveEntry}, {@link #closeArchiveEntry}, {@link
333         * #finish} or {@link #close} may throw a {@link
334         * Zip64RequiredException} if the entry's size or the total size
335         * of the archive exceeds 4GB or there are more than 65536 entries
336         * inside the archive.  Any archive created in this mode will be
337         * readable by implementations that don't support Zip64.</p>
338         * <p/>
339         * <p>When setting the mode to {@link Zip64Mode#Always Always},
340         * Zip64 extensions will be used for all entries.  Any archive
341         * created in this mode may be unreadable by implementations that
342         * don't support Zip64 even if all its contents would be.</p>
343         * <p/>
344         * <p>When setting the mode to {@link Zip64Mode#AsNeeded
345         * AsNeeded}, Zip64 extensions will transparently be used for
346         * those entries that require them.  This mode can only be used if
347         * the uncompressed size of the {@link ZipArchiveEntry} is known
348         * when calling {@link #putArchiveEntry} or the archive is written
349         * to a seekable output (i.e. you have used the {@link
350         * #ZipArchiveOutputStream(java.io.File) File-arg constructor}) -
351         * this mode is not valid when the output stream is not seekable
352         * and the uncompressed size is unknown when {@link
353         * #putArchiveEntry} is called.</p>
354         * <p/>
355         * <p>If no entry inside the resulting archive requires Zip64
356         * extensions then {@link Zip64Mode#Never Never} will create the
357         * smallest archive.  {@link Zip64Mode#AsNeeded AsNeeded} will
358         * create a slightly bigger archive if the uncompressed size of
359         * any entry has initially been unknown and create an archive
360         * identical to {@link Zip64Mode#Never Never} otherwise.  {@link
361         * Zip64Mode#Always Always} will create an archive that is at
362         * least 24 bytes per entry bigger than the one {@link
363         * Zip64Mode#Never Never} would create.</p>
364         * <p/>
365         * <p>Defaults to {@link Zip64Mode#AsNeeded AsNeeded} unless
366         * {@link #putArchiveEntry} is called with an entry of unknown
367         * size and data is written to a non-seekable stream - in this
368         * case the default is {@link Zip64Mode#Never Never}.</p>
369         */
370        public void setUseZip64(Zip64Mode mode) {
371            zip64Mode = mode;
372        }

374        /**
375         * @throws Zip64RequiredException if the archive's size exceeds 4
376         *                                GByte or there are more than 65535 entries inside the archive
377         *                                and {@link #setUseZip64} is {@link Zip64Mode#Never}.
378         */
379        @Override
380        public void finish() throws IOException {
381            if (finished) {
382                throw new IOException("This archive has already been finished");
383            }
384    
385            if (entry != null) {
386                throw new IOException("This archives contains unclosed entries.");
387            }
388    
389            cdOffset = written;
390            for (ZipArchiveEntry ze : entries) {
391                writeCentralFileHeader(ze);
392            }
393            cdLength = written - cdOffset;
394            writeZip64CentralDirectory();
395            writeCentralDirectoryEnd();
396            offsets.clear();
397            entries.clear();
398            def.end();
399            finished = true;
400        }
401    
402        /**
403         * Writes all necessary data for this entry.
404         *
405         * @throws java.io.IOException    on error
406         * @throws Zip64RequiredException if the entry's uncompressed or
407         *                                compressed size exceeds 4 GByte and {@link #setUseZip64}
408         *                                is {@link Zip64Mode#Never}.
409         */
410        @Override
411        public void closeArchiveEntry() throws IOException {
412            if (finished) {
413                throw new IOException("Stream has already been finished");
414            }
415    
416            if (entry == null) {
417                throw new IOException("No current entry to close");
418            }
419    
420            if (!entry.hasWritten) {
421                write(new byte[0], 0, 0);
422            }
423    
424            flushDeflater();
425    
426            final Zip64Mode effectiveMode = getEffectiveZip64Mode(entry.entry);
427            long bytesWritten = written - entry.dataStart;
428            long realCrc = crc.getValue();
429            crc.reset();
430    
431            final boolean actuallyNeedsZip64 =
432                    handleSizesAndCrc(bytesWritten, realCrc, effectiveMode);
433    
434            if (raf != null) {
435                rewriteSizesAndCrc(actuallyNeedsZip64);
436            }
437    
438            writeDataDescriptor(entry.entry);
439            entry = null;
440        }
441    
442        /**
443         * Ensures all bytes sent to the deflater are written to the stream.
444         */
445        private void flushDeflater() throws IOException {
446            if (entry.entry.getMethod() == DEFLATED) {
447                def.finish();
448                while (!def.finished()) {
449                    deflate();
450                }
451            }
452        }
453    
454        /**
455         * Ensures the current entry's size and CRC information is set to
456         * the values just written, verifies it isn't too big in the
457         * Zip64Mode.Never case and returns whether the entry would
458         * require a Zip64 extra field.
459         */
460        private boolean handleSizesAndCrc(long bytesWritten, long crc,
461                                          Zip64Mode effectiveMode)
462                throws ZipException {
463            if (entry.entry.getMethod() == DEFLATED) {
464                /* It turns out def.getBytesRead() returns wrong values if
465                 * the size exceeds 4 GB on Java < Java7
466                entry.entry.setSize(def.getBytesRead());
467                */
468                entry.entry.setSize(entry.bytesRead);
469                entry.entry.setCompressedSize(bytesWritten);
470                entry.entry.setCrc(crc);
471    
472                def.reset();
473            } else if (raf == null) {
474                if (entry.entry.getCrc() != crc) {
475                    throw new ZipException("bad CRC checksum for entry "
476                            + entry.entry.getName() + ": "
477                            + Long.toHexString(entry.entry.getCrc())
478                            + " instead of "
479                            + Long.toHexString(crc));
480                }
481    
482                if (entry.entry.getSize() != bytesWritten) {
483                    throw new ZipException("bad size for entry "
484                            + entry.entry.getName() + ": "
485                            + entry.entry.getSize()
486                            + " instead of "
487                            + bytesWritten);
488                }
489            } else { /* method is STORED and we used RandomAccessFile */
490                entry.entry.setSize(bytesWritten);
491                entry.entry.setCompressedSize(bytesWritten);
492                entry.entry.setCrc(crc);
493            }
494    
495            final boolean actuallyNeedsZip64 = effectiveMode == Zip64Mode.Always
496                    || entry.entry.getSize() >= ZIP64_MAGIC
497                    || entry.entry.getCompressedSize() >= ZIP64_MAGIC;
498            if (actuallyNeedsZip64 && effectiveMode == Zip64Mode.Never) {
499                throw new Zip64RequiredException(Zip64RequiredException
500                        .getEntryTooBigMessage(entry.entry));
501            }
502            return actuallyNeedsZip64;
503        }
504    
505        /**
506         * When using random access output, write the local file header
507         * and potentiall the ZIP64 extra containing the correct CRC and
508         * compressed/uncompressed sizes.
509         */
510        private void rewriteSizesAndCrc(boolean actuallyNeedsZip64)
511                throws IOException {
512            long save = raf.getFilePointer();
513    
514            raf.seek(entry.localDataStart);
515            writeOut(ZipLong.getBytes(entry.entry.getCrc()));
516            if (!hasZip64Extra(entry.entry) || !actuallyNeedsZip64) {
517                writeOut(ZipLong.getBytes(entry.entry.getCompressedSize()));
518                writeOut(ZipLong.getBytes(entry.entry.getSize()));
519            } else {
520                writeOut(ZipLong.ZIP64_MAGIC.getBytes());
521                writeOut(ZipLong.ZIP64_MAGIC.getBytes());
522            }
523    
524            if (hasZip64Extra(entry.entry)) {
525                // seek to ZIP64 extra, skip header and size information
526                raf.seek(entry.localDataStart + 3 * WORD + 2 * SHORT
527                        + getName(entry.entry).limit() + 2 * SHORT);
528                // inside the ZIP64 extra uncompressed size comes
529                // first, unlike the LFH, CD or data descriptor
530                writeOut(ZipEightByteInteger.getBytes(entry.entry.getSize()));
531                writeOut(ZipEightByteInteger.getBytes(entry.entry.getCompressedSize()));
532    
533                if (!actuallyNeedsZip64) {
534                    // do some cleanup:
535                    // * rewrite version needed to extract
536                    raf.seek(entry.localDataStart - 5 * SHORT);
537                    writeOut(ZipShort.getBytes(INITIAL_VERSION));
538    
539                    // * remove ZIP64 extra so it doesn't get written
540                    //   to the central directory
541                    entry.entry.removeExtraField(Zip64ExtendedInformationExtraField
542                            .HEADER_ID);
543                    entry.entry.setExtra();
544    
545                    // * reset hasUsedZip64 if it has been set because
546                    //   of this entry
547                    if (entry.causedUseOfZip64) {
548                        hasUsedZip64 = false;
549                    }
550                }
551            }
552            raf.seek(save);
553        }

555        /**
556         * {@inheritDoc}
557         *
558         * @throws ClassCastException     if entry is not an instance of ZipArchiveEntry
559         * @throws Zip64RequiredException if the entry's uncompressed or
560         *                                compressed size is known to exceed 4 GByte and {@link #setUseZip64}
561         *                                is {@link Zip64Mode#Never}.
562         */
563        @Override
564        public void putArchiveEntry(ArchiveEntry archiveEntry) throws IOException {
565            if (finished) {
566                throw new IOException("Stream has already been finished");
567            }
568    
569            if (entry != null) {
570                closeArchiveEntry();
571            }
572    
573            entry = new CurrentEntry((ZipArchiveEntry) archiveEntry);
574            entries.add(entry.entry);
575    
576            setDefaults(entry.entry);
577    
578            final Zip64Mode effectiveMode = getEffectiveZip64Mode(entry.entry);
579            validateSizeInformation(effectiveMode);
580    
581            if (shouldAddZip64Extra(entry.entry, effectiveMode)) {
582    
583                Zip64ExtendedInformationExtraField z64 = getZip64Extra(entry.entry);
584    
585                // just a placeholder, real data will be in data
586                // descriptor or inserted later via RandomAccessFile
587                ZipEightByteInteger size = ZipEightByteInteger.ZERO;
588                if (entry.entry.getMethod() == STORED
589                        && entry.entry.getSize() != ArchiveEntry.SIZE_UNKNOWN) {
590                    // actually, we already know the sizes
591                    size = new ZipEightByteInteger(entry.entry.getSize());
592                }
593                z64.setSize(size);
594                z64.setCompressedSize(size);
595                entry.entry.setExtra();
596            }
597    
598            if (entry.entry.getMethod() == DEFLATED && hasCompressionLevelChanged) {
599                def.setLevel(level);
600                hasCompressionLevelChanged = false;
601            }
602            writeLocalFileHeader(entry.entry);
603        }

605        /**
606         * Provides default values for compression method and last
607         * modification time.
608         */
609        private void setDefaults(ZipArchiveEntry entry) {
610            if (entry.getMethod() == -1) { // not specified
611                entry.setMethod(method);
612            }
613    
614            if (entry.getTime() == -1) { // not specified
615                entry.setTime(System.currentTimeMillis());
616            }
617        }

619        /**
620         * Throws an exception if the size is unknown for a stored entry
621         * that is written to a non-seekable output or the entry is too
622         * big to be written without Zip64 extra but the mode has been set
623         * to Never.
624         */
625        private void validateSizeInformation(Zip64Mode effectiveMode)
626                throws ZipException {
627            // Size/CRC not required if RandomAccessFile is used
628            if (entry.entry.getMethod() == STORED && raf == null) {
629                if (entry.entry.getSize() == ArchiveEntry.SIZE_UNKNOWN) {
630                    throw new ZipException("uncompressed size is required for"
631                            + " STORED method when not writing to a"
632                            + " file");
633                }
634                if (entry.entry.getCrc() == -1) {
635                    throw new ZipException("crc checksum is required for STORED"
636                            + " method when not writing to a file");
637                }
638                entry.entry.setCompressedSize(entry.entry.getSize());
639            }
640    
641            if ((entry.entry.getSize() >= ZIP64_MAGIC
642                    || entry.entry.getCompressedSize() >= ZIP64_MAGIC)
643                    && effectiveMode == Zip64Mode.Never) {
644                throw new Zip64RequiredException(Zip64RequiredException
645                        .getEntryTooBigMessage(entry.entry));
646            }
647        }

649        /**
650         * Whether to addd a Zip64 extended information extra field to the
651         * local file header.
652         * <p/>
653         * <p>Returns true if</p>
654         * <p/>
655         * <ul>
656         * <li>mode is Always</li>
657         * <li>or we already know it is going to be needed</li>
658         * <li>or the size is unknown and we can ensure it won't hurt
659         * other implementations if we add it (i.e. we can erase its
660         * usage</li>
661         * </ul>
662         */
663        private boolean shouldAddZip64Extra(ZipArchiveEntry entry, Zip64Mode mode) {
664            return mode == Zip64Mode.Always
665                    || entry.getSize() >= ZIP64_MAGIC
666                    || entry.getCompressedSize() >= ZIP64_MAGIC
667                    || (entry.getSize() == ArchiveEntry.SIZE_UNKNOWN
668                    && raf != null && mode != Zip64Mode.Never);
669        }

711         * Writes bytes to ZIP entry.
712         *
713         * @param b      the byte array to write
714         * @param offset the start position to write from
715         * @param length the number of bytes to write
716         * @throws java.io.IOException on error
717         */
718        @Override
719        public void write(byte[] b, int offset, int length) throws IOException {
720            ZipUtil.checkRequestedFeatures(entry.entry);
721            entry.hasWritten = true;
722            if (entry.entry.getMethod() == DEFLATED) {
723                writeDeflated(b, offset, length);
724            } else {
725                writeOut(b, offset, length);
726                written += length;
727            }
728            crc.update(b, offset, length);
729            count(length);
730        }

732        /**
733         * write implementation for DEFLATED entries.
734         */
735        private void writeDeflated(byte[] b, int offset, int length)
736                throws IOException {
737            if (length > 0 && !def.finished()) {
738                entry.bytesRead += length;
739                if (length <= DEFLATER_BLOCK_SIZE) {
740                    def.setInput(b, offset, length);
741                    deflateUntilInputIsNeeded();
742                } else {
743                    final int fullblocks = length / DEFLATER_BLOCK_SIZE;
744                    for (int i = 0; i < fullblocks; i++) {
745                        def.setInput(b, offset + i * DEFLATER_BLOCK_SIZE,
746                                DEFLATER_BLOCK_SIZE);
747                        deflateUntilInputIsNeeded();
748                    }
749                    final int done = fullblocks * DEFLATER_BLOCK_SIZE;
750                    if (done < length) {
751                        def.setInput(b, offset + done, length - done);
752                        deflateUntilInputIsNeeded();
753                    }
754                }
755            }
756        }

762         * @throws java.io.IOException    if an I/O error occurs.

829        /**
830         * Writes the local file header entry
831         *
832         * @param ze the entry to write
833         * @throws java.io.IOException on error
834         */
835        protected void writeLocalFileHeader(ZipArchiveEntry ze) throws IOException {
836    
837            boolean encodable = zipEncoding.canEncode(ze.getName());
838            ByteBuffer name = getName(ze);
839    
840            if (createUnicodeExtraFields != UnicodeExtraFieldPolicy.NEVER) {
841                addUnicodeExtraFields(ze, encodable, name);
842            }
843    
844            offsets.put(ze, Long.valueOf(written));
845    
846            writeOut(LFH_SIG);
847            written += WORD;
848    
849            //store method in local variable to prevent multiple method calls
850            final int zipMethod = ze.getMethod();
851    
852            writeVersionNeededToExtractAndGeneralPurposeBits(zipMethod,
853                    !encodable
854                            && fallbackToUTF8,
855                    hasZip64Extra(ze));
856            written += WORD;
857    
858            // compression method
859            writeOut(ZipShort.getBytes(zipMethod));
860            written += SHORT;
861    
862            // last mod. time and date
863            writeOut(ZipUtil.toDosTime(ze.getTime()));
864            written += WORD;
865    
866            // CRC
867            // compressed length
868            // uncompressed length
869            entry.localDataStart = written;
870            if (zipMethod == DEFLATED || raf != null) {
871                writeOut(LZERO);
872                if (hasZip64Extra(entry.entry)) {
873                    // point to ZIP64 extended information extra field for
874                    // sizes, may get rewritten once sizes are known if
875                    // stream is seekable
876                    writeOut(ZipLong.ZIP64_MAGIC.getBytes());
877                    writeOut(ZipLong.ZIP64_MAGIC.getBytes());
878                } else {
879                    writeOut(LZERO);
880                    writeOut(LZERO);
881                }
882            } else {
883                writeOut(ZipLong.getBytes(ze.getCrc()));
884                byte[] size = ZipLong.ZIP64_MAGIC.getBytes();
885                if (!hasZip64Extra(ze)) {
886                    size = ZipLong.getBytes(ze.getSize());
887                }
888                writeOut(size);
889                writeOut(size);
890            }
891            // CheckStyle:MagicNumber OFF
892            written += 12;
893            // CheckStyle:MagicNumber ON
894    
895            // file name length
896            writeOut(ZipShort.getBytes(name.limit()));
897            written += SHORT;
898    
899            // extra field length
900            byte[] extra = ze.getLocalFileDataExtra();
901            writeOut(ZipShort.getBytes(extra.length));
902            written += SHORT;
903    
904            // file name
905            writeOut(name.array(), name.arrayOffset(), name.limit());
906            written += name.limit();
907    
908            // extra field
909            writeOut(extra);
910            written += extra.length;
911    
912            entry.dataStart = written;
913        }

915        /**
916         * Adds UnicodeExtra fields for name and file comment if mode is
917         * ALWAYS or the data cannot be encoded using the configured
918         * encoding.
919         */
920        private void addUnicodeExtraFields(ZipArchiveEntry ze, boolean encodable,
921                                           ByteBuffer name)
922                throws IOException {
923            if (createUnicodeExtraFields == UnicodeExtraFieldPolicy.ALWAYS
924                    || !encodable) {
925                ze.addExtraField(new UnicodePathExtraField(ze.getName(),
926                        name.array(),
927                        name.arrayOffset(),
928                        name.limit()));
929            }
930    
931            String comm = ze.getComment();
932            if (comm != null && !"".equals(comm)) {
933    
934                boolean commentEncodable = zipEncoding.canEncode(comm);
935    
936                if (createUnicodeExtraFields == UnicodeExtraFieldPolicy.ALWAYS
937                        || !commentEncodable) {
938                    ByteBuffer commentB = getEntryEncoding(ze).encode(comm);
939                    ze.addExtraField(new UnicodeCommentExtraField(comm,
940                            commentB.array(),
941                            commentB.arrayOffset(),
942                            commentB.limit())
943                    );
944                }
945            }
946        }

948        /**
949         * Writes the data descriptor entry.
950         *
951         * @param ze the entry to write
952         * @throws java.io.IOException on error
953         */
954        protected void writeDataDescriptor(ZipArchiveEntry ze) throws IOException {
955            if (ze.getMethod() != DEFLATED || raf != null) {
956                return;
957            }
958            writeOut(DD_SIG);
959            writeOut(ZipLong.getBytes(ze.getCrc()));
960            int sizeFieldSize = WORD;
961            if (!hasZip64Extra(ze)) {
962                writeOut(ZipLong.getBytes(ze.getCompressedSize()));
963                writeOut(ZipLong.getBytes(ze.getSize()));
964            } else {
965                sizeFieldSize = DWORD;
966                writeOut(ZipEightByteInteger.getBytes(ze.getCompressedSize()));
967                writeOut(ZipEightByteInteger.getBytes(ze.getSize()));
968            }
969            written += 2 * WORD + 2 * sizeFieldSize;
970        }

972        /**
973         * Writes the central file header entry.
974         *
975         * @param ze the entry to write
976         * @throws java.io.IOException    on error
977         * @throws Zip64RequiredException if the archive's size exceeds 4
978         *                                GByte and {@link Zip64Mode #setUseZip64} is {@link
979         *                                Zip64Mode#Never}.
980         */
981        protected void writeCentralFileHeader(ZipArchiveEntry ze) throws IOException {
982            writeOut(CFH_SIG);
983            written += WORD;
984    
985            final long lfhOffset = offsets.get(ze).longValue();
986            final boolean needsZip64Extra = hasZip64Extra(ze)
987                    || ze.getCompressedSize() >= ZIP64_MAGIC
988                    || ze.getSize() >= ZIP64_MAGIC
989                    || lfhOffset >= ZIP64_MAGIC;
990    
991            if (needsZip64Extra && zip64Mode == Zip64Mode.Never) {
992                // must be the offset that is too big, otherwise an
993                // exception would have been throw in putArchiveEntry or
994                // closeArchiveEntry
995                throw new Zip64RequiredException(Zip64RequiredException
996                        .ARCHIVE_TOO_BIG_MESSAGE);
997            }
998    
999            handleZip64Extra(ze, lfhOffset, needsZip64Extra);
1000    
1001            // version made by
1002            // CheckStyle:MagicNumber OFF
1003            writeOut(ZipShort.getBytes((ze.getPlatform() << 8) |
1004                    (!hasUsedZip64 ? DATA_DESCRIPTOR_MIN_VERSION
1005                            : ZIP64_MIN_VERSION)));
1006            written += SHORT;
1007    
1008            final int zipMethod = ze.getMethod();
1009            final boolean encodable = zipEncoding.canEncode(ze.getName());
1010            writeVersionNeededToExtractAndGeneralPurposeBits(zipMethod,
1011                    !encodable
1012                            && fallbackToUTF8,
1013                    needsZip64Extra);
1014            written += WORD;
1015    
1016            // compression method
1017            writeOut(ZipShort.getBytes(zipMethod));
1018            written += SHORT;
1019    
1020            // last mod. time and date
1021            writeOut(ZipUtil.toDosTime(ze.getTime()));
1022            written += WORD;
1023    
1024            // CRC
1025            // compressed length
1026            // uncompressed length
1027            writeOut(ZipLong.getBytes(ze.getCrc()));
1028            if (ze.getCompressedSize() >= ZIP64_MAGIC
1029                    || ze.getSize() >= ZIP64_MAGIC) {
1030                writeOut(ZipLong.ZIP64_MAGIC.getBytes());
1031                writeOut(ZipLong.ZIP64_MAGIC.getBytes());
1032            } else {
1033                writeOut(ZipLong.getBytes(ze.getCompressedSize()));
1034                writeOut(ZipLong.getBytes(ze.getSize()));
1035            }
1036            // CheckStyle:MagicNumber OFF
1037            written += 12;
1038            // CheckStyle:MagicNumber ON
1039    
1040            ByteBuffer name = getName(ze);
1041    
1042            writeOut(ZipShort.getBytes(name.limit()));
1043            written += SHORT;
1044    
1045            // extra field length
1046            byte[] extra = ze.getCentralDirectoryExtra();
1047            writeOut(ZipShort.getBytes(extra.length));
1048            written += SHORT;
1049    
1050            // file comment length
1051            String comm = ze.getComment();
1052            if (comm == null) {
1053                comm = "";
1054            }
1055    
1056            ByteBuffer commentB = getEntryEncoding(ze).encode(comm);
1057    
1058            writeOut(ZipShort.getBytes(commentB.limit()));
1059            written += SHORT;
1060    
1061            // disk number start
1062            writeOut(ZERO);
1063            written += SHORT;
1064    
1065            // internal file attributes
1066            writeOut(ZipShort.getBytes(ze.getInternalAttributes()));
1067            written += SHORT;
1068    
1069            // external file attributes
1070            writeOut(ZipLong.getBytes(ze.getExternalAttributes()));
1071            written += WORD;
1072    
1073            // relative offset of LFH
1074            writeOut(ZipLong.getBytes(Math.min(lfhOffset, ZIP64_MAGIC)));
1075            written += WORD;
1076    
1077            // file name
1078            writeOut(name.array(), name.arrayOffset(), name.limit());
1079            written += name.limit();
1080    
1081            // extra field
1082            writeOut(extra);
1083            written += extra.length;
1084    
1085            // file comment
1086            writeOut(commentB.array(), commentB.arrayOffset(), commentB.limit());
1087            written += commentB.limit();
1088        }

1090        /**
1091         * If the entry needs Zip64 extra information inside the central
1092         * directory then configure its data.
1093         */
1094        private void handleZip64Extra(ZipArchiveEntry ze, long lfhOffset,
1095                                      boolean needsZip64Extra) {
1096            if (needsZip64Extra) {
1097                Zip64ExtendedInformationExtraField z64 = getZip64Extra(ze);
1098                if (ze.getCompressedSize() >= ZIP64_MAGIC
1099                        || ze.getSize() >= ZIP64_MAGIC) {
1100                    z64.setCompressedSize(new ZipEightByteInteger(ze.getCompressedSize()));
1101                    z64.setSize(new ZipEightByteInteger(ze.getSize()));
1102                } else {
1103                    // reset value that may have been set for LFH
1104                    z64.setCompressedSize(null);
1105                    z64.setSize(null);
1106                }
1107                if (lfhOffset >= ZIP64_MAGIC) {
1108                    z64.setRelativeHeaderOffset(new ZipEightByteInteger(lfhOffset));
1109                }
1110                ze.setExtra();
1111            }
1112        }

1114        /**
1115         * Writes the &quot;End of central dir record&quot;.
1116         *
1117         * @throws java.io.IOException    on error
1118         * @throws Zip64RequiredException if the archive's size exceeds 4
1119         *                                GByte or there are more than 65535 entries inside the archive
1120         *                                and {@link Zip64Mode #setUseZip64} is {@link Zip64Mode#Never}.
1121         */
1122        protected void writeCentralDirectoryEnd() throws IOException {
1123            writeOut(EOCD_SIG);
1124    
1125            // disk numbers
1126            writeOut(ZERO);
1127            writeOut(ZERO);
1128    
1129            // number of entries
1130            int numberOfEntries = entries.size();
1131            if (numberOfEntries > ZIP64_MAGIC_SHORT
1132                    && zip64Mode == Zip64Mode.Never) {
1133                throw new Zip64RequiredException(Zip64RequiredException
1134                        .TOO_MANY_ENTRIES_MESSAGE);
1135            }
1136            if (cdOffset > ZIP64_MAGIC && zip64Mode == Zip64Mode.Never) {
1137                throw new Zip64RequiredException(Zip64RequiredException
1138                        .ARCHIVE_TOO_BIG_MESSAGE);
1139            }
1140    
1141            byte[] num = ZipShort.getBytes(Math.min(numberOfEntries,
1142                    ZIP64_MAGIC_SHORT));
1143            writeOut(num);
1144            writeOut(num);
1145    
1146            // length and location of CD
1147            writeOut(ZipLong.getBytes(Math.min(cdLength, ZIP64_MAGIC)));
1148            writeOut(ZipLong.getBytes(Math.min(cdOffset, ZIP64_MAGIC)));
1149    
1150            // ZIP file comment
1151            ByteBuffer data = this.zipEncoding.encode(comment);
1152            writeOut(ZipShort.getBytes(data.limit()));
1153            writeOut(data.array(), data.arrayOffset(), data.limit());
1154        }

1156        private static final byte[] ONE = ZipLong.getBytes(1L);

1158        /**
1159         * Writes the &quot;ZIP64 End of central dir record&quot; and
1160         * &quot;ZIP64 End of central dir locator&quot;.
1161         *
1162         * @throws java.io.IOException on error
1163         */
1164        protected void writeZip64CentralDirectory() throws IOException {
1165            if (zip64Mode == Zip64Mode.Never) {
1166                return;
1167            }
1168    
1169            if (!hasUsedZip64
1170                    && (cdOffset >= ZIP64_MAGIC || cdLength >= ZIP64_MAGIC
1171                    || entries.size() >= ZIP64_MAGIC_SHORT)) {
1172                // actually "will use"
1173                hasUsedZip64 = true;
1174            }
1175    
1176            if (!hasUsedZip64) {
1177                return;
1178            }
1179    
1180            long offset = written;
1181    
1182            writeOut(ZIP64_EOCD_SIG);
1183            // size, we don't have any variable length as we don't support
1184            // the extensible data sector, yet
1185            writeOut(ZipEightByteInteger
1186                    .getBytes(SHORT   /* version made by */
1187                            + SHORT /* version needed to extract */
1188                            + WORD  /* disk number */
1189                            + WORD  /* disk with central directory */
1190                            + DWORD /* number of entries in CD on this disk */
1191                            + DWORD /* total number of entries */
1192                            + DWORD /* size of CD */
1193                            + DWORD /* offset of CD */
1194                    ));
1195    
1196            // version made by and version needed to extract
1197            writeOut(ZipShort.getBytes(ZIP64_MIN_VERSION));
1198            writeOut(ZipShort.getBytes(ZIP64_MIN_VERSION));
1199    
1200            // disk numbers - four bytes this time
1201            writeOut(LZERO);
1202            writeOut(LZERO);
1203    
1204            // number of entries
1205            byte[] num = ZipEightByteInteger.getBytes(entries.size());
1206            writeOut(num);
1207            writeOut(num);
1208    
1209            // length and location of CD
1210            writeOut(ZipEightByteInteger.getBytes(cdLength));
1211            writeOut(ZipEightByteInteger.getBytes(cdOffset));
1212    
1213            // no "zip64 extensible data sector" for now
1214    
1215            // and now the "ZIP64 end of central directory locator"
1216            writeOut(ZIP64_EOCD_LOC_SIG);
1217    
1218            // disk number holding the ZIP64 EOCD record
1219            writeOut(LZERO);
1220            // relative offset of ZIP64 EOCD record
1221            writeOut(ZipEightByteInteger.getBytes(offset));
1222            // total number of disks
1223            writeOut(ONE);
1224        }

1259        private void writeVersionNeededToExtractAndGeneralPurposeBits(final int
1260                                                                              zipMethod,
1261                                                                      final boolean
1262                                                                              utfFallback,
1263                                                                      final boolean
1264                                                                              zip64)
1265                throws IOException {
1266    
1267            // CheckStyle:MagicNumber OFF
1268            int versionNeededToExtract = INITIAL_VERSION;
1269            GeneralPurposeBit b = new GeneralPurposeBit();
1270            b.useUTF8ForNames(useUTF8Flag || utfFallback);
1271            if (zipMethod == DEFLATED && raf == null) {
1272                // requires version 2 as we are going to store length info
1273                // in the data descriptor
1274                versionNeededToExtract = DATA_DESCRIPTOR_MIN_VERSION;
1275                b.useDataDescriptor(true);
1276            }
1277            if (zip64) {
1278                versionNeededToExtract = ZIP64_MIN_VERSION;
1279            }
1280            // CheckStyle:MagicNumber ON
1281    
1282            // version needed to extract
1283            writeOut(ZipShort.getBytes(versionNeededToExtract));
1284            // general purpose bit flag
1285            writeOut(b.encode());
1286        }

1290         * create a new one and add it to the entry.
1291         */
1292        private Zip64ExtendedInformationExtraField
1293        getZip64Extra(ZipArchiveEntry ze) {
1294            if (entry != null) {
1295                entry.causedUseOfZip64 = !hasUsedZip64;
1296            }
1297            hasUsedZip64 = true;
1298            Zip64ExtendedInformationExtraField z64 =
1299                    (Zip64ExtendedInformationExtraField)
1300                            ze.getExtraField(Zip64ExtendedInformationExtraField
1301                                    .HEADER_ID);
1302            if (z64 == null) {
1303                /*
1304                  System.err.println("Adding z64 for " + ze.getName()
1305                  + ", method: " + ze.getMethod()
1306                  + " (" + (ze.getMethod() == STORED) + ")"
1307                  + ", raf: " + (raf != null));
1308                */
1309                z64 = new Zip64ExtendedInformationExtraField();
1310            }
1311    
1312            // even if the field is there already, make sure it is the first one
1313            ze.addAsFirstExtraField(z64);
1314    
1315            return z64;
