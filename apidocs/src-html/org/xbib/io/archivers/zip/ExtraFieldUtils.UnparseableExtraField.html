015        private static final int WORD = 4;
016    
017        /**
018         * Static registry of known extra fields.
019         */
020        private static final Map<ZipShort, Class<?>> implementations;
021    
022        static {
023            implementations = new HashMap<ZipShort, Class<?>>();
024            register(AsiExtraField.class);
025            register(JarMarker.class);
026            register(UnicodePathExtraField.class);
027            register(UnicodeCommentExtraField.class);
028            register(Zip64ExtendedInformationExtraField.class);
029        }

031        /**
032         * Register a ZipExtraField implementation.
033         * <p/>
034         * <p>The given class must have a no-arg constructor and implement
035         * the {@link ZipExtraField ZipExtraField interface}.</p>
036         *
037         * @param c the class to register
038         */
039        public static void register(Class<?> c) {
040            try {
041                ZipExtraField ze = (ZipExtraField) c.newInstance();
042                implementations.put(ze.getHeaderId(), c);
043            } catch (ClassCastException cc) {
044                throw new RuntimeException(c + " doesn\'t implement ZipExtraField");
045            } catch (InstantiationException ie) {
046                throw new RuntimeException(c + " is not a concrete class");
047            } catch (IllegalAccessException ie) {
048                throw new RuntimeException(c + "\'s no-arg constructor is not public");
049            }
050        }
051    
052        /**
053         * Create an instance of the approriate ExtraField, falls back to
054         * {@link UnrecognizedExtraField UnrecognizedExtraField}.
055         *
056         * @param headerId the header identifier
057         * @return an instance of the appropiate ExtraField
058         * @throws InstantiationException if unable to instantiate the class
059         * @throws IllegalAccessException if not allowed to instatiate the class
060         */
061        public static ZipExtraField createExtraField(ZipShort headerId)
062                throws InstantiationException, IllegalAccessException {
063            Class<?> c = implementations.get(headerId);
064            if (c != null) {
065                return (ZipExtraField) c.newInstance();
066            }
067            UnrecognizedExtraField u = new UnrecognizedExtraField();
068            u.setHeaderId(headerId);
069            return u;
070        }

072        /**
073         * Split the array into ExtraFields and populate them with the
074         * given data as local file data, throwing an exception if the
075         * data cannot be parsed.
076         *
077         * @param data an array of bytes as it appears in local file data
078         * @return an array of ExtraFields
079         * @throws java.util.zip.ZipException on error
080         */
081        public static ZipExtraField[] parse(byte[] data) throws ZipException {
082            return parse(data, true, UnparseableExtraField.THROW);
083        }
084    
085        /**
086         * Split the array into ExtraFields and populate them with the
087         * given data, throwing an exception if the data cannot be parsed.
088         *
089         * @param data  an array of bytes
090         * @param local whether data originates from the local file data
091         *              or the central directory
092         * @return an array of ExtraFields
093         * @throws java.util.zip.ZipException on error
094         */
095        public static ZipExtraField[] parse(byte[] data, boolean local)
096                throws ZipException {
097            return parse(data, local, UnparseableExtraField.THROW);
098        }
099    
100        /**
101         * Split the array into ExtraFields and populate them with the
102         * given data.
103         *
104         * @param data              an array of bytes
105         * @param local             whether data originates from the local file data
106         *                          or the central directory
107         * @param onUnparseableData what to do if the extra field data
108         *                          cannot be parsed.
109         * @return an array of ExtraFields
110         * @throws java.util.zip.ZipException on error
111         */
112        public static ZipExtraField[] parse(byte[] data, boolean local,
113                                            UnparseableExtraField onUnparseableData)
114                throws ZipException {
115            List<ZipExtraField> v = new ArrayList<ZipExtraField>();
116            int start = 0;
117            LOOP:
118            while (start <= data.length - WORD) {
119                ZipShort headerId = new ZipShort(data, start);
120                int length = (new ZipShort(data, start + 2)).getValue();
121                if (start + WORD + length > data.length) {
122                    switch (onUnparseableData.getKey()) {
123                        case UnparseableExtraField.THROW_KEY:
124                            throw new ZipException("bad extra field starting at "
125                                    + start + ".  Block length of "
126                                    + length + " bytes exceeds remaining"
127                                    + " data of "
128                                    + (data.length - start - WORD)
129                                    + " bytes.");
130                        case UnparseableExtraField.READ_KEY:
131                            UnparseableExtraFieldData field =
132                                    new UnparseableExtraFieldData();
133                            if (local) {
134                                field.parseFromLocalFileData(data, start,
135                                        data.length - start);
136                            } else {
137                                field.parseFromCentralDirectoryData(data, start,
138                                        data.length - start);
139                            }
140                            v.add(field);
141                            //$FALL-THROUGH$
142                        case UnparseableExtraField.SKIP_KEY:
143                            // since we cannot parse the data we must assume
144                            // the extra field consumes the whole rest of the
145                            // available data
146                            break LOOP;
147                        default:
148                            throw new ZipException("unknown UnparseableExtraField key: "
149                                    + onUnparseableData.getKey());
150                    }
151                }
152                try {
153                    ZipExtraField ze = createExtraField(headerId);
154                    if (local) {
155                        ze.parseFromLocalFileData(data, start + WORD, length);
156                    } else {
157                        ze.parseFromCentralDirectoryData(data, start + WORD,
158                                length);
159                    }
160                    v.add(ze);
161                } catch (InstantiationException ie) {
162                    throw new ZipException(ie.getMessage());
163                } catch (IllegalAccessException iae) {
164                    throw new ZipException(iae.getMessage());
165                }
166                start += (length + WORD);
167            }
168    
169            ZipExtraField[] result = new ZipExtraField[v.size()];
170            return v.toArray(result);
171        }

173        /**
174         * Merges the local file data fields of the given ZipExtraFields.
175         *
176         * @param data an array of ExtraFiles
177         * @return an array of bytes
178         */
179        public static byte[] mergeLocalFileDataData(ZipExtraField[] data) {
180            final boolean lastIsUnparseableHolder = data.length > 0
181                    && data[data.length - 1] instanceof UnparseableExtraFieldData;
182            int regularExtraFieldCount =
183                    lastIsUnparseableHolder ? data.length - 1 : data.length;
184    
185            int sum = WORD * regularExtraFieldCount;
186            for (ZipExtraField element : data) {
187                sum += element.getLocalFileDataLength().getValue();
188            }
189    
190            byte[] result = new byte[sum];
191            int start = 0;
192            for (int i = 0; i < regularExtraFieldCount; i++) {
193                System.arraycopy(data[i].getHeaderId().getBytes(),
194                        0, result, start, 2);
195                System.arraycopy(data[i].getLocalFileDataLength().getBytes(),
196                        0, result, start + 2, 2);
197                byte[] local = data[i].getLocalFileDataData();
198                System.arraycopy(local, 0, result, start + WORD, local.length);
199                start += (local.length + WORD);
200            }
201            if (lastIsUnparseableHolder) {
202                byte[] local = data[data.length - 1].getLocalFileDataData();
203                System.arraycopy(local, 0, result, start, local.length);
204            }
205            return result;
206        }

208        /**
209         * Merges the central directory fields of the given ZipExtraFields.
210         *
211         * @param data an array of ExtraFields
212         * @return an array of bytes
213         */
214        public static byte[] mergeCentralDirectoryData(ZipExtraField[] data) {
215            final boolean lastIsUnparseableHolder = data.length > 0
216                    && data[data.length - 1] instanceof UnparseableExtraFieldData;
217            int regularExtraFieldCount =
218                    lastIsUnparseableHolder ? data.length - 1 : data.length;
219    
220            int sum = WORD * regularExtraFieldCount;
221            for (ZipExtraField element : data) {
222                sum += element.getCentralDirectoryLength().getValue();
223            }
224            byte[] result = new byte[sum];
225            int start = 0;
226            for (int i = 0; i < regularExtraFieldCount; i++) {
227                System.arraycopy(data[i].getHeaderId().getBytes(),
228                        0, result, start, 2);
229                System.arraycopy(data[i].getCentralDirectoryLength().getBytes(),
230                        0, result, start + 2, 2);
231                byte[] local = data[i].getCentralDirectoryData();
232                System.arraycopy(local, 0, result, start + WORD, local.length);
233                start += (local.length + WORD);
234            }
235            if (lastIsUnparseableHolder) {
236                byte[] local = data[data.length - 1].getCentralDirectoryData();
237                System.arraycopy(local, 0, result, start, local.length);
238            }
239            return result;
240        }

242        /**
243         * "enum" for the possible actions to take if the extra field
244         * cannot be parsed.
245         */
246        public static final class UnparseableExtraField {
247            /**
248             * Key for "throw an exception" action.
249             */
250            public static final int THROW_KEY = 0;
251            /**
252             * Key for "skip" action.
253             */
254            public static final int SKIP_KEY = 1;
255            /**
256             * Key for "read" action.
257             */
258            public static final int READ_KEY = 2;
259    
260            /**
261             * Throw an exception if field cannot be parsed.
262             */
263            public static final UnparseableExtraField THROW
264                    = new UnparseableExtraField(THROW_KEY);
265    
266            /**
267             * Skip the extra field entirely and don't make its data
268             * available - effectively removing the extra field data.
269             */
270            public static final UnparseableExtraField SKIP
271                    = new UnparseableExtraField(SKIP_KEY);
272    
273            /**
274             * Read the extra field data into an instance of {@link
275             * UnparseableExtraFieldData UnparseableExtraFieldData}.
276             */
277            public static final UnparseableExtraField READ
278                    = new UnparseableExtraField(READ_KEY);
279    
280            private final int key;
281    
282            private UnparseableExtraField(int k) {
283                key = k;
284            }
285    
286            /**
287             * Key of the action to take.
288             */
289            public int getKey() {
290                return key;
291            }
292        }
