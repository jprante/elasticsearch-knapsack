027    /**
028     * Replacement for <code>java.util.ZipFile</code>.
029     * <p/>
030     * <p>This class adds support for file name encodings other than UTF-8
031     * (which is required to work on ZIP files created by native zip tools
032     * and is able to skip a preamble like the one found in self
033     * extracting archives.  Furthermore it returns instances of
034     * <code>org.apache.commons.compress.archivers.zip.ZipArchiveEntry</code>
035     * instead of <code>java.util.zip.ZipEntry</code>.</p>
036     * <p/>
037     * <p>It doesn't extend <code>java.util.zip.ZipFile</code> as it would
038     * have to reimplement all methods anyway.  Like
039     * <code>java.util.ZipFile</code>, it uses RandomAccessFile under the
040     * covers and supports compressed and uncompressed entries. This code
041     * also transparently supports Zip64
042     * extensions and thus individual entries and archives larger than 4
043     * GB or with more than 65536 entries.</p>
044     * <p/>
045     * <p>The method signatures mimic the ones of
046     * <code>java.util.zip.ZipFile</code>, with a couple of exceptions:
047     * <p/>
048     * <ul>
049     * <li>There is no getName method.</li>
050     * <li>entries has been renamed to getEntries.</li>
051     * <li>getEntries and getEntry return
052     * <code>org.apache.commons.compress.archivers.zip.ZipArchiveEntry</code>
053     * instances.</li>
054     * <li>close is allowed to throw IOException.</li>
055     * </ul>
056     */
057    public class ZipFile {

059        static final int NIBLET_MASK = 0x0f;
060        static final int BYTE_SHIFT = 8;
061        private static final int POS_0 = 0;

063        private static final int POS_2 = 2;
064        private static final int POS_3 = 3;

164        /**
165         * Opens the given file for reading, assuming the specified
166         * encoding for file names.
167         *
168         * @param f                     the archive.
169         * @param encoding              the encoding to use for file names, use null
170         *                              for the platform's default encoding
171         * @param useUnicodeExtraFields whether to use InfoZIP Unicode
172         *                              Extra Fields (if present) to set the file names.
173         * @throws java.io.IOException if an error occurs while reading the file.
174         */
175        public ZipFile(File f, String encoding, boolean useUnicodeExtraFields)
176                throws IOException {
177            this.archiveName = f.getAbsolutePath();
178            this.encoding = encoding;
179            this.zipEncoding = ZipEncodingHelper.getZipEncoding(encoding);
180            this.useUnicodeExtraFields = useUnicodeExtraFields;
181            archive = new RandomAccessFile(f, "r");
182            boolean success = false;
183            try {
184                Map<ZipArchiveEntry, NameAndComment> entriesWithoutUTF8Flag =
185                        populateFromCentralDirectory();
186                resolveLocalFileHeaderData(entriesWithoutUTF8Flag);
187                success = true;
188            } finally {
189                if (!success) {
190                    try {
191                        closed = true;
192                        archive.close();
193                    } catch (IOException e2) {
194                        // swallow, throw the original exception instead
195                    }
196                }
197            }
198        }

288        /**
289         * Returns an InputStream for reading the contents of the given entry.
290         *
291         * @param ze the entry to get the stream for.
292         * @return a stream to read the entry from.
293         * @throws java.io.IOException        if unable to create an input stream from the zipenty
294         * @throws java.util.zip.ZipException if the zipentry uses an unsupported feature
295         */
296        public InputStream getInputStream(ZipArchiveEntry ze)
297                throws IOException, ZipException {
298            OffsetEntry offsetEntry = entries.get(ze);
299            if (offsetEntry == null) {
300                return null;
301            }
302            ZipUtil.checkRequestedFeatures(ze);
303            long start = offsetEntry.dataOffset;
304            BoundedInputStream bis =
305                    new BoundedInputStream(start, ze.getCompressedSize());
306            switch (ze.getMethod()) {
307                case ZipEntry.STORED:
308                    return bis;
309                case ZipEntry.DEFLATED:
310                    bis.addDummy();
311                    final Inflater inflater = new Inflater(true);
312                    return new InflaterInputStream(bis, inflater) {
313                        @Override
314                        public void close() throws IOException {
315                            super.close();
316                            inflater.end();
317                        }
318                    };
319                default:
320                    throw new ZipException("Found unsupported compression method "
321                            + ze.getMethod());
322            }
323        }

326         * Ensures that the close method of this zipfile is called when
327         * there are no more references to it.
328         *
329         * @see #close()
330         */
331        @Override
332        protected void finalize() throws Throwable {
333            try {
334                if (!closed) {
335                    System.err.println("Cleaning up unclosed ZipFile for archive "
336                            + archiveName);
337                    close();
338                }
339            } finally {
340                super.finalize();
341            }
342        }

344        /**
345         * Length of a "central directory" entry structure without file
346         * name, extra fields or comment.
347         */
348        private static final int CFH_LEN =
349            /* version made by                 */ SHORT
350            /* version needed to extract       */ + SHORT
351            /* general purpose bit flag        */ + SHORT
352            /* compression method              */ + SHORT
353            /* last mod file time              */ + SHORT
354            /* last mod file date              */ + SHORT
355            /* crc-32                          */ + WORD
356            /* compressed size                 */ + WORD
357            /* uncompressed size               */ + WORD
358            /* filename length                 */ + SHORT
359            /* extra field length              */ + SHORT
360            /* file comment length             */ + SHORT
361            /* disk number start               */ + SHORT
362            /* internal file attributes        */ + SHORT
363            /* external file attributes        */ + WORD
364            /* relative offset of local header */ + WORD;
365    
366        private static final long CFH_SIG =
367                ZipLong.getValue(ZipArchiveOutputStream.CFH_SIG);
368    
369        /**
370         * Reads the central directory of the given archive and populates
371         * the internal tables with ZipArchiveEntry instances.
372         * <p/>
373         * <p>The ZipArchiveEntrys will know all data that can be obtained from
374         * the central directory alone, but not the data that requires the
375         * local file header or additional data to be read.</p>
376         *
377         * @return a map of zipentries that didn't have the language
378         * encoding flag set when read.
379         */
380        private Map<ZipArchiveEntry, NameAndComment> populateFromCentralDirectory()
381                throws IOException {
382            HashMap<ZipArchiveEntry, NameAndComment> noUTF8Flag =
383                    new HashMap<ZipArchiveEntry, NameAndComment>();
384    
385            positionAtCentralDirectory();
386    
387            byte[] signatureBytes = new byte[WORD];
388            archive.readFully(signatureBytes);
389            long sig = ZipLong.getValue(signatureBytes);
390    
391            if (sig != CFH_SIG && startsWithLocalFileHeader()) {
392                throw new IOException("central directory is empty, can't expand"
393                        + " corrupt archive.");
394            }
395    
396            while (sig == CFH_SIG) {
397                readCentralDirectoryEntry(noUTF8Flag);
398                archive.readFully(signatureBytes);
399                sig = ZipLong.getValue(signatureBytes);
400            }
401            return noUTF8Flag;
402        }
403    
404        /**
405         * Reads an individual entry of the central directory, creats an
406         * ZipArchiveEntry from it and adds it to the global maps.
407         *
408         * @param noUTF8Flag map used to collect entries that don't have
409         *                   their UTF-8 flag set and whose name will be set by data read
410         *                   from the local file header later.  The current entry may be
411         *                   added to this map.
412         */
413        private void
414        readCentralDirectoryEntry(Map<ZipArchiveEntry, NameAndComment> noUTF8Flag)
415                throws IOException {
416            byte[] cfh = new byte[CFH_LEN];
417    
418            archive.readFully(cfh);
419            int off = 0;
420            ZipArchiveEntry ze = new ZipArchiveEntry();
421    
422            int versionMadeBy = ZipShort.getValue(cfh, off);
423            off += SHORT;
424            ze.setPlatform((versionMadeBy >> BYTE_SHIFT) & NIBLET_MASK);
425    
426            off += SHORT; // skip version info
427    
428            final GeneralPurposeBit gpFlag = GeneralPurposeBit.parse(cfh, off);
429            final boolean hasUTF8Flag = gpFlag.usesUTF8ForNames();
430            final ZipEncoding entryEncoding =
431                    hasUTF8Flag ? ZipEncodingHelper.UTF8_ZIP_ENCODING : zipEncoding;
432            ze.setGeneralPurposeBit(gpFlag);
433    
434            off += SHORT;
435    
436            ze.setMethod(ZipShort.getValue(cfh, off));
437            off += SHORT;
438    
439            long time = ZipUtil.dosToJavaTime(ZipLong.getValue(cfh, off));
440            ze.setTime(time);
441            off += WORD;
442    
443            ze.setCrc(ZipLong.getValue(cfh, off));
444            off += WORD;
445    
446            ze.setCompressedSize(ZipLong.getValue(cfh, off));
447            off += WORD;
448    
449            ze.setSize(ZipLong.getValue(cfh, off));
450            off += WORD;
451    
452            int fileNameLen = ZipShort.getValue(cfh, off);
453            off += SHORT;
454    
455            int extraLen = ZipShort.getValue(cfh, off);
456            off += SHORT;
457    
458            int commentLen = ZipShort.getValue(cfh, off);
459            off += SHORT;
460    
461            int diskStart = ZipShort.getValue(cfh, off);
462            off += SHORT;
463    
464            ze.setInternalAttributes(ZipShort.getValue(cfh, off));
465            off += SHORT;
466    
467            ze.setExternalAttributes(ZipLong.getValue(cfh, off));
468            off += WORD;
469    
470            byte[] fileName = new byte[fileNameLen];
471            archive.readFully(fileName);
472            ze.setName(entryEncoding.decode(fileName), fileName);
473    
474            // LFH offset,
475            OffsetEntry offset = new OffsetEntry();
476            offset.headerOffset = ZipLong.getValue(cfh, off);
477            // data offset will be filled later
478            entries.put(ze, offset);
479    
480            nameMap.put(ze.getName(), ze);
481    
482            byte[] cdExtraData = new byte[extraLen];
483            archive.readFully(cdExtraData);
484            ze.setCentralDirectoryExtra(cdExtraData);
485    
486            setSizesAndOffsetFromZip64Extra(ze, offset, diskStart);
487    
488            byte[] comment = new byte[commentLen];
489            archive.readFully(comment);
490            ze.setComment(entryEncoding.decode(comment));
491    
492            if (!hasUTF8Flag && useUnicodeExtraFields) {
493                noUTF8Flag.put(ze, new NameAndComment(fileName, comment));
494            }
495        }

497        /**
498         * If the entry holds a Zip64 extended information extra field,
499         * read sizes from there if the entry's sizes are set to
500         * 0xFFFFFFFFF, do the same for the offset of the local file
501         * header.
502         * <p/>
503         * <p>Ensures the Zip64 extra either knows both compressed and
504         * uncompressed size or neither of both as the internal logic in
505         * ExtraFieldUtils forces the field to create local header data
506         * even if they are never used - and here a field with only one
507         * size would be invalid.</p>
508         */
509        private void setSizesAndOffsetFromZip64Extra(ZipArchiveEntry ze,
510                                                     OffsetEntry offset,
511                                                     int diskStart)
512                throws IOException {
513            Zip64ExtendedInformationExtraField z64 =
514                    (Zip64ExtendedInformationExtraField)
515                            ze.getExtraField(Zip64ExtendedInformationExtraField.HEADER_ID);
516            if (z64 != null) {
517                boolean hasUncompressedSize = ze.getSize() == ZIP64_MAGIC;
518                boolean hasCompressedSize = ze.getCompressedSize() == ZIP64_MAGIC;
519                boolean hasRelativeHeaderOffset =
520                        offset.headerOffset == ZIP64_MAGIC;
521                z64.reparseCentralDirectoryData(hasUncompressedSize,
522                        hasCompressedSize,
523                        hasRelativeHeaderOffset,
524                        diskStart == ZIP64_MAGIC_SHORT);
525    
526                if (hasUncompressedSize) {
527                    ze.setSize(z64.getSize().getLongValue());
528                } else if (hasCompressedSize) {
529                    z64.setSize(new ZipEightByteInteger(ze.getSize()));
530                }
531    
532                if (hasCompressedSize) {
533                    ze.setCompressedSize(z64.getCompressedSize().getLongValue());
534                } else if (hasUncompressedSize) {
535                    z64.setCompressedSize(new ZipEightByteInteger(ze.getCompressedSize()));
536                }
537    
538                if (hasRelativeHeaderOffset) {
539                    offset.headerOffset =
540                            z64.getRelativeHeaderOffset().getLongValue();
541                }
542            }
543        }

637        /**
638         * Searches for either the &quot;Zip64 end of central directory
639         * locator&quot; or the &quot;End of central dir record&quot;, parses
640         * it and positions the stream at the first central directory
641         * record.
642         */
643        private void positionAtCentralDirectory()
644                throws IOException {
645            boolean found = tryToLocateSignature(MIN_EOCD_SIZE + ZIP64_EOCDL_LENGTH,
646                    MAX_EOCD_SIZE + ZIP64_EOCDL_LENGTH,
647                    ZipArchiveOutputStream
648                            .ZIP64_EOCD_LOC_SIG);
649            if (!found) {
650                // not a ZIP64 archive
651                positionAtCentralDirectory32();
652            } else {
653                positionAtCentralDirectory64();
654            }
655        }
656    
657        /**
658         * Parses the &quot;Zip64 end of central directory locator&quot;,
659         * finds the &quot;Zip64 end of central directory record&quot; using the
660         * parsed information, parses that and positions the stream at the
661         * first central directory record.
662         */
663        private void positionAtCentralDirectory64()
664                throws IOException {
665            skipBytes(ZIP64_EOCDL_LOCATOR_OFFSET);
666            byte[] zip64EocdOffset = new byte[DWORD];
667            archive.readFully(zip64EocdOffset);
668            archive.seek(ZipEightByteInteger.getLongValue(zip64EocdOffset));
669            byte[] sig = new byte[WORD];
670            archive.readFully(sig);
671            if (sig[POS_0] != ZipArchiveOutputStream.ZIP64_EOCD_SIG[POS_0]
672                    || sig[POS_1] != ZipArchiveOutputStream.ZIP64_EOCD_SIG[POS_1]
673                    || sig[POS_2] != ZipArchiveOutputStream.ZIP64_EOCD_SIG[POS_2]
674                    || sig[POS_3] != ZipArchiveOutputStream.ZIP64_EOCD_SIG[POS_3]
675                    ) {
676                throw new ZipException("archive's ZIP64 end of central "
677                        + "directory locator is corrupt.");
678            }
679            skipBytes(ZIP64_EOCD_CFD_LOCATOR_OFFSET
680                    - WORD /* signature has already been read */);
681            byte[] cfdOffset = new byte[DWORD];
682            archive.readFully(cfdOffset);
683            archive.seek(ZipEightByteInteger.getLongValue(cfdOffset));
684        }

686        /**
687         * Searches for the &quot;End of central dir record&quot;, parses
688         * it and positions the stream at the first central directory
689         * record.
690         */
691        private void positionAtCentralDirectory32()
692                throws IOException {
693            boolean found = tryToLocateSignature(MIN_EOCD_SIZE, MAX_EOCD_SIZE,
694                    ZipArchiveOutputStream.EOCD_SIG);
695            if (!found) {
696                throw new ZipException("archive is not a ZIP archive");
697            }
698            skipBytes(CFD_LOCATOR_OFFSET);
699            byte[] cfdOffset = new byte[WORD];
700            archive.readFully(cfdOffset);
701            archive.seek(ZipLong.getValue(cfdOffset));
702        }

704        /**
705         * Searches the archive backwards from minDistance to maxDistance
706         * for the given signature, positions the RandomaccessFile right
707         * at the signature if it has been found.
708         */
709        private boolean tryToLocateSignature(long minDistanceFromEnd,
710                                             long maxDistanceFromEnd,
711                                             byte[] sig) throws IOException {
712            boolean found = false;
713            long off = archive.length() - minDistanceFromEnd;
714            final long stopSearching =
715                    Math.max(0L, archive.length() - maxDistanceFromEnd);
716            if (off >= 0) {
717                for (; off >= stopSearching; off--) {
718                    archive.seek(off);
719                    int curr = archive.read();
720                    if (curr == -1) {
721                        break;
722                    }
723                    if (curr == sig[POS_0]) {
724                        curr = archive.read();
725                        if (curr == sig[POS_1]) {
726                            curr = archive.read();
727                            if (curr == sig[POS_2]) {
728                                curr = archive.read();
729                                if (curr == sig[POS_3]) {
730                                    found = true;
731                                    break;
732                                }
733                            }
734                        }
735                    }
736                }
737            }
738            if (found) {
739                archive.seek(off);
740            }
741            return found;
742        }

744        /**
745         * Skips the given number of bytes or throws an EOFException if
746         * skipping failed.
747         */
748        private void skipBytes(final int count) throws IOException {
749            int totalSkipped = 0;
750            while (totalSkipped < count) {
751                int skippedNow = archive.skipBytes(count - totalSkipped);
752                if (skippedNow <= 0) {
753                    throw new EOFException();
754                }
755                totalSkipped += skippedNow;
756            }
757        }

759        /**
760         * Number of bytes in local file header up to the &quot;length of
761         * filename&quot; entry.
762         */
763        private static final long LFH_OFFSET_FOR_FILENAME_LENGTH =
764            /* local file header signature     */ WORD
765            /* version needed to extract       */ + SHORT
766            /* general purpose bit flag        */ + SHORT
767            /* compression method              */ + SHORT
768            /* last mod file time              */ + SHORT
769            /* last mod file date              */ + SHORT
770            /* crc-32                          */ + WORD
771            /* compressed size                 */ + WORD
772            /* uncompressed size               */ + WORD;
773    
774        /**
775         * Walks through all recorded entries and adds the data available
776         * from the local file header.
777         * <p/>
778         * <p>Also records the offsets for the data to read from the
779         * entries.</p>
780         */
781        private void resolveLocalFileHeaderData(Map<ZipArchiveEntry, NameAndComment>
782                                                        entriesWithoutUTF8Flag)
783                throws IOException {
784            // changing the name of a ZipArchiveEntry is going to change
785            // the hashcode - see COMPRESS-164
786            // Map needs to be reconstructed in order to keep central
787            // directory order
788            Map<ZipArchiveEntry, OffsetEntry> origMap =
789                    new LinkedHashMap<ZipArchiveEntry, OffsetEntry>(entries);
790            entries.clear();
791            for (Map.Entry<ZipArchiveEntry, OffsetEntry> ent : origMap.entrySet()) {
792                ZipArchiveEntry ze = ent.getKey();
793                OffsetEntry offsetEntry = ent.getValue();
794                long offset = offsetEntry.headerOffset;
795                archive.seek(offset + LFH_OFFSET_FOR_FILENAME_LENGTH);
796                byte[] b = new byte[SHORT];
797                archive.readFully(b);
798                int fileNameLen = ZipShort.getValue(b);
799                archive.readFully(b);
800                int extraFieldLen = ZipShort.getValue(b);
801                int lenToSkip = fileNameLen;
802                while (lenToSkip > 0) {
803                    int skipped = archive.skipBytes(lenToSkip);
804                    if (skipped <= 0) {
805                        throw new IOException("failed to skip file name in"
806                                + " local file header");
807                    }
808                    lenToSkip -= skipped;
809                }
810                byte[] localExtraData = new byte[extraFieldLen];
811                archive.readFully(localExtraData);
812                ze.setExtra(localExtraData);
813                offsetEntry.dataOffset = offset + LFH_OFFSET_FOR_FILENAME_LENGTH
814                        + SHORT + SHORT + fileNameLen + extraFieldLen;
815    
816                if (entriesWithoutUTF8Flag.containsKey(ze)) {
817                    String orig = ze.getName();
818                    NameAndComment nc = entriesWithoutUTF8Flag.get(ze);
819                    ZipUtil.setNameAndCommentFromExtraFields(ze, nc.name,
820                            nc.comment);
821                    if (!orig.equals(ze.getName())) {
822                        nameMap.remove(orig);
823                        nameMap.put(ze.getName(), ze);
824                    }
825                }
826                entries.put(ze, offsetEntry);
827            }
828        }

830        /**
831         * Checks whether the archive starts with a LFH.  If it doesn't,
832         * it may be an empty archive.
833         */
834        private boolean startsWithLocalFileHeader() throws IOException {
835            archive.seek(0);
836            final byte[] start = new byte[WORD];
837            archive.readFully(start);
838            for (int i = 0; i < start.length; i++) {
839                if (start[i] != ZipArchiveOutputStream.LFH_SIG[i]) {
840                    return false;
841                }
842            }
843            return true;
844        }

846        /**
847         * InputStream that delegates requests to the underlying
848         * RandomAccessFile, making sure that only bytes from a certain
849         * range can be read.
850         */
851        private class BoundedInputStream extends InputStream {
852            private long remaining;
853            private long loc;
854            private boolean addDummyByte = false;
855    
856            BoundedInputStream(long start, long remaining) {
857                this.remaining = remaining;
858                loc = start;
859            }
860    
861            @Override
862            public int read() throws IOException {
863                if (remaining-- <= 0) {
864                    if (addDummyByte) {
865                        addDummyByte = false;
866                        return 0;
867                    }
868                    return -1;
869                }
870                synchronized (archive) {
871                    archive.seek(loc++);
872                    return archive.read();
873                }
874            }
875    
876            @Override
877            public int read(byte[] b, int off, int len) throws IOException {
878                if (remaining <= 0) {
879                    if (addDummyByte) {
880                        addDummyByte = false;
881                        b[off] = 0;
882                        return 1;
883                    }
884                    return -1;
885                }
886    
887                if (len <= 0) {
888                    return 0;
889                }
890    
891                if (len > remaining) {
892                    len = (int) remaining;
893                }
894                int ret = -1;
895                synchronized (archive) {
896                    archive.seek(loc);
897                    ret = archive.read(b, off, len);
898                }
899                if (ret > 0) {
900                    loc += ret;
901                    remaining -= ret;
902                }
903                return ret;
904            }
905    
906            /**
907             * Inflater needs an extra dummy byte for nowrap - see
908             * Inflater's javadocs.
909             */
910            void addDummy() {
911                addDummyByte = true;
912            }
913        }

916            private final byte[] name;

918    
919            private NameAndComment(byte[] name, byte[] comment) {
920                this.name = name;
921                this.comment = comment;
922            }

925        /**
926         * Compares two ZipArchiveEntries based on their offset within the archive.
927         * <p/>
928         * <p>Won't return any meaningful results if one of the entries
929         * isn't part of the archive at all.</p>
930         */
931        private final Comparator<ZipArchiveEntry> OFFSET_COMPARATOR =
932                new Comparator<ZipArchiveEntry>() {
933                    public int compare(ZipArchiveEntry e1, ZipArchiveEntry e2) {
934                        if (e1 == e2) {
935                            return 0;
936                        }
937    
938                        OffsetEntry off1 = entries.get(e1);
939                        OffsetEntry off2 = entries.get(e2);
940                        if (off1 == null) {
941                            return 1;
942                        }
943                        if (off2 == null) {
944                            return -1;
945                        }
946                        long val = (off1.headerOffset - off2.headerOffset);
947                        return val == 0 ? 0 : val < 0 ? -1 : +1;
948                    }
949                };
