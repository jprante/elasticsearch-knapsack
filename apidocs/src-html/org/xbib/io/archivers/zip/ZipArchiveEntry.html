014/**
015 * Extension that adds better handling of extra fields and provides
016 * access to the internal and external file attributes.
017 * <p/>
018 * <p>The extra data is expected to follow the recommendation of
019 * {@link <a href="http://www.pkware.com/documents/casestudies/APPNOTE.TXT">
020 * APPNOTE.txt</a>}:</p>
021 * <ul>
022 * <li>the extra byte array consists of a sequence of extra fields</li>
023 * <li>each extra fields starts by a two byte header id followed by
024 * a two byte sequence holding the length of the remainder of
025 * data.</li>
026 * </ul>
027 * <p/>
028 * <p>Any extra data that cannot be parsed by the rules above will be
029 * consumed as "unparseable" extra data and treated differently by the
030 * methods of this class.</p>
031 */
032public class ZipArchiveEntry extends java.util.zip.ZipEntry implements ArchiveEntry {
033
034    public static final int PLATFORM_UNIX = 3;
035
036    public static final int PLATFORM_FAT = 0;
037
038    private static final int SHORT_MASK = 0xFFFF;
039
040    private static final int SHORT_SHIFT = 16;

042    /**
043     * The {@link java.util.zip.ZipEntry} base class only supports
044     * the compression methods STORED and DEFLATED. We override the
045     * field so that any compression methods can be used.
046     * <p/>
047     * The default value -1 means that the method has not been specified.
048     *
049     * @see <a href="https://issues.apache.org/jira/browse/COMPRESS-93"
050     * >COMPRESS-93</a>
051     */
052    private int method = -1;

057     * than 2GB for Java versions < 7.  Need to keep our own size
058     * information for Zip64 support.
059     */

095    /**
096     * Creates a new zip entry with fields taken from the specified zip entry.
097     * <p/>
098     * <p>Assumes the entry represents a directory if and only if the
099     * name ends with a forward slash "/".</p>
100     *
101     * @param entry the entry to get fields from
102     * @throws java.util.zip.ZipException on error
103     */
104    public ZipArchiveEntry(java.util.zip.ZipEntry entry) throws ZipException {
105        super(entry);
106        setName(entry.getName());
107        byte[] extra = entry.getExtra();
108        if (extra != null) {
109            setExtraFields(ExtraFieldUtils.parse(extra, true,
110                    ExtraFieldUtils
111                            .UnparseableExtraField.READ));
112        } else {
113            // initializes extra data to an empty byte array
114            setExtra();
115        }
116        setMethod(entry.getMethod());
117        setEntrySize(entry.getSize());
118    }

120    /**
121     * Creates a new zip entry with fields taken from the specified zip entry.
122     * <p/>
123     * <p>Assumes the entry represents a directory if and only if the
124     * name ends with a forward slash "/".</p>
125     *
126     * @param entry the entry to get fields from
127     * @throws java.util.zip.ZipException on error
128     */
129    public ZipArchiveEntry(ZipArchiveEntry entry) throws ZipException {
130        this((java.util.zip.ZipEntry) entry);
131        setInternalAttributes(entry.getInternalAttributes());
132        setExternalAttributes(entry.getExternalAttributes());
133        setExtraFields(entry.getExtraFields(true));
134    }

136    /**
137     * Creates a new zip entry taking some information from the given
138     * file and using the provided name.
139     * <p/>
140     * <p>The name will be adjusted to end with a forward slash "/" if
141     * the file is a directory.  If the file is not a directory a
142     * potential trailing forward slash will be stripped from the
143     * entry name.</p>
144     */
145    public ZipArchiveEntry(File inputFile, String entryName) {
146        this(inputFile.isDirectory() && !entryName.endsWith("/") ?
147                entryName + "/" : entryName);
148        if (inputFile.isFile()) {
149            setSize(inputFile.length());
150        }
151        setTime(inputFile.lastModified());
152        // TODO are there any other fields we can set here?
153    }

231    /**
232     * Sets Unix permissions in a way that is understood by Info-Zip's
233     * unzip command.
234     *
235     * @param mode an <code>int</code> value
236     */
237    public void setUnixMode(int mode) {
238        // CheckStyle:MagicNumberCheck OFF - no point
239        setExternalAttributes((mode << SHORT_SHIFT)
240                // MS-DOS read-only attribute
241                | ((mode & 0200) == 0 ? 1 : 0)
242                // MS-DOS directory flag
243                | (isDirectory() ? 0x10 : 0));
244        // CheckStyle:MagicNumberCheck ON
245        platform = PLATFORM_UNIX;
246    }
247
248    /**
249     * Unix permission.
250     *
251     * @return the unix permissions
252     */
253    public int getUnixMode() {
254        return platform != PLATFORM_UNIX ? 0 :
255                (int) ((getExternalAttributes() >> SHORT_SHIFT) & SHORT_MASK);
256    }

278    /**
279     * Replaces all currently attached extra fields with the new array.
280     *
281     * @param fields an array of extra fields
282     */
283    public void setExtraFields(ZipExtraField[] fields) {
284        extraFields = new LinkedHashMap<ZipShort, ZipExtraField>();
285        for (ZipExtraField field : fields) {
286            if (field instanceof UnparseableExtraFieldData) {
287                unparseableExtra = (UnparseableExtraFieldData) field;
288            } else {
289                extraFields.put(field.getHeaderId(), field);
290            }
291        }
292        setExtra();
293    }
294
295    /**
296     * Retrieves all extra fields that have been parsed successfully.
297     *
298     * @return an array of the extra fields
299     */
300    public ZipExtraField[] getExtraFields() {
301        return getExtraFields(false);
302    }
303
304    /**
305     * Retrieves extra fields.
306     *
307     * @param includeUnparseable whether to also return unparseable
308     *                           extra fields as {@link UnparseableExtraFieldData} if such data
309     *                           exists.
310     * @return an array of the extra fields
311     */
312    public ZipExtraField[] getExtraFields(boolean includeUnparseable) {
313        if (extraFields == null) {
314            return !includeUnparseable || unparseableExtra == null
315                    ? new ZipExtraField[0]
316                    : new ZipExtraField[]{unparseableExtra};
317        }
318        List<ZipExtraField> result =
319                new ArrayList<ZipExtraField>(extraFields.values());
320        if (includeUnparseable && unparseableExtra != null) {
321            result.add(unparseableExtra);
322        }
323        return result.toArray(new ZipExtraField[0]);
324    }

326    /**
327     * Adds an extra field - replacing an already present extra field
328     * of the same type.
329     * <p/>
330     * <p>If no extra field of the same type exists, the field will be
331     * added as last field.</p>
332     *
333     * @param ze an extra field
334     */
335    public void addExtraField(ZipExtraField ze) {
336        if (ze instanceof UnparseableExtraFieldData) {
337            unparseableExtra = (UnparseableExtraFieldData) ze;
338        } else {
339            if (extraFields == null) {
340                extraFields = new LinkedHashMap<ZipShort, ZipExtraField>();
341            }
342            extraFields.put(ze.getHeaderId(), ze);
343        }
344        setExtra();
345    }
346
347    /**
348     * Adds an extra field - replacing an already present extra field
349     * of the same type.
350     * <p/>
351     * <p>The new extra field will be the first one.</p>
352     *
353     * @param ze an extra field
354     */
355    public void addAsFirstExtraField(ZipExtraField ze) {
356        if (ze instanceof UnparseableExtraFieldData) {
357            unparseableExtra = (UnparseableExtraFieldData) ze;
358        } else {
359            LinkedHashMap<ZipShort, ZipExtraField> copy = extraFields;
360            extraFields = new LinkedHashMap<ZipShort, ZipExtraField>();
361            extraFields.put(ze.getHeaderId(), ze);
362            if (copy != null) {
363                copy.remove(ze.getHeaderId());
364                extraFields.putAll(copy);
365            }
366        }
367        setExtra();
368    }

370    /**
371     * Remove an extra field.
372     *
373     * @param type the type of extra field to remove
374     */
375    public void removeExtraField(ZipShort type) {
376        if (extraFields == null) {
377            throw new java.util.NoSuchElementException();
378        }
379        if (extraFields.remove(type) == null) {
380            throw new java.util.NoSuchElementException();
381        }
382        setExtra();
383    }

387     */
388    public void removeUnparseableExtraFieldData() {
389        if (unparseableExtra == null) {
390            throw new java.util.NoSuchElementException();
391        }
392        unparseableExtra = null;
393        setExtra();
394    }

417    /**
418     * Parses the given bytes as extra field data and consumes any
419     * unparseable data as an {@link UnparseableExtraFieldData}
420     * instance.
421     *
422     * @param extra an array of bytes to be parsed into extra fields
423     * @throws RuntimeException if the bytes cannot be parsed
424     * @throws RuntimeException on error
425     */
426    @Override
427    public void setExtra(byte[] extra) throws RuntimeException {
428        try {
429            ZipExtraField[] local =
430                    ExtraFieldUtils.parse(extra, true,
431                            ExtraFieldUtils.UnparseableExtraField.READ);
432            mergeExtraFields(local, true);
433        } catch (ZipException e) {
434            throw new RuntimeException("Error parsing extra fields for entry: "
435                    + getName() + " - " + e.getMessage(), e);
436        }
437    }

564        if (rawName != null) {

601    /**
602     * If there are no extra fields, use the given fields as new extra
603     * data - otherwise merge the fields assuming the existing fields
604     * and the new fields stem from different locations inside the
605     * archive.
606     *
607     * @param f     the extra fields to merge
608     * @param local whether the new fields originate from local data
609     */
610    private void mergeExtraFields(ZipExtraField[] f, boolean local)
611            throws ZipException {
612        if (extraFields == null) {
613            setExtraFields(f);
614        } else {
615            for (ZipExtraField element : f) {
616                ZipExtraField existing;
617                if (element instanceof UnparseableExtraFieldData) {
618                    existing = unparseableExtra;
619                } else {
620                    existing = getExtraField(element.getHeaderId());
621                }
622                if (existing == null) {
623                    addExtraField(element);
624                } else {
625                    if (local) {
626                        byte[] b = element.getLocalFileDataData();
627                        existing.parseFromLocalFileData(b, 0, b.length);
628                    } else {
629                        byte[] b = element.getCentralDirectoryData();
630                        existing.parseFromCentralDirectoryData(b, 0, b.length);
631                    }
632                }
633            }
634            setExtra();
635        }
636    }

647    @Override
648    public boolean equals(Object obj) {
649        if (this == obj) {
650            return true;
651        }
652        if (obj == null || getClass() != obj.getClass()) {
653            return false;
654        }
655        ZipArchiveEntry other = (ZipArchiveEntry) obj;
656        String myName = getName();
657        String otherName = other.getName();
658        if (myName == null) {
659            if (otherName != null) {
660                return false;
661            }
662        } else if (!myName.equals(otherName)) {
663            return false;
664        }
665        String myComment = getComment();
666        String otherComment = other.getComment();
667        if (myComment == null) {
668            if (otherComment != null) {
669                return false;
670            }
671        } else if (!myComment.equals(otherComment)) {
672            return false;
673        }
674        return getTime() == other.getTime()
675                && getInternalAttributes() == other.getInternalAttributes()
676                && getPlatform() == other.getPlatform()
677                && getExternalAttributes() == other.getExternalAttributes()
678                && getMethod() == other.getMethod()
679                && getEntrySize() == other.getEntrySize()
680                && getCrc() == other.getCrc()
681                && getCompressedSize() == other.getCompressedSize()
682                && Arrays.equals(getCentralDirectoryExtra(),
683                other.getCentralDirectoryExtra())
684                && Arrays.equals(getLocalFileDataExtra(),
685                other.getLocalFileDataExtra())
686                && gpb.equals(other.gpb);
687    }
