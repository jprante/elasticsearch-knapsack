002    package org.xbib.io.archivers.zip;
003    
004    import static org.xbib.io.archivers.zip.ZipConstants.BYTE_MASK;
005    import static org.xbib.io.archivers.zip.ZipConstants.WORD;
006    
007    /**
008     * Utility class that represents a four byte integer with conversion
009     * rules for the big endian byte order of ZIP files.
010     */
011    public final class ZipLong implements Cloneable {
012    
013        private static final int BYTE_1 = 1;
014        private static final int BYTE_1_MASK = 0xFF00;
015        private static final int BYTE_1_SHIFT = 8;
016    
017        private static final int BYTE_2 = 2;
018        private static final int BYTE_2_MASK = 0xFF0000;
019        private static final int BYTE_2_SHIFT = 16;
020    
021        private static final int BYTE_3 = 3;
022        private static final long BYTE_3_MASK = 0xFF000000L;
023        private static final int BYTE_3_SHIFT = 24;
024    
025        private final long value;
026    
027        /**
028         * Central File Header Signature
029         */
030        public static final ZipLong CFH_SIG = new ZipLong(0X02014B50L);
031    
032        /**
033         * Local File Header Signature
034         */
035        public static final ZipLong LFH_SIG = new ZipLong(0X04034B50L);
036    
037        /**
038         * Data Descriptor signature
039         */
040        public static final ZipLong DD_SIG = new ZipLong(0X08074B50L);
041    
042        /**
043         * Value stored in size and similar fields if ZIP64 extensions are
044         * used.
045         */
046        static final ZipLong ZIP64_MAGIC = new ZipLong(ZipConstants.ZIP64_MAGIC);
047    
048        /**
049         * Create instance from a number.
050         *
051         * @param value the long to store as a ZipLong
052         */
053        public ZipLong(long value) {
054            this.value = value;
055        }
056    
057        /**
058         * Create instance from bytes.
059         *
060         * @param bytes the bytes to store as a ZipLong
061         */
062        public ZipLong(byte[] bytes) {
063            this(bytes, 0);
064        }
065    
066        /**
067         * Create instance from the four bytes starting at offset.
068         *
069         * @param bytes  the bytes to store as a ZipLong
070         * @param offset the offset to start
071         */
072        public ZipLong(byte[] bytes, int offset) {
073            value = ZipLong.getValue(bytes, offset);
074        }
075    
076        /**
077         * Get value as four bytes in big endian byte order.
078         *
079         * @return value as four bytes in big endian order
080         */
081        public byte[] getBytes() {
082            return ZipLong.getBytes(value);
083        }
084    
085        /**
086         * Get value as Java long.
087         *
088         * @return value as a long
089         */
090        public long getValue() {
091            return value;
092        }
093    
094        /**
095         * Get value as four bytes in big endian byte order.
096         *
097         * @param value the value to convert
098         * @return value as four bytes in big endian byte order
099         */
100        public static byte[] getBytes(long value) {
101            byte[] result = new byte[WORD];
102            result[0] = (byte) ((value & BYTE_MASK));
103            result[BYTE_1] = (byte) ((value & BYTE_1_MASK) >> BYTE_1_SHIFT);
104            result[BYTE_2] = (byte) ((value & BYTE_2_MASK) >> BYTE_2_SHIFT);
105            result[BYTE_3] = (byte) ((value & BYTE_3_MASK) >> BYTE_3_SHIFT);
106            return result;
107        }
108    
109        /**
110         * Helper method to get the value as a Java long from four bytes starting at given array offset
111         *
112         * @param bytes  the array of bytes
113         * @param offset the offset to start
114         * @return the corresponding Java long value
115         */
116        public static long getValue(byte[] bytes, int offset) {
117            long value = (bytes[offset + BYTE_3] << BYTE_3_SHIFT) & BYTE_3_MASK;
118            value += (bytes[offset + BYTE_2] << BYTE_2_SHIFT) & BYTE_2_MASK;
119            value += (bytes[offset + BYTE_1] << BYTE_1_SHIFT) & BYTE_1_MASK;
120            value += (bytes[offset] & BYTE_MASK);
121            return value;
122        }
123    
124        /**
125         * Helper method to get the value as a Java long from a four-byte array
126         *
127         * @param bytes the array of bytes
128         * @return the corresponding Java long value
129         */
130        public static long getValue(byte[] bytes) {
131            return getValue(bytes, 0);
132        }
133    
134        /**
135         * Override to make two instances with same value equal.
136         *
137         * @param o an object to compare
138         * @return true if the objects are equal
139         */
140        @Override
141        public boolean equals(Object o) {
142            if (o == null || !(o instanceof ZipLong)) {
143                return false;
144            }
145            return value == ((ZipLong) o).getValue();
146        }
147    
148        /**
149         * Override to make two instances with same value equal.
150         *
151         * @return the value stored in the ZipLong
152         */
153        @Override
154        public int hashCode() {
155            return (int) value;
156        }
157    
158        @Override
159        public Object clone() {
160            try {
161                return super.clone();
162            } catch (CloneNotSupportedException cnfe) {
163                // impossible
164                throw new RuntimeException(cnfe);
165            }
166        }
167    
168        @Override
169        public String toString() {
170            return "ZipLong value: " + value;
171        }
172    }
