001    
002    package org.xbib.io.archivers.zip;
003    
004    /**
005     * Parser/encoder for the "general purpose bit" field in ZIP's local
006     * file and central directory headers.
007     */
008    public final class GeneralPurposeBit {
009        /**
010         * Indicates that the file is encrypted.
011         */
012        private static final int ENCRYPTION_FLAG = 1;
013    
014        /**
015         * Indicates that a data descriptor stored after the file contents
016         * will hold CRC and size information.
017         */
018        private static final int DATA_DESCRIPTOR_FLAG = 1 << 3;
019    
020        /**
021         * Indicates strong encryption.
022         */
023        private static final int STRONG_ENCRYPTION_FLAG = 1 << 6;
024    
025        /**
026         * Indicates that filenames are written in utf-8.
027         */
028        protected static final int UFT8_NAMES_FLAG = 1 << 11;
029    
030        private boolean languageEncodingFlag = false;
031    
032        private boolean dataDescriptorFlag = false;
033    
034        private boolean encryptionFlag = false;
035    
036        private boolean strongEncryptionFlag = false;
037    
038        public GeneralPurposeBit() {
039        }
040    
041        /**
042         * whether the current entry uses UTF8 for file name and comment.
043         */
044        public boolean usesUTF8ForNames() {
045            return languageEncodingFlag;
046        }
047    
048        /**
049         * whether the current entry will use UTF8 for file name and comment.
050         */
051        public void useUTF8ForNames(boolean b) {
052            languageEncodingFlag = b;
053        }
054    
055        /**
056         * whether the current entry uses the data descriptor to store CRC
057         * and size information
058         */
059        public boolean usesDataDescriptor() {
060            return dataDescriptorFlag;
061        }
062    
063        /**
064         * whether the current entry will use the data descriptor to store
065         * CRC and size information
066         */
067        public void useDataDescriptor(boolean b) {
068            dataDescriptorFlag = b;
069        }
070    
071        /**
072         * whether the current entry is encrypted
073         */
074        public boolean usesEncryption() {
075            return encryptionFlag;
076        }
077    
078        /**
079         * whether the current entry will be encrypted
080         */
081        public void useEncryption(boolean b) {
082            encryptionFlag = b;
083        }
084    
085        /**
086         * whether the current entry is encrypted using strong encryption
087         */
088        public boolean usesStrongEncryption() {
089            return encryptionFlag && strongEncryptionFlag;
090        }
091    
092        /**
093         * whether the current entry will be encrypted  using strong encryption
094         */
095        public void useStrongEncryption(boolean b) {
096            strongEncryptionFlag = b;
097            if (b) {
098                useEncryption(true);
099            }
100        }
101    
102        /**
103         * Encodes the set bits in a form suitable for ZIP archives.
104         */
105        public byte[] encode() {
106            return
107                    ZipShort.getBytes((dataDescriptorFlag ? DATA_DESCRIPTOR_FLAG : 0)
108                            |
109                            (languageEncodingFlag ? UFT8_NAMES_FLAG : 0)
110                            |
111                            (encryptionFlag ? ENCRYPTION_FLAG : 0)
112                            |
113                            (strongEncryptionFlag ? STRONG_ENCRYPTION_FLAG : 0)
114                    );
115        }
116    
117        /**
118         * Parses the supported flags from the given archive data.
119         *
120         * @param data   local file header or a central directory entry.
121         * @param offset offset at which the general purpose bit starts
122         */
123        public static GeneralPurposeBit parse(final byte[] data, final int offset) {
124            final int generalPurposeFlag = ZipShort.getValue(data, offset);
125            GeneralPurposeBit b = new GeneralPurposeBit();
126            b.useDataDescriptor((generalPurposeFlag & DATA_DESCRIPTOR_FLAG) != 0);
127            b.useUTF8ForNames((generalPurposeFlag & UFT8_NAMES_FLAG) != 0);
128            b.useStrongEncryption((generalPurposeFlag & STRONG_ENCRYPTION_FLAG)
129                    != 0);
130            b.useEncryption((generalPurposeFlag & ENCRYPTION_FLAG) != 0);
131            return b;
132        }
133    
134        @Override
135        public int hashCode() {
136            return 3 * (7 * (13 * (17 * (encryptionFlag ? 1 : 0)
137                    + (strongEncryptionFlag ? 1 : 0))
138                    + (languageEncodingFlag ? 1 : 0))
139                    + (dataDescriptorFlag ? 1 : 0));
140        }
141    
142        @Override
143        public boolean equals(Object o) {
144            if (!(o instanceof GeneralPurposeBit)) {
145                return false;
146            }
147            GeneralPurposeBit g = (GeneralPurposeBit) o;
148            return g.encryptionFlag == encryptionFlag
149                    && g.strongEncryptionFlag == strongEncryptionFlag
150                    && g.languageEncodingFlag == languageEncodingFlag
151                    && g.dataDescriptorFlag == dataDescriptorFlag;
152        }
153    }
