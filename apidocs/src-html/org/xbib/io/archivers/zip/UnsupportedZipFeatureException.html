002package org.xbib.io.archivers.zip;
003
004import java.util.zip.ZipException;
005
006/**
007 * Exception thrown when attempting to read or write data for a zip
008 * entry that uses ZIP features not supported by this library.
009 */
010public class UnsupportedZipFeatureException extends ZipException {
011
012    private final Feature reason;
013
014    private final ZipArchiveEntry entry;
015
016    /**
017     * Creates an exception.
018     *
019     * @param reason the feature that is not supported
020     * @param entry  the entry using the feature
021     */
022    public UnsupportedZipFeatureException(Feature reason,
023                                          ZipArchiveEntry entry) {
024        super("unsupported feature " + reason + " used in entry "
025                + entry.getName());
026        this.reason = reason;
027        this.entry = entry;
028    }
029
030    /**
031     * The unsupported feature that has been used.
032     */
033    public Feature getFeature() {
034        return reason;
035    }
036
037    /**
038     * The entry using the unsupported feature.
039     */
040    public ZipArchiveEntry getEntry() {
041        return entry;
042    }
043
044    /**
045     * ZIP Features that may or may not be supported.
046     */
047    public static class Feature {
048        /**
049         * The entry is encrypted.
050         */
051        public static final Feature ENCRYPTION = new Feature("encryption");
052        /**
053         * The entry used an unsupported compression method.
054         */
055        public static final Feature METHOD = new Feature("compression method");
056        /**
057         * The entry uses a data descriptor.
058         */
059        public static final Feature DATA_DESCRIPTOR = new Feature("data descriptor");
060
061        private final String name;
062
063        private Feature(String name) {
064            this.name = name;
065        }
066
067        @Override
068        public String toString() {
069            return name;
070        }
071    }
072}
