002    package org.xbib.io.archivers;
003    
004    import java.util.Date;
005    
006    /**
007     * Represents an entry of an archive.
008     */
009    public interface ArchiveEntry {
010    
011        /**
012         * Special value indicating that the size is unknown
013         */
014        long SIZE_UNKNOWN = -1;
015    
016        ArchiveEntry setName(String name);
017    
018        /**
019         * The name of the entry in the archive. May refer to a file or directory or other item
020         */
021        String getName();
022    
023        /**
024         * Set the (uncompressed) entry size in bytes
025         */
026        ArchiveEntry setEntrySize(long size);
027    
028        /**
029         * The (uncompressed) size of the entry. May be -1 (SIZE_UNKNOWN) if the size is unknown
030         */
031        long getEntrySize();
032    
033        ArchiveEntry setLastModified(Date date);
034    
035        /**
036         * The last modified date of the entry.
037         */
038        Date getLastModified();
039    
040        /**
041         * True if the entry refers to a directory
042         */
043        boolean isDirectory();
044    
045    }
