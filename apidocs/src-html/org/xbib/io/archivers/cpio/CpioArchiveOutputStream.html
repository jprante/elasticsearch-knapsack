002    package org.xbib.io.archivers.cpio;
003    
004    import org.xbib.io.archivers.ArchiveEntry;
005    import org.xbib.io.archivers.ArchiveOutputStream;
006    import org.xbib.io.archivers.ArchiveUtils;
007    
008    import java.io.IOException;
009    import java.io.OutputStream;
010    import java.util.HashMap;
011    
012    /**
013     * CPIOArchiveOutputStream is a stream for writing CPIO streams. All formats of
014     * CPIO are supported (old ASCII, old binary, new portable format and the new
015     * portable format with CRC).
016     * <p/>
017     * <p/>
018     * An entry can be written by creating an instance of CpioArchiveEntry and fill
019     * it with the necessary values and put it into the CPIO stream. Afterwards
020     * write the contents of the file into the CPIO stream. Either close the stream
021     * by calling finish() or put a next entry into the cpio stream.
022     * <p/>
023     * <code><pre>
024     * CpioArchiveOutputStream out = new CpioArchiveOutputStream(
025     *         new FileOutputStream(new File("test.cpio")));
026     * CpioArchiveEntry entry = new CpioArchiveEntry();
027     * entry.setName("testfile");
028     * String contents = &quot;12345&quot;;
029     * entry.setFileSize(contents.length());
030     * entry.setMode(CpioConstants.C_ISREG); // regular file
031     * ... set other attributes, e.g. time, number of links
032     * out.putArchiveEntry(entry);
033     * out.write(testContents.getBytes());
034     * out.close();
035     * </pre></code>
036     * <p/>
037     * Note: This implementation should be compatible to cpio 2.5
038     */
039    public class CpioArchiveOutputStream extends ArchiveOutputStream implements
040            CpioConstants {
041    
042        private CpioArchiveEntry entry;
043    
044        private boolean closed = false;
045    
046        /**
047         * indicates if this archive is finished
048         */
049        private boolean finished;
050    
051        private final short entryFormat;
052    
053        private final HashMap<String, CpioArchiveEntry> names =
054                new HashMap<String, CpioArchiveEntry>();
055    
056        private long crc = 0;
057    
058        private long written;
059    
060        private final OutputStream out;
061    
062        private final int blockSize;
063    
064        private long nextArtificalDeviceAndInode = 1;
065    
066        /**
067         * Construct the cpio output stream with a specified format and a
068         * blocksize of {@link CpioConstants#BLOCK_SIZE BLOCK_SIZE}.
069         *
070         * @param out    The cpio stream
071         * @param format The format of the stream
072         */
073        public CpioArchiveOutputStream(final OutputStream out, final short format) {
074            this(out, format, BLOCK_SIZE);
075        }
076    
077        /**
078         * Construct the cpio output stream with a specified format
079         *
080         * @param out       The cpio stream
081         * @param format    The format of the stream
082         * @param blockSize The block size of the archive.
083         */
084        public CpioArchiveOutputStream(final OutputStream out, final short format,
085                                       final int blockSize) {
086            this.out = out;
087            switch (format) {
088                case FORMAT_NEW:
089                case FORMAT_NEW_CRC:
090                case FORMAT_OLD_ASCII:
091                case FORMAT_OLD_BINARY:
092                    break;
093                default:
094                    throw new IllegalArgumentException("Unknown format: " + format);
095    
096            }
097            this.entryFormat = format;
098            this.blockSize = blockSize;
099        }
100    
101        /**
102         * Construct the cpio output stream. The format for this CPIO stream is the
103         * "new" format
104         *
105         * @param out The cpio stream
106         */
107        public CpioArchiveOutputStream(final OutputStream out) {
108            this(out, FORMAT_NEW);
109        }
110    
111        /**
112         * Check to make sure that this stream has not been closed
113         *
114         * @throws java.io.IOException if the stream is already closed
115         */
116        private void ensureOpen() throws IOException {
117            if (this.closed) {
118                throw new IOException("Stream closed");
119            }
120        }
121    
122        /**
123         * Begins writing a new CPIO file entry and positions the stream to the
124         * start of the entry data. Closes the current entry if still active. The
125         * current time will be used if the entry has no set modification time and
126         * the default header format will be used if no other format is specified in
127         * the entry.
128         *
129         * @param entry the CPIO cpioEntry to be written
130         * @throws java.io.IOException if an I/O error has occurred or if a CPIO file error has
131         *                             occurred
132         * @throws ClassCastException  if entry is not an instance of CpioArchiveEntry
133         */
134        @Override
135        public void putArchiveEntry(ArchiveEntry entry) throws IOException {
136            if (finished) {
137                throw new IOException("Stream has already been finished");
138            }
139    
140            CpioArchiveEntry e = (CpioArchiveEntry) entry;
141            ensureOpen();
142            if (this.entry != null) {
143                closeArchiveEntry(); // close previous entry
144            }
145            if (e.getTime() == -1) {
146                e.setTime(System.currentTimeMillis() / 1000);
147            }
148    
149            final short format = e.getFormat();
150            if (format != this.entryFormat) {
151                throw new IOException("Header format: " + format + " does not match existing format: " + this.entryFormat);
152            }
153    
154            if (this.names.put(e.getName(), e) != null) {
155                throw new IOException("duplicate entry: " + e.getName());
156            }
157    
158            writeHeader(e);
159            this.entry = e;
160            this.written = 0;
161        }
162    
163        private void writeHeader(final CpioArchiveEntry e) throws IOException {
164            switch (e.getFormat()) {
165                case FORMAT_NEW:
166                    out.write(ArchiveUtils.toAsciiBytes(MAGIC_NEW));
167                    count(6);
168                    writeNewEntry(e);
169                    break;
170                case FORMAT_NEW_CRC:
171                    out.write(ArchiveUtils.toAsciiBytes(MAGIC_NEW_CRC));
172                    count(6);
173                    writeNewEntry(e);
174                    break;
175                case FORMAT_OLD_ASCII:
176                    out.write(ArchiveUtils.toAsciiBytes(MAGIC_OLD_ASCII));
177                    count(6);
178                    writeOldAsciiEntry(e);
179                    break;
180                case FORMAT_OLD_BINARY:
181                    boolean swapHalfWord = true;
182                    writeBinaryLong(MAGIC_OLD_BINARY, 2, swapHalfWord);
183                    writeOldBinaryEntry(e, swapHalfWord);
184                    break;
185            }
186        }
187    
188        private void writeNewEntry(final CpioArchiveEntry entry) throws IOException {
189            long inode = entry.getInode();
190            long devMin = entry.getDeviceMin();
191            if (CPIO_TRAILER.equals(entry.getName())) {
192                inode = devMin = 0;
193            } else {
194                if (inode == 0 && devMin == 0) {
195                    inode = nextArtificalDeviceAndInode & 0xFFFFFFFF;
196                    devMin = (nextArtificalDeviceAndInode++ >> 32) & 0xFFFFFFFF;
197                } else {
198                    nextArtificalDeviceAndInode =
199                            Math.max(nextArtificalDeviceAndInode,
200                                    inode + 0x100000000L * devMin) + 1;
201                }
202            }
203    
204            writeAsciiLong(inode, 8, 16);
205            writeAsciiLong(entry.getMode(), 8, 16);
206            writeAsciiLong(entry.getUID(), 8, 16);
207            writeAsciiLong(entry.getGID(), 8, 16);
208            writeAsciiLong(entry.getNumberOfLinks(), 8, 16);
209            writeAsciiLong(entry.getTime(), 8, 16);
210            writeAsciiLong(entry.getEntrySize(), 8, 16);
211            writeAsciiLong(entry.getDeviceMaj(), 8, 16);
212            writeAsciiLong(devMin, 8, 16);
213            writeAsciiLong(entry.getRemoteDeviceMaj(), 8, 16);
214            writeAsciiLong(entry.getRemoteDeviceMin(), 8, 16);
215            writeAsciiLong(entry.getName().length() + 1, 8, 16);
216            writeAsciiLong(entry.getChksum(), 8, 16);
217            writeCString(entry.getName());
218            pad(entry.getHeaderPadCount());
219        }
220    
221        private void writeOldAsciiEntry(final CpioArchiveEntry entry)
222                throws IOException {
223            long inode = entry.getInode();
224            long device = entry.getDevice();
225            if (CPIO_TRAILER.equals(entry.getName())) {
226                inode = device = 0;
227            } else {
228                if (inode == 0 && device == 0) {
229                    inode = nextArtificalDeviceAndInode & 0777777;
230                    device = (nextArtificalDeviceAndInode++ >> 18) & 0777777;
231                } else {
232                    nextArtificalDeviceAndInode =
233                            Math.max(nextArtificalDeviceAndInode,
234                                    inode + 01000000 * device) + 1;
235                }
236            }
237    
238            writeAsciiLong(device, 6, 8);
239            writeAsciiLong(inode, 6, 8);
240            writeAsciiLong(entry.getMode(), 6, 8);
241            writeAsciiLong(entry.getUID(), 6, 8);
242            writeAsciiLong(entry.getGID(), 6, 8);
243            writeAsciiLong(entry.getNumberOfLinks(), 6, 8);
244            writeAsciiLong(entry.getRemoteDevice(), 6, 8);
245            writeAsciiLong(entry.getTime(), 11, 8);
246            writeAsciiLong(entry.getName().length() + 1, 6, 8);
247            writeAsciiLong(entry.getEntrySize(), 11, 8);
248            writeCString(entry.getName());
249        }
250    
251        private void writeOldBinaryEntry(final CpioArchiveEntry entry,
252                                         final boolean swapHalfWord) throws IOException {
253            long inode = entry.getInode();
254            long device = entry.getDevice();
255            if (CPIO_TRAILER.equals(entry.getName())) {
256                inode = device = 0;
257            } else {
258                if (inode == 0 && device == 0) {
259                    inode = nextArtificalDeviceAndInode & 0xFFFF;
260                    device = (nextArtificalDeviceAndInode++ >> 16) & 0xFFFF;
261                } else {
262                    nextArtificalDeviceAndInode =
263                            Math.max(nextArtificalDeviceAndInode,
264                                    inode + 0x10000 * device) + 1;
265                }
266            }
267    
268            writeBinaryLong(device, 2, swapHalfWord);
269            writeBinaryLong(inode, 2, swapHalfWord);
270            writeBinaryLong(entry.getMode(), 2, swapHalfWord);
271            writeBinaryLong(entry.getUID(), 2, swapHalfWord);
272            writeBinaryLong(entry.getGID(), 2, swapHalfWord);
273            writeBinaryLong(entry.getNumberOfLinks(), 2, swapHalfWord);
274            writeBinaryLong(entry.getRemoteDevice(), 2, swapHalfWord);
275            writeBinaryLong(entry.getTime(), 4, swapHalfWord);
276            writeBinaryLong(entry.getName().length() + 1, 2, swapHalfWord);
277            writeBinaryLong(entry.getEntrySize(), 4, swapHalfWord);
278            writeCString(entry.getName());
279            pad(entry.getHeaderPadCount());
280        }
281    
282        @Override
283        public void closeArchiveEntry() throws IOException {
284            if (finished) {
285                throw new IOException("Stream has already been finished");
286            }
287    
288            ensureOpen();
289    
290            if (entry == null) {
291                throw new IOException("Trying to close non-existent entry");
292            }
293    
294            if (this.entry.getEntrySize() != this.written) {
295                throw new IOException("invalid entry size (expected "
296                        + this.entry.getEntrySize() + " but got " + this.written
297                        + " bytes)");
298            }
299            pad(this.entry.getDataPadCount());
300            if (this.entry.getFormat() == FORMAT_NEW_CRC
301                    && this.crc != this.entry.getChksum()) {
302                throw new IOException("CRC Error");
303            }
304            this.entry = null;
305            this.crc = 0;
306            this.written = 0;
307        }
308    
309        /**
310         * Writes an array of bytes to the current CPIO entry data. This method will
311         * block until all the bytes are written.
312         *
313         * @param b   the data to be written
314         * @param off the start offset in the data
315         * @param len the number of bytes that are written
316         * @throws java.io.IOException if an I/O error has occurred or if a CPIO file error has
317         *                             occurred
318         */
319        @Override
320        public void write(final byte[] b, final int off, final int len)
321                throws IOException {
322            ensureOpen();
323            if (off < 0 || len < 0 || off > b.length - len) {
324                throw new IndexOutOfBoundsException();
325            } else if (len == 0) {
326                return;
327            }
328    
329            if (this.entry == null) {
330                throw new IOException("no current CPIO entry");
331            }
332            if (this.written + len > this.entry.getEntrySize()) {
333                throw new IOException("attempt to write past end of STORED entry");
334            }
335            out.write(b, off, len);
336            this.written += len;
337            if (this.entry.getFormat() == FORMAT_NEW_CRC) {
338                for (int pos = 0; pos < len; pos++) {
339                    this.crc += b[pos] & 0xFF;
340                }
341            }
342            count(len);
343        }
344    
345        /**
346         * Finishes writing the contents of the CPIO output stream without closing
347         * the underlying stream. Use this method when applying multiple filters in
348         * succession to the same output stream.
349         *
350         * @throws java.io.IOException if an I/O exception has occurred or if a CPIO file error has
351         *                             occurred
352         */
353        @Override
354        public void finish() throws IOException {
355            ensureOpen();
356            if (finished) {
357                throw new IOException("This archive has already been finished");
358            }
359    
360            if (this.entry != null) {
361                throw new IOException("This archive contains unclosed entries.");
362            }
363            this.entry = new CpioArchiveEntry(this.entryFormat);
364            this.entry.setName(CPIO_TRAILER);
365            this.entry.setNumberOfLinks(1);
366            writeHeader(this.entry);
367            closeArchiveEntry();
368    
369            int lengthOfLastBlock = (int) (getBytesWritten() % blockSize);
370            if (lengthOfLastBlock != 0) {
371                pad(blockSize - lengthOfLastBlock);
372            }
373    
374            finished = true;
375        }
376    
377        /**
378         * Closes the CPIO output stream as well as the stream being filtered.
379         *
380         * @throws java.io.IOException if an I/O error has occurred or if a CPIO file error has
381         *                             occurred
382         */
383        @Override
384        public void close() throws IOException {
385            if (!finished) {
386                finish();
387            }
388    
389            if (!this.closed) {
390                out.close();
391                this.closed = true;
392            }
393        }
394    
395        private void pad(int count) throws IOException {
396            if (count > 0) {
397                byte buff[] = new byte[count];
398                out.write(buff);
399                count(count);
400            }
401        }
402    
403        private void writeBinaryLong(final long number, final int length,
404                                     final boolean swapHalfWord) throws IOException {
405            byte tmp[] = CpioUtil.long2byteArray(number, length, swapHalfWord);
406            out.write(tmp);
407            count(tmp.length);
408        }
409    
410        private void writeAsciiLong(final long number, final int length,
411                                    final int radix) throws IOException {
412            StringBuffer tmp = new StringBuffer();
413            String tmpStr;
414            if (radix == 16) {
415                tmp.append(Long.toHexString(number));
416            } else if (radix == 8) {
417                tmp.append(Long.toOctalString(number));
418            } else {
419                tmp.append(Long.toString(number));
420            }
421    
422            if (tmp.length() <= length) {
423                long insertLength = length - tmp.length();
424                for (int pos = 0; pos < insertLength; pos++) {
425                    tmp.insert(0, "0");
426                }
427                tmpStr = tmp.toString();
428            } else {
429                tmpStr = tmp.substring(tmp.length() - length);
430            }
431            byte[] b = ArchiveUtils.toAsciiBytes(tmpStr);
432            out.write(b);
433            count(b.length);
434        }
435    
436        /**
437         * Writes an ASCII string to the stream followed by \0
438         *
439         * @param str the String to write
440         * @throws java.io.IOException if the string couldn't be written
441         */
442        private void writeCString(final String str) throws IOException {
443            byte[] b = ArchiveUtils.toAsciiBytes(str);
444            out.write(b);
445            out.write('\0');
446            count(b.length + 1);
447        }
