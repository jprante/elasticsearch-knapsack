008import java.io.EOFException;
009import java.io.IOException;
010import java.io.InputStream;

012/**
013 * CPIOArchiveInputStream is a stream for reading cpio streams. All formats of
014 * cpio are supported (old ascii, old binary, new portable format and the new
015 * portable format with crc).
016 * <p/>
017 * <p/>
018 * The stream can be read by extracting a cpio entry (containing all
019 * informations about a entry) and afterwards reading from the stream the file
020 * specified by the entry.
021 * <p/>
022 * <code><pre>
023 * CPIOArchiveInputStream cpioIn = new CPIOArchiveInputStream(
024 *         new FileInputStream(new File(&quot;test.cpio&quot;)));
025 * CPIOArchiveEntry cpioEntry;
026 * <p/>
027 * while ((cpioEntry = cpioIn.getNextEntry()) != null) {
028 *     System.out.println(cpioEntry.getName());
029 *     int tmp;
030 *     StringBuilder buf = new StringBuilder();
031 *     while ((tmp = cpIn.read()) != -1) {
032 *         buf.append((char) tmp);
033 *     }
034 *     System.out.println(buf.toString());
035 * }
036 * cpioIn.close();
037 * </pre></code>
038 * <p/>
039 * Note: This implementation should be compatible to cpio 2.5
040 */
041
042public class CpioArchiveInputStream extends ArchiveInputStream implements
043        CpioConstants {
044
045    private boolean closed = false;
046
047    private CpioArchiveEntry entry;
048
049    private long entryBytesRead = 0;
050
051    private boolean entryEOF = false;
052
053    private final byte tmpbuf[] = new byte[4096];
054
055    private long crc = 0;
056
057    private final InputStream in;
058
059    /**
060     * Construct the cpio input stream
061     *
062     * @param in The cpio stream
063     */
064    public CpioArchiveInputStream(final InputStream in) {
065        this.in = in;
066        this.closed = false;
067    }
068
069    /**
070     * Returns 0 after EOF has reached for the current entry data, otherwise
071     * always return 1.
072     * <p/>
073     * Programs should not count on this method to return the actual number of
074     * bytes that could be read without blocking.
075     *
076     * @return 1 before EOF and 0 after EOF has reached for current entry.
077     * @throws java.io.IOException if an I/O error has occurred or if a CPIO file error has
078     *                             occurred
079     */
080    @Override
081    public int available() throws IOException {
082        ensureOpen();
083        if (this.entryEOF) {
084            return 0;
085        }
086        return 1;
087    }
088
089    /**
090     * Closes the CPIO input stream.
091     *
092     * @throws java.io.IOException if an I/O error has occurred
093     */
094    @Override
095    public void close() throws IOException {
096        if (!this.closed) {
097            in.close();
098            this.closed = true;
099        }
100    }
101
102    /**
103     * Closes the current CPIO entry and positions the stream for reading the
104     * next entry.
105     *
106     * @throws java.io.IOException if an I/O error has occurred or if a CPIO file error has
107     *                             occurred
108     */
109    private void closeEntry() throws IOException {
110        ensureOpen();
111        while (read(this.tmpbuf, 0, this.tmpbuf.length) != -1) {
112            // do nothing
113        }
114
115        this.entryEOF = true;
116    }
117
118    /**
119     * Check to make sure that this stream has not been closed
120     *
121     * @throws java.io.IOException if the stream is already closed
122     */
123    private void ensureOpen() throws IOException {
124        if (this.closed) {
125            throw new IOException("stream closed");
126        }
127    }
128
129    /**
130     * Reads the next CPIO file entry and positions stream at the beginning of
131     * the entry data.
132     *
133     * @return the CPIOArchiveEntry just read
134     * @throws java.io.IOException if an I/O error has occurred or if a CPIO file error has
135     *                             occurred
136     */
137    public CpioArchiveEntry getNextCPIOEntry() throws IOException {
138        ensureOpen();
139        if (this.entry != null) {
140            closeEntry();
141        }
142        byte magic[] = new byte[2];
143        readFully(magic, 0, magic.length);
144        if (CpioUtil.byteArray2long(magic, false) == MAGIC_OLD_BINARY) {
145            this.entry = readOldBinaryEntry(false);
146        } else if (CpioUtil.byteArray2long(magic, true) == MAGIC_OLD_BINARY) {
147            this.entry = readOldBinaryEntry(true);
148        } else {
149            byte more_magic[] = new byte[4];
150            readFully(more_magic, 0, more_magic.length);
151            byte tmp[] = new byte[6];
152            System.arraycopy(magic, 0, tmp, 0, magic.length);
153            System.arraycopy(more_magic, 0, tmp, magic.length,
154                    more_magic.length);
155            String magicString = ArchiveUtils.toAsciiString(tmp);
156            if (magicString.equals(MAGIC_NEW)) {
157                this.entry = readNewEntry(false);
158            } else if (magicString.equals(MAGIC_NEW_CRC)) {
159                this.entry = readNewEntry(true);
160            } else if (magicString.equals(MAGIC_OLD_ASCII)) {
161                this.entry = readOldAsciiEntry();
162            } else {
163                throw new IOException("Unknown magic [" + magicString + "]. Occured at byte: " + getBytesRead());
164            }
165        }
166
167        this.entryBytesRead = 0;
168        this.entryEOF = false;
169        this.crc = 0;
170
171        if (this.entry.getName().equals(CPIO_TRAILER)) {
172            this.entryEOF = true;
173            return null;
174        }
175        return this.entry;
176    }
177
178    private void skip(int bytes) throws IOException {
179        final byte[] buff = new byte[4]; // Cannot be more than 3 bytes
180        if (bytes > 0) {
181            readFully(buff, 0, bytes);
182        }
183    }
184
185    /**
186     * Reads from the current CPIO entry into an array of bytes. Blocks until
187     * some input is available.
188     *
189     * @param b   the buffer into which the data is read
190     * @param off the start offset of the data
191     * @param len the maximum number of bytes read
192     * @return the actual number of bytes read, or -1 if the end of the entry is
193     * reached
194     * @throws java.io.IOException if an I/O error has occurred or if a CPIO file error has
195     *                             occurred
196     */
197    @Override
198    public int read(final byte[] b, final int off, final int len)
199            throws IOException {
200        ensureOpen();
201        if (off < 0 || len < 0 || off > b.length - len) {
202            throw new IndexOutOfBoundsException();
203        } else if (len == 0) {
204            return 0;
205        }
206
207        if (this.entry == null || this.entryEOF) {
208            return -1;
209        }
210        if (this.entryBytesRead == this.entry.getEntrySize()) {
211            skip(entry.getDataPadCount());
212            this.entryEOF = true;
213            if (this.entry.getFormat() == FORMAT_NEW_CRC
214                    && this.crc != this.entry.getChksum()) {
215                throw new IOException("CRC Error. Occured at byte: "
216                        + getBytesRead());
217            }
218            return -1; // EOF for this entry
219        }
220        int tmplength = (int) Math.min(len, this.entry.getEntrySize()
221                - this.entryBytesRead);
222        if (tmplength < 0) {
223            return -1;
224        }
225
226        int tmpread = readFully(b, off, tmplength);
227        if (this.entry.getFormat() == FORMAT_NEW_CRC) {
228            for (int pos = 0; pos < tmpread; pos++) {
229                this.crc += b[pos] & 0xFF;
230            }
231        }
232        this.entryBytesRead += tmpread;
233
234        return tmpread;
235    }
236
237    private int readFully(final byte[] b, final int off, final int len)
238            throws IOException {
239        if (len < 0) {
240            throw new IndexOutOfBoundsException();
241        }
242        int n = 0;
243        while (n < len) {
244            int count = this.in.read(b, off + n, len - n);
245            count(count);
246            if (count < 0) {
247                throw new EOFException();
248            }
249            n += count;
250        }
251        return n;
252    }
253
254    private long readBinaryLong(final int length, final boolean swapHalfWord)
255            throws IOException {
256        byte tmp[] = new byte[length];
257        readFully(tmp, 0, tmp.length);
258        return CpioUtil.byteArray2long(tmp, swapHalfWord);
259    }
260
261    private long readAsciiLong(final int length, final int radix)
262            throws IOException {
263        byte tmpBuffer[] = new byte[length];
264        readFully(tmpBuffer, 0, tmpBuffer.length);
265        return Long.parseLong(ArchiveUtils.toAsciiString(tmpBuffer), radix);
266    }
267
268    private CpioArchiveEntry readNewEntry(final boolean hasCrc)
269            throws IOException {
270        CpioArchiveEntry ret;
271        if (hasCrc) {
272            ret = new CpioArchiveEntry(FORMAT_NEW_CRC);
273        } else {
274            ret = new CpioArchiveEntry(FORMAT_NEW);
275        }
276
277        ret.setInode(readAsciiLong(8, 16));
278        long mode = readAsciiLong(8, 16);
279        if (mode != 0) { // mode is initialised to 0
280            ret.setMode(mode);
281        }
282        ret.setUID(readAsciiLong(8, 16));
283        ret.setGID(readAsciiLong(8, 16));
284        ret.setNumberOfLinks(readAsciiLong(8, 16));
285        ret.setTime(readAsciiLong(8, 16));
286        ret.setEntrySize(readAsciiLong(8, 16));
287        ret.setDeviceMaj(readAsciiLong(8, 16));
288        ret.setDeviceMin(readAsciiLong(8, 16));
289        ret.setRemoteDeviceMaj(readAsciiLong(8, 16));
290        ret.setRemoteDeviceMin(readAsciiLong(8, 16));
291        long namesize = readAsciiLong(8, 16);
292        ret.setChksum(readAsciiLong(8, 16));
293        String name = readCString((int) namesize);
294        ret.setName(name);
295        if (mode == 0 && !name.equals(CPIO_TRAILER)) {
296            throw new IOException("Mode 0 only allowed in the trailer. Found entry name: " + name + " Occured at byte: " + getBytesRead());
297        }
298        skip(ret.getHeaderPadCount());
299
300        return ret;
301    }
302
303    private CpioArchiveEntry readOldAsciiEntry() throws IOException {
304        CpioArchiveEntry ret = new CpioArchiveEntry(FORMAT_OLD_ASCII);
305
306        ret.setDevice(readAsciiLong(6, 8));
307        ret.setInode(readAsciiLong(6, 8));
308        final long mode = readAsciiLong(6, 8);
309        if (mode != 0) {
310            ret.setMode(mode);
311        }
312        ret.setUID(readAsciiLong(6, 8));
313        ret.setGID(readAsciiLong(6, 8));
314        ret.setNumberOfLinks(readAsciiLong(6, 8));
315        ret.setRemoteDevice(readAsciiLong(6, 8));
316        ret.setTime(readAsciiLong(11, 8));
317        long namesize = readAsciiLong(6, 8);
318        ret.setEntrySize(readAsciiLong(11, 8));
319        final String name = readCString((int) namesize);
320        ret.setName(name);
321        if (mode == 0 && !name.equals(CPIO_TRAILER)) {
322            throw new IOException("Mode 0 only allowed in the trailer. Found entry: " + name + " Occured at byte: " + getBytesRead());
323        }
324
325        return ret;
326    }

328    private CpioArchiveEntry readOldBinaryEntry(final boolean swapHalfWord)
329            throws IOException {
330        CpioArchiveEntry ret = new CpioArchiveEntry(FORMAT_OLD_BINARY);
331
332        ret.setDevice(readBinaryLong(2, swapHalfWord));
333        ret.setInode(readBinaryLong(2, swapHalfWord));
334        final long mode = readBinaryLong(2, swapHalfWord);
335        if (mode != 0) {
336            ret.setMode(mode);
337        }
338        ret.setUID(readBinaryLong(2, swapHalfWord));
339        ret.setGID(readBinaryLong(2, swapHalfWord));
340        ret.setNumberOfLinks(readBinaryLong(2, swapHalfWord));
341        ret.setRemoteDevice(readBinaryLong(2, swapHalfWord));
342        ret.setTime(readBinaryLong(4, swapHalfWord));
343        long namesize = readBinaryLong(2, swapHalfWord);
344        ret.setEntrySize(readBinaryLong(4, swapHalfWord));
345        final String name = readCString((int) namesize);
346        ret.setName(name);
347        if (mode == 0 && !name.equals(CPIO_TRAILER)) {
348            throw new IOException("Mode 0 only allowed in the trailer. Found entry: " + name + "Occured at byte: " + getBytesRead());
349        }
350        skip(ret.getHeaderPadCount());
351
352        return ret;
353    }

355    private String readCString(final int length) throws IOException {
356        byte tmpBuffer[] = new byte[length];
357        readFully(tmpBuffer, 0, tmpBuffer.length);
358        return new String(tmpBuffer, 0, tmpBuffer.length - 1); // TODO default charset?
359    }

361    /**
362     * Skips specified number of bytes in the current CPIO entry.
363     *
364     * @param n the number of bytes to skip
365     * @return the actual number of bytes skipped
366     * @throws java.io.IOException      if an I/O error has occurred
367     * @throws IllegalArgumentException if n < 0
368     */
369    @Override
370    public long skip(final long n) throws IOException {
371        if (n < 0) {
372            throw new IllegalArgumentException("negative skip length");
373        }
374        ensureOpen();
375        int max = (int) Math.min(n, Integer.MAX_VALUE);
376        int total = 0;
377
378        while (total < max) {
379            int len = max - total;
380            if (len > this.tmpbuf.length) {
381                len = this.tmpbuf.length;
382            }
383            len = read(this.tmpbuf, 0, len);
384            if (len == -1) {
385                this.entryEOF = true;
386                break;
387            }
388            total += len;
389        }
390        return total;
391    }
392
393    @Override
394    public ArchiveEntry getNextEntry() throws IOException {
395        return getNextCPIOEntry();
396    }

398    /**
399     * Checks if the signature matches one of the following magic values:
400     * <p/>
401     * Strings:
402     * <p/>
403     * "070701" - MAGIC_NEW
404     * "070702" - MAGIC_NEW_CRC
405     * "070707" - MAGIC_OLD_ASCII
406     * <p/>
407     * Octal Binary value:
408     * <p/>
409     * 070707 - MAGIC_OLD_BINARY (held as a short) = 0x71C7 or 0xC771
410     */
411    public static boolean matches(byte[] signature, int length) {
412        if (length < 6) {
413            return false;
414        }
415
416        // Check binary values
417        if (signature[0] == 0x71 && (signature[1] & 0xFF) == 0xc7) {
418            return true;
419        }
420        if (signature[1] == 0x71 && (signature[0] & 0xFF) == 0xc7) {
421            return true;
422        }
423
424        // Check Ascii (String) values
425        // 3037 3037 30nn
426        if (signature[0] != 0x30) {
427            return false;
428        }
429        if (signature[1] != 0x37) {
430            return false;
431        }
432        if (signature[2] != 0x30) {
433            return false;
434        }
435        if (signature[3] != 0x37) {
436            return false;
437        }
438        if (signature[4] != 0x30) {
439            return false;
440        }
441        // Check last byte
442        if (signature[5] == 0x31) {
443            return true;
444        }
445        if (signature[5] == 0x32) {
446            return true;
447        }
448        if (signature[5] == 0x37) {
449            return true;
450        }
451
452        return false;
453    }
