009    /**
010     * A cpio archive consists of a sequence of files. There are several types of
011     * headers defided in two categories of new and old format. The headers are
012     * recognized by magic numbers:
013     * <p/>
014     * <ul>
015     * <li>"070701" ASCII for new portable format</li>
016     * <li>"070702" ASCII for new portable format with CRC format</li>
017     * <li>"070707" ASCII for old ascii (also known as Portable ASCII, odc or old
018     * character format</li>
019     * <li>070707 binary for old binary</li>
020     * </ul>
021     * <p/>
022     * <p>The old binary format is limited to 16 bits for user id, group
023     * id, device, and inode numbers. It is limited to 4 gigabyte file
024     * sizes.
025     * <p/>
026     * The old ASCII format is limited to 18 bits for the user id, group
027     * id, device, and inode numbers. It is limited to 8 gigabyte file
028     * sizes.
029     * <p/>
030     * The new ASCII format is limited to 4 gigabyte file sizes.
031     * <p/>
032     * CPIO 2.5 knows also about tar, but it is not recognized here.</p>
033     * <p/>
034     * <p/>
035     * <h3>OLD FORMAT</h3>
036     * <p/>
037     * <p>Each file has a 76 (ascii) / 26 (binary) byte header, a variable
038     * length, NUL terminated filename, and variable length file data. A
039     * header for a filename "TRAILER!!!" indicates the end of the
040     * archive.</p>
041     * <p/>
042     * <p>All the fields in the header are ISO 646 (approximately ASCII)
043     * strings of octal numbers, left padded, not NUL terminated.</p>
044     * <p/>
045     * <pre>
046     * FIELDNAME        NOTES
047     * c_magic          The integer value octal 070707.  This value can be used to deter-
048     *                  mine whether this archive is written with little-endian or big-
049     *                  endian integers.
050     * c_dev            Device that contains a directory entry for this file
051     * c_ino            I-node number that identifies the input file to the file system
052     * c_mode           The mode specifies both the regular permissions and the file type.
053     * c_uid            Numeric User ID of the owner of the input file
054     * c_gid            Numeric Group ID of the owner of the input file
055     * c_nlink          Number of links that are connected to the input file
056     * c_rdev           For block special and character special entries, this field
057     *                  contains the associated device number.  For all other entry types,
058     *                  it should be set to zero by writers and ignored by readers.
059     * c_mtime[2]       Modification time of the file, indicated as the number of seconds
060     *                  since the start of the epoch, 00:00:00 UTC January 1, 1970.  The
061     *                  four-byte integer is stored with the most-significant 16 bits
062     *                  first followed by the least-significant 16 bits.  Each of the two
063     *                  16 bit values are stored in machine-native byte order.
064     * c_namesize       Length of the path name, including the terminating null byte
065     * c_filesize[2]    Length of the file in bytes. This is the length of the data
066     *                  section that follows the header structure. Must be 0 for
067     *                  FIFOs and directories
068     *
069     * All fields are unsigned short fields with 16-bit integer values
070     * apart from c_mtime and c_filesize which are 32-bit integer values
071     * </pre>
072     * <p/>
073     * <p>If necessary, the filename and file data are padded with a NUL byte to an even length</p>
074     * <p/>
075     * <p>Special files, directories, and the trailer are recorded with
076     * the h_filesize field equal to 0.</p>
077     * <p/>
078     * <p>In the ASCII version of this format, the 16-bit entries are represented as 6-byte octal numbers,
079     * and the 32-bit entries are represented as 11-byte octal numbers. No padding is added.</p>
080     * <p/>
081     * <h3>NEW FORMAT</h3>
082     * <p/>
083     * <p>Each file has a 110 byte header, a variable length, NUL
084     * terminated filename, and variable length file data. A header for a
085     * filename "TRAILER!!!" indicates the end of the archive. All the
086     * fields in the header are ISO 646 (approximately ASCII) strings of
087     * hexadecimal numbers, left padded, not NUL terminated.</p>
088     * <p/>
089     * <pre>
090     * FIELDNAME        NOTES
091     * c_magic[6]       The string 070701 for new ASCII, the string 070702 for new ASCII with CRC
092     * c_ino[8]
093     * c_mode[8]
094     * c_uid[8]
095     * c_gid[8]
096     * c_nlink[8]
097     * c_mtim[8]
098     * c_filesize[8]    must be 0 for FIFOs and directories
099     * c_maj[8]
100     * c_min[8]
101     * c_rmaj[8]        only valid for chr and blk special files
102     * c_rmin[8]        only valid for chr and blk special files
103     * c_namesize[8]    count includes terminating NUL in pathname
104     * c_check[8]       0 for "new" portable format; for CRC format
105     *                  the sum of all the bytes in the file
106     * </pre>
107     * <p/>
108     * <p>New ASCII Format The "new" ASCII format uses 8-byte hexadecimal
109     * fields for all numbers and separates device numbers into separate
110     * fields for major and minor numbers.</p>
111     * <p/>
112     * <p>The pathname is followed by NUL bytes so that the total size of
113     * the fixed header plus pathname is a multiple of four. Likewise, the
114     * file data is padded to a multiple of four bytes.</p>
115     * <p/>
116     * <p>This class uses mutable fields and is not considered to be
117     * threadsafe.</p>
118     * <p/>
119     * <p>Based on code from the jRPM project (http://jrpm.sourceforge.net).</p>
120     * <p/>
121     * <p>The MAGIC numbers and other constants are defined in {@link CpioConstants}</p>
122     * <p/>
123     * <p>
124     * N.B. does not handle the cpio "tar" format
125     * </p>
126     *
127     * @see <a href="http://people.freebsd.org/~kientzle/libarchive/man/cpio.5.txt">CPIO man page</a>
128     */

180        /**
181         * Creates a CPIOArchiveEntry with a specified format.
182         *
183         * @param format The cpio format for this entry.
184         *               <br/>
185         *               Possible format values are:
186         *               <p/>
187         *               CpioConstants.FORMAT_NEW<br/>
188         *               CpioConstants.FORMAT_NEW_CRC<br/>
189         *               CpioConstants.FORMAT_OLD_BINARY<br/>
190         *               CpioConstants.FORMAT_OLD_ASCII<br/>
191         */
192        public CpioArchiveEntry(final short format) {
193            switch (format) {
194                case FORMAT_NEW:
195                    this.headerSize = 110;
196                    this.alignmentBoundary = 4;
197                    break;
198                case FORMAT_NEW_CRC:
199                    this.headerSize = 110;
200                    this.alignmentBoundary = 4;
201                    break;
202                case FORMAT_OLD_ASCII:
203                    this.headerSize = 76;
204                    this.alignmentBoundary = 0;
205                    break;
206                case FORMAT_OLD_BINARY:
207                    this.headerSize = 26;
208                    this.alignmentBoundary = 2;
209                    break;
210                default:
211                    throw new IllegalArgumentException("Unknown header type");
212            }
213            this.fileFormat = format;
214        }

294         * @param entryName The name of this entry.

298         *                  CpioConstants.FORMAT_NEW<br/>
299         *                  CpioConstants.FORMAT_NEW_CRC<br/>

301         *                  CpioConstants.FORMAT_OLD_ASCII<br/>

303        public CpioArchiveEntry(final short format, File inputFile,
304                                String entryName) {
305            this(format, entryName, inputFile.isFile() ? inputFile.length() : 0);
306            long mode = 0;
307            if (inputFile.isDirectory()) {
308                mode |= C_ISDIR;
309            } else if (inputFile.isFile()) {
310                mode |= C_ISREG;
311            } else {
312                throw new IllegalArgumentException("Cannot determine type of file "
313                        + inputFile.getName());
314            }
315            // TODO set other fields as needed
316            setMode(mode);
317            setTime(inputFile.lastModified() / 1000);
318        }

430        /**
431         * Get the number of bytes needed to pad the header to the alignment boundary.
432         *
433         * @return the number of bytes needed to pad the header (0,1,2,3)
434         */
435        public int getHeaderPadCount() {
436            if (this.alignmentBoundary == 0) {
437                return 0;
438            }
439            int size = this.headerSize + this.name.length() + 1; // Name has terminating null
440            int remain = size % this.alignmentBoundary;
441            if (remain > 0) {
442                return this.alignmentBoundary - remain;
443            }
444            return 0;
445        }
446    
447        /**
448         * Get the number of bytes needed to pad the data to the alignment boundary.
449         *
450         * @return the number of bytes needed to pad the data (0,1,2,3)
451         */
452        public int getDataPadCount() {
453            if (this.alignmentBoundary == 0) {
454                return 0;
455            }
456            long size = this.filesize;
457            int remain = (int) (size % this.alignmentBoundary);
458            if (remain > 0) {
459                return this.alignmentBoundary - remain;
460            }
461            return 0;
462        }

712        /**
713         * Set the mode of this entry (e.g. directory, regular file).
714         *
715         * @param mode The mode to set.
716         */
717        public void setMode(final long mode) {
718            final long maskedMode = mode & S_IFMT;
719            switch ((int) maskedMode) {
720                case C_ISDIR:
721                case C_ISLNK:
722                case C_ISREG:
723                case C_ISFIFO:
724                case C_ISCHR:
725                case C_ISBLK:
726                case C_ISSOCK:
727                case C_ISNWK:
728                    break;
729                default:
730                    throw new IllegalArgumentException(
731                            "Unknown mode. "
732                                    + "Full: " + Long.toHexString(mode)
733                                    + " Masked: " + Long.toHexString(maskedMode));
734            }
735    
736            this.mode = mode;
737        }
