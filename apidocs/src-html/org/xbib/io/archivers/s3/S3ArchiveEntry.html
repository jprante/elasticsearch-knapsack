002    package org.xbib.io.archivers.s3;
003    
004    import org.xbib.io.archivers.ArchiveEntry;
005    
006    import java.util.Date;
007    
008    public class S3ArchiveEntry implements ArchiveEntry {
009    
010        private String name;
011    
012        private long size;
013    
014        private Date lastmodified;
015    
016        public S3ArchiveEntry() {
017        }
018    
019        @Override
020        public ArchiveEntry setName(String name) {
021            this.name = name;
022            return this;
023        }
024    
025        @Override
026        public String getName() {
027            return name;
028        }
029    
030        @Override
031        public ArchiveEntry setEntrySize(long size) {
032            this.size = size;
033            return this;
034        }
035    
036        @Override
037        public long getEntrySize() {
038            return size;
039        }
040    
041        @Override
042        public ArchiveEntry setLastModified(Date date) {
043            this.lastmodified = date;
044            return this;
045        }
046    
047        @Override
048        public Date getLastModified() {
049            return lastmodified;
050        }
051    
052        @Override
053        public boolean isDirectory() {
054            return false;
055        }
056    
057    }
