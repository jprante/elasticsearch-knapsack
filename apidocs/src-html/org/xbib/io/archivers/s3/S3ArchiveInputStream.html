002package org.xbib.io.archivers.s3;
003
004import com.amazonaws.services.s3.AmazonS3Client;
005import com.amazonaws.services.s3.model.GetObjectRequest;
006import com.amazonaws.services.s3.model.S3Object;
007import org.xbib.io.archivers.ArchiveEntry;
008import org.xbib.io.archivers.ArchiveInputStream;
009
010import java.io.IOException;
011import java.io.InputStream;
012
013public class S3ArchiveInputStream extends ArchiveInputStream {
014
015    private final AmazonS3Client client;
016
017    private final String bucketName;
018
019    private final String key;
020
021    private InputStream in;
022
023    private S3Object object;
024
025    S3ArchiveInputStream(AmazonS3Client client, String bucketName, String key) {
026        this.client = client;
027        this.bucketName = bucketName;
028        this.key = key;
029    }
030
031    @Override
032    public ArchiveEntry getNextEntry() throws IOException {
033        GetObjectRequest request = new GetObjectRequest(bucketName, key);
034        object = client.getObject(request);
035        S3ArchiveEntry entry = new S3ArchiveEntry();
036        entry.setName(object.getKey());
037        entry.setLastModified(object.getObjectMetadata().getLastModified());
038        entry.setEntrySize(object.getObjectMetadata().getContentLength());
039        in = object.getObjectContent();
040        return entry;
041    }
042
043    public InputStream getInputStream() {
044        return in;
045    }
046
047    public void close() throws IOException {
048        super.close();
049        if (object != null) {
050            object.close();
051        }
052    }
053
054}
