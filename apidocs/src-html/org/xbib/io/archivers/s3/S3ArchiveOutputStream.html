002    package org.xbib.io.archivers.s3;
003    
004    import com.amazonaws.services.s3.AmazonS3Client;
005    import com.amazonaws.services.s3.model.ObjectMetadata;
006    import com.amazonaws.services.s3.model.PutObjectRequest;
007    import org.xbib.io.archivers.ArchiveEntry;
008    import org.xbib.io.archivers.ArchiveOutputStream;
009    
010    import java.io.FileInputStream;
011    import java.io.IOException;
012    
013    public class S3ArchiveOutputStream extends ArchiveOutputStream {
014    
015        private final AmazonS3Client client;
016    
017        private final String bucketName;
018    
019        private final String key;
020    
021        S3ArchiveOutputStream(AmazonS3Client client, String bucketName, String key) {
022            this.client = client;
023            this.bucketName = bucketName;
024            this.key = key;
025        }
026    
027        @Override
028        public void putArchiveEntry(ArchiveEntry entry) throws IOException {
029            ObjectMetadata metadata = new ObjectMetadata();
030            metadata.setContentLength(entry.getEntrySize());
031            metadata.setLastModified(entry.getLastModified());
032            FileInputStream input = new FileInputStream(entry.getName());
033            PutObjectRequest request = new PutObjectRequest(bucketName, key, input, metadata);
034            client.putObject(request);
035        }
036    
037        @Override
038        public void closeArchiveEntry() throws IOException {
039            // do nothing
040        }
041    
042        @Override
043        public void finish() throws IOException {
044            // do nothing
045        }
046    
047    }
