126        /**
127         * Get the available data that can be read from the current
128         * entry in the archive. This does not indicate how much data
129         * is left in the entire archive, only in the current entry.
130         * This value is determined from the entry's size header field
131         * and the amount of data already read from the current entry.
132         * Integer.MAX_VALUE is returen in case more than Integer.MAX_VALUE
133         * bytes are left in the current entry in the archive.
134         *
135         * @return The number of available bytes for the current entry.
136         * @throws java.io.IOException for signature
137         */
138        @Override
139        public int available() throws IOException {
140            if (entrySize - entryOffset > Integer.MAX_VALUE) {
141                return Integer.MAX_VALUE;
142            }
143            return (int) (entrySize - entryOffset);
144        }

146        /**
147         * Skip bytes in the input buffer. This skips bytes in the
148         * current entry's data, not the entire archive, and will
149         * stop at the end of the current entry's data if the number
150         * to skip extends beyond that point.
151         *
152         * @param numToSkip The number of bytes to skip.
153         * @return the number actually skipped
154         * @throws java.io.IOException on error
155         */
156        @Override
157        public long skip(long numToSkip) throws IOException {
158            // REVIEW
159            // This is horribly inefficient, but it ensures that we
160            // properly skip over bytes via the TarBuffer...
161            //
162            byte[] skipBuf = new byte[BUFFER_SIZE];
163            long skip = numToSkip;
164            while (skip > 0) {
165                int realSkip = (int) (skip > skipBuf.length ? skipBuf.length : skip);
166                int numRead = read(skipBuf, 0, realSkip);
167                if (numRead == -1) {
168                    break;
169                }
170                skip -= numRead;
171            }
172            return (numToSkip - skip);
173        }

182        /**
183         * Get the next entry in this tar archive. This will skip
184         * over any remaining data in the current entry, if there
185         * is one, and place the input stream at the header of the
186         * next entry, and read the header and instantiate a new
187         * TarEntry from the header bytes and return that entry.
188         * If there are no more entries in the archive, null will
189         * be returned to indicate that the end of the archive has
190         * been reached.
191         *
192         * @return The next TarEntry in the archive, or null.
193         * @throws java.io.IOException on error
194         */
195        public TarArchiveEntry getNextTarEntry() throws IOException {
196            if (hasHitEOF) {
197                return null;
198            }
199    
200            if (currEntry != null) {
201                long numToSkip = entrySize - entryOffset;
202    
203                while (numToSkip > 0) {
204                    long skipped = skip(numToSkip);
205                    if (skipped <= 0) {
206                        throw new RuntimeException("failed to skip current tar entry");
207                    }
208                    numToSkip -= skipped;
209                }
210    
211                readBuf = null;
212            }
213    
214            byte[] headerBuf = getRecord();
215    
216            if (hasHitEOF) {
217                currEntry = null;
218                return null;
219            }
220    
221            try {
222                currEntry = new TarArchiveEntry(headerBuf, encoding);
223            } catch (IllegalArgumentException e) {
224                IOException ioe = new IOException("Error detected parsing the header");
225                ioe.initCause(e);
226                throw ioe;
227            }
228            entryOffset = 0;
229            entrySize = currEntry.getEntrySize();
230    
231            if (currEntry.isGNULongNameEntry()) {
232                // read in the name
233                StringBuffer longName = new StringBuffer();
234                byte[] buf = new byte[SMALL_BUFFER_SIZE];
235                int length = 0;
236                while ((length = read(buf)) >= 0) {
237                    longName.append(new String(buf, 0, length)); // TODO default charset?
238                }
239                getNextEntry();
240                if (currEntry == null) {
241                    // Bugzilla: 40334
242                    // Malformed tar file - long entry name not followed by entry
243                    return null;
244                }
245                // remove trailing null terminator
246                if (longName.length() > 0
247                        && longName.charAt(longName.length() - 1) == 0) {
248                    longName.deleteCharAt(longName.length() - 1);
249                }
250                currEntry.setName(longName.toString());
251            }
252    
253            if (currEntry.isPaxHeader()) { // Process Pax headers
254                paxHeaders();
255            }
256    
257            if (currEntry.isGNUSparse()) { // Process sparse files
258                readGNUSparse();
259            }
260    
261            // If the size of the next element in the archive has changed
262            // due to a new size being reported in the posix header
263            // information, we update entrySize here so that it contains
264            // the correct value.
265            entrySize = currEntry.getEntrySize();
266            return currEntry;
267        }

269        /**
270         * Get the next record in this tar archive. This will skip
271         * over any remaining data in the current entry, if there
272         * is one, and place the input stream at the header of the
273         * next entry.
274         * If there are no more entries in the archive, null will
275         * be returned to indicate that the end of the archive has
276         * been reached.
277         *
278         * @return The next header in the archive, or null.
279         * @throws java.io.IOException on error
280         */
281        private byte[] getRecord() throws IOException {
282            if (hasHitEOF) {
283                return null;
284            }
285    
286            byte[] headerBuf = buffer.readRecord();
287    
288            if (headerBuf == null) {
289                hasHitEOF = true;
290            } else if (buffer.isEOFRecord(headerBuf)) {
291                hasHitEOF = true;
292            }
293    
294            return hasHitEOF ? null : headerBuf;
295        }

297        private void paxHeaders() throws IOException {
298            Map<String, String> headers = parsePaxHeaders(this);
299            getNextEntry(); // Get the actual file entry
300            applyPaxHeadersToCurrentEntry(headers);
301        }
302    
303        Map<String, String> parsePaxHeaders(InputStream i) throws IOException {
304            Map<String, String> headers = new HashMap<String, String>();
305            // Format is "length keyword=value\n";
306            while (true) { // get length
307                int ch;
308                int len = 0;
309                int read = 0;
310                while ((ch = i.read()) != -1) {
311                    read++;
312                    if (ch == ' ') { // End of length string
313                        // Get keyword
314                        ByteArrayOutputStream coll = new ByteArrayOutputStream();
315                        while ((ch = i.read()) != -1) {
316                            read++;
317                            if (ch == '=') { // end of keyword
318                                String keyword = coll.toString("UTF-8");
319                                // Get rest of entry
320                                byte[] rest = new byte[len - read];
321                                int got = i.read(rest);
322                                if (got != len - read) {
323                                    throw new IOException("Failed to read "
324                                            + "Paxheader. Expected "
325                                            + (len - read)
326                                            + " bytes, read "
327                                            + got);
328                                }
329                                // Drop trailing NL
330                                String value = new String(rest, 0,
331                                        len - read - 1, Charset.forName("UTF-8"));
332                                headers.put(keyword, value);
333                                break;
334                            }
335                            coll.write((byte) ch);
336                        }
337                        break; // Processed single header
338                    }
339                    len *= 10;
340                    len += ch - '0';
341                }
342                if (ch == -1) { // EOF
343                    break;
344                }
345            }
346            return headers;
347        }

349        private void applyPaxHeadersToCurrentEntry(Map<String, String> headers) {
350            /*
351             * The following headers are defined for Pax.
352             * atime, ctime, charset: cannot use these without changing TarArchiveEntry fields
353             * mtime
354             * comment
355             * gid, gname
356             * linkpath
357             * size
358             * uid,uname
359             * SCHILY.devminor, SCHILY.devmajor: don't have setters/getters for those
360             */
361            for (Entry<String, String> ent : headers.entrySet()) {
362                String key = ent.getKey();
363                String val = ent.getValue();
364                if ("path".equals(key)) {
365                    currEntry.setName(val);
366                } else if ("linkpath".equals(key)) {
367                    currEntry.setLinkName(val);
368                } else if ("gid".equals(key)) {
369                    currEntry.setGroupId(Integer.parseInt(val));
370                } else if ("gname".equals(key)) {
371                    currEntry.setGroupName(val);
372                } else if ("uid".equals(key)) {
373                    currEntry.setUserId(Integer.parseInt(val));
374                } else if ("uname".equals(key)) {
375                    currEntry.setUserName(val);
376                } else if ("size".equals(key)) {
377                    currEntry.setEntrySize(Long.parseLong(val));
378                } else if ("mtime".equals(key)) {
379                    long mtime = (long) (Double.parseDouble(val) * 1000);
380                    currEntry.setLastModified(new Date(mtime));
381                } else if ("SCHILY.devminor".equals(key)) {
382                    currEntry.setDevMinor(Integer.parseInt(val));
383                } else if ("SCHILY.devmajor".equals(key)) {
384                    currEntry.setDevMajor(Integer.parseInt(val));
385                }
386            }
387        }

389        /**
390         * Adds the sparse chunks from the current entry to the sparse chunks,
391         * including any additional sparse entries following the current entry.
392         *
393         * @throws java.io.IOException on error
394         */
395        private void readGNUSparse() throws IOException {
396            /* we do not really process sparse files yet
397            sparses = new ArrayList();
398            sparses.addAll(currEntry.getSparses());
399            */
400            if (currEntry.isExtended()) {
401                TarArchiveSparseEntry entry;
402                do {
403                    byte[] headerBuf = getRecord();
404                    if (hasHitEOF) {
405                        currEntry = null;
406                        break;
407                    }
408                    entry = new TarArchiveSparseEntry(headerBuf);
409                    /* we do not really process sparse files yet
410                    sparses.addAll(entry.getSparses());
411                    */
412                } while (entry.isExtended());
413            }
414        }

421        /**
422         * Reads bytes from the current tar archive entry.
423         * <p/>
424         * This method is aware of the boundaries of the current
425         * entry in the archive and will deal with them as if they
426         * were this stream's start and EOF.
427         *
428         * @param buf       The buffer into which to place bytes read.
429         * @param offset    The offset at which to place bytes read.
430         * @param numToRead The number of bytes to read.
431         * @return The number of bytes read, or -1 at EOF.
432         * @throws java.io.IOException on error
433         */
434        @Override
435        public int read(byte[] buf, int offset, int numToRead) throws IOException {
436            int totalRead = 0;
437    
438            if (entryOffset >= entrySize) {
439                return -1;
440            }
441    
442            if ((numToRead + entryOffset) > entrySize) {
443                numToRead = (int) (entrySize - entryOffset);
444            }
445    
446            if (readBuf != null) {
447                int sz = (numToRead > readBuf.length) ? readBuf.length
448                        : numToRead;
449    
450                System.arraycopy(readBuf, 0, buf, offset, sz);
451    
452                if (sz >= readBuf.length) {
453                    readBuf = null;
454                } else {
455                    int newLen = readBuf.length - sz;
456                    byte[] newBuf = new byte[newLen];
457    
458                    System.arraycopy(readBuf, sz, newBuf, 0, newLen);
459    
460                    readBuf = newBuf;
461                }
462    
463                totalRead += sz;
464                numToRead -= sz;
465                offset += sz;
466            }
467    
468            while (numToRead > 0) {
469                byte[] rec = buffer.readRecord();
470    
471                if (rec == null) {
472                    // Unexpected EOF!
473                    throw new IOException("unexpected EOF with " + numToRead
474                            + " bytes unread. Occured at byte: " + getBytesRead());
475                }
476                count(rec.length);
477                int sz = numToRead;
478                int recLen = rec.length;
479    
480                if (recLen > sz) {
481                    System.arraycopy(rec, 0, buf, offset, sz);
482    
483                    readBuf = new byte[recLen - sz];
484    
485                    System.arraycopy(rec, sz, readBuf, 0, recLen - sz);
486                } else {
487                    sz = recLen;
488    
489                    System.arraycopy(rec, 0, buf, offset, recLen);
490                }
491    
492                totalRead += sz;
493                numToRead -= sz;
494                offset += sz;
495            }
496    
497            entryOffset += totalRead;
498    
499            return totalRead;
500        }

518        /**
519         * Checks if the signature matches what is expected for a tar file.
520         *
521         * @param signature the bytes to check
522         * @param length    the number of bytes to check
523         * @return true, if this stream is a tar archive stream, false otherwise
524         */
525        public static boolean matches(byte[] signature, int length) {
526            return length >= TarConstants.VERSION_OFFSET + TarConstants.VERSIONLEN
527                    && (ArchiveUtils.matchAsciiBuffer(TarConstants.MAGIC_POSIX, signature, TarConstants.MAGIC_OFFSET, TarConstants.MAGICLEN)
528                    && ArchiveUtils.matchAsciiBuffer(TarConstants.VERSION_POSIX, signature, TarConstants.VERSION_OFFSET, TarConstants.VERSIONLEN)
529                    || ArchiveUtils.matchAsciiBuffer(TarConstants.MAGIC_GNU, signature, TarConstants.MAGIC_OFFSET, TarConstants.MAGICLEN)
530                    && (ArchiveUtils.matchAsciiBuffer(TarConstants.VERSION_GNU_SPACE, signature, TarConstants.VERSION_OFFSET, TarConstants.VERSIONLEN)
531                    || ArchiveUtils.matchAsciiBuffer(TarConstants.VERSION_GNU_ZERO, signature, TarConstants.VERSION_OFFSET, TarConstants.VERSIONLEN))
532                    || ArchiveUtils.matchAsciiBuffer(TarConstants.MAGIC_ANT, signature, TarConstants.MAGIC_OFFSET, TarConstants.MAGICLEN)
533                    && ArchiveUtils.matchAsciiBuffer(TarConstants.VERSION_ANT, signature, TarConstants.VERSION_OFFSET, TarConstants.VERSIONLEN));
534    
535        }
