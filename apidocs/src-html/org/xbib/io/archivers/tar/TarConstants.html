002package org.xbib.io.archivers.tar;
003
004/**
005 * This interface contains all the definitions used in the package.
006 * <p/>
007 * For tar formats (FORMAT_OLDGNU, FORMAT_POSIX, etc.) see GNU tar
008 * <I>tar.h</I> type <I>enum archive_format</I>
009 */
010public interface TarConstants {
011
012    /**
013     * GNU format as per before tar 1.12.
014     */
015    int FORMAT_OLDGNU = 2;
016
017    /**
018     * Pure Posix format.
019     */
020    int FORMAT_POSIX = 3;
021
022    /**
023     * The length of the name field in a header buffer.
024     */
025    int NAMELEN = 100;
026
027    /**
028     * The length of the mode field in a header buffer.
029     */
030    int MODELEN = 8;
031
032    /**
033     * The length of the user id field in a header buffer.
034     */
035    int UIDLEN = 8;
036
037    /**
038     * The length of the group id field in a header buffer.
039     */
040    int GIDLEN = 8;
041
042    /**
043     * The maximum value of gid/uid in a tar archive which can
044     * be expressed in octal char notation (that's 7 sevens, octal).
045     */
046    long MAXID = 07777777L;
047
048    /**
049     * The length of the checksum field in a header buffer.
050     */
051    int CHKSUMLEN = 8;
052
053    /**
054     * The length of the size field in a header buffer.
055     * Includes the trailing space or NUL.
056     */
057    int SIZELEN = 12;
058
059    /**
060     * The maximum size of a file in a tar archive
061     * which can be expressed in octal char notation (that's 11 sevens, octal).
062     */
063    long MAXSIZE = 077777777777L;
064
065    /**
066     * Offset of start of magic field within header record
067     */
068    int MAGIC_OFFSET = 257;
069    /**
070     * The length of the magic field in a header buffer.
071     */
072    int MAGICLEN = 6;
073
074    /**
075     * Offset of start of magic field within header record
076     */
077    int VERSION_OFFSET = 263;
078    /**
079     * Previously this was regarded as part of "magic" field, but it is separate.
080     */
081    int VERSIONLEN = 2;
082
083    /**
084     * The length of the modification time field in a header buffer.
085     */
086    int MODTIMELEN = 12;
087
088    /**
089     * The length of the user name field in a header buffer.
090     */
091    int UNAMELEN = 32;
092
093    /**
094     * The length of the group name field in a header buffer.
095     */
096    int GNAMELEN = 32;
097
098    /**
099     * The length of each of the device fields (major and minor) in a header buffer.
100     */
101    int DEVLEN = 8;
102
103    /**
104     * Length of the prefix field.
105     */
106    int PREFIXLEN = 155;
107
108    /**
109     * The length of the access time field in an old GNU header buffer.
110     */
111    int ATIMELEN_GNU = 12;
112
113    /**
114     * The length of the created time field in an old GNU header buffer.
115     */
116    int CTIMELEN_GNU = 12;
117
118    /**
119     * The length of the multivolume start offset field in an old GNU header buffer.
120     */
121    int OFFSETLEN_GNU = 12;
122
123    /**
124     * The length of the long names field in an old GNU header buffer.
125     */
126    int LONGNAMESLEN_GNU = 4;
127
128    /**
129     * The length of the padding field in an old GNU header buffer.
130     */
131    int PAD2LEN_GNU = 1;
132
133    /**
134     * The sum of the length of all sparse headers in an old GNU header buffer.
135     */
136    int SPARSELEN_GNU = 96;
137
138    /**
139     * The length of the is extension field in an old GNU header buffer.
140     */
141    int ISEXTENDEDLEN_GNU = 1;
142
143    /**
144     * The length of the real size field in an old GNU header buffer.
145     */
146    int REALSIZELEN_GNU = 12;
147
148    /**
149     * The sum of the length of all sparse headers in a sparse header buffer.
150     */
151    int SPARSELEN_GNU_SPARSE = 504;
152
153    /**
154     * The length of the is extension field in a sparse header buffer.
155     */
156    int ISEXTENDEDLEN_GNU_SPARSE = 1;
157
158    /**
159     * LF_ constants represent the "link flag" of an entry, or more commonly,
160     * the "entry type". This is the "old way" of indicating a normal file.
161     */
162    byte LF_OLDNORM = 0;
163
164    /**
165     * Normal file type.
166     */
167    byte LF_NORMAL = (byte) '0';
168
169    /**
170     * Link file type.
171     */
172    byte LF_LINK = (byte) '1';
173
174    /**
175     * Symbolic link file type.
176     */
177    byte LF_SYMLINK = (byte) '2';
178
179    /**
180     * Character device file type.
181     */
182    byte LF_CHR = (byte) '3';
183
184    /**
185     * Block device file type.
186     */
187    byte LF_BLK = (byte) '4';
188
189    /**
190     * Directory file type.
191     */
192    byte LF_DIR = (byte) '5';
193
194    /**
195     * FIFO (pipe) file type.
196     */
197    byte LF_FIFO = (byte) '6';
198
199    /**
200     * Contiguous file type.
201     */
202    byte LF_CONTIG = (byte) '7';
203
204    /**
205     * Identifies the *next* file on the tape as having a long name.
206     */
207    byte LF_GNUTYPE_LONGNAME = (byte) 'L';
208
209    /**
210     * Sparse file type.
211     */
212    byte LF_GNUTYPE_SPARSE = (byte) 'S';
213
214    // See "http://www.opengroup.org/onlinepubs/009695399/utilities/pax.html#tag_04_100_13_02"
215
216    /**
217     * Identifies the entry as a Pax extended header.
218     */
219    byte LF_PAX_EXTENDED_HEADER_LC = (byte) 'x';
220
221    /**
222     * Identifies the entry as a Pax extended header (SunOS tar -E).
223     */
224    byte LF_PAX_EXTENDED_HEADER_UC = (byte) 'X';
225
226    /**
227     * Identifies the entry as a Pax global extended header.
228     */
229    byte LF_PAX_GLOBAL_EXTENDED_HEADER = (byte) 'g';
230
231    /**
232     * The magic tag representing a POSIX tar archive.
233     */
234    String MAGIC_POSIX = "ustar\0";
235    String VERSION_POSIX = "00";
236
237    /**
238     * The magic tag representing a GNU tar archive.
239     */
240    String MAGIC_GNU = "ustar ";
241    // Appear to be two possible GNU versions
242    String VERSION_GNU_SPACE = " \0";
243    String VERSION_GNU_ZERO = "0\0";
244
245    /**
246     * The magic tag representing an Ant tar archive.
247     */
248    String MAGIC_ANT = "ustar\0";
249
250    /**
251     * The "version" representing an Ant tar archive.
252     */
253    // Does not appear to have a version, however Ant does write 8 bytes,
254    // so assume the version is 2 nulls
255    String VERSION_ANT = "\0\0";
256
257    /**
258     * The name of the GNU tar entry which contains a long name.
259     */
260    String GNU_LONGLINK = "././@LongLink";
261
262}
