013/**
014 * This class represents an entry in a Tar archive. It consists
015 * of the entry's header, as well as the entry's File. Entries
016 * can be instantiated in one of three ways, depending on how
017 * they are to be used.
018 * <p/>
019 * TarEntries that are created from the header bytes read from
020 * an archive are instantiated with the TarEntry( byte[] )
021 * constructor. These entries will be used when extracting from
022 * or listing the contents of an archive. These entries have their
023 * header filled in using the header bytes. They also set the File
024 * to null, since they reference an archive entry not a file.
025 * <p/>
026 * TarEntries that are created from Files that are to be written
027 * into an archive are instantiated with the TarEntry( File )
028 * constructor. These entries have their header filled in using
029 * the File's information. They also keep a reference to the File
030 * for convenience when writing entries.
031 * <p/>
032 * Finally, TarEntries can be constructed from nothing but a name.
033 * This allows the programmer to construct the entry by hand, for
034 * instance when only an InputStream is available for writing to
035 * the archive, and the header information is constructed from
036 * other information. In this case the header fields are set to
037 * defaults and the File is set to null.
038 * <p/>
039 * <p/>
040 * The C structure for a Tar Entry's header is:
041 * <pre>
042 * struct header {
043 * char name[100];     // TarConstants.NAMELEN    - offset   0
044 * char mode[8];       // TarConstants.MODELEN    - offset 100
045 * char uid[8];        // TarConstants.UIDLEN     - offset 108
046 * char gid[8];        // TarConstants.GIDLEN     - offset 116
047 * char size[12];      // TarConstants.SIZELEN    - offset 124
048 * char mtime[12];     // TarConstants.MODTIMELEN - offset 136
049 * char chksum[8];     // TarConstants.CHKSUMLEN  - offset 148
050 * char linkflag[1];   //                         - offset 156
051 * char linkname[100]; // TarConstants.NAMELEN    - offset 157
052 * The following fields are only present in new-style POSIX tar archives:
053 * char magic[6];      // TarConstants.MAGICLEN   - offset 257
054 * char version[2];    // TarConstants.VERSIONLEN - offset 263
055 * char uname[32];     // TarConstants.UNAMELEN   - offset 265
056 * char gname[32];     // TarConstants.GNAMELEN   - offset 297
057 * char devmajor[8];   // TarConstants.DEVLEN     - offset 329
058 * char devminor[8];   // TarConstants.DEVLEN     - offset 337
059 * char prefix[155];   // TarConstants.PREFIXLEN  - offset 345
060 * // Used if "name" field is not long enough to hold the path
061 * char pad[12];       // NULs                    - offset 500
062 * } header;
063 * All unused bytes are set to null.
064 * New-style GNU tar files are slightly different from the above.
065 * For values of size larger than 077777777777L (11 7s)
066 * or uid and gid larger than 07777777L (7 7s)
067 * the sign bit of the first byte is set, and the rest of the
068 * field is the binary representation of the number.
069 * See TarUtils.parseOctalOrBinary.
070 * </pre>
071 * <p/>
072 * <p/>
073 * The C structure for a old GNU Tar Entry's header is:
074 * <pre>
075 * struct oldgnu_header {
076 * char unused_pad1[345]; // TarConstants.PAD1LEN_GNU       - offset 0
077 * char atime[12];        // TarConstants.ATIMELEN_GNU      - offset 345
078 * char ctime[12];        // TarConstants.CTIMELEN_GNU      - offset 357
079 * char offset[12];       // TarConstants.OFFSETLEN_GNU     - offset 369
080 * char longnames[4];     // TarConstants.LONGNAMESLEN_GNU  - offset 381
081 * char unused_pad2;      // TarConstants.PAD2LEN_GNU       - offset 385
082 * struct sparse sp[4];   // TarConstants.SPARSELEN_GNU     - offset 386
083 * char isextended;       // TarConstants.ISEXTENDEDLEN_GNU - offset 482
084 * char realsize[12];     // TarConstants.REALSIZELEN_GNU   - offset 483
085 * char unused_pad[17];   // TarConstants.PAD3LEN_GNU       - offset 495
086 * };
087 * </pre>
088 * Whereas, "struct sparse" is:
089 * <pre>
090 * struct sparse {
091 * char offset[12];   // offset 0
092 * char numbytes[12]; // offset 12
093 * };
094 * </pre>
095 */

206    public TarArchiveEntry() {
207        this.magic = MAGIC_POSIX;
208        this.version = VERSION_POSIX;
209        this.name = "";
210        this.linkName = "";
211        String user = System.getProperty("user.name", "");
212        if (user.length() > MAX_NAMELEN) {
213            user = user.substring(0, MAX_NAMELEN);
214        }
215        this.userId = 0;
216        this.groupId = 0;
217        this.userName = user;
218        this.groupName = "";
219        this.file = null;

234     * to construct the entry's header "by hand". File is set to null.

241        this();
242
243        name = normalizeFileName(name, preserveLeadingSlashes);
244        boolean isDir = name.endsWith("/");
245
246        this.devMajor = 0;
247        this.devMinor = 0;
248        this.name = name;
249        this.mode = isDir ? DEFAULT_DIR_MODE : DEFAULT_FILE_MODE;
250        this.linkFlag = isDir ? LF_DIR : LF_NORMAL;
251        this.userId = 0;
252        this.groupId = 0;
253        this.size = 0;
254        this.modTime = (new Date()).getTime() / MILLIS_PER_SECOND;
255        this.linkName = "";
256        this.userName = "";
257        this.groupName = "";
258    }

286    /**
287     * Construct an entry for a file. File is set to file, and the
288     * header is constructed from information from the file.
289     *
290     * @param file     The file that the entry represents.
291     * @param fileName the name to be used for the entry.
292     */
293    public TarArchiveEntry(File file, String fileName) {
294        this();
295
296        this.file = file;
297
298        this.linkName = "";
299
300        if (file.isDirectory()) {
301            this.mode = DEFAULT_DIR_MODE;
302            this.linkFlag = LF_DIR;
303
304            int nameLength = fileName.length();
305            if (nameLength == 0 || fileName.charAt(nameLength - 1) != '/') {
306                this.name = fileName + "/";
307            } else {
308                this.name = fileName;
309            }
310            this.size = 0;
311        } else {
312            this.mode = DEFAULT_FILE_MODE;
313            this.linkFlag = LF_NORMAL;
314            this.size = file.length();
315            this.name = fileName;
316        }
317
318        this.modTime = file.lastModified() / MILLIS_PER_SECOND;
319        this.devMajor = 0;
320        this.devMinor = 0;
321    }

762    /**
763     * If this entry represents a file, and the file is a directory, return
764     * an array of TarEntries for this entry's children.
765     *
766     * @return An array of TarEntry's for this entry's children.
767     */
768    public TarArchiveEntry[] getDirectoryEntries() {
769        if (file == null || !file.isDirectory()) {
770            return new TarArchiveEntry[0];
771        }
772
773        String[] list = file.list();
774        TarArchiveEntry[] result = new TarArchiveEntry[list.length];
775
776        for (int i = 0; i < list.length; ++i) {
777            result[i] = new TarArchiveEntry(new File(file, list[i]));
778        }
779
780        return result;
781    }

783    /**
784     * Write an entry's header information to a header buffer.
785     * <p/>
786     * <p>This method does not use the star/GNU tar/BSD tar extensions.</p>
787     *
788     * @param outbuf The tar entry header buffer to fill in.
789     */
790    public void writeEntryHeader(byte[] outbuf) {
791        try {
792            writeEntryHeader(outbuf, TarUtils.DEFAULT_ENCODING, false);
793        } catch (IOException ex) {
794            try {
795                writeEntryHeader(outbuf, TarUtils.FALLBACK_ENCODING, false);
796            } catch (IOException ex2) {
797                // impossible
798                throw new RuntimeException(ex2);
799            }
800        }
801    }
802
803    /**
804     * Write an entry's header information to a header buffer.
805     *
806     * @param outbuf   The tar entry header buffer to fill in.
807     * @param encoding encoding to use when writing the file name.
808     * @param starMode whether to use the star/GNU tar/BSD tar
809     *                 extension for numeric fields if their value doesn't fit in the
810     *                 maximum size of standard tar archives
811     */
812    public void writeEntryHeader(byte[] outbuf, ZipEncoding encoding,
813                                 boolean starMode) throws IOException {
814        int offset = 0;
815
816        offset = TarUtils.formatNameBytes(name, outbuf, offset, NAMELEN,
817                encoding);
818        offset = writeEntryHeaderField(mode, outbuf, offset, MODELEN, starMode);
819        offset = writeEntryHeaderField(userId, outbuf, offset, UIDLEN,
820                starMode);
821        offset = writeEntryHeaderField(groupId, outbuf, offset, GIDLEN,
822                starMode);
823        offset = writeEntryHeaderField(size, outbuf, offset, SIZELEN, starMode);
824        offset = writeEntryHeaderField(modTime, outbuf, offset, MODTIMELEN,
825                starMode);
826
827        int csOffset = offset;
828
829        for (int c = 0; c < CHKSUMLEN; ++c) {
830            outbuf[offset++] = (byte) ' ';
831        }
832
833        outbuf[offset++] = linkFlag;
834        offset = TarUtils.formatNameBytes(linkName, outbuf, offset, NAMELEN,
835                encoding);
836        offset = TarUtils.formatNameBytes(magic, outbuf, offset, MAGICLEN);
837        offset = TarUtils.formatNameBytes(version, outbuf, offset, VERSIONLEN);
838        offset = TarUtils.formatNameBytes(userName, outbuf, offset, UNAMELEN,
839                encoding);
840        offset = TarUtils.formatNameBytes(groupName, outbuf, offset, GNAMELEN,
841                encoding);
842        offset = writeEntryHeaderField(devMajor, outbuf, offset, DEVLEN,
843                starMode);
844        offset = writeEntryHeaderField(devMinor, outbuf, offset, DEVLEN,
845                starMode);
846
847        while (offset < outbuf.length) {
848            outbuf[offset++] = 0;
849        }
850
851        long chk = TarUtils.computeCheckSum(outbuf);
852
853        TarUtils.formatCheckSumOctalBytes(chk, outbuf, csOffset, CHKSUMLEN);
854    }
855
856    private int writeEntryHeaderField(long value, byte[] outbuf, int offset,
857                                      int length, boolean starMode) {
858        if (!starMode && (value < 0
859                || value >= (1l << (3 * (length - 1))))) {
860            // value doesn't fit into field when written as octal
861            // number, will be written to PAX header or causes an
862            // error
863            return TarUtils.formatLongOctalBytes(0, outbuf, offset, length);
864        }
865        return TarUtils.formatLongOctalOrBinaryBytes(value, outbuf, offset,
866                length);
867    }

870     * Parse an entry's header information from a header buffer.
871     *
872     * @param header The tar entry header buffer to get information from.
873     * @throws IllegalArgumentException if any of the numeric fields have an invalid format
874     */
875    public void parseTarHeader(byte[] header) {
876        try {
877            parseTarHeader(header, TarUtils.DEFAULT_ENCODING);
878        } catch (IOException ex) {
879            try {
880                parseTarHeader(header, TarUtils.DEFAULT_ENCODING, true);
881            } catch (IOException ex2) {
882                // not really possible
883                throw new RuntimeException(ex2);
884            }
885        }

888    /**
889     * Parse an entry's header information from a header buffer.
890     *
891     * @param header   The tar entry header buffer to get information from.
892     * @param encoding encoding to use for file names
893     * @throws IllegalArgumentException if any of the numeric fields
894     *                                  have an invalid format
895     */
896    public void parseTarHeader(byte[] header, ZipEncoding encoding)
897            throws IOException {
898        parseTarHeader(header, encoding, false);
899    }
900
901    private void parseTarHeader(byte[] header, ZipEncoding encoding,
902                                final boolean oldStyle)
903            throws IOException {
904        int offset = 0;
905
906        name = oldStyle ? TarUtils.parseName(header, offset, NAMELEN)
907                : TarUtils.parseName(header, offset, NAMELEN, encoding);
908        offset += NAMELEN;
909        mode = (int) TarUtils.parseOctalOrBinary(header, offset, MODELEN);
910        offset += MODELEN;
911        userId = (int) TarUtils.parseOctalOrBinary(header, offset, UIDLEN);
912        offset += UIDLEN;
913        groupId = (int) TarUtils.parseOctalOrBinary(header, offset, GIDLEN);
914        offset += GIDLEN;
915        size = TarUtils.parseOctalOrBinary(header, offset, SIZELEN);
916        offset += SIZELEN;
917        modTime = TarUtils.parseOctalOrBinary(header, offset, MODTIMELEN);
918        offset += MODTIMELEN;
919        offset += CHKSUMLEN;
920        linkFlag = header[offset++];
921        linkName = oldStyle ? TarUtils.parseName(header, offset, NAMELEN)
922                : TarUtils.parseName(header, offset, NAMELEN, encoding);
923        offset += NAMELEN;
924        magic = TarUtils.parseName(header, offset, MAGICLEN);
925        offset += MAGICLEN;
926        version = TarUtils.parseName(header, offset, VERSIONLEN);
927        offset += VERSIONLEN;
928        userName = oldStyle ? TarUtils.parseName(header, offset, UNAMELEN)
929                : TarUtils.parseName(header, offset, UNAMELEN, encoding);
930        offset += UNAMELEN;
931        groupName = oldStyle ? TarUtils.parseName(header, offset, GNAMELEN)
932                : TarUtils.parseName(header, offset, GNAMELEN, encoding);
933        offset += GNAMELEN;
934        devMajor = (int) TarUtils.parseOctalOrBinary(header, offset, DEVLEN);
935        offset += DEVLEN;
936        devMinor = (int) TarUtils.parseOctalOrBinary(header, offset, DEVLEN);
937        offset += DEVLEN;
938
939        int type = evaluateType(header);
940        switch (type) {
941            case FORMAT_OLDGNU: {
942                offset += ATIMELEN_GNU;
943                offset += CTIMELEN_GNU;
944                offset += OFFSETLEN_GNU;
945                offset += LONGNAMESLEN_GNU;
946                offset += PAD2LEN_GNU;
947                offset += SPARSELEN_GNU;
948                isExtended = TarUtils.parseBoolean(header, offset);
949                offset += ISEXTENDEDLEN_GNU;
950                realSize = TarUtils.parseOctal(header, offset, REALSIZELEN_GNU);
951                offset += REALSIZELEN_GNU;
952                break;
953            }
954            case FORMAT_POSIX:
955            default: {
956                String prefix = oldStyle
957                        ? TarUtils.parseName(header, offset, PREFIXLEN)
958                        : TarUtils.parseName(header, offset, PREFIXLEN, encoding);
959                // SunOS tar -E does not add / to directory names, so fix
960                // up to be consistent
961                if (isDirectory() && !name.endsWith("/")) {
962                    name = name + "/";
963                }
964                if (prefix.length() > 0) {
965                    name = prefix + "/" + name;
966                }
967            }
968        }
969    }

971    /**
972     * Strips Windows' drive letter as well as any leading slashes,
973     * turns path separators into forward slahes.
974     */
975    private static String normalizeFileName(String fileName,
976                                            boolean preserveLeadingSlashes) {
977        String osname = System.getProperty("os.name").toLowerCase(Locale.ENGLISH);
978
979        // Strip off drive letters!
980        // REVIEW Would a better check be "(File.separator == '\')"?
981
982        if (osname.startsWith("windows")) {
983            if (fileName.length() > 2) {
984                char ch1 = fileName.charAt(0);
985                char ch2 = fileName.charAt(1);
986
987                if (ch2 == ':'
988                        && ((ch1 >= 'a' && ch1 <= 'z')
989                        || (ch1 >= 'A' && ch1 <= 'Z'))) {
990                    fileName = fileName.substring(2);
991                }
992            }
993        } else if (osname.indexOf("netware") > -1) {
994            int colon = fileName.indexOf(':');
995            if (colon != -1) {
996                fileName = fileName.substring(colon + 1);
997            }
998        }
999
1000        fileName = fileName.replace(File.separatorChar, '/');
1001
1002        // No absolute pathnames
1003        // Windows (and Posix?) paths can start with "\\NetworkDrive\",
1004        // so we loop on starting /'s.
1005        while (!preserveLeadingSlashes && fileName.startsWith("/")) {
1006            fileName = fileName.substring(1);
1007        }
1008        return fileName;
1009    }

1011    /**
1012     * Evaluate an entry's header format from a header buffer.
1013     *
1014     * @param header The tar entry header buffer to evaluate the format for.
1015     * @return format type
1016     */
1017    private int evaluateType(byte[] header) {
1018        if (ArchiveUtils.matchAsciiBuffer(MAGIC_GNU, header, MAGIC_OFFSET, MAGICLEN)) {
1019            return FORMAT_OLDGNU;
1020        }
1021        if (ArchiveUtils.matchAsciiBuffer(MAGIC_POSIX, header, MAGIC_OFFSET, MAGICLEN)) {
1022            return FORMAT_POSIX;
1023        }
1024        return 0;
1025    }
