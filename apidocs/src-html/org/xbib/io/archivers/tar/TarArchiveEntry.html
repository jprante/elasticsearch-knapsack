013    /**
014     * This class represents an entry in a Tar archive. It consists
015     * of the entry's header, as well as the entry's File. Entries
016     * can be instantiated in one of three ways, depending on how
017     * they are to be used.
018     * <p/>
019     * TarEntries that are created from the header bytes read from
020     * an archive are instantiated with the TarEntry( byte[] )
021     * constructor. These entries will be used when extracting from
022     * or listing the contents of an archive. These entries have their
023     * header filled in using the header bytes. They also set the File
024     * to null, since they reference an archive entry not a file.
025     * <p/>
026     * TarEntries that are created from Files that are to be written
027     * into an archive are instantiated with the TarEntry( File )
028     * constructor. These entries have their header filled in using
029     * the File's information. They also keep a reference to the File
030     * for convenience when writing entries.
031     * <p/>
032     * Finally, TarEntries can be constructed from nothing but a name.
033     * This allows the programmer to construct the entry by hand, for
034     * instance when only an InputStream is available for writing to
035     * the archive, and the header information is constructed from
036     * other information. In this case the header fields are set to
037     * defaults and the File is set to null.
038     * <p/>
039     * <p/>
040     * The C structure for a Tar Entry's header is:
041     * <pre>
042     * struct header {
043     * char name[100];     // TarConstants.NAMELEN    - offset   0
044     * char mode[8];       // TarConstants.MODELEN    - offset 100
045     * char uid[8];        // TarConstants.UIDLEN     - offset 108
046     * char gid[8];        // TarConstants.GIDLEN     - offset 116
047     * char size[12];      // TarConstants.SIZELEN    - offset 124
048     * char mtime[12];     // TarConstants.MODTIMELEN - offset 136
049     * char chksum[8];     // TarConstants.CHKSUMLEN  - offset 148
050     * char linkflag[1];   //                         - offset 156
051     * char linkname[100]; // TarConstants.NAMELEN    - offset 157
052     * The following fields are only present in new-style POSIX tar archives:
053     * char magic[6];      // TarConstants.MAGICLEN   - offset 257
054     * char version[2];    // TarConstants.VERSIONLEN - offset 263
055     * char uname[32];     // TarConstants.UNAMELEN   - offset 265
056     * char gname[32];     // TarConstants.GNAMELEN   - offset 297
057     * char devmajor[8];   // TarConstants.DEVLEN     - offset 329
058     * char devminor[8];   // TarConstants.DEVLEN     - offset 337
059     * char prefix[155];   // TarConstants.PREFIXLEN  - offset 345
060     * // Used if "name" field is not long enough to hold the path
061     * char pad[12];       // NULs                    - offset 500
062     * } header;
063     * All unused bytes are set to null.
064     * New-style GNU tar files are slightly different from the above.
065     * For values of size larger than 077777777777L (11 7s)
066     * or uid and gid larger than 07777777L (7 7s)
067     * the sign bit of the first byte is set, and the rest of the
068     * field is the binary representation of the number.
069     * See TarUtils.parseOctalOrBinary.
070     * </pre>
071     * <p/>
072     * <p/>
073     * The C structure for a old GNU Tar Entry's header is:
074     * <pre>
075     * struct oldgnu_header {
076     * char unused_pad1[345]; // TarConstants.PAD1LEN_GNU       - offset 0
077     * char atime[12];        // TarConstants.ATIMELEN_GNU      - offset 345
078     * char ctime[12];        // TarConstants.CTIMELEN_GNU      - offset 357
079     * char offset[12];       // TarConstants.OFFSETLEN_GNU     - offset 369
080     * char longnames[4];     // TarConstants.LONGNAMESLEN_GNU  - offset 381
081     * char unused_pad2;      // TarConstants.PAD2LEN_GNU       - offset 385
082     * struct sparse sp[4];   // TarConstants.SPARSELEN_GNU     - offset 386
083     * char isextended;       // TarConstants.ISEXTENDEDLEN_GNU - offset 482
084     * char realsize[12];     // TarConstants.REALSIZELEN_GNU   - offset 483
085     * char unused_pad[17];   // TarConstants.PAD3LEN_GNU       - offset 495
086     * };
087     * </pre>
088     * Whereas, "struct sparse" is:
089     * <pre>
090     * struct sparse {
091     * char offset[12];   // offset 0
092     * char numbytes[12]; // offset 12
093     * };
094     * </pre>
095     */

203        /**
204         * Construct an empty entry and prepares the header values.
205         */
206        public TarArchiveEntry() {
207            this.magic = MAGIC_POSIX;
208            this.version = VERSION_POSIX;
209            this.name = "";
210            this.linkName = "";
211            String user = System.getProperty("user.name", "");
212            if (user.length() > MAX_NAMELEN) {
213                user = user.substring(0, MAX_NAMELEN);
214            }
215            this.userId = 0;
216            this.groupId = 0;
217            this.userName = user;
218            this.groupName = "";
219            this.file = null;
220        }

286        /**
287         * Construct an entry for a file. File is set to file, and the
288         * header is constructed from information from the file.
289         *
290         * @param file     The file that the entry represents.
291         * @param fileName the name to be used for the entry.
292         */
293        public TarArchiveEntry(File file, String fileName) {
294            this();
295    
296            this.file = file;
297    
298            this.linkName = "";
299    
300            if (file.isDirectory()) {
301                this.mode = DEFAULT_DIR_MODE;
302                this.linkFlag = LF_DIR;
303    
304                int nameLength = fileName.length();
305                if (nameLength == 0 || fileName.charAt(nameLength - 1) != '/') {
306                    this.name = fileName + "/";
307                } else {
308                    this.name = fileName;
309                }
310                this.size = 0;
311            } else {
312                this.mode = DEFAULT_FILE_MODE;
313                this.linkFlag = LF_NORMAL;
314                this.size = file.length();
315                this.name = fileName;
316            }
317    
318            this.modTime = file.lastModified() / MILLIS_PER_SECOND;
319            this.devMajor = 0;
320            this.devMinor = 0;
321        }

765        /**
766         * If this entry represents a file, and the file is a directory, return
767         * an array of TarEntries for this entry's children.
768         *
769         * @return An array of TarEntry's for this entry's children.
770         */
771        public TarArchiveEntry[] getDirectoryEntries() {
772            if (file == null || !file.isDirectory()) {
773                return new TarArchiveEntry[0];
774            }
775    
776            String[] list = file.list();
777            TarArchiveEntry[] result = new TarArchiveEntry[list.length];
778    
779            for (int i = 0; i < list.length; ++i) {
780                result[i] = new TarArchiveEntry(new File(file, list[i]));
781            }
782    
783            return result;
784        }

786        /**
787         * Write an entry's header information to a header buffer.
788         * <p/>
789         * <p>This method does not use the star/GNU tar/BSD tar extensions.</p>
790         *
791         * @param outbuf The tar entry header buffer to fill in.
792         */
793        public void writeEntryHeader(byte[] outbuf) {
794            try {
795                writeEntryHeader(outbuf, TarUtils.DEFAULT_ENCODING, false);
796            } catch (IOException ex) {
797                try {
798                    writeEntryHeader(outbuf, TarUtils.FALLBACK_ENCODING, false);
799                } catch (IOException ex2) {
800                    // impossible
801                    throw new RuntimeException(ex2);
802                }
803            }
804        }

806        /**
807         * Write an entry's header information to a header buffer.
808         *
809         * @param outbuf   The tar entry header buffer to fill in.
810         * @param encoding encoding to use when writing the file name.
811         * @param starMode whether to use the star/GNU tar/BSD tar
812         *                 extension for numeric fields if their value doesn't fit in the
813         *                 maximum size of standard tar archives
814         */
815        public void writeEntryHeader(byte[] outbuf, ZipEncoding encoding,
816                                     boolean starMode) throws IOException {
817            int offset = 0;
818    
819            offset = TarUtils.formatNameBytes(name, outbuf, offset, NAMELEN,
820                    encoding);
821            offset = writeEntryHeaderField(mode, outbuf, offset, MODELEN, starMode);
822            offset = writeEntryHeaderField(userId, outbuf, offset, UIDLEN,
823                    starMode);
824            offset = writeEntryHeaderField(groupId, outbuf, offset, GIDLEN,
825                    starMode);
826            offset = writeEntryHeaderField(size, outbuf, offset, SIZELEN, starMode);
827            offset = writeEntryHeaderField(modTime, outbuf, offset, MODTIMELEN,
828                    starMode);
829    
830            int csOffset = offset;
831    
832            for (int c = 0; c < CHKSUMLEN; ++c) {
833                outbuf[offset++] = (byte) ' ';
834            }
835    
836            outbuf[offset++] = linkFlag;
837            offset = TarUtils.formatNameBytes(linkName, outbuf, offset, NAMELEN,
838                    encoding);
839            offset = TarUtils.formatNameBytes(magic, outbuf, offset, MAGICLEN);
840            offset = TarUtils.formatNameBytes(version, outbuf, offset, VERSIONLEN);
841            offset = TarUtils.formatNameBytes(userName, outbuf, offset, UNAMELEN,
842                    encoding);
843            offset = TarUtils.formatNameBytes(groupName, outbuf, offset, GNAMELEN,
844                    encoding);
845            offset = writeEntryHeaderField(devMajor, outbuf, offset, DEVLEN,
846                    starMode);
847            offset = writeEntryHeaderField(devMinor, outbuf, offset, DEVLEN,
848                    starMode);
849    
850            while (offset < outbuf.length) {
851                outbuf[offset++] = 0;
852            }
853    
854            long chk = TarUtils.computeCheckSum(outbuf);
855    
856            TarUtils.formatCheckSumOctalBytes(chk, outbuf, csOffset, CHKSUMLEN);
857        }
858    
859        private int writeEntryHeaderField(long value, byte[] outbuf, int offset,
860                                          int length, boolean starMode) {
861            if (!starMode && (value < 0
862                    || value >= (1l << (3 * (length - 1))))) {
863                // value doesn't fit into field when written as octal
864                // number, will be written to PAX header or causes an
865                // error
866                return TarUtils.formatLongOctalBytes(0, outbuf, offset, length);
867            }
868            return TarUtils.formatLongOctalOrBinaryBytes(value, outbuf, offset,
869                    length);
870        }

873         * Parse an entry's header information from a header buffer.
874         *
875         * @param header The tar entry header buffer to get information from.
876         * @throws IllegalArgumentException if any of the numeric fields have an invalid format
877         */
878        public void parseTarHeader(byte[] header) {
879            try {
880                parseTarHeader(header, TarUtils.DEFAULT_ENCODING);
881            } catch (IOException ex) {
882                try {
883                    parseTarHeader(header, TarUtils.DEFAULT_ENCODING, true);
884                } catch (IOException ex2) {
885                    // not really possible
886                    throw new RuntimeException(ex2);
887                }
888            }
889        }

891        /**
892         * Parse an entry's header information from a header buffer.
893         *
894         * @param header   The tar entry header buffer to get information from.
895         * @param encoding encoding to use for file names
896         * @throws IllegalArgumentException if any of the numeric fields
897         *                                  have an invalid format
898         */
899        public void parseTarHeader(byte[] header, ZipEncoding encoding)
900                throws IOException {
901            parseTarHeader(header, encoding, false);
902        }

904        private void parseTarHeader(byte[] header, ZipEncoding encoding,
905                                    final boolean oldStyle)
906                throws IOException {
907            int offset = 0;
908    
909            name = oldStyle ? TarUtils.parseName(header, offset, NAMELEN)
910                    : TarUtils.parseName(header, offset, NAMELEN, encoding);
911            offset += NAMELEN;
912            mode = (int) TarUtils.parseOctalOrBinary(header, offset, MODELEN);
913            offset += MODELEN;
914            userId = (int) TarUtils.parseOctalOrBinary(header, offset, UIDLEN);
915            offset += UIDLEN;
916            groupId = (int) TarUtils.parseOctalOrBinary(header, offset, GIDLEN);
917            offset += GIDLEN;
918            size = TarUtils.parseOctalOrBinary(header, offset, SIZELEN);
919            offset += SIZELEN;
920            modTime = TarUtils.parseOctalOrBinary(header, offset, MODTIMELEN);
921            offset += MODTIMELEN;
922            offset += CHKSUMLEN;
923            linkFlag = header[offset++];
924            linkName = oldStyle ? TarUtils.parseName(header, offset, NAMELEN)
925                    : TarUtils.parseName(header, offset, NAMELEN, encoding);
926            offset += NAMELEN;
927            magic = TarUtils.parseName(header, offset, MAGICLEN);
928            offset += MAGICLEN;
929            version = TarUtils.parseName(header, offset, VERSIONLEN);
930            offset += VERSIONLEN;
931            userName = oldStyle ? TarUtils.parseName(header, offset, UNAMELEN)
932                    : TarUtils.parseName(header, offset, UNAMELEN, encoding);
933            offset += UNAMELEN;
934            groupName = oldStyle ? TarUtils.parseName(header, offset, GNAMELEN)
935                    : TarUtils.parseName(header, offset, GNAMELEN, encoding);
936            offset += GNAMELEN;
937            devMajor = (int) TarUtils.parseOctalOrBinary(header, offset, DEVLEN);
938            offset += DEVLEN;
939            devMinor = (int) TarUtils.parseOctalOrBinary(header, offset, DEVLEN);
940            offset += DEVLEN;
941    
942            int type = evaluateType(header);
943            switch (type) {
944                case FORMAT_OLDGNU: {
945                    offset += ATIMELEN_GNU;
946                    offset += CTIMELEN_GNU;
947                    offset += OFFSETLEN_GNU;
948                    offset += LONGNAMESLEN_GNU;
949                    offset += PAD2LEN_GNU;
950                    offset += SPARSELEN_GNU;
951                    isExtended = TarUtils.parseBoolean(header, offset);
952                    offset += ISEXTENDEDLEN_GNU;
953                    realSize = TarUtils.parseOctal(header, offset, REALSIZELEN_GNU);
954                    offset += REALSIZELEN_GNU;
955                    break;
956                }
957                case FORMAT_POSIX:
958                default: {
959                    String prefix = oldStyle
960                            ? TarUtils.parseName(header, offset, PREFIXLEN)
961                            : TarUtils.parseName(header, offset, PREFIXLEN, encoding);
962                    // SunOS tar -E does not add / to directory names, so fix
963                    // up to be consistent
964                    if (isDirectory() && !name.endsWith("/")) {
965                        name = name + "/";
966                    }
967                    if (prefix.length() > 0) {
968                        name = prefix + "/" + name;
969                    }
970                }
971            }
972        }

974        /**
975         * Strips Windows' drive letter as well as any leading slashes,
976         * turns path separators into forward slahes.
977         */
978        private static String normalizeFileName(String fileName,
979                                                boolean preserveLeadingSlashes) {
980            String osname = System.getProperty("os.name").toLowerCase(Locale.ENGLISH);
981    
982            // Strip off drive letters!
983            // REVIEW Would a better check be "(File.separator == '\')"?
984    
985            if (osname.startsWith("windows")) {
986                if (fileName.length() > 2) {
987                    char ch1 = fileName.charAt(0);
988                    char ch2 = fileName.charAt(1);
989    
990                    if (ch2 == ':'
991                            && ((ch1 >= 'a' && ch1 <= 'z')
992                            || (ch1 >= 'A' && ch1 <= 'Z'))) {
993                        fileName = fileName.substring(2);
994                    }
995                }
996            } else if (osname.indexOf("netware") > -1) {
997                int colon = fileName.indexOf(':');
998                if (colon != -1) {
999                    fileName = fileName.substring(colon + 1);
1000                }
1001            }
1002    
1003            fileName = fileName.replace(File.separatorChar, '/');
1004    
1005            // No absolute pathnames
1006            // Windows (and Posix?) paths can start with "\\NetworkDrive\",
1007            // so we loop on starting /'s.
1008            while (!preserveLeadingSlashes && fileName.startsWith("/")) {
1009                fileName = fileName.substring(1);
1010            }
1011            return fileName;
1012        }

1014        /**
1015         * Evaluate an entry's header format from a header buffer.
1016         *
1017         * @param header The tar entry header buffer to evaluate the format for.
1018         * @return format type
1019         */
1020        private int evaluateType(byte[] header) {
1021            if (ArchiveUtils.matchAsciiBuffer(MAGIC_GNU, header, MAGIC_OFFSET, MAGICLEN)) {
1022                return FORMAT_OLDGNU;
1023            }
1024            if (ArchiveUtils.matchAsciiBuffer(MAGIC_POSIX, header, MAGIC_OFFSET, MAGICLEN)) {
1025                return FORMAT_POSIX;
1026            }
1027            return 0;
1028        }
