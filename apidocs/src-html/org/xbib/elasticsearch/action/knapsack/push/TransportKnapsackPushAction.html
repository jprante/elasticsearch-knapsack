069public class TransportKnapsackPushAction extends TransportAction<KnapsackPushRequest, KnapsackPushResponse> {
070
071    private final static ESLogger logger = ESLoggerFactory.getLogger(KnapsackPushAction.class.getSimpleName());
072
073    private final Environment environment;
074
075    private final Client client;
076
077    private final NodeService nodeService;
078
079    private final KnapsackService knapsack;
080
081    @Inject
082    public TransportKnapsackPushAction(Settings settings, Environment environment,
083                                       ThreadPool threadPool,
084                                       Client client, NodeService nodeService, ActionFilters actionFilters,
085                                       IndexNameExpressionResolver indexNameExpressionResolver,
086                                       KnapsackService knapsack) {
087        super(settings, KnapsackPushAction.NAME, threadPool, actionFilters, indexNameExpressionResolver);
088        this.environment = environment;
089        this.client = client;
090        this.nodeService = nodeService;
091        this.knapsack = knapsack;
092    }
093
094    @Override
095    protected void doExecute(final KnapsackPushRequest request, ActionListener<KnapsackPushResponse> listener) {
096        final KnapsackState state = new KnapsackState()
097                .setMode("push")
098                .setNodeName(nodeService.nodeName());
099        final KnapsackPushResponse response = new KnapsackPushResponse()
100                .setState(state);
101        try {
102            final BulkTransportClient bulkClient = new BulkTransportClient();
103            bulkClient.flushIngestInterval(TimeValue.timeValueSeconds(5))
104                    .maxActionsPerRequest(request.getMaxActionsPerBulkRequest())
105                    .maxConcurrentRequests(request.getMaxBulkConcurrency())
106                    .init(clientSettings(client, environment, request));
107            state.setTimestamp(new DateTime());
108            response.setRunning(true);
109            knapsack.submit(new Thread() {
110                public void run() {
111                    performPush(request, state, bulkClient);
112                }
113            });
114            // ensure to add export to state before response is sent
115            knapsack.addExport(state);
116            listener.onResponse(response);
117        } catch (Throwable e) {
118            logger.error(e.getMessage(), e);
119            listener.onFailure(e);
120        }
121    }
122
123    /**
124     * Push action thread
125     *
126     * @param request    the request
127     * @param state      the state
128     * @param bulkClient the bulk client
129     */
130    final void performPush(final KnapsackPushRequest request,
131                           final KnapsackState state,
132                           final BulkTransportClient bulkClient) {
133        try {
134            logger.info("start of push: {}", state);
135            long count = 0L;
136            Map<String, Set<String>> indices = new HashMap<>();
137            for (String s : Strings.commaDelimitedListToSet(request.getIndex())) {
138                indices.put(s, Strings.commaDelimitedListToSet(request.getType()));
139            }
140            if (request.withMetadata()) {
141                logger.info("map={}", request.getIndexTypeNames());
142                if (request.getIndexTypeNames() != null) {
143                    for (Object spec : request.getIndexTypeNames().keySet()) {
144                        if (spec == null) {
145                            continue;
146                        }
147                        String[] s = spec.toString().split("/");
148                        String index = s[0];
149                        String type = s.length > 1 ? s[1] : null;
150                        if (!"_all".equals(index)) {
151                            Set<String> types = indices.get(index);
152                            if (types == null) {
153                                types = new HashSet<>();
154                            }
155                            if (type != null) {
156                                types.add(type);
157                            }
158                            indices.put(index, types);
159                        }
160                    }
161                }
162                // get settings for all indices
163                logger.info("getting settings for indices {}", indices.keySet());
164                Set<String> settingsIndices = new HashSet<>(indices.keySet());
165                settingsIndices.remove("_all");
166                Map<String, String> settings = getSettings(client,
167                        settingsIndices.toArray(new String[settingsIndices.size()]));
168                logger.info("found indices: {}", settings.keySet());
169                // we resolved the specs in indices to the real indices in the settings
170                // get mapping and alias per index and create index if copy mode is enabled
171                for (String index : settings.keySet()) {
172                    CreateIndexRequest createIndexRequest = createIndexRequest(mapIndex(request, index));
173                    Set<String> types = indices.get(index);
174                    createIndexRequest.settings(settings.get(index));
175                    logger.info("getting mappings for index {} and types {}", index, types);
176                    Map<String, String> mappings = getMapping(client, index,
177                            types != null ? new HashSet<>(types) : null);
178                    logger.info("found mappings: {}", mappings.keySet());
179                    for (String type : mappings.keySet()) {
180                        logger.info("adding mapping: {}", mapType(request, index, type));
181                        createIndexRequest.mapping(mapType(request, index, type), mappings.get(type));
182                    }
183                    // create index
184                    logger.info("creating index: {}", mapIndex(request, index));
185                    try {
186                        bulkClient.client().execute(CreateIndexAction.INSTANCE, createIndexRequest).actionGet();
187                        logger.info("index created: {}", mapIndex(request, index));
188                    } catch (Exception e) {
189                        // maybe an index already exists exception, check if index is empty, throw exception only if not empty
190                        CountResponse countResponse =
191                                bulkClient.client().execute(CountAction.INSTANCE, new CountRequest(mapIndex(request, index))).actionGet();
192                        logger.info("count={} status={}", countResponse.getCount(), countResponse.status());
193                        if (countResponse.getCount() > 0L) {
194                            throw e;
195                        }
196                    }
197                    logger.info("getting aliases for index {}", index);
198                    Map<String, String> aliases = getAliases(client, index);
199                    logger.info("found {} aliases", aliases.size());
200                    if (!aliases.isEmpty()) {
201                        IndicesAliasesRequestBuilder requestBuilder = new IndicesAliasesRequestBuilder(client, IndicesAliasesAction.INSTANCE);
202                        for (String alias : aliases.keySet()) {
203                            if (aliases.get(alias).isEmpty()) {
204                                requestBuilder.addAlias(index, alias);
205                            } else {
206                                requestBuilder.addAlias(index, alias, aliases.get(alias)); // with filter
207                            }
208                        }
209                        requestBuilder.execute().actionGet();
210                        logger.info("aliases created", aliases.size());
211                    }
212                }
213            }
214            SearchRequest searchRequest = request.getSearchRequest();
215            if (searchRequest == null) {
216                searchRequest = new SearchRequestBuilder(client, SearchAction.INSTANCE)
217                        .setQuery(QueryBuilders.matchAllQuery()).request();
218            }
219            for (String index : indices.keySet()) {
220                searchRequest.searchType(SearchType.SCAN).scroll(request.getTimeout());
221                if (!"_all".equals(index)) {
222                    searchRequest.indices(index);
223                }
224                Set<String> types = indices.get(index);
225                if (types != null) {
226                    searchRequest.types(types.toArray(new String[types.size()]));
227                }
228                // use local node client here
229                SearchResponse searchResponse = client.execute(SearchAction.INSTANCE, searchRequest).actionGet();
230                long total = 0L;
231                while (searchResponse.getScrollId() != null && !Thread.interrupted()) {
232                    searchResponse = client.execute(SearchScrollAction.INSTANCE,
233                                new SearchScrollRequest(searchResponse.getScrollId()).scroll(request.getTimeout())).actionGet();
234                    long hits = searchResponse.getHits().getHits().length;
235                    if (hits == 0) {
236                        break;
237                    }
238                    total += hits;
239                    logger.debug("total={} hits={} took={}", total, hits, searchResponse.getTookInMillis());
240                    for (SearchHit hit : searchResponse.getHits()) {
241                        IndexRequest indexRequest = new IndexRequest(mapIndex(request, hit.getIndex()),
242                                mapType(request, hit.getIndex(), hit.getType()), hit.getId());
243                        for (String f : hit.getFields().keySet()) {
244                            switch (f) {
245                                case "_parent":
246                                    indexRequest.parent(hit.getFields().get(f).getValue().toString());
247                                    break;
248                                case "_routing":
249                                    indexRequest.routing(hit.getFields().get(f).getValue().toString());
250                                    break;
251                                case "_timestamp":
252                                    indexRequest.timestamp(hit.getFields().get(f).getValue().toString());
253                                    break;
254                                case "_version":
255                                    indexRequest.versionType(VersionType.EXTERNAL)
256                                            .version(Long.parseLong(hit.getFields().get(f).getValue().toString()));
257                                    break;
258                                case "_source":
259                                    indexRequest.source(hit.getSourceAsString());
260                                    break;
261                                default:
262                                    indexRequest.source(f, hit.getFields().get(f).getValue().toString());
263                                    break;
264                            }
265                            count++;
266                        }
267                        if (!hit.getFields().keySet().contains("_source")) {
268                            indexRequest.source(hit.getSourceAsString());
269                            count++;
270                        }
271                        bulkClient.bulkIndex(indexRequest);
272                    }
273                }
274            }
275            bulkClient.flushIngest();
276            bulkClient.waitForResponses(TimeValue.timeValueSeconds(60));
277            for (String index : indices.keySet()) {
278                bulkClient.refreshIndex(index);
279            }
280            bulkClient.shutdown();
281            logger.info("end of push: {}, count = {}", state, count);
282        } catch (Throwable e) {
283            logger.error(e.getMessage(), e);
284        } finally {
285            try {
286                knapsack.removeExport(state);
287            } catch (IOException e) {
288                logger.error(e.getMessage(), e);
289            }
290        }
291    }
