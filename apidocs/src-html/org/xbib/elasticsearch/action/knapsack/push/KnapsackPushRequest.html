016package org.xbib.elasticsearch.action.knapsack.push;
017
018import org.elasticsearch.action.ActionRequest;
019import org.elasticsearch.action.ActionRequestValidationException;
020import org.elasticsearch.action.search.SearchRequest;
021import org.elasticsearch.common.io.stream.StreamInput;
022import org.elasticsearch.common.io.stream.StreamOutput;
023import org.elasticsearch.common.unit.TimeValue;
024import org.xbib.elasticsearch.knapsack.KnapsackRequest;
025
026import java.io.IOException;
027import java.util.HashMap;
028import java.util.Map;
029
030public class KnapsackPushRequest extends ActionRequest<KnapsackPushRequest>
031        implements KnapsackRequest {
032
033    private String host;
034
035    private int port;
036
037    private String cluster;
038
039    private boolean sniff;
040
041    private TimeValue timeout;
042
043    private int limit;
044
045    private int maxActionsPerBulkRequest = 1000;
046
047    private int maxBulkConcurrency = 2 * Runtime.getRuntime().availableProcessors();
048
049    private Map<String, Object> indexTypeNames = new HashMap<>();
050
051    private boolean withMetadata;
052
053    private String index = "_all";
054
055    private String type;
056
057    private SearchRequest searchRequest;
058
059    public KnapsackPushRequest setHost(String host) {
060        this.host = host;
061        return this;
062    }
063
064    public String getHost() {
065        return host;
066    }
067
068    public KnapsackPushRequest setPort(int port) {
069        this.port = port;
070        return this;
071    }
072
073    public int getPort() {
074        return port;
075    }
076
077    public KnapsackPushRequest setCluster(String cluster) {
078        this.cluster = cluster;
079        return this;
080    }
081
082    public String getCluster() {
083        return cluster;
084    }
085
086
087    public KnapsackPushRequest setSniff(boolean sniff) {
088        this.sniff = sniff;
089        return this;
090    }
091
092    public boolean getSniff() {
093        return sniff;
094    }
095
096    public KnapsackPushRequest setIndex(String index) {
097        this.index = index;
098        return this;
099    }
100
101    public String getIndex() {
102        return index;
103    }
104
105    public KnapsackPushRequest setType(String type) {
106        this.type = type;
107        return this;
108    }
109
110    public String getType() {
111        return type;
112    }
113
114    public KnapsackPushRequest setTimeout(TimeValue timeValue) {
115        this.timeout = timeValue;
116        return this;
117    }
118
119    public TimeValue getTimeout() {
120        return timeout;
121    }
122
123    public KnapsackPushRequest setLimit(int limit) {
124        this.limit = limit;
125        return this;
126    }
127
128    public int getLimit() {
129        return limit;
130    }
131
132    public KnapsackPushRequest setMaxActionsPerBulkRequest(int maxActionsPerBulkRequest) {
133        this.maxActionsPerBulkRequest = maxActionsPerBulkRequest;
134        return this;
135    }
136
137    public int getMaxActionsPerBulkRequest() {
138        return maxActionsPerBulkRequest;
139    }
140
141    public KnapsackPushRequest setMaxBulkConcurrency(int maxBulkConcurrency) {
142        this.maxBulkConcurrency = maxBulkConcurrency;
143        return this;
144    }
145
146    public int getMaxBulkConcurrency() {
147        return maxBulkConcurrency;
148    }
149
150    public KnapsackPushRequest setIndexTypeNames(Map<String, Object> indexTypeNames) {
151        this.indexTypeNames = indexTypeNames;
152        return this;
153    }
154
155    public Map<String, Object> getIndexTypeNames() {
156        return indexTypeNames;
157    }
158
159    public KnapsackPushRequest withMetadata(boolean withMetadata) {
160        this.withMetadata = withMetadata;
161        return this;
162    }
163
164    public boolean withMetadata() {
165        return withMetadata;
166    }
167
168    public KnapsackPushRequest setSearchRequest(SearchRequest searchRequest) {
169        this.searchRequest = searchRequest;
170        return this;
171    }
172
173    public SearchRequest getSearchRequest() {
174        return searchRequest;
175    }
176
177    @Override
178    public void writeTo(StreamOutput out) throws IOException {
179        super.writeTo(out);
180        out.writeString(host);
181        out.writeInt(port);
182        out.writeString(cluster);
183        out.writeBoolean(sniff);
184        if (timeout != null) {
185            out.writeBoolean(true);
186            timeout.writeTo(out);
187        } else {
188            out.writeBoolean(false);
189        }
190        out.writeInt(limit);
191        out.writeInt(maxActionsPerBulkRequest);
192        out.writeInt(maxBulkConcurrency);
193        out.writeMap(indexTypeNames);
194        out.writeBoolean(withMetadata);
195        out.writeString(index);
196        out.writeString(type);
197        if (searchRequest != null) {
198            out.writeBoolean(true);
199            searchRequest.writeTo(out);
200        } else {
201            out.writeBoolean(false);
202        }
203    }
204
205    @Override
206    public ActionRequestValidationException validate() {
207        return null;
208    }
209
210    @Override
211    public void readFrom(StreamInput in) throws IOException {
212        super.readFrom(in);
213        host = in.readString();
214        port = in.readInt();
215        cluster = in.readString();
216        sniff = in.readBoolean();
217        if (in.readBoolean()) {
218            timeout = TimeValue.readTimeValue(in);
219        }
220        limit = in.readInt();
221        maxActionsPerBulkRequest = in.readInt();
222        maxBulkConcurrency = in.readInt();
223        indexTypeNames = in.readMap();
224        withMetadata = in.readBoolean();
225        index = in.readString();
226        type = in.readString();
227        if (in.readBoolean()) {
228            searchRequest = new SearchRequest();
229            searchRequest.readFrom(in);
230        }
231    }
232}
