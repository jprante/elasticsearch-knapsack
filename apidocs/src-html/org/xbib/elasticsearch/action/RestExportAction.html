070/**
071 * The knapsack export action performs a scan/scroll action over a user defined query
072 * and stores the result in an archive or transfers it to an ES cluster
073 */
074public class RestExportAction extends BaseRestHandler {
075
076    private final ClusterName clusterName;
077
078    private final KnapsackHelper knapsackHelper;
079
080    private final SettingsFilter settingsFilter;
081
082    @Inject
083    public RestExportAction(Settings settings,
084                            Client client,
085                            RestController controller,
086                            SettingsFilter settingsFilter,
087                            ClusterService clusterService,
088                            ClusterName clusterName) {
089        super(settings, client);
090        this.settingsFilter = settingsFilter;
091        this.clusterName = clusterName;
092        this.knapsackHelper = new KnapsackHelper(client, clusterService);
093
094        controller.registerHandler(POST, "/_export", this);
095        controller.registerHandler(POST, "/{index}/_export", this);
096        controller.registerHandler(POST, "/{index}/{type}/_export", this);
097
098        controller.registerHandler(POST, "/_export/{mode}", this);
099        controller.registerHandler(POST, "/{index}/_export/{mode}", this);
100        controller.registerHandler(POST, "/{index}/{type}/_export/{mode}", this);
101
102        controller.registerHandler(GET, "/_export/state", new RestKnapsackExportState());
103    }
104
105    @Override
106    public void handleRequest(final RestRequest request, RestChannel channel) {
107        try {
108            final XContentBuilder builder = restContentBuilder(request);
109            final boolean copy = "copy".equals(request.param("mode"));
110            if (copy) {
111                final URI destination = KnapsackHelper.getAddress(client, clusterName, request.param("host"), request.paramAsInt("port", 0),
112                        request.param("cluster"));
113                EsExecutors.daemonThreadFactory(settings, "knapsack-export-[" + destination + "]")
114                        .newThread(new ExportThread(request, destination))
115                        .start();
116                builder.startObject()
117                        .field("running", true)
118                        .field("mode", request.param("mode") != null ? request.param("mode") : "export")
119                        .field("path", destination)
120                        .endObject();
121            } else {
122                final String index = request.param("index", "_all");
123                final String type = request.param("type");
124                final String defaultSpec = index + (type != null ? "_" + type : "") + ".tar.gz";
125                final String path = request.param("path", defaultSpec);
126                URI uri = URI.create("file:" + path);
127                ConnectionService<Session<KnapsackPacket>> service = ConnectionService.getInstance();
128                ConnectionFactory<Session<KnapsackPacket>> factory = service.getConnectionFactory(uri);
129                final Connection<Session<KnapsackPacket>> connection = factory.getConnection(uri);
130                final Session<KnapsackPacket> session = connection.createSession();
131                session.open(Session.Mode.WRITE);
132                if (session.isOpen()) {
133                    EsExecutors.daemonThreadFactory(settings, "knapsack-export-[" + connection.getURI() + "]")
134                            .newThread(new ExportThread(request, connection, session))
135                            .start();
136                    builder.startObject()
137                            .field("running", true)
138                            .field("mode", request.param("mode") != null ? request.param("mode") : "export")
139                            .field("archive", factory.getName())
140                            .field("path", connection.getURI())
141                            .endObject();
142                } else {
143                    builder.startObject()
144                            .field("running", false)
145                            .endObject();
146                    channel.sendResponse(new XContentRestResponse(request, BAD_REQUEST, builder));
147                    return;
148                }
149            }
150            channel.sendResponse(new XContentRestResponse(request, OK, builder));
151        } catch (Throwable ex) {
152            try {
153                logger.error(ex.getMessage(), ex);
154                channel.sendResponse(new XContentThrowableRestResponse(request, ex));
155            } catch (Exception ex2) {
156                logger.error(ex2.getMessage(), ex2);
157            }
158        }
159    }

161    private class ExportThread extends Thread {
162
163        private final RestRequest request;
164
165        private BulkClient bulkClient;
166
167        private Connection<Session<KnapsackPacket>> connection;
168
169        private Session<KnapsackPacket> session;
170
171        private TimeValue timeout;
172
173        private URI destination;
174
175        private Map<String,Object> map;
176
177        ExportThread(RestRequest request, URI destination) {
178            this.request = request;
179            this.destination = destination;
180        }
181
182        ExportThread(RestRequest request, Connection<Session<KnapsackPacket>> connection, Session<KnapsackPacket> session) {
183            this.request = request;
184            this.connection = connection;
185            this.session = session;
186        }
187
188        @Override
189        public void run() {
190            this.bulkClient = new BulkClient();
191            logger.info("params = {}", request.params());
192            this.timeout = request.paramAsTime("timeout", TimeValue.timeValueSeconds(30L));
193            final int maxActionsPerBulkRequest = request.paramAsInt("maxActionsPerBulkRequest", 1000);
194            final int maxBulkConcurrency = request.paramAsInt("maxBulkConcurrency",
195                    Runtime.getRuntime().availableProcessors() * 2);
196            final boolean skipIndexCreation = request.paramAsBoolean("skipIndexCreation", false);
197            map = toMap(request.param("map"));
198            logger.info("index/type map = {}", map);
199            String path = connection != null ? connection.getURI().getSchemeSpecificPart() : null;
200            boolean copy = "copy".equals(request.param("mode"));
201            boolean s3 = "s3".equals(request.param("mode"));
202            final KnapsackStatus status = new KnapsackStatus()
203                    .setTimestamp(new Date())
204                    .setMap(map)
205                    .setPath(path)
206                    .setURI(destination)
207                    .setCopy(copy)
208                    .setS3(s3);
209            try {
210                logger.info("starting export: {}", status);
211                knapsackHelper.addExport(status);
212                Map<String,Set<String>> indices = newHashMap();
213                for (String s : Strings.commaDelimitedListToSet(request.param("index", "_all"))) {
214                    indices.put(s, Strings.commaDelimitedListToSet(request.param("type")));
215                }
216                if (copy) {
217                    bulkClient.flushInterval(TimeValue.timeValueSeconds(5))
218                            .maxActionsPerBulkRequest(maxActionsPerBulkRequest)
219                            .maxConcurrentBulkRequests(maxBulkConcurrency)
220                            .newClient(destination, KnapsackHelper.clientSettings(destination));
221                    logger.info("waiting for healthy cluster...");
222                    bulkClient.waitForCluster(ClusterHealthStatus.YELLOW, timeout);
223                    logger.info("... cluster is ready");
224                }
225                if (!skipIndexCreation) {
226                    if (map != null) {
227                        for (String spec : map.keySet()) {
228                            if (spec == null) {
229                                continue;
230                            }
231                            String[] s = spec.split("/");
232                            String index = s[0];
233                            String type = s.length > 1 ? s[1] : null;
234                            if (!"_all".equals(index)) {
235                                Set<String> types = indices.get(index);
236                                if (types == null) {
237                                    types = newHashSet();
238                                }
239                                if (type != null) {
240                                    types.add(type);
241                                }
242                                indices.put(index, types);
243                            }
244                        }
245                    }
246                    // get settings for all indices
247                    logger.info("getting settings for indices {}", indices.keySet());
248                    Set<String> settingsIndices = newHashSet(indices.keySet());
249                    settingsIndices.remove("_all");
250                    Map<String, String> settings = getSettings(settingsIndices.toArray(new String[settingsIndices.size()]));
251                    logger.info("found indices: {}", settings.keySet());
252                    // we resolved the specs in indices to the real indices in the settings
253                    // get mapping per index, create index
254                    for (String index : settings.keySet()) {
255                        CreateIndexRequest createIndexRequest = createIndexRequest(mapIndex(index));
256                        if (!copy) {
257                            session.write(new KnapsackPacket(mapIndex(index),
258                                    "_settings",
259                                    null,
260                                    null,
261                                    settings.get(index)));
262                        }
263                        Set<String> types = indices.get(index);
264                        createIndexRequest.settings(settings.get(index));
265                        logger.info("getting mappings for index {} and types {}", index, types);
266                        Map<String, String> mappings = getMapping(index, types != null ? ImmutableSet.copyOf(types) : null);
267                        logger.info("found mappings: {}", mappings.keySet());
268                        for (String type : mappings.keySet()) {
269                            if (!copy) {
270                                 session.write(new KnapsackPacket(mapIndex(index),
271                                         mapType(index, type),
272                                         "_mapping",
273                                         null,
274                                         mappings.get(type)));
275                            }
276                            logger.info("adding mapping: {}", mapType(index, type));
277                            createIndexRequest.mapping(mapType(index, type), mappings.get(type));
278                        }
279                        if (copy) {
280                            // create index
281                            logger.info("creating index: {}", mapIndex(index));
282                            bulkClient.client().admin().indices().create(createIndexRequest).actionGet();
283                            logger.info("index created: {}", mapIndex(index));
284                        }
285                    }
286                }
287                SearchRequest searchRequest;
288                // override size = 10 default
289                request.params().put("size", request.param("maxActionsPerBulkRequest", "1000"));
290                searchRequest = RestSearchAction.parseSearchRequest(request);
291                searchRequest.listenerThreaded(false);
292                SearchOperationThreading operationThreading =
293                        SearchOperationThreading.fromString(request.param("operation_threading"), null);
294                if (operationThreading != null) {
295                    if (operationThreading == SearchOperationThreading.NO_THREADS) {
296                        operationThreading = SearchOperationThreading.SINGLE_THREAD;
297                    }
298                    searchRequest.operationThreading(operationThreading);
299                }
300                for (String index : indices.keySet()) {
301                    searchRequest.searchType(SearchType.SCAN).scroll(timeout);
302                    if (!"_all".equals(index)) {
303                        searchRequest.indices(index);
304                    }
305                    Set<String> types = indices.get(index);
306                    if (types != null) {
307                        searchRequest.types(types.toArray(new String[types.size()]));
308                    }
309                    // use local node client
310                    SearchResponse searchResponse = client.search(searchRequest).actionGet();
311                    long total = 0L;
312                    while (searchResponse.getScrollId() != null) {
313                        searchResponse = client.prepareSearchScroll(searchResponse.getScrollId())
314                                .setScroll(timeout)
315                                .execute()
316                                .actionGet();
317                        long hits = searchResponse.getHits().getHits().length;
318                        if (hits == 0) {
319                            break;
320                        }
321                        total += hits;
322                        if (logger.isDebugEnabled()) {
323                            logger.debug("total={} hits={} took={}", total, hits, searchResponse.getTookInMillis());
324                        }
325                        for (SearchHit hit : searchResponse.getHits()) {
326                            if (copy) {
327                                IndexRequest indexRequest = new IndexRequest(mapIndex(hit.getIndex()),
328                                        mapType(hit.getIndex(), hit.getType()), hit.getId());
329                                for (String f : hit.getFields().keySet()) {
330                                    if (f.equals("_parent")) {
331                                        indexRequest.parent(hit.getFields().get(f).getValue().toString());
332                                    } else if (f.equals("_routing")) {
333                                        indexRequest.routing(hit.getFields().get(f).getValue().toString());
334                                    } else if (f.equals("_timestamp")) {
335                                        indexRequest.timestamp(hit.getFields().get(f).getValue().toString());
336                                    } else if (f.equals("_version")) {
337                                        indexRequest.versionType(VersionType.EXTERNAL)
338                                                .version(Long.parseLong(hit.getFields().get(f).getValue().toString()));
339                                    } else if (f.equals("_source")) {
340                                        indexRequest.source(hit.getSourceAsString());
341                                    } else {
342                                        indexRequest.source(f, hit.getFields().get(f).getValue().toString());
343                                    }
344                                }
345                                if (!hit.getFields().keySet().contains("_source")) {
346                                    indexRequest.source(hit.getSourceAsString());
347                                }
348                                bulkClient.index(indexRequest);
349                            } else {
350                                for (String f : hit.getFields().keySet()) {
351                                    session.write(new KnapsackPacket(mapIndex(hit.getIndex()),
352                                        mapType(hit.getIndex(), hit.getType()),
353                                        hit.getId(), f, hit.getFields().get(f).getValue().toString()));
354                                }
355                                if (!hit.getFields().keySet().contains("_source")) {
356                                    session.write(new KnapsackPacket(mapIndex(hit.getIndex()),
357                                        mapType(hit.getIndex(), hit.getType()),
358                                        hit.getId(), "_source", hit.getSourceAsString()));
359                                }
360                            }
361                        }
362                    }
363                }
364                logger.info("end of export: {}", status);
365                if (!copy) {
366                    session.close();
367                }
368                if (copy) {
369                    bulkClient.refresh().shutdown();
370                    bulkClient = null;
371                }
372                if (s3 && request.param("uri") != null) {
373                    // s3://auth:host, s3://host
374                    URI s3URI = URI.create(request.param("uri")
375                            + "?bucketName=" + request.param("bucketName")
376                            + "&key=" + request.param("key"));
377                    logger.info("trying to transfer file {} to s3 {}", path, s3URI);
378                    ConnectionService<Session<ObjectPacket>> service = ConnectionService.getInstance();
379                    ConnectionFactory<Session<ObjectPacket>> factory = service.getConnectionFactory(s3URI);
380                    final Connection<Session<ObjectPacket>> connection = factory.getConnection(s3URI);
381                    final Session<ObjectPacket> session = connection.createSession();
382                    session.open(Session.Mode.WRITE);
383                    session.write(new ObjectPacket().name(path));
384                    session.close();
385                }
386            } catch (Exception e) {
387                logger.error(e.getMessage(), e);
388            } finally {
389                try {
390                    knapsackHelper.removeExport(status);
391                } catch (IOException e) {
392                    logger.error(e.getMessage(), e);
393                }
394            }
395        }

397        private Map<String, Object> toMap(String s) {
398            if (s == null) {
399                logger.warn("no map defined");
400                return newHashMap();
401            }
402            BytesArray ref = new BytesArray(s);
403            try {
404                byte[] b = ref.hasArray() ? ref.array() : ref.toBytes();
405                XContentParser parser = XContentFactory.xContent(XContentType.JSON).createParser(b, 0, b.length);
406                return parser.mapAndClose();
407            } catch (IOException e) {
408                logger.warn("parse error while reading map", e);
409                return newHashMap();
410            }
411        }
412
413        private String mapIndex(String index) {
414            return map.containsKey(index) ? map.get(index).toString() : index;
415        }
416
417        private String mapType(String index, String type) {
418            String s = index + "/" + type;
419            return map.containsKey(s) ? map.get(s).toString() : type;
420        }
421    }

423    private Map<String, String> getSettings(String... index) throws IOException {
424        Map<String, String> settings = newHashMap();
425        ClusterStateRequestBuilder request = client.admin().cluster().prepareState()
426                .setFilterRoutingTable(true)
427                .setFilterNodes(true)
428                .setFilterIndices(index);
429        ClusterStateResponse response = request.execute().actionGet();
430        MetaData metaData = response.getState().metaData();
431        if (!metaData.getIndices().isEmpty()) {
432            // filter out the settings from the metadata
433            for (IndexMetaData indexMetaData : metaData) {
434                final XContentBuilder builder = jsonBuilder();
435                builder.startObject();
436                for (Map.Entry<String, String> entry :
437                        settingsFilter.filterSettings(indexMetaData.getSettings()).getAsMap().entrySet()) {
438                    builder.field(entry.getKey(), entry.getValue());
439                }
440                builder.endObject();
441                settings.put(indexMetaData.getIndex(), builder.string());
442            }
443        }
444        return settings;
445    }

447    private Map<String, String> getMapping(String index, Set<String> types) throws IOException {
448        Map<String, String> mappings = newHashMap();
449        ClusterStateRequestBuilder request = client.admin().cluster().prepareState()
450                .setFilterRoutingTable(true)
451                .setFilterNodes(true);
452        if (!"_all".equals(index)) {
453            request.setFilterIndices(index);
454        }
455        ClusterStateResponse response = request.execute().actionGet();
456        MetaData metaData = response.getState().getMetaData();
457        if (!metaData.getIndices().isEmpty()) {
458            for (IndexMetaData indexMetaData : metaData) {
459                for (ObjectCursor<MappingMetaData> c : indexMetaData.getMappings().values()) {
460                    MappingMetaData mappingMetaData = c.value;
461                    if (types == null || types.isEmpty() || types.contains(mappingMetaData.type())) {
462                        final XContentBuilder builder = jsonBuilder();
463                        builder.startObject();
464                        builder.field(mappingMetaData.type());
465                        builder.map(mappingMetaData.sourceAsMap());
466                        builder.endObject();
467                        mappings.put(mappingMetaData.type(), builder.string());
468                    }
469                }
470            }
471        }
472        return mappings;
473    }

475    private class RestKnapsackExportState implements RestHandler {
476
477        @Override
478        public void handleRequest(RestRequest request, RestChannel channel) {
479            try {
480                XContentBuilder builder = restContentBuilder(request);
481                builder.startObject()
482                        .field("exports")
483                        .startArray();
484                for (KnapsackStatus export : knapsackHelper.getExports()) {
485                    export.toXContent(builder, EMPTY_PARAMS);
486                }
487                builder.endArray()
488                        .endObject();
489                channel.sendResponse(new XContentRestResponse(request, OK, builder));
490            } catch (IOException ioe) {
491                try {
492                    channel.sendResponse(new XContentThrowableRestResponse(request, ioe));
493                } catch (IOException e) {
494                    logger.error("unable to send response to client");
495                }
496            }
497        }
498    }
