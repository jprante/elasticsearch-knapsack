001    
002    package org.xbib.elasticsearch.plugin.knapsack;
003    
004    import org.xbib.io.Packet;
005    import org.xbib.io.URIUtil;
006    
007    import java.io.File;
008    
009    public class KnapsackPacket implements Packet<String> {
010    
011        private String name;
012    
013        private String packet;
014    
015        public KnapsackPacket(String index, String type, String id, String field, String packet) {
016            this.name = encodeName(new String[]{index, type, id, field});
017            this.packet = packet;
018        }
019    
020        @Override
021        public KnapsackPacket name(String name) {
022            this.name = name;
023            return this;
024        }
025    
026        @Override
027        public String name() {
028            return name;
029        }
030    
031        @Override
032        public KnapsackPacket packet(String packet) {
033            this.packet = packet;
034            return this;
035        }
036    
037        @Override
038        public String packet() {
039            return packet;
040        }
041    
042        @Override
043        public String toString() {
044            return packet;
045        }
046    
047        public static String encodeName(String[] components) {
048            StringBuilder sb = new StringBuilder();
049            for (int i = 0; i < components.length; i++) {
050                if (i > 0) {
051                    sb.append(File.separator);
052                }
053                sb.append(URIUtil.encode(components[i] != null ? components[i] : "", URIUtil.UTF8));
054            }
055            return sb.toString();
056        }
057    
058        public static String[] decodeName(String component) {
059            String[] components = component.split(File.separator);
060            for (int i = 0; i < components.length; i++) {
061                components[i] = URIUtil.decode(components[i] != null ? components[i] : "", URIUtil.UTF8);
062            }
063            return components;
064        }
065    }
