002package org.xbib.elasticsearch.plugin.knapsack;
003
004import org.elasticsearch.cluster.settings.ClusterDynamicSettingsModule;
005import org.elasticsearch.plugins.AbstractPlugin;
006import org.elasticsearch.rest.RestModule;
007import org.xbib.elasticsearch.action.RestExportAction;
008import org.xbib.elasticsearch.action.RestImportAction;
009
010public class KnapsackPlugin extends AbstractPlugin {
011
012    @Override
013    public String name() {
014        return "knapsack";
015    }
016
017    @Override
018    public String description() {
019        return "Knapsack plugin";
020    }
021
022    public void onModule(RestModule module) {
023        module.addRestAction(RestExportAction.class);
024        module.addRestAction(RestImportAction.class);
025    }
026
027    public void onModule(ClusterDynamicSettingsModule module) {
028        module.addDynamicSettings(KnapsackHelper.EXPORT_STATE_SETTING_NAME);
029        module.addDynamicSettings(KnapsackHelper.IMPORT_STATE_SETTING_NAME);
030    }
031
032}
