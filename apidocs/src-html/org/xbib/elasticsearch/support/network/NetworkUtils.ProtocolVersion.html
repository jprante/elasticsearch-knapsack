016public abstract class NetworkUtils {
017
018    public static final String IPv4_SETTING = "java.net.preferIPv4Stack";
019    public static final String IPv6_SETTING = "java.net.preferIPv6Addresses";
020    private final static InetAddress localAddress;
021
022    static {
023        InetAddress address;
024        try {
025            address = InetAddress.getLocalHost();
026        } catch (Throwable e) {
027            address = InetAddress.getLoopbackAddress();
028        }
029        localAddress = address;
030    }

032    private NetworkUtils() {
033    }

035    public static InetAddress getLocalAddress() {
036        return localAddress;
037    }
038
039    public static InetAddress getFirstNonLoopbackAddress(ProtocolVersion ip_version) throws SocketException {
040        InetAddress address;
041        for (NetworkInterface networkInterface : getNetworkInterfaces()) {
042            try {
043                if (!networkInterface.isUp() || networkInterface.isLoopback()) {
044                    continue;
045                }
046            } catch (Exception e) {
047                continue;
048            }
049            address = getFirstNonLoopbackAddress(networkInterface, ip_version);
050            if (address != null) {
051                return address;
052            }
053        }
054        return null;
055    }
056
057    public static InetAddress getFirstNonLoopbackAddress(NetworkInterface networkInterface, ProtocolVersion ipVersion) throws SocketException {
058        if (networkInterface == null) {
059            throw new IllegalArgumentException("network interface is null");
060        }
061        for (Enumeration<InetAddress> addresses = networkInterface.getInetAddresses(); addresses.hasMoreElements(); ) {
062            InetAddress address = addresses.nextElement();
063            if (!address.isLoopbackAddress()) {
064                if ((address instanceof Inet4Address && ipVersion == ProtocolVersion.IPv4) ||
065                        (address instanceof Inet6Address && ipVersion == ProtocolVersion.IPv6)) {
066                    return address;
067                }
068            }
069        }
070        return null;
071    }
072
073    public static InetAddress getFirstAddress(NetworkInterface networkInterface, ProtocolVersion ipVersion) throws SocketException {
074        if (networkInterface == null) {
075            throw new IllegalArgumentException("network interface is null");
076        }
077        for (Enumeration<InetAddress> addresses = networkInterface.getInetAddresses(); addresses.hasMoreElements(); ) {
078            InetAddress address = addresses.nextElement();
079            if ((address instanceof Inet4Address && ipVersion == ProtocolVersion.IPv4) ||
080                    (address instanceof Inet6Address && ipVersion == ProtocolVersion.IPv6)) {
081                return address;
082            }
083        }
084        return null;
085    }
086
087    public static List<NetworkInterface> getAllAvailableInterfaces() throws SocketException {
088        List<NetworkInterface> allInterfaces = new ArrayList<>();
089        for (Enumeration<NetworkInterface> interfaces = NetworkInterface.getNetworkInterfaces(); interfaces.hasMoreElements(); ) {
090            NetworkInterface networkInterface = interfaces.nextElement();
091            allInterfaces.add(networkInterface);
092            Enumeration<NetworkInterface> subInterfaces = networkInterface.getSubInterfaces();
093            if (subInterfaces.hasMoreElements()) {
094                while (subInterfaces.hasMoreElements()) {
095                    allInterfaces.add(subInterfaces.nextElement());
096                }
097            }
098        }
099        sortInterfaces(allInterfaces);
100        return allInterfaces;
101    }
102
103    public static List<InetAddress> getAllAvailableAddresses() throws SocketException {
104        List<InetAddress> allAddresses = new ArrayList<>();
105        for (NetworkInterface networkInterface : getNetworkInterfaces()) {
106            Enumeration<InetAddress> addrs = networkInterface.getInetAddresses();
107            while (addrs.hasMoreElements()) {
108                allAddresses.add(addrs.nextElement());
109            }
110        }
111        sortAddresses(allAddresses);
112        return allAddresses;
113    }
114
115    public static ProtocolVersion getProtocolVersion() throws SocketException {
116        switch (findAvailableProtocols()) {
117            case IPv4:
118                return ProtocolVersion.IPv4;
119            case IPv6:
120                return ProtocolVersion.IPv6;
121            case IPv46:
122                if (Boolean.getBoolean(System.getProperty(IPv4_SETTING))) {
123                    return ProtocolVersion.IPv4;
124                }
125                if (Boolean.getBoolean(System.getProperty(IPv6_SETTING))) {
126                    return ProtocolVersion.IPv6;
127                }
128                return ProtocolVersion.IPv6;
129        }
130        return ProtocolVersion.NONE;
131    }
132
133    public static ProtocolVersion findAvailableProtocols() throws SocketException {
134        boolean hasIPv4 = false;
135        boolean hasIPv6 = false;
136        for (InetAddress addr : getAllAvailableAddresses()) {
137            if (addr instanceof Inet4Address) {
138                hasIPv4 = true;
139            }
140            if (addr instanceof Inet6Address) {
141                hasIPv6 = true;
142            }
143        }
144        if (hasIPv4 && hasIPv6) {
145            return ProtocolVersion.IPv46;
146        }
147        if (hasIPv4) {
148            return ProtocolVersion.IPv4;
149        }
150        if (hasIPv6) {
151            return ProtocolVersion.IPv6;
152        }
153        return ProtocolVersion.NONE;
154    }

156    public static InetAddress resolveInetAddress(String host, String defaultValue) throws IOException {
157        if (host == null) {
158            host = defaultValue;
159        }
160        String origHost = host;
161        int pos = host.indexOf(':');
162        if (pos > 0) {
163            host = host.substring(0, pos - 1);
164        }
165        if ((host.startsWith("#") && host.endsWith("#")) || (host.startsWith("_") && host.endsWith("_"))) {
166            host = host.substring(1, host.length() - 1);
167            if (host.equals("local")) {
168                return getLocalAddress();
169            } else if (host.startsWith("non_loopback")) {
170                if (host.toLowerCase(Locale.ROOT).endsWith(":ipv4")) {
171                    return getFirstNonLoopbackAddress(ProtocolVersion.IPv4);
172                } else if (host.toLowerCase(Locale.ROOT).endsWith(":ipv6")) {
173                    return getFirstNonLoopbackAddress(ProtocolVersion.IPv6);
174                } else {
175                    return getFirstNonLoopbackAddress(getProtocolVersion());
176                }
177            } else {
178                ProtocolVersion protocolVersion = getProtocolVersion();
179                if (host.toLowerCase(Locale.ROOT).endsWith(":ipv4")) {
180                    protocolVersion = ProtocolVersion.IPv4;
181                    host = host.substring(0, host.length() - 5);
182                } else if (host.toLowerCase(Locale.ROOT).endsWith(":ipv6")) {
183                    protocolVersion = ProtocolVersion.IPv6;
184                    host = host.substring(0, host.length() - 5);
185                }
186                for (NetworkInterface ni : getAllAvailableInterfaces()) {
187                    if (!ni.isUp()) {
188                        continue;
189                    }
190                    if (host.equals(ni.getName()) || host.equals(ni.getDisplayName())) {
191                        if (ni.isLoopback()) {
192                            return getFirstAddress(ni, protocolVersion);
193                        } else {
194                            return getFirstNonLoopbackAddress(ni, protocolVersion);
195                        }
196                    }
197                }
198            }
199            throw new IOException("failed to find network interface for [" + origHost + "]");
200        }
201        return InetAddress.getByName(host);
202    }
203
204    private static List<NetworkInterface> getNetworkInterfaces() throws SocketException {
205        List<NetworkInterface> networkInterfaces = new ArrayList<>();
206        Enumeration<NetworkInterface> interfaces = NetworkInterface.getNetworkInterfaces();
207        while (interfaces.hasMoreElements()) {
208            NetworkInterface networkInterface = interfaces.nextElement();
209            networkInterfaces.add(networkInterface);
210            Enumeration<NetworkInterface> subInterfaces = networkInterface.getSubInterfaces();
211            if (subInterfaces.hasMoreElements()) {
212                while (subInterfaces.hasMoreElements()) {
213                    networkInterfaces.add(subInterfaces.nextElement());
214                }
215            }
216        }
217        sortInterfaces(networkInterfaces);
218        return networkInterfaces;
219    }
220
221    private static void sortInterfaces(List<NetworkInterface> interfaces) {
222        Collections.sort(interfaces, new Comparator<NetworkInterface>() {
223            @Override
224            public int compare(NetworkInterface o1, NetworkInterface o2) {
225                return Integer.compare(o1.getIndex(), o2.getIndex());
226            }
227        });
228    }
229
230    private static void sortAddresses(List<InetAddress> addressList) {
231        Collections.sort(addressList, new Comparator<InetAddress>() {
232            @Override
233            public int compare(InetAddress o1, InetAddress o2) {
234                return compareBytes(o1.getAddress(), o2.getAddress());
235            }
236        });
237    }
238
239    private static int compareBytes(byte[] left, byte[] right) {
240        for (int i = 0, j = 0; i < left.length && j < right.length; i++, j++) {
241            int a = (left[i] & 0xff);
242            int b = (right[j] & 0xff);
243            if (a != b) {
244                return a - b;
245            }
246        }
247        return left.length - right.length;
248    }

250    public static enum ProtocolVersion {
251        IPv4, IPv6, IPv46, NONE
252    }
