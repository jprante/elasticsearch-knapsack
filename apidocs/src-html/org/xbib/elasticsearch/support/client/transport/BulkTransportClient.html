082    @Override
083    public BulkTransportClient init(Settings settings) throws IOException {
084        super.init(settings);
085        resetSettings();
086        BulkProcessor.Listener listener = new BulkProcessor.Listener() {
087            @Override
088            public void beforeBulk(long executionId, BulkRequest request) {
089                metric.getCurrentIngest().inc();
090                long l = metric.getCurrentIngest().count();
091                int n = request.numberOfActions();
092                metric.getSubmitted().inc(n);
093                metric.getCurrentIngestNumDocs().inc(n);
094                metric.getTotalIngestSizeInBytes().inc(request.estimatedSizeInBytes());
095                logger.debug("before bulk [{}] [actions={}] [bytes={}] [concurrent requests={}]",
096                        executionId,
097                        request.numberOfActions(),
098                        request.estimatedSizeInBytes(),
099                        l);
100            }
101
102            @Override
103            public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
104                metric.getCurrentIngest().dec();
105                long l = metric.getCurrentIngest().count();
106                metric.getSucceeded().inc(response.getItems().length);
107                metric.getTotalIngest().inc(response.getTookInMillis());
108                int n = 0;
109                for (BulkItemResponse itemResponse : response.getItems()) {
110                    if (itemResponse.isFailed()) {
111                        n++;
112                        metric.getSucceeded().dec(1);
113                        metric.getFailed().inc(1);
114                    }
115                }
116                logger.debug("after bulk [{}] [succeeded={}] [failed={}] [{}ms] [concurrent requests={}]",
117                        executionId,
118                        metric.getSucceeded().count(),
119                        metric.getFailed().count(),
120                        response.getTook().millis(),
121                        l);
122                if (n > 0) {
123                    logger.error("bulk [{}] failed with {} failed items, failure message = {}",
124                            executionId, n, response.buildFailureMessage());
125                } else {
126                    metric.getCurrentIngestNumDocs().dec(response.getItems().length);
127                }
128            }
129
130            @Override
131            public void afterBulk(long executionId, BulkRequest requst, Throwable failure) {
132                metric.getCurrentIngest().dec();
133                throwable = failure;
134                closed = true;
135                logger.error("bulk [" + executionId + "] error", failure);
136            }
137        };
138        BulkProcessor.Builder builder = BulkProcessor.builder(client, listener)
139                .setBulkActions(maxActionsPerRequest)
140                .setConcurrentRequests(maxConcurrentRequests)
141                .setFlushInterval(flushInterval);
142        if (maxVolumePerRequest != null) {
143            builder.setBulkSize(maxVolumePerRequest);
144        }
145        this.bulkProcessor = builder.build();
146        this.closed = false;
147        return this;
148    }

217    @Override
218    public BulkTransportClient index(String index, String type, String id, String source) {
219        if (closed) {
220            throw new ElasticsearchException("client is closed");
221        }
222        try {
223            metric.getCurrentIngest().inc();
224            bulkProcessor.add(new IndexRequest(index).type(type).id(id).create(false).source(source));
225        } catch (Exception e) {
226            throwable = e;
227            closed = true;
228            logger.error("bulk add of index request failed: " + e.getMessage(), e);
229        } finally {
230            metric.getCurrentIngest().dec();
231        }
232        return this;
233    }

290    @Override
291    public synchronized BulkTransportClient flushIngest() {
292        if (closed) {
293            throw new ElasticsearchException("client is closed");
294        }
295        if (client == null) {
296            logger.warn("no client");
297            return this;
298        }
299        logger.debug("flushing bulk processor");
300        BulkProcessorHelper.flush(bulkProcessor);
301        return this;
302    }
303
304    @Override
305    public synchronized BulkTransportClient waitForResponses(TimeValue maxWaitTime) throws InterruptedException {
306        if (closed) {
307            throw new ElasticsearchException("client is closed");
308        }
309        if (client == null) {
310            logger.warn("no client");
311            return this;
312        }
313        BulkProcessorHelper.waitFor(bulkProcessor, maxWaitTime);
314        return this;
315    }

333    @Override
334    public synchronized void shutdown() {
335        if (closed) {
336            super.shutdown();
337            throw new ElasticsearchException("client is closed");
338        }
339        if (client == null) {
340            logger.warn("no client");
341            return;
342        }
343        try {
344            if (bulkProcessor != null) {
345                logger.debug("closing bulk processor...");
346                bulkProcessor.close();
347            }
348            if (metric != null && metric.indices() != null && !metric.indices().isEmpty()) {
349                logger.debug("stopping bulk mode for indices {}...", metric.indices());
350                for (String index : ImmutableSet.copyOf(metric.indices())) {
351                    stopBulk(index);
352                }
353            }
354            logger.debug("shutting down...");
355            super.shutdown();
356            logger.debug("shutting down completed");
357        } catch (Exception e) {
358            logger.error(e.getMessage(), e);
359        }
360    }
