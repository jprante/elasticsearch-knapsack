001package org.xbib.elasticsearch.support.client;
002
003import org.elasticsearch.common.metrics.CounterMetric;
004import org.elasticsearch.common.metrics.MeanMetric;
005
006import java.util.HashMap;
007import java.util.HashSet;
008import java.util.Map;
009import java.util.Set;
010
011public class Metric {
012
013    private final Set<String> indexNames = new HashSet<String>();
014    private final Map<String, Long> startBulkRefreshIntervals = new HashMap<String, Long>();
015    private final Map<String, Long> stopBulkRefreshIntervals = new HashMap<String, Long>();
016    private final MeanMetric totalIngest = new MeanMetric();
017    private final CounterMetric totalIngestSizeInBytes = new CounterMetric();
018    private final CounterMetric currentIngest = new CounterMetric();
019    private final CounterMetric currentIngestNumDocs = new CounterMetric();
020    private final CounterMetric submitted = new CounterMetric();
021    private final CounterMetric succeeded = new CounterMetric();
022    private final CounterMetric failed = new CounterMetric();
023    private long started;
024
025    public MeanMetric getTotalIngest() {
026        return totalIngest;
027    }
028
029    public CounterMetric getTotalIngestSizeInBytes() {
030        return totalIngestSizeInBytes;
031    }
032
033    public CounterMetric getCurrentIngest() {
034        return currentIngest;
035    }
036
037    public CounterMetric getCurrentIngestNumDocs() {
038        return currentIngestNumDocs;
039    }
040
041    public CounterMetric getSubmitted() {
042        return submitted;
043    }
044
045    public CounterMetric getSucceeded() {
046        return succeeded;
047    }
048
049    public CounterMetric getFailed() {
050        return failed;
051    }
052
053    public Metric start() {
054        this.started = System.nanoTime();
055        return this;
056    }
057
058    public long elapsed() {
059        return System.nanoTime() - started;
060    }
061
062    public Metric setupBulk(String indexName, long startRefreshInterval, long stopRefreshInterval) {
063        synchronized (indexNames) {
064            indexNames.add(indexName);
065            startBulkRefreshIntervals.put(indexName, startRefreshInterval);
066            stopBulkRefreshIntervals.put(indexName, stopRefreshInterval);
067        }
068        return this;
069    }
070
071    public boolean isBulk(String indexName) {
072        return indexNames.contains(indexName);
073    }
074
075    public Metric removeBulk(String indexName) {
076        synchronized (indexNames) {
077            indexNames.remove(indexName);
078        }
079        return this;
080    }
081
082    public Set<String> indices() {
083        return indexNames;
084    }
085
086    public Map<String, Long> getStartBulkRefreshIntervals() {
087        return startBulkRefreshIntervals;
088    }
089
090    public Map<String, Long> getStopBulkRefreshIntervals() {
091        return stopBulkRefreshIntervals;
092    }
093
094}
