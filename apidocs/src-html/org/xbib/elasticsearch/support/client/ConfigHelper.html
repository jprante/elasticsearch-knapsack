001package org.xbib.elasticsearch.support.client;
002
003import org.elasticsearch.common.io.Streams;
004import org.elasticsearch.common.settings.Settings;
005
006import java.io.IOException;
007import java.io.InputStream;
008import java.io.InputStreamReader;
009import java.io.StringWriter;
010import java.util.HashMap;
011import java.util.Map;
012
013public class ConfigHelper {
014
015    private Settings.Builder settingsBuilder;
016
017    private Settings settings;
018
019    private Map<String, String> mappings = new HashMap<>();
020
021    public ConfigHelper reset() {
022        settingsBuilder = Settings.settingsBuilder();
023        settings = null;
024        mappings = new HashMap<>();
025        return this;
026    }
027
028    public ConfigHelper settings(Settings settings) {
029        this.settings = settings;
030        return this;
031    }
032
033    public ConfigHelper setting(String key, String value) {
034        if (settingsBuilder == null) {
035            settingsBuilder = Settings.settingsBuilder();
036        }
037        settingsBuilder.put(key, value);
038        return this;
039    }
040
041    public ConfigHelper setting(String key, Boolean value) {
042        if (settingsBuilder == null) {
043            settingsBuilder = Settings.settingsBuilder();
044        }
045        settingsBuilder.put(key, value);
046        return this;
047    }
048
049    public ConfigHelper setting(String key, Integer value) {
050        if (settingsBuilder == null) {
051            settingsBuilder = Settings.settingsBuilder();
052        }
053        settingsBuilder.put(key, value);
054        return this;
055    }
056
057    public ConfigHelper setting(InputStream in) throws IOException {
058        settingsBuilder = Settings.settingsBuilder().loadFromStream(".json", in);
059        return this;
060    }
061
062    public Settings.Builder settingsBuilder() {
063        return settingsBuilder != null ? settingsBuilder : Settings.settingsBuilder();
064    }
065
066    public Settings settings() {
067        if (settings != null) {
068            return settings;
069        }
070        if (settingsBuilder == null) {
071            settingsBuilder = Settings.settingsBuilder();
072        }
073        return settingsBuilder.build();
074    }
075
076    public ConfigHelper mapping(String type, String mapping) throws IOException {
077        mappings.put(type, mapping);
078        return this;
079    }
080
081    public ConfigHelper mapping(String type, InputStream in) throws IOException {
082        if (type == null) {
083            return this;
084        }
085        StringWriter sw = new StringWriter();
086        Streams.copy(new InputStreamReader(in), sw);
087        mappings.put(type, sw.toString());
088        return this;
089    }
090
091    public Map<String, String> mappings() {
092        return mappings.isEmpty() ? null : mappings;
093    }
094
095}
