001package org.xbib.elasticsearch.support.client;
002
003import org.elasticsearch.action.admin.cluster.state.ClusterStateRequest;
004import org.elasticsearch.action.admin.cluster.state.ClusterStateResponse;
005import org.elasticsearch.client.ElasticsearchClient;
006import org.elasticsearch.client.transport.NoNodeAvailableException;
007import org.elasticsearch.client.transport.TransportClient;
008import org.elasticsearch.cluster.node.DiscoveryNode;
009import org.elasticsearch.cluster.node.DiscoveryNodes;
010import org.elasticsearch.common.logging.ESLogger;
011import org.elasticsearch.common.logging.ESLoggerFactory;
012import org.elasticsearch.common.settings.Settings;
013import org.elasticsearch.common.transport.InetSocketTransportAddress;
014import org.xbib.elasticsearch.support.network.NetworkUtils;
015
016import java.io.IOException;
017import java.io.InputStream;
018import java.net.InetAddress;
019import java.net.UnknownHostException;
020import java.util.ArrayList;
021import java.util.Collection;
022import java.util.List;
023import java.util.Map;
024
025public abstract class BaseTransportClient {
026
027    private final static ESLogger logger = ESLoggerFactory.getLogger(BaseTransportClient.class.getName());
028
029    protected TransportClient client;
030
031    protected ConfigHelper configHelper = new ConfigHelper();
032
033    private boolean isShutdown;
034
035    protected void createClient(Map<String, String> settings) throws IOException {
036        createClient(Settings.builder().put(settings).build());
037    }
038
039    protected void createClient(Settings settings) throws IOException {
040        if (client != null) {
041            logger.warn("client is open, closing...");
042            client.close();
043            client.threadPool().shutdown();
044            logger.warn("client is closed");
045            client = null;
046        }
047        if (settings != null) {
048            String version = System.getProperty("os.name")
049                    + " " + System.getProperty("java.vm.name")
050                    + " " + System.getProperty("java.vm.vendor")
051                    + " " + System.getProperty("java.runtime.version")
052                    + " " + System.getProperty("java.vm.version");
053            logger.info("creating transport client on {} with effective settings {}",
054                    version, settings.getAsMap());
055            this.client = TransportClient.builder().settings(settings).build();
056            Collection<InetSocketTransportAddress> addrs = findAddresses(settings);
057            if (!connect(addrs, settings.getAsBoolean("autodiscover", false))) {
058                throw new NoNodeAvailableException("no cluster nodes available, check settings "
059                        + settings.getAsMap());
060            }
061        }
062    }
063
064    public ElasticsearchClient client() {
065        return client;
066    }
067
068    public synchronized void shutdown() {
069        if (client != null) {
070            logger.debug("shutdown started");
071            client.close();
072            client.threadPool().shutdown();
073            client = null;
074            logger.debug("shutdown complete");
075        }
076        isShutdown = true;
077    }
078
079    public boolean isShutdown() {
080        return isShutdown;
081    }
082
083    protected Settings findSettings() {
084        Settings.Builder settingsBuilder = Settings.settingsBuilder();
085        settingsBuilder.put("host", "localhost");
086        try {
087            String hostname = InetAddress.getLocalHost().getHostName();
088            logger.debug("the hostname is {}", hostname);
089            settingsBuilder.put("host", hostname)
090                    .put("port", 9300);
091        } catch (UnknownHostException e) {
092            logger.warn("can't resolve host name, probably something wrong with network config: " + e.getMessage(), e);
093        } catch (Exception e) {
094            logger.warn(e.getMessage(), e);
095        }
096        return settingsBuilder.build();
097    }
098
099    protected Collection<InetSocketTransportAddress> findAddresses(Settings settings) throws IOException {
100        String[] hostnames = settings.getAsArray("host", new String[]{"localhost"});
101        int port = settings.getAsInt("port", 9300);
102        Collection<InetSocketTransportAddress> addresses = new ArrayList<>();
103        for (String hostname : hostnames) {
104            String[] splitHost = hostname.split(":", 2);
105            if (splitHost.length == 2) {
106                String host = splitHost[0];
107                InetAddress inetAddress = NetworkUtils.resolveInetAddress(host, null);
108                try {
109                    port = Integer.parseInt(splitHost[1]);
110                } catch (Exception e) {
111                    // ignore
112                }
113                addresses.add(new InetSocketTransportAddress(inetAddress, port));
114            }
115            if (splitHost.length == 1) {
116                String host = splitHost[0];
117                InetAddress inetAddress = NetworkUtils.resolveInetAddress(host, null);
118                addresses.add(new InetSocketTransportAddress(inetAddress, port));
119            }
120        }
121        return addresses;
122    }
123
124    protected boolean connect(Collection<InetSocketTransportAddress> addresses, boolean autodiscover) {
125        logger.info("trying to connect to {}", addresses);
126        for (InetSocketTransportAddress address : addresses) {
127            client.addTransportAddress(address);
128        }
129        if (client.connectedNodes() != null) {
130            List<DiscoveryNode> nodes = client.connectedNodes();
131            if (!nodes.isEmpty()) {
132                logger.info("connected to {}", nodes);
133                if (autodiscover) {
134                    logger.info("trying to auto-discover all cluster nodes...");
135                    ClusterStateResponse clusterStateResponse = client.admin().cluster().state(new ClusterStateRequest()).actionGet();
136                    DiscoveryNodes discoveryNodes = clusterStateResponse.getState().getNodes();
137                    for (DiscoveryNode node : discoveryNodes) {
138                        logger.info("connecting to auto-discovered node {}", node);
139                        try {
140                            client.addTransportAddress(node.address());
141                        } catch (Exception e) {
142                            logger.warn("can't connect to node " + node, e);
143                        }
144                    }
145                    logger.info("after auto-discovery connected to {}", client.connectedNodes());
146                }
147                return true;
148            }
149            return false;
150        }
151        return false;
152    }
153
154    public Settings.Builder getSettingsBuilder() {
155        return configHelper.settingsBuilder();
156    }
157
158    public void resetSettings() {
159        configHelper.reset();
160    }
161
162    public void setting(InputStream in) throws IOException {
163        configHelper.setting(in);
164    }
165
166    public void addSetting(String key, String value) {
167        configHelper.setting(key, value);
168    }
169
170    public void addSetting(String key, Boolean value) {
171        configHelper.setting(key, value);
172    }
173
174    public void addSetting(String key, Integer value) {
175        configHelper.setting(key, value);
176    }
177
178    public void mapping(String type, String mapping) throws IOException {
179        configHelper.mapping(type, mapping);
180    }
181
182    public void mapping(String type, InputStream in) throws IOException {
183        configHelper.mapping(type, in);
184    }
185
186    public Map<String, String> getMappings() {
187        return configHelper.mappings();
188    }
189
190}
