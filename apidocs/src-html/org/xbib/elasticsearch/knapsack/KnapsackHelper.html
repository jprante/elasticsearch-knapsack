053    public static Map<String, Object> toMap(String s, ESLogger logger) throws IOException {
054        Map<String, Object> map = s == null ? new HashMap<String, Object>()
055                : XContentFactory.xContent(XContentType.JSON).createParser(s).map();
056        if (map.isEmpty() && s != null && s.length() > 0) {
057            logger.warn("can not parse map, check URI escape, got param: {}", s);
058        }
059        return map;
060    }
061
062    public static Map<String, String> getSettings(ElasticsearchClient client, String... index) throws IOException {
063        Map<String, String> settings = new HashMap<>();
064        ClusterStateRequestBuilder request = new ClusterStateRequestBuilder(client, ClusterStateAction.INSTANCE)
065                .setIndices(index);
066        ClusterStateResponse response = request.execute().actionGet();
067        MetaData metaData = response.getState().metaData();
068        if (!metaData.getIndices().isEmpty()) {
069            // filter out the settings from the metadata
070            for (IndexMetaData indexMetaData : metaData) {
071                final XContentBuilder builder = jsonBuilder();
072                indexMetaData.getSettings().toXContent(builder, ToXContent.EMPTY_PARAMS);
073                settings.put(indexMetaData.getIndex(), builder.string());
074            }
075        }
076        return settings;
077    }
078
079    public static Map<String, String> getMapping(ElasticsearchClient client, String index, Set<String> types) throws IOException {
080        Map<String, String> mappings = new HashMap<>();
081        ClusterStateRequestBuilder request = new ClusterStateRequestBuilder(client, ClusterStateAction.INSTANCE);
082        if (!"_all".equals(index)) {
083            request.setIndices(index);
084        }
085        ClusterStateResponse response = request.execute().actionGet();
086        MetaData metaData = response.getState().getMetaData();
087        if (!metaData.getIndices().isEmpty()) {
088            for (IndexMetaData indexMetaData : metaData) {
089                for (ObjectCursor<MappingMetaData> c : indexMetaData.getMappings().values()) {
090                    MappingMetaData mappingMetaData = c.value;
091                    if (types == null || types.isEmpty() || types.contains(mappingMetaData.type())) {
092                        final XContentBuilder builder = jsonBuilder();
093                        builder.startObject();
094                        builder.field(mappingMetaData.type());
095                        builder.map(mappingMetaData.sourceAsMap());
096                        builder.endObject();
097                        mappings.put(mappingMetaData.type(), builder.string());
098                    }
099                }
100            }
101        }
102        return mappings;
103    }

105    public static Map<String, String> getAliases(ElasticsearchClient client, String index) throws IOException {
106        Map<String, String> aliases = new HashMap<>();
107        ClusterStateRequestBuilder request = new ClusterStateRequestBuilder(client, ClusterStateAction.INSTANCE);
108        if (!"_all".equals(index)) {
109            request.setIndices(index);
110        }
111        ClusterStateResponse response = request.execute().actionGet();
112        MetaData metaData = response.getState().getMetaData();
113        if (!metaData.getIndices().isEmpty()) {
114            for (IndexMetaData indexMetaData : metaData) {
115                for (ObjectCursor<AliasMetaData> c : indexMetaData.getAliases().values()) {
116                    AliasMetaData aliasMetaData = c.value;
117                    String alias = aliasMetaData.getAlias();
118                    if (aliasMetaData.getFilter() != null) {
119                        aliases.put(alias, new String(aliasMetaData.getFilter().uncompressed()));
120                    } else {
121                        aliases.put(alias, ""); // empty string = no filter
122                    }
123                }
124            }
125        }
126        return aliases;
127    }

138    public static Settings clientSettings(ElasticsearchClient client, Environment environment, KnapsackRequest request) {
139        String cluster = request.getCluster();
140        String host = request.getHost();
141        int port = request.getPort();
142        if (host == null) {
143            NodesInfoResponse response = client.execute(NodesInfoAction.INSTANCE, new NodesInfoRequest().transport(true)).actionGet();
144            InetSocketTransportAddress address = (InetSocketTransportAddress) response.iterator().next()
145                    .getTransport().getAddress().publishAddress();
146            host = address.address().getAddress().getHostAddress();
147            port = address.address().getPort();
148            if (cluster == null) {
149                cluster = response.getClusterName().value();
150            }
151        }
152        if (cluster == null) {
153            NodesInfoResponse response = client.execute(NodesInfoAction.INSTANCE, new NodesInfoRequest().transport(true)).actionGet();
154            cluster = response.getClusterName().value();
155        }
156        return Settings.settingsBuilder()
157                .put("host", host)
158                .put("port", port)
159                .put("cluster.name", cluster)
160                .put("timeout", request.getTimeout()) // our timeout for instantiating TransportClient
161                .put("client.transport.sniff", request.getSniff()) // sniff = look for other nodes
162                .put("client.transport.ping_timeout", request.getTimeout()) // timeout for the transport connection
163                .put("client.transport.ignore_cluster_name", true) // we want to connect to other clusters, not ours
164                .put("path.plugins", ".dontexist") // this disables site plugins when instantiating TransportClient
165                //.classLoader(getClassLoader(environment)) // this disables all jvm plugins when instantiating TransportClient
166                .build();
167    }
