016package org.xbib.elasticsearch.knapsack;
017
018import org.elasticsearch.common.io.stream.StreamInput;
019import org.elasticsearch.common.io.stream.StreamOutput;
020import org.elasticsearch.common.io.stream.Streamable;
021import org.elasticsearch.common.joda.DateMathParser;
022import org.elasticsearch.common.joda.Joda;
023import org.elasticsearch.common.xcontent.ToXContent;
024import org.elasticsearch.common.xcontent.XContentBuilder;
025import org.elasticsearch.common.xcontent.XContentParser;
026import org.elasticsearch.common.xcontent.XContentParser.Token;
027import org.joda.time.DateTime;
028
029import java.io.IOException;
030import java.net.URI;
031import java.nio.file.Path;
032import java.nio.file.Paths;
033import java.util.Date;
034import java.util.concurrent.Callable;
035
036import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
037import static org.elasticsearch.common.xcontent.XContentParser.Token.END_OBJECT;
038import static org.elasticsearch.common.xcontent.XContentParser.Token.FIELD_NAME;
039import static org.elasticsearch.common.xcontent.XContentParser.Token.VALUE_NULL;
040
041public class KnapsackState implements Streamable, ToXContent {
042
043    /**
044     * The mode of the knapsack operation
045     */
046    private String mode;
047
048    /**
049     * The time stamp of this Knapsack operation
050     */
051    private DateTime timestamp;
052
053    /**
054     * The path of the archive involved in this Knapsack operation
055     */
056    private Path path;
057
058    /**
059     * The address URI of the Elasticsearch cluster
060     */
061    private String address;
062
063    /**
064     * The node name where the knapsack operation is executed
065     */
066    private String nodeName;
067
068    public KnapsackState() {
069    }
070
071    public KnapsackState setMode(String mode) {
072        this.mode = mode;
073        return this;
074    }
075
076    public String getMode() {
077        return mode;
078    }
079
080    public KnapsackState setTimestamp(DateTime timestamp) {
081        this.timestamp = timestamp;
082        return this;
083    }
084
085    public DateTime getTimestamp() {
086        return timestamp;
087    }
088
089    public KnapsackState setPath(Path path) {
090        this.path = path;
091        return this;
092    }
093
094    public Path getPath() {
095        return path;
096    }
097
098    public KnapsackState setClusterAddress(String address) {
099        this.address = address;
100        return this;
101    }
102
103    public String getClusterAddress() {
104        return address;
105    }
106
107
108    public KnapsackState setNodeName(String nodeName) {
109        this.nodeName = nodeName;
110        return this;
111    }
112
113    public String getNodeName() {
114        return nodeName;
115    }
116
117    private static Callable<Long> now() {
118        return new Callable<Long>() {
119            @Override
120            public Long call() {
121                return 0L;
122            }
123        };
124    }
125
126    private final static DateMathParser dateParser = new DateMathParser(Joda.forPattern("dateOptionalTime"));
127
128    public KnapsackState fromXContent(XContentParser parser) throws IOException {
129        Long startTimestamp = new Date().getTime();
130        Path path = null;
131        String address = null;
132        String nodeName = null;
133        String currentFieldName = null;
134        Token token;
135        while ((token = parser.nextToken()) != END_OBJECT) {
136            if (token == FIELD_NAME) {
137                currentFieldName = parser.currentName();
138            } else if (token.isValue() || token == VALUE_NULL) {
139                switch (currentFieldName) {
140                    case "mode":
141                        mode = parser.text();
142                        break;
143                    case "started":
144                        startTimestamp = parser.text() != null ? dateParser.parse(parser.text(), now()) : null;
145                        break;
146                    case "path":
147                        path = parser.text() != null ? Paths.get(URI.create(parser.text())) : null;
148                        break;
149                    case "cluster_address":
150                        address = parser.text();
151                        break;
152                    case "node_name":
153                        nodeName = parser.text();
154                        break;
155                }
156            }
157        }
158        return new KnapsackState()
159                .setMode(mode)
160                .setTimestamp(new DateTime(startTimestamp))
161                .setPath(path)
162                .setClusterAddress(address)
163                .setNodeName(nodeName);
164    }
165
166    @Override
167    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
168        builder.startObject()
169                .field("mode", mode);
170        if (timestamp != null) {
171            builder.field("started", timestamp);
172        }
173        if (path != null) {
174            builder.field("path", path.toUri().toString());
175        }
176        if (address != null) {
177            builder.field("cluster_address", address);
178        }
179        if (nodeName != null) {
180            builder.field("node_name", nodeName);
181        }
182        builder.endObject();
183        return builder;
184    }
185
186    public String id() {
187        StringBuilder sb = new StringBuilder();
188        sb.append(mode)
189                .append(timestamp)
190                .append(path)
191                .append(address)
192                .append(nodeName);
193        return sb.toString();
194    }
195
196    @Override
197    public int hashCode() {
198        return id().hashCode();
199    }
200
201    @Override
202    public boolean equals(Object obj) {
203        if (this == obj) {
204            return true;
205        }
206        if (obj == null) {
207            return false;
208        }
209        if (getClass() != obj.getClass()) {
210            return false;
211        }
212        KnapsackState other = (KnapsackState) obj;
213        return id().equals(other.id());
214    }
215
216    @Override
217    public void readFrom(StreamInput in) throws IOException {
218        mode = in.readString();
219        timestamp = new DateTime(in.readLong());
220        path = Paths.get(URI.create(in.readString()));
221        address = in.readString();
222        nodeName = in.readString();
223    }
224
225    @Override
226    public void writeTo(StreamOutput out) throws IOException {
227        out.writeString(mode);
228        out.writeLong(timestamp.getMillis());
229        out.writeString(path.toUri().toString());
230        out.writeString(address);
231        out.writeString(nodeName);
232    }
233
234    @Override
235    public String toString() {
236        try {
237            return toXContent(jsonBuilder(), EMPTY_PARAMS).string();
238        } catch (IOException e) {
239            // ignore
240        }
241        return "";
242    }
243}
