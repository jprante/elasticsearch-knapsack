002    package org.xbib.classloader.url;
003    
004    import org.xbib.classloader.ResourceLocation;
005    
006    import java.net.URL;
007    
008    public abstract class AbstractURLResourceLocation implements ResourceLocation {
009    
010        private final URL codeSource;
011    
012        public AbstractURLResourceLocation(URL codeSource) {
013            this.codeSource = codeSource;
014        }
015    
016        public final URL getCodeSource() {
017            return codeSource;
018        }
019    
020        public void close() {
021        }
022    
023        public final boolean equals(Object o) {
024            if (this == o) {
025                return true;
026            }
027            if (o == null || getClass() != o.getClass()) {
028                return false;
029            }
030            AbstractURLResourceLocation that = (AbstractURLResourceLocation) o;
031            return codeSource.equals(that.codeSource);
032        }
033    
034        public final int hashCode() {
035            return codeSource.hashCode();
036        }
037    
038        public final String toString() {
039            return "[" + getClass().getName() + ": " + codeSource + "]";
040        }
041    }
