002package org.xbib.classloader;
003
004import java.io.ByteArrayOutputStream;
005import java.io.IOException;
006import java.io.InputStream;
007import java.security.cert.Certificate;
008import java.util.jar.Attributes;
009import java.util.jar.Manifest;
010
011public abstract class AbstractResourceHandle implements ResourceHandle {
012
013    public byte[] getBytes() throws IOException {
014        InputStream in = getInputStream();
015        try {
016            return load(in);
017        } finally {
018            if (in != null) {
019                try {
020                    in.close();
021                } catch (Exception ignored) {
022                    // ignore
023                }
024            }
025        }
026    }
027
028    public Manifest getManifest() throws IOException {
029        return null;
030    }
031
032    public Certificate[] getCertificates() {
033        return null;
034    }
035
036    public Attributes getAttributes() throws IOException {
037        Manifest m = getManifest();
038        if (m == null) {
039            return null;
040        }
041
042        String entry = getUrl().getFile();
043        return m.getAttributes(entry);
044    }
045
046    public void close() {
047    }
048
049    public String toString() {
050        return "[" + getName() + ": " + getUrl() + "; code source: " + getCodeSourceUrl() + "]";
051    }
052
053    static byte[] load(InputStream inputStream) throws IOException {
054        try {
055            byte[] buffer = new byte[4096];
056            ByteArrayOutputStream out = new ByteArrayOutputStream();
057            for (int count = inputStream.read(buffer); count >= 0; count = inputStream.read(buffer)) {
058                out.write(buffer, 0, count);
059            }
060            return out.toByteArray();
061        } finally {
062            if (inputStream != null) {
063                try {
064                    inputStream.close();
065                } catch (Exception ignored) {
066                }
067            }
068        }
069    }
070}
