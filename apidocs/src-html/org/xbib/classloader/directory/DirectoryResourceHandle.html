002package org.xbib.classloader.directory;
003
004import org.xbib.classloader.AbstractResourceHandle;
005
006import java.io.File;
007import java.io.FileInputStream;
008import java.io.IOException;
009import java.io.InputStream;
010import java.net.MalformedURLException;
011import java.net.URL;
012import java.security.cert.Certificate;
013import java.util.jar.Attributes;
014import java.util.jar.Manifest;
015
016public class DirectoryResourceHandle extends AbstractResourceHandle {
017
018    private final String name;
019
020    private final File file;
021
022    private final Manifest manifest;
023
024    private final URL url;
025
026    private final URL codeSource;
027
028    public DirectoryResourceHandle(String name, File file, File codeSource, Manifest manifest) throws MalformedURLException {
029        this.name = name;
030        this.file = file;
031        this.codeSource = codeSource.toURI().toURL();
032        this.manifest = manifest;
033        url = file.toURI().toURL();
034    }
035
036    @Override
037    public String getName() {
038        return name;
039    }
040
041    @Override
042    public URL getUrl() {
043        return url;
044    }
045
046    @Override
047    public URL getCodeSourceUrl() {
048        return codeSource;
049    }
050
051    @Override
052    public boolean isDirectory() {
053        return file.isDirectory();
054    }
055
056    @Override
057    public InputStream getInputStream() throws IOException {
058        if (file.isDirectory()) {
059            return new EmptyInputStream();
060        }
061        return new FileInputStream(file);
062    }
063
064    @Override
065    public int getContentLength() {
066        if (file.isDirectory() || file.length() > Integer.MAX_VALUE) {
067            return -1;
068        } else {
069            return (int) file.length();
070        }
071    }
072
073    @Override
074    public Manifest getManifest() throws IOException {
075        return manifest;
076    }
077
078    @Override
079    public Attributes getAttributes() throws IOException {
080        if (manifest == null) {
081            return null;
082        }
083        return manifest.getAttributes(getName());
084    }
085
086    /**
087     * Always return null. This could be implementd by verifing the signatures
088     * in the manifest file against the actual file, but we don't need this
089     * right now.
090     *
091     * @return null
092     */
093    public Certificate[] getCertificates() {
094        return null;
095    }
096
097    static final class EmptyInputStream extends InputStream {
098
099        public int read() {
100            return -1;
101        }
102
103        public int read(byte b[]) {
104            return -1;
105        }
106
107        public int read(byte b[], int off, int len) {
108            return -1;
109        }
110
111        public long skip(long n) {
112            return 0;
113        }
114
115        public int available() {
116            return 0;
117        }
118
119        public void close() {
120        }
121
122        public synchronized void mark(int readlimit) {
123        }
124
125        public synchronized void reset() {
126        }
127
128        public boolean markSupported() {
129            return false;
130        }
131    }
132}
