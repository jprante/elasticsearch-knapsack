002    package org.xbib.classloader;
003    
004    import java.net.URL;
005    import java.util.Collection;
006    import java.util.Enumeration;
007    import java.util.Iterator;
008    import java.util.NoSuchElementException;
009    
010    public class ResourceEnumeration implements Enumeration<URL> {
011    
012        private Iterator iterator;
013    
014        private final String resourceName;
015    
016        private URL next;
017    
018        public ResourceEnumeration(Collection resourceLocations, String resourceName) {
019            this.iterator = resourceLocations.iterator();
020            this.resourceName = resourceName;
021        }
022    
023        public boolean hasMoreElements() {
024            fetchNext();
025            return (next != null);
026        }
027    
028        public URL nextElement() {
029            fetchNext();
030            // save next into a local variable and clear the next field
031            URL next = this.next;
032            this.next = null;
033            // if we didn't have a next throw an exception
034            if (next == null) {
035                throw new NoSuchElementException();
036            }
037            return next;
038        }
039    
040        private void fetchNext() {
041            if (iterator == null) {
042                return;
043            }
044            if (next != null) {
045                return;
046            }
047            try {
048                while (iterator.hasNext()) {
049                    ResourceLocation resourceLocation = (ResourceLocation) iterator.next();
050                    ResourceHandle resourceHandle = resourceLocation.getResourceHandle(resourceName);
051                    if (resourceHandle != null) {
052                        next = resourceHandle.getUrl();
053                        return;
054                    }
055                }
056                // no more elements
057                // clear the iterator so it can be GCed
058                iterator = null;
059            } catch (IllegalStateException e) {
060                // Jar file was closed... this means the resource finder was destroyed
061                // clear the iterator so it can be GCed
062                iterator = null;
063                throw e;
064            }
065        }
066    }
